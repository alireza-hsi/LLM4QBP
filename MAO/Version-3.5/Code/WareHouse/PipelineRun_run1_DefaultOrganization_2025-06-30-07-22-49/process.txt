        <branch condition='Blood glucose > 11 mmol/L' id='branch3'>
          <activity role='ED medical staff' action='Repeat blood glucose measurement' objects='patient' id='activity4'/>
        </branch>
        <branch condition='Blood glucose 4-11 mmol/L' id='branch4'>
          <activity role='ED medical staff' action='Continue standard care' objects='patient' id='activity5'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch id='branch5'>

          <exclusiveGateway id='exclusivegateway3'>

              <activity role='ED medical staff' action='Order brain CT scan' objects='patient' id='activity13-1'/>
              <activity role='ED medical staff' action='Review CT scan results' objects='patient' id='activity13-2'/>
              <activity role='ED medical staff' action='Initiate antiplatelet therapy if no contraindication' objects='patient' id='activity13-3'/>
              <exclusiveGateway id='exclusivegateway4'>
                <branch condition='Immediate access to specialized TIA/minor stroke clinic' id='branch11'>
                  <activity role='ED medical staff' action='Provide immediate care at specialized TIA/minor stroke clinic' objects='patient' id='activity14'/>
                </branch>
                <branch condition='Delay to specialized clinic poses risk' id='branch12'>
                  <activity role='ED medical staff' action='Perform evaluation and management in appropriately resourced ED; consider inpatient evaluation for stroke prevention' objects='patient' id='activity15'/>
                </branch>
                <branch condition='Medical attention sought > 48 hours from symptom onset' id='branch13'>
                  <activity role='ED medical staff' action='Refer for evaluation at specialized TIA/minor stroke clinic or outpatient clinic with stroke-prevention services within 24 hours of initial presentation' objects='patient' id='activity16'/>
                </branch>
              </exclusiveGateway>
            </branch>
            <branch condition='TIA without high-risk features' id='branch14'>
              <activity role='ED medical staff' action='Order brain CT scan' objects='patient' id='activity17-1'/>

          </exclusiveGateway>
        </branch>
        <branch id='branch9-2'>
          <inclusiveGateway id='inclusivegateway1'>
            <branch condition='Patient presents with fluctuating or crescendo hemibody motor weakness, sensory loss, or speech difficulty' id='branch15'>

            <branch condition='Patient not on antiplatelet agent at presentation and brain imaging excludes intracranial hemorrhage' id='branch16'>
              <exclusiveGateway id='exclusivegateway5'>

                </branch>
                <branch condition='Swallowing screening passed' id='branch18'>
                  <exclusiveGateway id='exclusivegateway6'>
                    <branch condition='ASA selected' id='branch19'>
                      <activity role='ED medical staff' action='Administer ASA 160 mg loading dose, then enteric coated ASA (81–325 mg) daily (maintenance: 81 mg/day)' objects='patient' id='activity21'/>

                    <branch condition='Clopidogrel selected' id='branch20'>
                      <activity role='ED medical staff' action='Administer clopidogrel 300 mg loading dose, then 75 mg/day' objects='patient' id='activity22'/>
