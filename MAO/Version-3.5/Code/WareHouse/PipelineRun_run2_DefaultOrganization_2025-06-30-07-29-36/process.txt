<process>
  <activity role='ED medical staff' action='Obtain patient history' objects='patient' id='activity1-1'/>
  <activity role='ED medical staff' action='Perform physical examination' objects='patient' id='activity1-2'/>
  <activity role='ED medical staff' action='Assess neurological status' objects='patient' id='activity1-3'/>
  <parallelGateway id='parallelgateway1'>
    <branch id='branch1'>
      <activity role='ED medical staff' action='Measure blood glucose' objects='patient' id='activity2'/>
      <exclusiveGateway id='exclusivegateway1'>
        <branch condition='Hypoglycemia detected' id='branch2'>
          <activity role='ED medical staff' action='Correct hypoglycemia immediately' objects='patient' id='activity3'/>
        </branch>
        <branch condition='First random glucose > 11 mmol/L' id='branch3'>
          <activity role='ED medical staff' action='Repeat blood glucose measurement' objects='patient' id='activity4'/>
        </branch>
        <branch condition='Glucose normal and ≤ 11 mmol/L' id='branch4'>
          <activity role='ED medical staff' action='Continue with routine care' objects='patient' id='activity5'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch id='branch5'>
      <activity role='ED medical staff' action='Evaluate body temperature as part of routine vital signs every 4 hours for first 48 hours' objects='patient' id='activity6'/>
      <exclusiveGateway id='exclusivegateway2'>
        <branch condition='Temperature > 37.5°C' id='branch6'>
          <parallelGateway id='parallelgateway2'>
            <branch id='branch7'>
              <activity role='ED medical staff' action='Initiate temperature-reducing measures' objects='patient' id='activity7'/>
            </branch>
            <branch id='branch8'>
              <activity role='ED medical staff' action='Order infection-related laboratory tests' objects='patient' id='activity8-1'/>
              <activity role='ED medical staff' action='Review laboratory test results' objects='patient' id='activity8-2'/>
            </branch>
            <branch id='branch9'>
              <activity role='ED medical staff' action='Initiate antipyretic and antimicrobial therapy as required' objects='patient' id='activity9'/>
            </branch>
          </parallelGateway>
        </branch>
        <branch condition='Temperature ≤ 37.5°C' id='branch10'>
          <activity role='ED medical staff' action='Continue with routine monitoring' objects='patient' id='activity10'/>
        </branch>
      </exclusiveGateway>
    </branch>
  </parallelGateway>
  <activity role='ED medical staff' action='Assess for OHTAC recommendations: TIA/minor stroke management' objects='patient' id='activity12'/>
  <exclusiveGateway id='exclusivegateway3'>
    <branch condition='TIA with high-risk features or minor stroke' id='branch13'>
      <activity role='ED medical staff' action='Order brain CT scan' objects='patient' id='activity13-1'/>
      <activity role='ED medical staff' action='Review CT scan results' objects='patient' id='activity13-2'/>
      <activity role='ED medical staff' action='Initiate antiplatelet therapy if not contraindicated' objects='patient' id='activity13-3'/>
      <exclusiveGateway id='exclusivegateway4'>
        <branch condition='Immediate care available at specialized TIA/minor stroke clinic' id='branch14'>
          <activity role='ED medical staff' action='Provide immediate care at specialized TIA/minor stroke clinic' objects='patient' id='activity14'/>
        </branch>
        <branch condition='Delay to specialized clinic poses risk' id='branch15'>
          <activity role='ED medical staff' action='Perform evaluation at appropriately resourced ED and consider inpatient evaluation/management for stroke prevention' objects='patient' id='activity15'/>
        </branch>
        <branch condition='Medical attention sought after 48 hours from symptom onset' id='branch16'>
          <activity role='ED medical staff' action='Refer for evaluation at specialized TIA/minor stroke clinic or outpatient clinic with stroke-prevention services within 24 hours of initial presentation' objects='patient' id='activity16'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='TIA without high-risk features' id='branch17'>
      <activity role='ED medical staff' action='Order brain CT scan' objects='patient' id='activity17-1'/>
      <activity role='ED medical staff' action='Review CT scan results' objects='patient' id='activity17-2'/>
      <activity role='ED medical staff' action='Initiate antiplatelet therapy if not contraindicated' objects='patient' id='activity17-3'/>
      <activity role='ED medical staff' action='Refer to outpatient clinic with stroke-prevention services for comprehensive evaluation and management within 1 month of symptom onset' objects='patient' id='activity18'/>
    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='exclusivegateway5'>
    <branch condition='TIA patient with fluctuating/crescendo hemibody motor weakness, sensory loss, or speech difficulty' id='branch18'>
      <activity role='ED medical staff' action='Evaluate in ED and consider for hospital admission' objects='patient' id='activity19'/>
    </branch>
    <branch condition='No fluctuating/crescendo symptoms' id='branch19'>
      <activity role='ED medical staff' action='Continue with outpatient management as indicated' objects='patient' id='activity20'/>
    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='exclusivegateway6'>
    <branch condition='Patient not on antiplatelet agent at presentation' id='branch20'>
      <exclusiveGateway id='exclusivegateway7'>
        <branch condition='Brain imaging excludes intracranial hemorrhage' id='branch21'>
          <inclusiveGateway id='inclusivegateway1'>
            <branch condition='ASA selected' id='branch22'>
              <activity role='ED medical staff' action='Administer ASA 160 mg loading dose, then enteric coated ASA (81–325 mg) daily (maintenance 81 mg/day for most)' objects='patient' id='activity21'/>
            </branch>
            <branch condition='Clopidogrel selected' id='branch23'>
              <activity role='ED medical staff' action='Administer clopidogrel 300 mg loading dose, then 75 mg/day' objects='patient' id='activity22'/>
            </branch>
            <branch condition='Extended-release dipyridamole/ASA selected' id='branch24'>
              <activity role='ED medical staff' action='Administer extended-release dipyridamole 200 mg/ASA 25 mg bid (load with ASA 160 mg first)' objects='patient' id='activity23'/>
            </branch>
          </inclusiveGateway>
        </branch>
        <branch condition='Brain imaging shows intracranial hemorrhage' id='branch25'>
          <activity role='ED medical staff' action='Do not initiate antiplatelet therapy; manage per hemorrhagic stroke protocol' objects='patient' id='activity24'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='Patient already on antiplatelet agent' id='branch26'>
      <activity role='ED medical staff' action='Continue current antiplatelet therapy as appropriate' objects='patient' id='activity25'/>
    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='exclusivegateway8'>
    <branch condition='Patient fails swallowing screening' id='branch27'>
      <activity role='ED medical staff' action='Offer rectal or gastric ASA as necessary' objects='patient' id='activity26'/>
    </branch>
    <branch condition='Patient passes swallowing screening' id='branch28'>
      <activity role='ED medical staff' action='Administer oral antiplatelet therapy as indicated' objects='patient' id='activity27'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED medical staff' action='Provide cross-continuum secondary prevention assessments and therapies (per Modules 5 and 10)' objects='patient' id='activity11'/>
</process>
