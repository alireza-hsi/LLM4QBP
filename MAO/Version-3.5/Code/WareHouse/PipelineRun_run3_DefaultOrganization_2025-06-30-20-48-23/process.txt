  <parallelGateway id='parallelgateway1'>
    <branch id='branch1'>
      <activity role='ED medical staff' action='Measure blood glucose' objects='patient' id='activity2'/>
      <exclusiveGateway id='exclusivegateway1'>
        <branch condition='Hypoglycemia detected' id='branch2'>
          <activity role='ED medical staff' action='Correct hypoglycemia immediately' objects='patient' id='activity3'/>
        </branch>
        <branch condition='First random glucose value > 11 mmol/L' id='branch3'>
          <activity role='ED medical staff' action='Repeat blood glucose measurement' objects='patient' id='activity4'/>
        </branch>
        <branch condition='Glucose normal and ≤ 11 mmol/L' id='branch4'>
          <activity role='ED medical staff' action='Proceed to next step' objects='' id='activity5'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch id='branch5'>
      <activity role='ED medical staff' action='Evaluate body temperature as part of routine vital signs every 4 hours for first 48 hours' objects='patient' id='activity6'/>

        <branch condition='Temperature > 37.5°C' id='branch6'>
          <activity role='ED medical staff' action='Initiate temperature-reducing measures' objects='patient' id='activity7'/>
          <activity role='ED medical staff' action='Investigate potential infection' objects='patient' id='activity8'/>
          <activity role='ED medical staff' action='Initiate antipyretic and antimicrobial therapy as required' objects='patient' id='activity9'/>
        </branch>
        <branch condition='Temperature ≤ 37.5°C' id='branch7'>
          <activity role='ED medical staff' action='Continue routine monitoring' objects='patient' id='activity10'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch id='branch12'>
      <activity role='ED medical staff' action='Determine TIA/minor stroke risk features and timing of presentation' objects='patient' id='activity15'/>
      <exclusiveGateway id='exclusivegateway4'>
        <branch condition='TIA with high-risk features or minor stroke, presentation ≤ 48 hours from symptom onset' id='branch13'>
          <activity role='ED medical staff' action='Order brain CT scan' objects='patient' id='activity16-1'/>
          <activity role='ED medical staff' action='Review CT scan results' objects='patient, CT scan' id='activity16-2'/>
          <activity role='ED medical staff' action='Initiate antiplatelet therapy if no contraindication and no intracranial hemorrhage' objects='patient, medication' id='activity16-3'/>
          <exclusiveGateway id='exclusivegateway5'>
            <branch condition='Immediate care available at specialized TIA/minor stroke clinic' id='branch14'>
              <activity role='ED medical staff' action='Provide immediate care at specialized TIA/minor stroke clinic' objects='patient' id='activity17'/>
            </branch>
            <branch condition='Delay to specialized clinic poses risk' id='branch15'>
              <activity role='ED medical staff' action='Perform evaluation and initial management in appropriately resourced ED' objects='patient' id='activity18-1'/>
              <activity role='ED medical staff' action='Consider inpatient evaluation for stroke prevention' objects='patient' id='activity18-2'/>
            </branch>
          </exclusiveGateway>
        </branch>
        <branch condition='TIA with high-risk features or minor stroke, presentation > 48 hours from symptom onset' id='branch16'>
          <activity role='ED medical staff' action='Refer for evaluation at specialized TIA/minor stroke clinic or outpatient clinic with stroke-prevention services within 24 hours of initial presentation' objects='patient' id='activity19'/>
        </branch>
        <branch condition='TIA without high-risk features' id='branch17'>
          <activity role='ED medical staff' action='Order brain CT scan' objects='patient' id='activity20-1'/>
          <activity role='ED medical staff' action='Review CT scan results' objects='patient, CT scan' id='activity20-2'/>
          <activity role='ED medical staff' action='Initiate antiplatelet therapy if no contraindication and no intracranial hemorrhage' objects='patient, medication' id='activity20-3'/>
          <activity role='ED medical staff' action='Refer to outpatient clinic with stroke-prevention services for comprehensive evaluation and management within 1 month of symptom onset' objects='patient' id='activity21'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch id='branch18'>
      <activity role='ED medical staff' action='Check if patient is already on antiplatelet therapy at presentation' objects='patient' id='activity22'/>
      <exclusiveGateway id='exclusivegateway6'>
        <branch condition='Not on antiplatelet therapy' id='branch19'>
          <activity role='ED medical staff' action='Start antiplatelet therapy immediately after brain imaging has excluded intracranial hemorrhage (choose ASA, clopidogrel, or dipyridamole/ASA as per protocol)' objects='patient, medication' id='activity23'/>
        </branch>
        <branch condition='Already on antiplatelet therapy' id='branch20'>
          <activity role='ED medical staff' action='Continue current antiplatelet therapy as appropriate' objects='patient, medication' id='activity24'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch id='branch21'>
      <activity role='ED medical staff' action='Offer rectal or gastric ASA as necessary (e.g., for patients who fail swallowing screening)' objects='patient, rectal/gastric ASA' id='activity25'/>
    </branch>
  </parallelGateway>
  <activity role='ED medical staff' action='Assess for fluctuating/crescendo hemibody motor weakness, sensory loss, or speech difficulty' objects='patient' id='activity12'/>
  <exclusiveGateway id='exclusivegateway3'>
