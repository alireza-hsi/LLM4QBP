<process>
  <activity role='ED medical staff' action='Obtain patient history' objects='patient' id='activity1-1'/>
  <activity role='ED medical staff' action='Perform physical examination' objects='patient' id='activity1-2'/>
  <activity role='ED medical staff' action='Assess neurological status' objects='patient' id='activity1-3'/>
  <parallelGateway id='parallelgateway1'>
    <branch id='branch1'>
      <activity role='ED medical staff' action='Measure blood glucose' objects='patient' id='activity2'/>
      <exclusiveGateway id='exclusivegateway1'>
        <branch condition='If hypoglycemia detected' id='branch2'>
          <activity role='ED medical staff' action='Correct hypoglycemia immediately' objects='patient' id='activity3'/>
        </branch>
        <branch condition='If first random glucose value > 11 mmol/L' id='branch3'>
          <activity role='ED medical staff' action='Repeat blood glucose measurement' objects='patient' id='activity4'/>
        </branch>
        <branch condition='If blood glucose normal and ≤ 11 mmol/L' id='branch4'>
          <activity role='ED medical staff' action='Continue with standard care' objects='patient' id='activity5'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch id='branch5'>
      <activity role='ED medical staff' action='Measure body temperature as part of routine vital signs every 4 hours for first 48 hours' objects='patient' id='activity6'/>
      <exclusiveGateway id='exclusivegateway2'>
        <branch condition='If temperature > 37.5°C' id='branch6'>
          <parallelGateway id='parallelgateway2'>
            <branch id='branch7'>
              <activity role='ED medical staff' action='Initiate temperature-reducing measures' objects='patient' id='activity7'/>
            </branch>
            <branch id='branch8'>
              <activity role='ED medical staff' action='Investigate potential infection' objects='patient' id='activity8'/>
            </branch>
            <branch id='branch9'>
              <activity role='ED medical staff' action='Initiate antipyretic and antimicrobial therapy as required' objects='patient' id='activity9'/>
            </branch>
          </parallelGateway>
        </branch>
        <branch condition='If temperature ≤ 37.5°C' id='branch10'>
          <activity role='ED medical staff' action='Continue routine monitoring' objects='patient' id='activity10'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch id='branch12'>
      <exclusiveGateway id='exclusivegateway3'>
        <branch condition='If TIA with high-risk features or minor stroke' id='branch13'>
          <activity role='ED medical staff' action='Order brain CT scan' objects='patient' id='activity12-1'/>
          <activity role='ED medical staff' action='Review CT scan results' objects='patient' id='activity12-2'/>
          <activity role='ED medical staff' action='Initiate antiplatelet therapy if no contraindication' objects='patient' id='activity12-3'/>
          <exclusiveGateway id='exclusivegateway4'>
            <branch condition='If immediate care available at specialized TIA/minor stroke clinic' id='branch14'>
              <activity role='ED medical staff' action='Refer patient to specialized TIA/minor stroke clinic for rapid evaluation and management' objects='patient' id='activity13'/>
            </branch>
            <branch condition='If delay to specialized clinic poses risk' id='branch15'>
              <activity role='ED medical staff' action='Perform evaluation in ED' objects='patient' id='activity14-1'/>
              <activity role='ED medical staff' action='Consider inpatient evaluation for stroke prevention' objects='patient' id='activity14-2'/>
            </branch>
            <branch condition='If medical attention sought after 48 hours from symptom onset' id='branch16'>
              <activity role='ED medical staff' action='Refer for evaluation at specialized TIA/minor stroke clinic or outpatient clinic with stroke-prevention services within 24 hours of initial presentation' objects='patient' id='activity15'/>
            </branch>
          </exclusiveGateway>
        </branch>
        <branch condition='If TIA without high-risk features' id='branch17'>
          <activity role='ED medical staff' action='Order brain CT scan' objects='patient' id='activity16-1'/>
          <activity role='ED medical staff' action='Review CT scan results' objects='patient' id='activity16-2'/>
          <activity role='ED medical staff' action='Initiate antiplatelet therapy if no contraindication' objects='patient' id='activity16-3'/>
          <activity role='ED medical staff' action='Refer to outpatient clinic with stroke-prevention services for comprehensive evaluation and management within 1 month of symptom onset' objects='patient' id='activity17'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch id='branch25'>
      <inclusiveGateway id='inclusivegateway1'>
        <branch condition='If TIA patient presents with fluctuating/crescendo hemibody motor weakness, sensory loss, or speech difficulty' id='branch18'>
          <activity role='ED medical staff' action='Evaluate in ED and consider for hospital admission' objects='patient' id='activity18'/>
        </branch>
        <branch condition='If patient with TIA or minor stroke not on antiplatelet agent at presentation and brain imaging excludes intracranial hemorrhage' id='branch19'>
          <exclusiveGateway id='exclusivegateway5'>
            <branch condition='If patient can swallow' id='branch20'>
              <inclusiveGateway id='inclusivegateway2'>
                <branch condition='If ASA selected' id='branch21'>
                  <activity role='ED medical staff' action='Administer ASA 160 mg loading dose, then enteric coated ASA 81–325 mg daily (maintenance 81 mg/day)' objects='patient' id='activity19'/>
                </branch>
                <branch condition='If clopidogrel selected' id='branch22'>
                  <activity role='ED medical staff' action='Administer clopidogrel 300 mg loading dose, then 75 mg/day' objects='patient' id='activity20'/>
                </branch>
                <branch condition='If extended-release dipyridamole/ASA selected' id='branch23'>
                  <activity role='ED medical staff' action='Administer extended-release dipyridamole 200 mg/ASA 25 mg bid (load with ASA 160 mg first)' objects='patient' id='activity21'/>
                </branch>
              </inclusiveGateway>
            </branch>
            <branch condition='If patient fails swallowing screening' id='branch24'>
              <activity role='ED medical staff' action='Offer rectal or gastric ASA as necessary' objects='patient' id='activity22'/>
            </branch>
          </exclusiveGateway>
        </branch>
      </inclusiveGateway>
    </branch>
  </parallelGateway>
  <activity role='ED medical staff' action='Provide cross-continuum secondary prevention assessments and therapies (per Modules 5 and 10)' objects='patient' id='activity11'/>
</process>
