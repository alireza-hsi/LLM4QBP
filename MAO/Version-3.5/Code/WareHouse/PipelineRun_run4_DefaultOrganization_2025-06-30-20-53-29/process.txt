<process>
  <activity role='ED medical staff' action='Obtain patient history' objects='patient' id='activity1-1'/>
  <activity role='ED medical staff' action='Perform physical examination' objects='patient' id='activity1-2'/>
  <activity role='ED medical staff' action='Order initial investigations' objects='patient' id='activity1-3'/>
  <parallelGateway id='parallelgateway1'>
    <branch id='branch1'>
      <activity role='ED medical staff' action='Measure blood glucose' objects='patient' id='activity2'/>
      <exclusiveGateway id='exclusivegateway1'>
        <branch condition='If hypoglycemia detected' id='branch2'>
          <activity role='ED medical staff' action='Correct hypoglycemia immediately' objects='patient' id='activity3'/>
        </branch>
        <branch condition='If blood glucose > 11 mmol/L' id='branch3'>
          <activity role='ED medical staff' action='Repeat blood glucose measurement' objects='patient' id='activity4'/>
        </branch>
        <branch condition='If blood glucose normal (no hypoglycemia, ≤ 11 mmol/L)' id='branch4'>
          <activity role='ED medical staff' action='Continue with standard care' objects='patient' id='activity5'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch id='branch5'>
      <activity role='ED medical staff' action='Evaluate body temperature as part of routine vital signs every 4 hours for first 48 hours' objects='patient' id='activity6'/>
      <exclusiveGateway id='exclusivegateway2'>
        <branch condition='If temperature > 37.5°C' id='branch6'>
          <activity role='ED medical staff' action='Initiate temperature-reducing measures' objects='patient' id='activity7'/>
          <parallelGateway id='parallelgateway2'>
            <branch id='branch6-1'>
              <activity role='ED medical staff' action='Order infection-related laboratory tests' objects='patient' id='activity8-1'/>
            </branch>
            <branch id='branch6-2'>
              <activity role='ED medical staff' action='Perform clinical assessment for infection' objects='patient' id='activity8-2'/>
            </branch>
          </parallelGateway>
          <activity role='ED medical staff' action='Initiate antipyretic and antimicrobial therapy as required' objects='patient' id='activity9'/>
        </branch>
        <branch condition='If temperature ≤ 37.5°C' id='branch7'>
          <activity role='ED medical staff' action='Continue with standard care' objects='patient' id='activity10'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch id='branch9'>
      <activity role='ED medical staff' action='Assess for fluctuating or crescendo hemibody motor weakness, sensory loss, or speech difficulty' objects='patient' id='activity12'/>
      <exclusiveGateway id='exclusivegateway3'>
        <branch condition='If fluctuating/crescendo symptoms present' id='branch10'>
          <activity role='ED medical staff' action='Evaluate in ED for possible hospital admission' objects='patient' id='activity13'/>
        </branch>
        <branch condition='If fluctuating/crescendo symptoms absent' id='branch11'>
          <activity role='ED medical staff' action='Continue with outpatient pathway' objects='patient' id='activity14'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch id='branch12'>
      <activity role='ED medical staff' action='Determine TIA risk features and symptom timing' objects='patient' id='activity15'/>
      <exclusiveGateway id='exclusivegateway4'>
        <branch condition='TIA with high-risk features or minor stroke, and presentation ≤ 48 hours from symptom onset' id='branch13'>
          <activity role='ED medical staff' action='Order brain CT scan' objects='patient' id='activity16-1'/>
          <activity role='ED medical staff' action='Review CT scan for intracranial hemorrhage' objects='patient' id='activity16-2'/>
          <activity role='ED medical staff' action='Initiate antiplatelet therapy if no contraindication' objects='patient' id='activity16-3'/>
          <activity role='ED medical staff' action='Refer for immediate care at specialized TIA/minor stroke clinic; if delay, evaluate in ED and consider inpatient management' objects='patient' id='activity17'/>
        </branch>
        <branch condition='TIA with high-risk features or minor stroke, and presentation > 48 hours from symptom onset' id='branch14'>
          <activity role='ED medical staff' action='Refer for evaluation at specialized TIA/minor stroke clinic or outpatient clinic with stroke-prevention services within 24 hours of initial presentation' objects='patient' id='activity18'/>
        </branch>
        <branch condition='TIA without high-risk features' id='branch15'>
          <activity role='ED medical staff' action='Order brain CT scan' objects='patient' id='activity19-1'/>
          <activity role='ED medical staff' action='Review CT scan for intracranial hemorrhage' objects='patient' id='activity19-2'/>
          <activity role='ED medical staff' action='Initiate antiplatelet therapy if no contraindication' objects='patient' id='activity19-3'/>
          <activity role='ED medical staff' action='Refer to outpatient clinic with stroke-prevention services for comprehensive evaluation and management within 1 month of symptom onset' objects='patient' id='activity20'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch id='branch16'>
      <activity role='ED medical staff' action='Check if patient is on antiplatelet agent at presentation' objects='patient' id='activity21'/>
      <exclusiveGateway id='exclusivegateway5'>
        <branch condition='If not on antiplatelet agent and intracranial hemorrhage excluded by brain imaging' id='branch17'>
          <inclusiveGateway id='inclusivegateway1'>
            <branch condition='ASA selected' id='branch18'>
              <activity role='ED medical staff' action='Administer ASA 160 mg loading dose, then enteric coated ASA (81–325 mg) daily (maintenance: 81 mg/day)' objects='patient' id='activity22'/>
            </branch>
            <branch condition='Clopidogrel selected' id='branch19'>
              <activity role='ED medical staff' action='Administer clopidogrel 300 mg loading dose, then 75 mg/day' objects='patient' id='activity23'/>
            </branch>
            <branch condition='Extended-release dipyridamole/ASA selected' id='branch20'>
              <activity role='ED medical staff' action='Administer extended-release dipyridamole 200 mg/ASA 25 mg bid (load with ASA 160 mg first)' objects='patient' id='activity24'/>
            </branch>
          </inclusiveGateway>
        </branch>
        <branch condition='If already on antiplatelet agent or contraindicated' id='branch21'>
          <activity role='ED medical staff' action='Continue or adjust antiplatelet therapy as clinically indicated' objects='patient' id='activity25'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch id='branch22'>
      <activity role='ED medical staff' action='Offer rectal or gastric ASA as necessary (e.g., for patients who fail swallowing screening)' objects='patient' id='activity26'/>
    </branch>
  </parallelGateway>
  <activity role='ED medical staff' action='Provide cross-continuum secondary prevention assessments and therapies (per Modules 5 and 10), regardless of admission or discharge' objects='patient' id='activity11'/>
</process>
