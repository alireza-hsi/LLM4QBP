<process>
  <activity role='ED medical staff' action='Initial assessment of patient with suspected TIA or minor (nondisabling) stroke' objects='patient' id='activity1'/>
  <parallelGateway id='parallelgateway1'>
    <branch id='branch1'>
      <activity role='ED medical staff' action='Measure blood glucose' objects='patient' id='activity2'/>
    </branch>
    <branch id='branch2'>
      <activity role='ED medical staff' action='Evaluate body temperature as part of routine vital signs every 4 hours for first 48 hours' objects='patient' id='activity3'/>
    </branch>
  </parallelGateway>
  <exclusiveGateway id='exclusivegateway1'>
    <branch condition='Hypoglycemia detected' id='branch3'>
      <activity role='ED medical staff' action='Correct hypoglycemia immediately' objects='patient' id='activity4'/>
      <activity role='ED medical staff' action='Administer intravenous glucose' objects='patient' id='activity5'/>
      <activity role='ED medical staff' action='Monitor patient response' objects='patient' id='activity6'/>
    </branch>
    <branch condition='First random glucose value > 11 mmol/L' id='branch4'>
      <activity role='ED medical staff' action='Repeat blood glucose measurement' objects='patient' id='activity7'/>
    </branch>
    <branch condition='No hypoglycemia and glucose ≤ 11 mmol/L' id='branch5'>
      <activity role='ED medical staff' action='Proceed to next assessment' objects='patient' id='activity8'/>
    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='exclusivegateway2'>
    <branch condition='Temperature > 37.5°C' id='branch6'>
      <activity role='ED medical staff' action='Initiate temperature-reducing measures' objects='patient' id='activity9'/>
      <activity role='ED medical staff' action='Investigate potential infection' objects='patient' id='activity10'/>
      <parallelGateway id='parallelgateway2'>
        <branch id='branch7'>
          <activity role='ED medical staff' action='Obtain blood cultures' objects='patient' id='activity11'/>
        </branch>
        <branch id='branch8'>
          <activity role='ED medical staff' action='Order chest X-ray and urinalysis' objects='patient' id='activity12'/>
        </branch>
      </parallelGateway>
      <exclusiveGateway id='exclusivegateway3'>
        <branch condition='Fever only' id='branch9'>
          <activity role='ED medical staff' action='Administer antipyretic therapy' objects='patient' id='activity13'/>
        </branch>
        <branch condition='Infection confirmed' id='branch10'>
          <activity role='ED medical staff' action='Administer antimicrobial therapy' objects='patient' id='activity14'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='Temperature ≤ 37.5°C' id='branch11'>
      <activity role='ED medical staff' action='Continue routine monitoring' objects='patient' id='activity15'/>
    </branch>
  </exclusiveGateway>
  <parallelGateway id='parallelgateway3'>
    <branch id='branch9a'>
      <activity role='ED medical staff' action='Assess for fluctuating or crescendo hemibody motor weakness, sensory loss, or speech difficulty' objects='patient' id='activity16'/>
      <exclusiveGateway id='exclusivegateway4'>
        <branch condition='Fluctuating/crescendo symptoms present' id='branch12'>
          <activity role='ED medical staff' action='Evaluate in ED for possible hospital admission' objects='patient' id='activity17'/>
        </branch>
        <branch condition='No fluctuating/crescendo symptoms' id='branch13'>
          <activity role='ED medical staff' action='Proceed to risk stratification and imaging' objects='patient' id='activity18'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch id='branch9b'>
      <activity role='ED medical staff' action='Determine TIA/minor stroke risk features and timing of presentation' objects='patient' id='activity19'/>
      <exclusiveGateway id='exclusivegateway5'>
        <branch condition='TIA with high-risk features or minor stroke, presentation ≤ 48h from symptom onset' id='branch14'>
          <activity role='ED medical staff' action='Order brain CT scan' objects='patient' id='activity20'/>
          <activity role='ED medical staff' action='Wait for imaging to exclude intracranial hemorrhage' objects='patient, brain imaging' id='activity21'/>
          <exclusiveGateway id='exclusivegateway6'>
            <branch condition='No contraindication to antiplatelet therapy' id='branch15'>
              <activity role='ED medical staff' action='Initiate antiplatelet therapy as soon as possible and no later than 24 hours after symptom onset' objects='patient' id='activity22'/>
            </branch>
            <branch condition='Contraindication to antiplatelet therapy' id='branch16'>
              <activity role='ED medical staff' action='Do not initiate antiplatelet therapy' objects='patient' id='activity23'/>
            </branch>
          </exclusiveGateway>
          <exclusiveGateway id='exclusivegateway7'>
            <branch condition='Immediate access to specialized TIA/minor stroke clinic' id='branch17'>
              <activity role='ED medical staff' action='Provide immediate care at specialized TIA/minor stroke clinic' objects='patient' id='activity24'/>
            </branch>
            <branch condition='Delay to specialized clinic poses risk' id='branch18'>
              <activity role='ED medical staff' action='Evaluate in appropriately resourced ED and consider inpatient evaluation for stroke prevention' objects='patient' id='activity25'/>
            </branch>
          </exclusiveGateway>
        </branch>
        <branch condition='TIA with high-risk features or minor stroke, presentation > 48h from symptom onset' id='branch19'>
          <activity role='ED medical staff' action='Refer for evaluation at specialized TIA/minor stroke clinic or outpatient clinic with stroke-prevention services within 24 hours of initial presentation' objects='patient' id='activity26'/>
        </branch>
        <branch condition='TIA without high-risk features' id='branch20'>
          <activity role='ED medical staff' action='Order brain CT scan' objects='patient' id='activity27'/>
          <activity role='ED medical staff' action='Wait for imaging to exclude intracranial hemorrhage' objects='patient, brain imaging' id='activity28'/>
          <exclusiveGateway id='exclusivegateway8'>
            <branch condition='No contraindication to antiplatelet therapy' id='branch21'>
              <activity role='ED medical staff' action='Initiate antiplatelet therapy as soon as possible and no later than 24 hours after initial presentation' objects='patient' id='activity29'/>
            </branch>
            <branch condition='Contraindication to antiplatelet therapy' id='branch22'>
              <activity role='ED medical staff' action='Do not initiate antiplatelet therapy' objects='patient' id='activity30'/>
            </branch>
          </exclusiveGateway>
          <activity role='ED medical staff' action='Refer to outpatient clinic with stroke-prevention services for comprehensive evaluation and management within 1 month of symptom onset' objects='patient' id='activity31'/>
        </branch>
      </exclusiveGateway>
    </branch>
  </parallelGateway>
  <activity role='ED medical staff' action='Assess if patient is already on antiplatelet therapy at presentation' objects='patient' id='activity32'/>
  <exclusiveGateway id='exclusivegateway9'>
    <branch condition='Not on antiplatelet therapy' id='branch23'>
      <activity role='ED medical staff' action='Choose antiplatelet agent (ASA, clopidogrel, or dipyridamole/ASA) according to protocol' objects='patient' id='activity33'/>
      <activity role='ED medical staff' action='Administer selected antiplatelet agent' objects='patient' id='activity34'/>
    </branch>
    <branch condition='Already on antiplatelet therapy' id='branch24'>
      <activity role='ED medical staff' action='Continue current antiplatelet therapy as appropriate' objects='patient' id='activity35'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED medical staff' action='Screen for swallowing ability' objects='patient' id='activity36'/>
  <exclusiveGateway id='exclusivegateway10'>
    <branch condition='Swallowing screening failed' id='branch25'>
      <activity role='ED medical staff' action='Offer rectal or gastric ASA as necessary' objects='patient' id='activity37'/>
    </branch>
    <branch condition='Swallowing screening passed' id='branch26'>
      <activity role='ED medical staff' action='Continue with oral medications as appropriate' objects='patient' id='activity38'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED medical staff' action='Initiate cross-continuum secondary prevention assessments and therapies (per Modules 5 and 10)' objects='patient' id='activity39'/>
</process>
