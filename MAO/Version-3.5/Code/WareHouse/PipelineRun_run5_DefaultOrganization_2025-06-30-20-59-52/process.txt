<process>
  <activity role='ED medical staff' action='Obtain patient history' objects='patient' id='activity1-1'/>
  <activity role='ED medical staff' action='Perform physical examination' objects='patient' id='activity1-2'/>
  <activity role='ED medical staff' action='Assess neurological status' objects='patient' id='activity1-3'/>
  <parallelGateway id='parallelgateway1'>
    <branch id='branch1'>
      <activity role='ED medical staff' action='Measure blood glucose' objects='patient, glucometer' id='activity2'/>
      <exclusiveGateway id='exclusivegateway1'>
        <branch condition='Blood glucose < normal range (hypoglycemia)' id='branch2'>
          <activity role='ED medical staff' action='Correct hypoglycemia immediately' objects='patient, glucose' id='activity3'/>
        </branch>
        <branch condition='Blood glucose > 11 mmol/L' id='branch3'>
          <activity role='ED medical staff' action='Repeat blood glucose measurement' objects='patient, glucometer' id='activity4'/>
        </branch>
        <branch condition='Blood glucose within normal range' id='branch4'>
          <activity role='ED medical staff' action='Continue with standard care' objects='patient' id='activity5'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch id='branch5'>
      <activity role='ED medical staff' action='Evaluate body temperature every 4 hours for first 48 hours' objects='patient, thermometer' id='activity6'/>
      <exclusiveGateway id='exclusivegateway2'>
        <branch condition='Temperature > 37.5°C' id='branch6'>
          <parallelGateway id='parallelgateway2'>
            <branch id='branch7'>
              <activity role='ED medical staff' action='Initiate temperature-reducing measures' objects='patient, cooling methods' id='activity7'/>
            </branch>
            <branch id='branch8'>
              <parallelGateway id='parallelgateway3'>
                <branch id='branch8-1'>
                  <activity role='ED medical staff' action='Order laboratory tests for infection' objects='patient, lab requisition' id='activity8-1'/>
                </branch>
                <branch id='branch8-2'>
                  <activity role='ED medical staff' action='Order imaging studies if indicated' objects='patient, imaging requisition' id='activity8-2'/>
                </branch>
              </parallelGateway>
              <activity role='ED medical staff' action='Initiate antipyretic and antimicrobial therapy as required' objects='patient, medications' id='activity9'/>
            </branch>
          </parallelGateway>
        </branch>
        <branch condition='Temperature ≤ 37.5°C' id='branch9'>
          <activity role='ED medical staff' action='Continue routine monitoring' objects='patient' id='activity10'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch id='branch11'>
      <exclusiveGateway id='exclusivegateway3'>
        <branch condition='TIA with high-risk features or minor stroke' id='branch12'>
          <activity role='ED medical staff' action='Arrange brain CT scan' objects='patient, CT scanner' id='activity12-1'/>
          <activity role='ED medical staff' action='Review CT scan results' objects='patient, CT results' id='activity12-2'/>
          <activity role='ED medical staff' action='Initiate antiplatelet therapy if no contraindication' objects='patient, antiplatelet agents' id='activity12-3'/>
          <exclusiveGateway id='exclusivegateway4'>
            <branch condition='Immediate access to specialized TIA/minor stroke clinic' id='branch13'>
              <activity role='ED medical staff' action='Provide immediate care at specialized TIA/minor stroke clinic' objects='patient, clinic resources' id='activity13'/>
            </branch>
            <branch condition='Delay to specialized clinic poses risk' id='branch14'>
              <activity role='ED medical staff' action='Perform evaluation in appropriately resourced ED and consider inpatient evaluation/management for stroke prevention' objects='patient, ED resources' id='activity14'/>
            </branch>
            <branch condition='Medical attention sought after 48 hours from symptom onset' id='branch15'>
              <activity role='ED medical staff' action='Refer for evaluation at specialized TIA/minor stroke clinic or outpatient clinic with stroke-prevention services within 24 hours of initial presentation' objects='patient, referral' id='activity15'/>
            </branch>
          </exclusiveGateway>
        </branch>
        <branch condition='TIA without high-risk features' id='branch16'>
          <activity role='ED medical staff' action='Arrange brain CT scan' objects='patient, CT scanner' id='activity16-1'/>
          <activity role='ED medical staff' action='Review CT scan results' objects='patient, CT results' id='activity16-2'/>
          <activity role='ED medical staff' action='Initiate antiplatelet therapy if no contraindication' objects='patient, antiplatelet agents' id='activity16-3'/>
          <activity role='ED medical staff' action='Refer to outpatient clinic with stroke-prevention services for comprehensive evaluation and management within 1 month of symptom onset' objects='patient, referral' id='activity17'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch id='branch21'>
      <inclusiveGateway id='inclusivegateway1'>
        <branch condition='Patient presents with fluctuating or crescendo hemibody motor weakness, sensory loss, or speech difficulty' id='branch17'>
          <activity role='ED medical staff' action='Evaluate in ED for consideration of hospital admission' objects='patient' id='activity18'/>
        </branch>
        <branch condition='Patient not on antiplatelet agent at presentation and brain imaging excludes intracranial hemorrhage' id='branch18'>
          <exclusiveGateway id='exclusivegateway5'>
            <branch condition='Patient can swallow' id='branch19'>
              <activity role='ED medical staff' action='Start antiplatelet therapy immediately (choose one: ASA 160 mg loading dose then 81–325 mg daily, clopidogrel 300 mg loading dose then 75 mg daily, or extended-release dipyridamole 200 mg/ASA 25 mg bid with ASA 160 mg loading)' objects='patient, medications' id='activity19'/>
            </branch>
            <branch condition='Patient fails swallowing screening' id='branch20'>
              <activity role='ED medical staff' action='Offer rectal or gastric ASA as necessary' objects='patient, rectal/gastric ASA' id='activity20'/>
            </branch>
          </exclusiveGateway>
        </branch>
      </inclusiveGateway>
    </branch>
  </parallelGateway>
  <activity role='ED medical staff' action='Initiate cross-continuum secondary prevention assessments and therapies (refer to Modules 5 and 10)' objects='patient' id='activity11'/>
</process>
