<process>
  <activity role='ED medical staff' action='Obtain patient history' objects='patient' id='activity1-1'/>
  <activity role='ED medical staff' action='Perform physical examination' objects='patient' id='activity1-2'/>
  <activity role='ED medical staff' action='Assess neurological status' objects='patient' id='activity1-3'/>
  <activity role='ED medical staff' action='Measure blood glucose' objects='patient' id='activity2'/>
  <exclusiveGateway id='exclusivegateway1'>
    <branch condition='If hypoglycemia detected' id='branch1'>
      <activity role='ED medical staff' action='Correct hypoglycemia immediately' objects='patient, glucose correction agents' id='activity3'/>
    </branch>
    <branch condition='If first random glucose > 11 mmol/L' id='branch2'>
      <activity role='ED medical staff' action='Repeat blood glucose measurement' objects='patient' id='activity4'/>
    </branch>
    <branch condition='If glucose normal and ≤ 11 mmol/L' id='branch3'>
      <activity role='ED medical staff' action='Proceed to next step' objects='' id='activity5'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED medical staff' action='Measure body temperature as part of routine vital signs every 4 hours for first 48 hours' objects='patient, thermometer' id='activity6'/>
  <exclusiveGateway id='exclusivegateway2'>
    <branch condition='If temperature > 37.5°C' id='branch4'>
      <parallelGateway id='parallelgateway1'>
        <branch id='branch5'>
          <activity role='ED medical staff' action='Initiate temperature-reducing measures' objects='patient, cooling methods' id='activity7'/>
        </branch>
        <branch id='branch6'>
          <activity role='ED medical staff' action='Investigate potential infection' objects='patient' id='activity8'/>
        </branch>
      </parallelGateway>
      <exclusiveGateway id='exclusivegateway8'>
        <branch condition='Infection suspected or confirmed' id='branch27'>
          <activity role='ED medical staff' action='Initiate antipyretic and antimicrobial therapy as required' objects='patient, medications' id='activity9'/>
        </branch>
        <branch condition='No infection suspected or confirmed' id='branch28'>
          <activity role='ED medical staff' action='Continue monitoring and supportive care' objects='patient' id='activity29'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='If temperature ≤ 37.5°C' id='branch8'>
      <activity role='ED medical staff' action='Continue routine monitoring' objects='patient' id='activity10'/>
    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='exclusivegateway3'>
    <branch condition='TIA with high-risk features or minor stroke' id='branch9'>
      <parallelGateway id='parallelgateway2'>
        <branch id='branch10'>
          <activity role='ED medical staff' action='Order brain CT scan' objects='patient, CT scanner' id='activity11'/>
        </branch>
        <branch id='branch11'>
          <activity role='ED medical staff' action='Initiate antiplatelet therapy (if not contraindicated)' objects='patient, antiplatelet agents' id='activity12'/>
        </branch>
      </parallelGateway>
      <exclusiveGateway id='exclusivegateway4'>
        <branch condition='Immediate access to specialized TIA/minor stroke clinic' id='branch12'>
          <activity role='ED medical staff' action='Provide immediate care at specialized TIA/minor stroke clinic' objects='patient, clinic resources' id='activity13'/>
        </branch>
        <branch condition='Delay to specialized clinic poses risk' id='branch13'>
          <activity role='ED medical staff' action='Evaluate at appropriately resourced ED and consider inpatient evaluation and management for stroke prevention' objects='patient, ED resources' id='activity14'/>
        </branch>
        <branch condition='Medical attention sought after 48 hours from symptom onset' id='branch14'>
          <activity role='ED medical staff' action='Refer for evaluation at specialized TIA/minor stroke clinic or outpatient clinic with stroke-prevention services within 24 hours of initial presentation' objects='patient, referral forms' id='activity15'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='TIA without high-risk features' id='branch15'>
      <parallelGateway id='parallelgateway3'>
        <branch id='branch16'>
          <activity role='ED medical staff' action='Order brain CT scan' objects='patient, CT scanner' id='activity16'/>
        </branch>
        <branch id='branch17'>
          <activity role='ED medical staff' action='Initiate antiplatelet therapy (if not contraindicated)' objects='patient, antiplatelet agents' id='activity17'/>
        </branch>
      </parallelGateway>
      <activity role='ED medical staff' action='Refer to outpatient clinic with stroke-prevention services for comprehensive evaluation and management within 1 month of symptom onset' objects='patient, referral forms' id='activity18'/>
    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='exclusivegateway5'>
    <branch condition='TIA patient with fluctuating/crescendo hemibody motor weakness, sensory loss, or speech difficulty' id='branch18'>
      <activity role='ED medical staff' action='Monitor neurological status' objects='patient' id='activity19'/>
      <activity role='ED medical staff' action='Consult neurology for admission decision' objects='patient, neurology team' id='activity20'/>
    </branch>
    <branch condition='Other TIA/minor stroke patients' id='branch19'>
      <activity role='ED medical staff' action='Proceed with outpatient management as appropriate' objects='patient' id='activity21'/>
    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='exclusivegateway6'>
    <branch condition='Patient not on antiplatelet agent at presentation and brain imaging excludes intracranial hemorrhage' id='branch20'>
      <inclusiveGateway id='inclusivegateway1'>
        <branch condition='ASA selected' id='branch21'>
          <activity role='ED medical staff' action='Administer ASA 160 mg loading dose, then enteric coated ASA (81–325 mg) daily (maintenance dose 81 mg/day for most)' objects='patient, ASA' id='activity22'/>
        </branch>
        <branch condition='Clopidogrel selected' id='branch22'>
          <activity role='ED medical staff' action='Administer clopidogrel 300 mg loading dose, then 75 mg/day' objects='patient, clopidogrel' id='activity23'/>
        </branch>
        <branch condition='Extended-release dipyridamole/ASA selected' id='branch23'>
          <activity role='ED medical staff' action='Administer extended-release dipyridamole 200 mg/ASA 25 mg bid (load with ASA 160 mg first)' objects='patient, dipyridamole, ASA' id='activity24'/>
        </branch>
      </inclusiveGateway>
    </branch>
    <branch condition='Patient already on antiplatelet agent or contraindication present' id='branch24'>
      <activity role='ED medical staff' action='Continue or adjust antiplatelet therapy as clinically indicated' objects='patient, medication list' id='activity25'/>
    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='exclusivegateway7'>
    <branch condition='Patient fails swallowing screening' id='branch25'>
      <activity role='ED medical staff' action='Offer rectal or gastric ASA as necessary' objects='patient, rectal/gastric ASA' id='activity26'/>
    </branch>
    <branch condition='Patient passes swallowing screening' id='branch26'>
      <activity role='ED medical staff' action='Administer oral antiplatelet therapy as indicated' objects='patient, oral antiplatelet agents' id='activity27'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED medical staff' action='Provide cross-continuum secondary prevention assessments and therapies (refer to Modules 5 and 10)' objects='patient' id='activity28'/>
</process>
