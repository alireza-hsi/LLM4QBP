  <activity role='ED medical staff' action='Assess vital signs' objects='patient' id='activity1-3'/>
  <parallelGateway id='parallelgateway1'>
    <branch id='branch1'>

        <branch condition='Hypoglycemia detected' id='branch2'>
          <activity role='ED medical staff' action='Correct hypoglycemia immediately' objects='patient, glucose' id='activity3'/>
        </branch>
        <branch condition='Blood glucose > 11 mmol/L' id='branch3'>
          <activity role='ED medical staff' action='Repeat blood glucose measurement' objects='patient, glucometer' id='activity4'/>
        </branch>
        <branch condition='Blood glucose normal' id='branch4'>
          <activity role='ED medical staff' action='Proceed to next assessment' objects='patient' id='activity5'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch id='branch5'>
      <activity role='ED medical staff' action='Evaluate body temperature as part of routine vital signs every 4 hours for first 48 hours' objects='patient, thermometer' id='activity6'/>
      <exclusiveGateway id='exclusivegateway2'>
        <branch condition='Temperature > 37.5 C' id='branch6'>
          <parallelGateway id='parallelgateway2'>
            <branch id='branch7'>
              <activity role='ED medical staff' action='Initiate temperature-reducing measures' objects='patient, cooling methods' id='activity7'/>
            </branch>
            <branch id='branch8'>
              <activity role='ED medical staff' action='Order laboratory tests for infection' objects='patient, laboratory tests' id='activity8-1'/>
              <activity role='ED medical staff' action='Review laboratory results' objects='patient, laboratory results' id='activity8-2'/>
            </branch>
            <branch id='branch9'>
              <activity role='ED medical staff' action='Initiate antipyretic and antimicrobial therapy as required' objects='patient, medications' id='activity9'/>
            </branch>
          </parallelGateway>
        </branch>
        <branch condition='Temperature ≤ 37.5 C' id='branch10'>
          <activity role='ED medical staff' action='Continue routine monitoring' objects='patient' id='activity10'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch id='branch12'>
      <exclusiveGateway id='exclusivegateway3'>
        <branch condition='TIA with high-risk features or minor stroke' id='branch13'>
          <activity role='ED medical staff' action='Order brain CT scan' objects='patient, CT scanner' id='activity12-1'/>
          <activity role='ED medical staff' action='Review CT scan results' objects='patient, CT scan results' id='activity12-2'/>
          <activity role='ED medical staff' action='Initiate antiplatelet therapy if no contraindication' objects='patient, antiplatelet agents' id='activity12-3'/>
          <exclusiveGateway id='exclusivegateway4'>
            <branch condition='Immediate access to specialized TIA/minor stroke clinic' id='branch14'>
              <activity role='ED medical staff' action='Provide immediate care at specialized TIA/minor stroke clinic' objects='patient, clinic resources' id='activity13'/>
            </branch>
            <branch condition='Delay to specialized clinic poses risk' id='branch15'>
              <activity role='ED medical staff' action='Evaluate and manage in appropriately resourced ED; consider inpatient evaluation for stroke prevention' objects='patient, ED resources' id='activity14'/>
            </branch>
            <branch condition='Medical attention sought after 48 hours from symptom onset' id='branch16'>
              <activity role='ED medical staff' action='Refer for evaluation at specialized TIA/minor stroke clinic or outpatient clinic with stroke-prevention services within 24 hours of initial presentation' objects='patient, referral' id='activity15'/>
            </branch>
          </exclusiveGateway>
        </branch>
        <branch condition='TIA without high-risk features' id='branch17'>
          <activity role='ED medical staff' action='Order brain CT scan' objects='patient, CT scanner' id='activity16-1'/>
          <activity role='ED medical staff' action='Review CT scan results' objects='patient, CT scan results' id='activity16-2'/>
          <activity role='ED medical staff' action='Initiate antiplatelet therapy if no contraindication' objects='patient, antiplatelet agents' id='activity16-3'/>
          <activity role='ED medical staff' action='Refer to outpatient clinic with stroke-prevention services for comprehensive evaluation and management within 1 month of symptom onset' objects='patient, referral' id='activity17'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch id='branch_inclusive1'>
      <inclusiveGateway id='inclusivegateway1'>
        <branch condition='TIA patient with fluctuating or crescendo hemibody motor weakness, sensory loss, or speech difficulty' id='branch18'>
          <activity role='ED medical staff' action='Evaluate in ED for possible admission to hospital' objects='patient' id='activity18'/>
        </branch>
        <branch condition='TIA or minor stroke patient not on antiplatelet agent at presentation' id='branch19'>
          <exclusiveGateway id='exclusivegateway5'>
            <branch condition='Patient can swallow' id='branch20'>
              <activity role='ED medical staff' action='Start antiplatelet therapy immediately after brain imaging has excluded intracranial hemorrhage (choose one: ASA 160 mg loading dose then 81–325 mg daily, clopidogrel 300 mg loading then 75 mg/day, or extended-release dipyridamole 200 mg/ASA 25 mg bid with ASA 160 mg loading)' objects='patient, antiplatelet agents' id='activity19'/>
            </branch>
            <branch condition='Patient cannot swallow' id='branch21'>
              <activity role='ED medical staff' action='Offer rectal or gastric ASA as necessary' objects='patient, rectal/gastric ASA' id='activity20'/>
            </branch>
          </exclusiveGateway>
        </branch>
      </inclusiveGateway>
    </branch>
  </parallelGateway>
  <parallelGateway id='parallelgateway_converge1'>

      <activity role='ED medical staff' action='All diagnostic, treatment, and referral activities completed' objects='patient' id='activity_converge2'/>
