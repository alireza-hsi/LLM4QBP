<process>
  <activity role='ED medical staff' action='Perform initial neurological assessment' objects='patient' id='activity1-1'/>
  <activity role='ED medical staff' action='Stabilize airway, breathing, and circulation' objects='patient' id='activity1-2'/>
  <parallelGateway id='parallelgateway1'>
    <branch id='branch1'>
      <activity role='ED medical staff' action='Measure blood glucose' objects='patient, glucometer' id='activity2'/>
      <exclusiveGateway id='exclusivegateway1'>
        <branch condition='Blood glucose < normal range (hypoglycemia)' id='branch2'>
          <activity role='ED medical staff' action='Correct hypoglycemia immediately' objects='glucose, patient' id='activity3'/>
        </branch>
        <branch condition='Blood glucose > 11 mmol/L' id='branch3'>
          <activity role='ED medical staff' action='Repeat blood glucose measurement' objects='patient, glucometer' id='activity4'/>
        </branch>
        <branch condition='Blood glucose within normal range' id='branch4'>
          <activity role='ED medical staff' action='Proceed to next steps' objects='patient' id='activity5'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch id='branch5'>
      <activity role='ED medical staff' action='Evaluate body temperature as part of routine vital signs every 4 hours for first 48 hours' objects='patient, thermometer' id='activity6'/>
      <exclusiveGateway id='exclusivegateway2'>
        <branch condition='Temperature > 37.5°C' id='branch6'>
          <activity role='ED medical staff' action='Initiate temperature-reducing measures' objects='patient, antipyretics, cooling devices' id='activity7'/>
          <parallelGateway id='parallelgateway2-activity8'>
            <branch id='branch6-1'>
              <activity role='ED medical staff' action='Obtain relevant cultures (e.g., blood, urine)' objects='patient, culture bottles, urine container' id='activity8-1'/>
            </branch>
            <branch id='branch6-2'>
              <activity role='ED medical staff' action='Order chest X-ray if respiratory symptoms present' objects='patient, radiology order' id='activity8-2'/>
            </branch>
          </parallelGateway>
          <activity role='ED medical staff' action='Initiate antipyretic and antimicrobial therapy as required' objects='patient, medications' id='activity9'/>
        </branch>
        <branch condition='Temperature ≤ 37.5°C' id='branch7'>
          <activity role='ED medical staff' action='Continue routine monitoring' objects='patient' id='activity10'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch id='branch9'>
      <exclusiveGateway id='exclusivegateway3'>
        <branch condition='TIA with high-risk features or minor stroke' id='branch10'>
          <activity role='ED medical staff' action='Order brain CT scan' objects='patient, CT scanner' id='activity12-1'/>
          <activity role='ED medical staff' action='Initiate antiplatelet therapy if not contraindicated' objects='patient, antiplatelet agents' id='activity12-2'/>
          <exclusiveGateway id='exclusivegateway4'>
            <branch condition='Immediate access to specialized TIA/minor stroke clinic' id='branch11'>
              <activity role='ED medical staff' action='Provide immediate care at specialized TIA/minor stroke clinic' objects='patient, clinic resources' id='activity13'/>
            </branch>
            <branch condition='Delay to specialized clinic poses risk' id='branch12'>
              <activity role='ED medical staff' action='Perform evaluation in appropriately resourced ED and consider inpatient evaluation for stroke prevention' objects='patient, ED resources' id='activity14'/>
            </branch>
            <branch condition='Medical attention sought >48 hours after symptom onset' id='branch13'>
              <activity role='ED medical staff' action='Refer for evaluation at specialized TIA/minor stroke clinic or outpatient clinic with stroke-prevention services within 24 hours of initial presentation' objects='patient, referral' id='activity15'/>
            </branch>
          </exclusiveGateway>
        </branch>
        <branch condition='TIA without high-risk features' id='branch14'>
          <activity role='ED medical staff' action='Order brain CT scan' objects='patient, CT scanner' id='activity16-1'/>
          <activity role='ED medical staff' action='Initiate antiplatelet therapy if not contraindicated' objects='patient, antiplatelet agents' id='activity16-2'/>
          <activity role='ED medical staff' action='Refer to outpatient clinic with stroke-prevention services for comprehensive evaluation and management within 1 month of symptom onset' objects='patient, referral' id='activity17'/>
        </branch>
      </exclusiveGateway>
      <inclusiveGateway id='inclusivegateway1'>
        <branch condition='Patient presents with fluctuating or crescendo hemibody motor weakness, sensory loss, or speech difficulty' id='branch15'>
          <activity role='ED medical staff' action='Evaluate in ED for possible hospital admission' objects='patient' id='activity18'/>
        </branch>
        <branch condition='Patient not on antiplatelet agent at presentation and brain imaging has excluded intracranial hemorrhage' id='branch16'>
          <exclusiveGateway id='exclusivegateway5'>
            <branch condition='Patient can swallow' id='branch17'>
              <activity role='ED medical staff' action='Start antiplatelet therapy immediately with one of: ASA 160 mg loading dose then 81–325 mg daily (maintenance 81 mg/day), clopidogrel 300 mg loading then 75 mg/day, or extended-release dipyridamole 200 mg/ASA 25 mg bid (load with ASA 160 mg first)' objects='patient, antiplatelet agents' id='activity19'/>
            </branch>
            <branch condition='Patient cannot swallow' id='branch18'>
              <activity role='ED medical staff' action='Offer rectal or gastric ASA as necessary' objects='patient, rectal/gastric ASA' id='activity20'/>
            </branch>
          </exclusiveGateway>
        </branch>
      </inclusiveGateway>
    </branch>
  </parallelGateway>
  <activity role='ED medical staff' action='Provide cross-continuum secondary prevention assessments and therapies (per Modules 5 and 10)' objects='patient' id='activity11'/>
</process>
