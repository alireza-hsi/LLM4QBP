          <activity role='ED medical staff' action='Repeat blood glucose measurement' objects='patient' id='activity4'/>
        </branch>
        <branch condition='No hypoglycemia and glucose ≤ 11 mmol/L' id='branch4'>
          <activity role='ED medical staff' action='Continue with standard care' objects='patient' id='activity5'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch id='branch5'>
      <activity role='ED medical staff' action='Evaluate body temperature as part of routine vital signs every 4 hours for first 48 hours' objects='patient' id='activity6'/>
      <exclusiveGateway id='exclusivegateway2'>
        <branch condition='Temperature > 37.5°C' id='branch6'>
          <parallelGateway id='parallelgateway2'>
            <branch id='branch7'>
              <activity role='ED medical staff' action='Initiate temperature-reducing measures' objects='patient' id='activity7'/>
            </branch>
            <branch id='branch8'>
              <parallelGateway id='parallelgateway3'>
                <branch id='branch8-1'>
                  <activity role='ED medical staff' action='Order infection-related laboratory tests' objects='patient' id='activity8-1'/>

                <branch id='branch8-2'>
                  <activity role='ED medical staff' action='Assess for clinical signs of infection' objects='patient' id='activity8-2'/>
                </branch>
              </parallelGateway>
            </branch>
            <branch id='branch9'>
              <activity role='ED medical staff' action='Initiate antipyretic and antimicrobial therapy as required' objects='patient' id='activity9'/>
            </branch>
          </parallelGateway>
        </branch>
        <branch condition='Temperature ≤ 37.5°C' id='branch10'>
          <activity role='ED medical staff' action='Continue routine monitoring' objects='patient' id='activity10'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch id='branch11'>
      <exclusiveGateway id='exclusivegateway3'>
        <branch condition='TIA with high-risk features OR minor stroke' id='branch13'>
          <activity role='ED medical staff' action='Order brain CT scan' objects='patient' id='activity12-1'/>
          <activity role='ED medical staff' action='Review CT scan results' objects='patient' id='activity12-2'/>
          <activity role='ED medical staff' action='Initiate antiplatelet therapy if no contraindication and no intracranial hemorrhage' objects='patient' id='activity12-3'/>
          <exclusiveGateway id='exclusivegateway4'>
            <branch condition='Immediate access to specialized TIA/minor stroke clinic' id='branch14'>
              <activity role='ED medical staff' action='Provide immediate care at specialized TIA/minor stroke clinic' objects='patient' id='activity13'/>
            </branch>
            <branch condition='Delay to specialized clinic poses risk' id='branch15'>
              <activity role='ED medical staff' action='Perform evaluation in appropriately resourced ED and consider inpatient evaluation for stroke prevention' objects='patient' id='activity14'/>
            </branch>
            <branch condition='Medical attention sought after 48 hours from symptom onset' id='branch16'>
              <activity role='ED medical staff' action='Refer for evaluation at specialized TIA/minor stroke clinic or outpatient clinic with stroke-prevention services within 24 hours of initial presentation' objects='patient' id='activity15'/>
            </branch>
          </exclusiveGateway>
        </branch>
        <branch condition='TIA without high-risk features' id='branch17'>
          <activity role='ED medical staff' action='Order brain CT scan' objects='patient' id='activity16-1'/>
          <activity role='ED medical staff' action='Review CT scan results' objects='patient' id='activity16-2'/>
          <activity role='ED medical staff' action='Initiate antiplatelet therapy if no contraindication and no intracranial hemorrhage' objects='patient' id='activity16-3'/>

        </branch>
      </exclusiveGateway>
    </branch>
    <branch id='branch12'>
      <inclusiveGateway id='inclusivegateway1'>
        <branch condition='TIA patient with fluctuating/crescendo hemibody motor weakness, sensory loss, or speech difficulty' id='branch18'>
          <activity role='ED medical staff' action='Evaluate in ED for possible admission to hospital' objects='patient' id='activity18'/>

                <branch condition='ASA selected' id='branch21'>

                  <activity role='ED medical staff' action='Start clopidogrel 300 mg loading dose, then 75 mg/day' objects='patient' id='activity20'/>
                </branch>
                <branch condition='Extended-release dipyridamole/ASA selected' id='branch23'>
                  <activity role='ED medical staff' action='Load with ASA 160 mg, then start extended-release dipyridamole 200 mg/ASA 25 mg bid' objects='patient' id='activity21'/>
                </branch>
              </inclusiveGateway>
            </branch>
            <branch condition='Patient fails swallowing screening' id='branch24'>

        </branch>

  <activity role='ED medical staff' action='Provide cross-continuum secondary prevention assessments and therapies (per Modules 5 and 10)' objects='patient' id='activity11'/>
