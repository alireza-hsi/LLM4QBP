<process>
  <activity role='ED staff' action='Establish standardized priority protocol for suspected stroke patients' objects='protocol documentation' id='activity1-1'/>
  <activity role='ED staff' action='Admit suspected stroke patient to stroke unit within first few hours' objects='admission forms, stroke unit bed' id='activity1-2'/>
  <activity role='ED staff' action='Measure blood glucose on arrival' objects='blood glucose meter' id='activity2'/>
  <exclusiveGateway id='exclusivegateway1'>
    <branch condition='If hypoglycemia detected' id='branch2'>
      <activity role='ED staff' action='Correct hypoglycemia immediately' objects='glucose, IV access' id='activity3'/>
    </branch>
    <branch condition='If first random glucose > 11 mmol/L' id='branch3'>
      <activity role='ED staff' action='Repeat blood glucose measurement' objects='blood glucose meter' id='activity4'/>
    </branch>
    <branch condition='If glucose normal' id='branch4'>
      <activity role='ED staff' action='Proceed to next step' objects='' id='activity5'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED staff' action='Evaluate body temperature as part of routine vital signs every 4 hours for first 48 hours' objects='thermometer' id='activity6'/>
  <exclusiveGateway id='exclusivegateway2'>
    <branch condition='If temperature > 37.5°C' id='branch6'>
      <activity role='ED staff' action='Initiate temperature-reducing measures (including antipyretic therapy as required)' objects='cooling devices, antipyretics' id='activity7'/>
      <activity role='ED staff' action='Order laboratory tests and cultures' objects='laboratory requisition, specimen containers' id='activity8-1'/>
      <activity role='ED staff' action='Review laboratory results for infection' objects='laboratory results' id='activity8-2'/>
      <exclusiveGateway id='exclusivegateway3'>
        <branch condition='If infection confirmed or suspected' id='branch9'>
          <activity role='ED staff' action='Initiate antimicrobial therapy as required' objects='antimicrobials' id='activity9'/>
        </branch>
        <branch condition='If no infection' id='branch10'>
          <activity role='ED staff' action='Continue monitoring' objects='' id='activity10'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='If temperature ≤ 37.5°C' id='branch12'>
      <activity role='ED staff' action='Continue routine monitoring' objects='' id='activity12'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED staff' action='Perform neurological assessment for fluctuating/crescendo hemibody motor weakness, sensory loss, or speech difficulty' objects='neurological assessment' id='activity13'/>
  <exclusiveGateway id='exclusivegateway4'>
    <branch condition='If fluctuating/crescendo symptoms present' id='branch14'>
      <activity role='ED staff' action='Evaluate for hospital admission' objects='admission criteria' id='activity14'/>
    </branch>
    <branch condition='If not present' id='branch15'>
      <activity role='ED staff' action='Proceed to risk stratification and timing assessment' objects='' id='activity15'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED staff' action='Determine TIA/minor stroke risk features and timing of presentation' objects='patient history, clinical assessment' id='activity16'/>
  <exclusiveGateway id='exclusivegateway5'>
    <branch condition='TIA with high-risk features or minor stroke, presenting ≤ 48 hours from symptom onset' id='branch17'>
      <activity role='ED staff' action='Arrange brain CT scan' objects='CT scanner' id='activity17-1'/>
      <activity role='ED staff' action='Interpret CT scan results' objects='CT images' id='activity17-2'/>
      <activity role='ED staff' action='Initiate antiplatelet therapy if not contraindicated' objects='antiplatelet agents' id='activity17-3'/>
      <exclusiveGateway id='exclusivegateway6'>
        <branch condition='Immediate care available at specialized TIA/minor stroke clinic' id='branch18'>
          <activity role='ED staff' action='Provide immediate care at specialized TIA/minor stroke clinic' objects='clinic resources' id='activity18'/>
        </branch>
        <branch condition='Delay to specialized clinic poses risk' id='branch19'>
          <activity role='ED staff' action='Perform evaluation in appropriately resourced ED and consider inpatient evaluation for stroke prevention' objects='ED resources' id='activity19'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='TIA with high-risk features or minor stroke, presenting > 48 hours from symptom onset' id='branch20'>
      <activity role='ED staff' action='Refer for evaluation at specialized TIA/minor stroke clinic or outpatient clinic with stroke-prevention services within 24 hours of initial presentation' objects='referral documentation' id='activity20'/>
    </branch>
    <branch condition='TIA without high-risk features' id='branch21'>
      <activity role='ED staff' action='Arrange brain CT scan' objects='CT scanner' id='activity21-1'/>
      <activity role='ED staff' action='Interpret CT scan results' objects='CT images' id='activity21-2'/>
      <activity role='ED staff' action='Initiate antiplatelet therapy if not contraindicated' objects='antiplatelet agents' id='activity21-3'/>
      <activity role='ED staff' action='Refer to outpatient clinic with stroke-prevention services for comprehensive evaluation and management within 1 month of symptom onset' objects='referral documentation' id='activity22'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED staff' action='Interpret all test results immediately (including imaging and blood tests)' objects='test results' id='activity23'/>
  <activity role='ED staff' action='Assess if patient is currently on antiplatelet therapy' objects='medication history' id='activity24'/>
  <exclusiveGateway id='exclusivegateway7'>
    <branch condition='If not on antiplatelet therapy' id='branch24'>
      <activity role='ED staff' action='Start antiplatelet therapy immediately after brain imaging excludes intracranial hemorrhage' objects='ASA, clopidogrel, dipyridamole' id='activity25'/>
      <exclusiveGateway id='exclusivegateway8'>
        <branch condition='Patient fails swallowing screening' id='branch25'>
          <activity role='ED staff' action='Offer rectal or gastric ASA as necessary' objects='rectal ASA, gastric tube' id='activity26'/>
        </branch>
        <branch condition='Patient passes swallowing screening' id='branch26'>
          <activity role='ED staff' action='Administer oral antiplatelet therapy' objects='oral ASA, clopidogrel, dipyridamole' id='activity27'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='If already on antiplatelet therapy' id='branch27'>
      <activity role='ED staff' action='Continue current antiplatelet therapy as appropriate' objects='current medication' id='activity28'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED staff' action='Conduct secondary prevention assessment' objects='prevention assessment tools' id='activity29-1'/>
  <activity role='ED staff' action='Initiate secondary prevention therapies' objects='therapy protocols' id='activity29-2'/>
</process>
