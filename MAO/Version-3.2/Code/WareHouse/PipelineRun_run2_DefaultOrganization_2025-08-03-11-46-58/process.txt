<process>
  <activity role='ED staff' action='Establish standardized priority protocol for suspected stroke patients' objects='protocols, suspected TIA/minor stroke patient' id='activity1-1'/>
  <activity role='ED staff' action='Admit suspected TIA/minor stroke patient to stroke unit within first few hours' objects='suspected TIA/minor stroke patient, stroke unit' id='activity1-2'/>
  <activity role='ED staff' action='Perform initial clinical assessment for TIA/minor stroke and identify high-risk features or fluctuating/crescendo symptoms' objects='patient, clinical criteria' id='activity2-1'/>
  <activity role='ED staff' action='Measure blood glucose on arrival' objects='patient, blood glucose meter' id='activity2-2'/>
  <exclusiveGateway id='exclusivegateway1'>
    <branch condition='If hypoglycemia detected' id='branch2'>
      <activity role='ED staff' action='Correct hypoglycemia immediately' objects='patient, glucose administration' id='activity2-3'/>
    </branch>
    <branch condition='If first random glucose > 11 mmol/L' id='branch3'>
      <activity role='ED staff' action='Repeat blood glucose measurement' objects='patient, blood glucose meter' id='activity2-4'/>
    </branch>
    <branch condition='If normoglycemia' id='branch4'>
      <activity role='ED staff' action='Proceed to next step' objects='' id='activity2-5'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED staff' action='Evaluate body temperature as part of routine vital signs every 4 hours for first 48 hours' objects='patient, thermometer' id='activity2-6'/>
  <exclusiveGateway id='exclusivegateway2'>
    <branch condition='If temperature > 37.5°C' id='branch6'>
      <parallelGateway id='parallelgateway2'>
        <branch id='branch7'>
          <activity role='ED staff' action='Initiate temperature-reducing measures' objects='patient, cooling methods' id='activity2-7'/>
        </branch>
        <branch id='branch8'>
          <activity role='ED staff' action='Investigate potential infection' objects='patient, infection workup' id='activity2-8'/>
        </branch>
        <branch id='branch9'>
          <activity role='ED staff' action='Initiate antipyretic and antimicrobial therapy as required' objects='patient, antipyretics, antimicrobials' id='activity2-9'/>
        </branch>
      </parallelGateway>
    </branch>
    <branch condition='If temperature ≤ 37.5°C' id='branch10'>
      <activity role='ED staff' action='Continue routine monitoring' objects='patient' id='activity2-10'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED staff' action='Arrange urgent brain CT scan' objects='patient, CT scanner' id='activity2-11'/>
  <activity role='ED staff' action='Interpret imaging results immediately upon availability' objects='patient, imaging results' id='activity2-12'/>
  <activity role='ED staff' action='Interpret blood test results immediately upon availability' objects='patient, blood test results' id='activity2-13'/>
  <exclusiveGateway id='exclusivegateway3'>
    <branch condition='TIA with high-risk features or minor stroke' id='branch14'>
      <activity role='ED staff' action='Initiate antiplatelet therapy (if not contraindicated) within 24 hours' objects='patient, antiplatelet agents' id='activity3-1'/>
      <exclusiveGateway id='exclusivegateway4'>
        <branch condition='Fluctuating/crescendo hemibody motor weakness, sensory loss, or speech difficulty present' id='branch15'>
          <activity role='ED staff' action='Evaluate in ED for possible admission to hospital' objects='patient' id='activity3-2'/>
        </branch>
        <branch condition='No fluctuating/crescendo hemibody motor weakness, sensory loss, or speech difficulty' id='branch16'>
          <exclusiveGateway id='exclusivegateway5'>
            <branch condition='Immediate access to specialized TIA/minor stroke clinic' id='branch17'>
              <activity role='ED staff' action='Provide immediate care at specialized TIA/minor stroke clinic' objects='patient, clinic resources' id='activity3-3'/>
            </branch>
            <branch condition='Delay to specialized clinic poses risk' id='branch18'>
              <activity role='ED staff' action='Perform evaluation in appropriately resourced ED and consider inpatient evaluation for stroke prevention' objects='patient, ED resources' id='activity3-4'/>
            </branch>
            <branch condition='Medical attention sought after 48 hours from symptom onset' id='branch19'>
              <activity role='ED staff' action='Refer for evaluation at specialized TIA/minor stroke clinic or outpatient clinic with stroke-prevention services within 24 hours of initial presentation' objects='patient, referral' id='activity3-5'/>
            </branch>
          </exclusiveGateway>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='TIA without high-risk features' id='branch20'>
      <activity role='ED staff' action='Initiate antiplatelet therapy (if not contraindicated) within 24 hours' objects='patient, antiplatelet agents' id='activity3-6'/>
      <activity role='ED staff' action='Refer to outpatient clinic with stroke-prevention services for comprehensive evaluation and management within 1 month of symptom onset' objects='patient, referral' id='activity3-7'/>
    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='exclusivegateway6'>
    <branch condition='Patient not on antiplatelet agent at presentation' id='branch21'>
      <activity role='ED staff' action='Start antiplatelet therapy immediately after brain imaging excludes intracranial hemorrhage' objects='patient, antiplatelet agents' id='activity4-1'/>
      <exclusiveGateway id='exclusivegateway7'>
        <branch condition='Patient fails swallowing screening' id='branch22'>
          <activity role='ED staff' action='Offer rectal or gastric ASA as necessary' objects='patient, rectal/gastric ASA' id='activity4-2'/>
        </branch>
        <branch condition='Patient passes swallowing screening' id='branch23'>
          <activity role='ED staff' action='Administer oral antiplatelet therapy' objects='patient, oral antiplatelet agents' id='activity4-3'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='Patient already on antiplatelet agent' id='branch24'>
      <activity role='ED staff' action='Continue current antiplatelet therapy as appropriate' objects='patient, current medication' id='activity4-4'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED staff' action='Provide cross-continuum secondary prevention assessments and therapies (Modules 5 and 10) to all patients before discharge or transfer' objects='patient, prevention protocols' id='activity5-1'/>
</process>
