<process>
  <activity role='ED medical staff' action='Obtain patient history' objects='patient, clinical records' id='activity1-1'/>
  <activity role='ED medical staff' action='Perform physical examination' objects='patient' id='activity1-2'/>
  <activity role='ED medical staff' action='Review clinical records' objects='clinical records' id='activity1-3'/>
  <activity role='ED medical staff' action='Measure blood glucose' objects='patient, glucometer' id='activity2'/>
  <exclusiveGateway id='exclusivegateway1'>
    <branch condition='Hypoglycemia detected' id='branch2'>
      <activity role='ED medical staff' action='Correct hypoglycemia immediately' objects='patient, glucose administration' id='activity3'/>
    </branch>
    <branch condition='Blood glucose > 11 mmol/L' id='branch3'>
      <activity role='ED medical staff' action='Repeat blood glucose measurement' objects='patient, glucometer' id='activity4'/>
    </branch>
    <branch condition='Blood glucose normal (not hypoglycemic or >11 mmol/L)' id='branch4'>
      <activity role='ED medical staff' action='Proceed with next steps' objects='patient' id='activity5'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED medical staff' action='Evaluate body temperature as part of routine vital signs every 4 hours for first 48 hours' objects='patient, thermometer' id='activity6'/>
  <exclusiveGateway id='exclusivegateway2'>
    <branch condition='Temperature > 37.5°C' id='branch6'>
      <parallelGateway id='parallelgateway2'>
        <branch id='branch7'>
          <activity role='ED medical staff' action='Initiate temperature-reducing measures' objects='patient, cooling methods' id='activity7'/>
        </branch>
        <branch id='branch8'>
          <activity role='ED medical staff' action='Order laboratory tests for infection' objects='patient, laboratory tests' id='activity8-1'/>
          <activity role='ED medical staff' action='Review laboratory results' objects='laboratory results' id='activity8-2'/>
          <activity role='ED medical staff' action='Determine need for antimicrobial therapy' objects='patient, laboratory results' id='activity8-3'/>
          <activity role='ED medical staff' action='Initiate antipyretic and antimicrobial therapy as required' objects='patient, medications' id='activity9'/>
        </branch>
      </parallelGateway>
    </branch>
    <branch condition='Temperature ≤ 37.5°C' id='branch10'>
      <activity role='ED medical staff' action='Continue routine monitoring' objects='patient' id='activity10'/>
    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='exclusivegateway3'>
    <branch condition='TIA with high-risk features OR minor stroke' id='branch11'>
      <activity role='ED medical staff' action='Arrange brain CT scan' objects='patient, CT scanner' id='activity12-1'/>
      <activity role='ED medical staff' action='Review CT scan results' objects='CT scan results' id='activity12-2'/>
      <activity role='ED medical staff' action='Initiate antiplatelet therapy if no contraindication' objects='patient, antiplatelet medication' id='activity12-3'/>
      <exclusiveGateway id='exclusivegateway4'>
        <branch condition='Immediate access to specialized TIA/minor stroke clinic' id='branch12'>
          <activity role='ED medical staff' action='Provide immediate care at specialized TIA/minor stroke clinic' objects='patient, clinic resources' id='activity13'/>
        </branch>
        <branch condition='Delay to specialized clinic poses risk' id='branch13'>
          <activity role='ED medical staff' action='Perform evaluation in appropriately resourced ED and consider inpatient evaluation for stroke prevention' objects='patient, ED resources' id='activity14'/>
        </branch>
        <branch condition='Medical attention sought after 48 hours from symptom onset' id='branch14'>
          <activity role='ED medical staff' action='Refer for evaluation at specialized TIA/minor stroke clinic or outpatient clinic with stroke-prevention services within 24 hours of initial presentation' objects='patient, referral' id='activity15'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='TIA without high-risk features' id='branch15'>
      <activity role='ED medical staff' action='Arrange brain CT scan' objects='patient, CT scanner' id='activity16-1'/>
      <activity role='ED medical staff' action='Review CT scan results' objects='CT scan results' id='activity16-2'/>
      <activity role='ED medical staff' action='Initiate antiplatelet therapy if no contraindication' objects='patient, antiplatelet medication' id='activity16-3'/>
      <activity role='ED medical staff' action='Refer to outpatient clinic with stroke-prevention services for comprehensive evaluation and management within 1 month of symptom onset' objects='patient, referral' id='activity17'/>
    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='exclusivegateway5'>
    <branch condition='TIA patient with fluctuating/crescendo hemibody motor weakness, sensory loss, or speech difficulty' id='branch16'>
      <activity role='ED medical staff' action='Evaluate in ED for consideration of hospital admission' objects='patient, clinical assessment' id='activity18'/>
    </branch>
    <branch condition='Other TIA/minor stroke patients' id='branch17'>
      <activity role='ED medical staff' action='Continue with outpatient management pathway' objects='patient' id='activity19'/>
    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='exclusivegateway6'>
    <branch condition='Patient not on antiplatelet agent at presentation' id='branch18'>
      <exclusiveGateway id='exclusivegateway7'>
        <branch condition='Patient can swallow' id='branch19'>
          <inclusiveGateway id='inclusivegateway1'>
            <branch condition='ASA selected' id='branch20'>
              <activity role='ED medical staff' action='Start ASA 160 mg loading dose, then enteric coated ASA (81–325 mg) daily (maintenance 81 mg/day)' objects='patient, ASA' id='activity20'/>
            </branch>
            <branch condition='Clopidogrel selected' id='branch21'>
              <activity role='ED medical staff' action='Start clopidogrel 300 mg loading dose, then 75 mg/day' objects='patient, clopidogrel' id='activity21'/>
            </branch>
            <branch condition='Extended-release dipyridamole/ASA selected' id='branch22'>
              <activity role='ED medical staff' action='Load with ASA 160 mg, then start extended-release dipyridamole 200 mg/ASA 25 mg bid' objects='patient, dipyridamole/ASA' id='activity22'/>
            </branch>
          </inclusiveGateway>
        </branch>
        <branch condition='Patient cannot swallow' id='branch23'>
          <activity role='ED medical staff' action='Offer rectal or gastric ASA as necessary' objects='patient, rectal/gastric ASA' id='activity23'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='Patient already on antiplatelet agent' id='branch24'>
      <activity role='ED medical staff' action='Continue current antiplatelet therapy as appropriate' objects='patient, medication list' id='activity24'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED medical staff' action='Provide cross-continuum secondary prevention assessment and therapies (Modules 5 and 10)' objects='patient, prevention protocol' id='activity25'/>
</process>
