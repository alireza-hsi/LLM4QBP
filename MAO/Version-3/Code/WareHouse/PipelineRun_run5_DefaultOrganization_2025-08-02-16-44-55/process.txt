<process>
  <activity role='ED medical staff' action='Obtain clinical history' objects='patient, clinical history' id='activity1-1'/>
  <activity role='ED medical staff' action='Perform physical examination' objects='patient' id='activity1-2'/>
  <activity role='ED medical staff' action='Assess vital signs' objects='patient, vital signs' id='activity1-3'/>
  <parallelGateway id='parallelgateway1'>
    <branch id='branch1'>
      <activity role='ED medical staff' action='Check blood glucose' objects='patient, glucometer' id='activity2'/>
      <exclusiveGateway id='exclusivegateway1'>
        <branch condition='If hypoglycemia detected' id='branch2'>
          <activity role='ED medical staff' action='Correct hypoglycemia immediately' objects='patient, glucose administration' id='activity3'/>
        </branch>
        <branch condition='If first random glucose > 11 mmol/L' id='branch3'>
          <activity role='ED medical staff' action='Repeat blood glucose measurement' objects='patient, glucometer' id='activity4'/>
        </branch>
        <branch condition='If glucose normal' id='branch4'>
          <activity role='ED medical staff' action='Proceed to next assessment' objects='patient' id='activity5'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch id='branch5'>
      <activity role='ED medical staff' action='Evaluate body temperature as part of routine vital signs every 4 hours for first 48 hours' objects='patient, thermometer' id='activity6'/>
      <exclusiveGateway id='exclusivegateway2'>
        <branch condition='If temperature > 37.5°C' id='branch6'>
          <parallelGateway id='parallelgateway2'>
            <branch id='branch7'>
              <activity role='ED medical staff' action='Initiate temperature-reducing measures' objects='patient, cooling methods' id='activity7'/>
            </branch>
            <branch id='branch8'>
              <activity role='ED medical staff' action='Order laboratory tests for infection' objects='patient, laboratory tests' id='activity8-1'/>
              <activity role='ED medical staff' action='Review laboratory test results' objects='patient, laboratory results' id='activity8-2'/>
              <activity role='ED medical staff' action='Initiate antipyretic and antimicrobial therapy as required' objects='patient, medications' id='activity9'/>
            </branch>
          </parallelGateway>
        </branch>
        <branch condition='If temperature ≤ 37.5°C' id='branch10'>
          <activity role='ED medical staff' action='Continue routine monitoring' objects='patient' id='activity10'/>
        </branch>
      </exclusiveGateway>
    </branch>
  </parallelGateway>
  <exclusiveGateway id='exclusivegateway3'>
    <branch condition='If TIA with high-risk features OR minor stroke' id='branch13'>
      <activity role='ED medical staff' action='Order brain CT scan' objects='patient, CT scanner' id='activity12-1'/>
      <activity role='ED medical staff' action='Review CT scan results' objects='patient, CT results' id='activity12-2'/>
      <exclusiveGateway id='exclusivegateway4a'>
        <branch condition='If patient not on antiplatelet agent at presentation and brain imaging excludes intracranial hemorrhage' id='branch13a'>
          <exclusiveGateway id='exclusivegateway5a'>
            <branch condition='If patient can swallow' id='branch20a'>
              <activity role='ED medical staff' action='Start antiplatelet therapy immediately (choose one: ASA 160 mg loading then 81–325 mg daily, clopidogrel 300 mg loading then 75 mg daily, or extended-release dipyridamole 200 mg/ASA 25 mg bid after ASA 160 mg loading)' objects='patient, antiplatelet agents' id='activity19a'/>
            </branch>
            <branch condition='If patient fails swallowing screening' id='branch21a'>
              <activity role='ED medical staff' action='Offer rectal or gastric ASA as necessary' objects='patient, rectal/gastric ASA' id='activity20a'/>
            </branch>
          </exclusiveGateway>
        </branch>
        <branch condition='If patient already on antiplatelet agent or brain imaging shows intracranial hemorrhage' id='branch13b'>
          <activity role='ED medical staff' action='Proceed to next step' objects='patient' id='activity19b'/>
        </branch>
      </exclusiveGateway>
      <exclusiveGateway id='exclusivegateway4'>
        <branch condition='If immediate care available at specialized TIA/minor stroke clinic' id='branch14'>
          <activity role='ED medical staff' action='Refer patient to specialized TIA/minor stroke clinic for rapid evaluation and treatment' objects='patient, referral documentation' id='activity13'/>
        </branch>
        <branch condition='If delay to specialized clinic poses risk' id='branch15'>
          <activity role='ED medical staff' action='Perform evaluation in ED' objects='patient, ED resources' id='activity14-1'/>
          <activity role='ED medical staff' action='Consider inpatient evaluation and management for stroke prevention' objects='patient, admission criteria' id='activity14-2'/>
        </branch>
        <branch condition='If medical attention sought after 48 hours from symptom onset' id='branch16'>
          <activity role='ED medical staff' action='Refer for evaluation at specialized TIA/minor stroke clinic or outpatient clinic with stroke-prevention services within 24 hours of initial presentation' objects='patient, referral documentation' id='activity15'/>
        </branch>
      </exclusiveGateway>
      <exclusiveGateway id='exclusivegateway6'>
        <branch condition='If patient presents with fluctuating or crescendo hemibody motor weakness, sensory loss, or speech difficulty' id='branch18'>
          <activity role='ED medical staff' action='Evaluate in ED to consider admission to hospital' objects='patient, admission criteria' id='activity18'/>
        </branch>
        <branch condition='If patient does not present with these features' id='branch18b'>
          <activity role='ED medical staff' action='Proceed to next step' objects='patient' id='activity18b'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='If TIA without high-risk features' id='branch17'>
      <activity role='ED medical staff' action='Order brain CT scan' objects='patient, CT scanner' id='activity16-1'/>
      <activity role='ED medical staff' action='Review CT scan results' objects='patient, CT results' id='activity16-2'/>
      <exclusiveGateway id='exclusivegateway4b'>
        <branch condition='If patient not on antiplatelet agent at presentation and brain imaging excludes intracranial hemorrhage' id='branch17a'>
          <exclusiveGateway id='exclusivegateway5b'>
            <branch condition='If patient can swallow' id='branch20b'>
              <activity role='ED medical staff' action='Start antiplatelet therapy immediately (choose one: ASA 160 mg loading then 81–325 mg daily, clopidogrel 300 mg loading then 75 mg daily, or extended-release dipyridamole 200 mg/ASA 25 mg bid after ASA 160 mg loading)' objects='patient, antiplatelet agents' id='activity19b1'/>
            </branch>
            <branch condition='If patient fails swallowing screening' id='branch21b'>
              <activity role='ED medical staff' action='Offer rectal or gastric ASA as necessary' objects='patient, rectal/gastric ASA' id='activity20b'/>
            </branch>
          </exclusiveGateway>
        </branch>
        <branch condition='If patient already on antiplatelet agent or brain imaging shows intracranial hemorrhage' id='branch17b'>
          <activity role='ED medical staff' action='Proceed to next step' objects='patient' id='activity19b2'/>
        </branch>
      </exclusiveGateway>
      <activity role='ED medical staff' action='Refer to outpatient clinic with stroke-prevention services for comprehensive evaluation and management within 1 month of symptom onset' objects='patient, referral documentation' id='activity17'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED medical staff' action='Provide cross-continuum secondary prevention assessments and therapies (Modules 5 and 10)' objects='patient, prevention protocol' id='activity11'/>
</process>
