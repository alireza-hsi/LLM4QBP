<process>
  <activity role='ED medical staff' action='Obtain patient history' objects='patient' id='activity1-1'/>
  <activity role='ED medical staff' action='Perform physical examination' objects='patient' id='activity1-2'/>
  <activity role='ED medical staff' action='Assess neurological status' objects='patient' id='activity1-3'/>
  <parallelGateway id='parallelgateway1'>
    <branch id='branch1'>
      <activity role='ED medical staff' action='Measure blood glucose' objects='patient' id='activity2'/>
      <exclusiveGateway id='exclusivegateway1'>
        <branch condition='If hypoglycemia detected' id='branch2'>
          <activity role='ED medical staff' action='Correct hypoglycemia immediately' objects='patient, glucose correction agents' id='activity3'/>
        </branch>
        <branch condition='If first random glucose > 11 mmol/L' id='branch3'>
          <activity role='ED medical staff' action='Repeat blood glucose measurement' objects='patient' id='activity4'/>
        </branch>
        <branch condition='If normoglycemia' id='branch4'>
          <activity role='ED medical staff' action='Continue standard care' objects='patient' id='activity5'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch id='branch5'>
      <activity role='ED medical staff' action='Evaluate body temperature as part of routine vital signs every 4 hours for first 48 hours' objects='patient, thermometer' id='activity6'/>
      <exclusiveGateway id='exclusivegateway2'>
        <branch condition='If temperature > 37.5°C' id='branch6'>
          <parallelGateway id='parallelgateway2'>
            <branch id='branch7'>
              <activity role='ED medical staff' action='Initiate temperature-reducing measures' objects='patient, cooling methods' id='activity7'/>
            </branch>
            <branch id='branch8'>
              <activity role='ED medical staff' action='Investigate potential infection' objects='patient, laboratory tests' id='activity8'/>
              <activity role='ED medical staff' action='Initiate antipyretic and antimicrobial therapy as required' objects='patient, antipyretics, antimicrobials' id='activity9'/>
            </branch>
          </parallelGateway>
        </branch>
        <branch condition='If temperature ≤ 37.5°C' id='branch9'>
          <activity role='ED medical staff' action='Continue standard monitoring' objects='patient' id='activity10'/>
        </branch>
      </exclusiveGateway>
    </branch>
  </parallelGateway>
  <exclusiveGateway id='exclusivegateway3'>
    <branch condition='TIA with high-risk features or minor stroke' id='branch10'>
      <activity role='ED medical staff' action='Order brain CT scan' objects='patient, CT scanner' id='activity11-1'/>
      <activity role='ED medical staff' action='Review CT scan results' objects='patient, CT scan results' id='activity11-2'/>
      <exclusiveGateway id='exclusivegateway4'>
        <branch condition='No intracranial hemorrhage' id='branch11'>
          <activity role='ED medical staff' action='Initiate antiplatelet therapy if no contraindication' objects='patient, antiplatelet agents' id='activity11-3'/>
          <exclusiveGateway id='exclusivegateway5'>
            <branch condition='Immediate care available at specialized TIA/minor stroke clinic' id='branch12'>
              <activity role='ED medical staff' action='Refer patient to specialized TIA/minor stroke clinic for rapid diagnostic and medical evaluation' objects='patient, referral documentation' id='activity12'/>
            </branch>
            <branch condition='Delay to specialized clinic poses risk' id='branch13'>
              <activity role='ED medical staff' action='Perform evaluation at ED' objects='patient' id='activity13-1'/>
              <activity role='ED medical staff' action='Consider inpatient evaluation for stroke prevention' objects='patient' id='activity13-2'/>
            </branch>
            <branch condition='Medical attention sought after 48 hours from symptom onset' id='branch14'>
              <activity role='ED medical staff' action='Refer patient for evaluation at specialized TIA/minor stroke clinic or outpatient clinic with stroke-prevention services within 24 hours of initial presentation' objects='patient, referral documentation' id='activity14'/>
            </branch>
          </exclusiveGateway>
        </branch>
        <branch condition='Intracranial hemorrhage present' id='branch15'>
          <activity role='ED medical staff' action='Manage according to acute hemorrhagic stroke protocol' objects='patient' id='activity15'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='TIA without high-risk features' id='branch16'>
      <activity role='ED medical staff' action='Order brain CT scan' objects='patient, CT scanner' id='activity16-1'/>
      <activity role='ED medical staff' action='Review CT scan results' objects='patient, CT scan results' id='activity16-2'/>
      <exclusiveGateway id='exclusivegateway6'>
        <branch condition='No intracranial hemorrhage' id='branch17'>
          <activity role='ED medical staff' action='Initiate antiplatelet therapy if no contraindication' objects='patient, antiplatelet agents' id='activity16-3'/>
          <activity role='ED medical staff' action='Refer patient to outpatient clinic with stroke-prevention services for comprehensive evaluation and management within 1 month of symptom onset' objects='patient, referral documentation' id='activity17'/>
        </branch>
        <branch condition='Intracranial hemorrhage present' id='branch18'>
          <activity role='ED medical staff' action='Manage according to acute hemorrhagic stroke protocol' objects='patient' id='activity18'/>
        </branch>
      </exclusiveGateway>
    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='exclusivegateway7'>
    <branch condition='TIA patient with fluctuating/crescendo hemibody motor weakness, sensory loss, or speech difficulty' id='branch19'>
      <activity role='ED medical staff' action='Evaluate patient in ED and consider admission to hospital' objects='patient' id='activity19'/>
    </branch>
    <branch condition='Other TIA/minor stroke patients' id='branch20'>
      <activity role='ED medical staff' action='Continue with outpatient management as appropriate' objects='patient' id='activity20'/>
    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='exclusivegateway8'>
    <branch condition='Patient not on antiplatelet agent at presentation' id='branch21'>
      <exclusiveGateway id='exclusivegateway9'>
        <branch condition='Swallowing screening failed' id='branch22'>
          <activity role='ED medical staff' action='Offer rectal or gastric ASA as necessary' objects='patient, rectal/gastric ASA' id='activity21'/>
        </branch>
        <branch condition='Swallowing screening passed' id='branch23'>
          <activity role='ED medical staff' action='Administer oral antiplatelet therapy (ASA, clopidogrel, or dipyridamole/ASA as per protocol)' objects='patient, oral antiplatelet agents' id='activity22'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='Patient already on antiplatelet agent' id='branch24'>
      <activity role='ED medical staff' action='Continue current antiplatelet therapy as appropriate' objects='patient, medication list' id='activity23'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED medical staff' action='Provide cross-continuum secondary prevention assessment and therapies (per Modules 5 and 10)' objects='patient' id='activity24'/>
</process>
