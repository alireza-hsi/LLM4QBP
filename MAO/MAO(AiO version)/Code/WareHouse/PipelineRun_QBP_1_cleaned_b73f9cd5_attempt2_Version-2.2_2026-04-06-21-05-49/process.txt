<process>
  <exclusiveGateway id='exclusivegateway1'>
    <branch condition='patient arrives by EMS with suspected stroke or TIA' id='branch1'>
      <activity role='EMS' action='prenotify receiving hospital and transport or divert patient where feasible to a regional or district stroke centre' objects='suspected stroke or TIA patient information' id='activity1'/>
      <activity role='receiving hospital acute stroke team' action='prepare for patient arrival and attend the ED for appropriate patients' objects='EMS prenotification information' id='activity2'/>
    </branch>
    <branch condition='patient presents to outpatient clinic or is directly admitted to acute care or arrives at ED without EMS prenotification' id='branch2'>
      <activity role='receiving clinical team' action='initiate standardized stroke or TIA assessment and treatment protocol on presentation' objects='suspected stroke or TIA patient' id='activity3'/>
    </branch>
  </exclusiveGateway>

  <activity role='triage clinician' action='assess airway' objects='airway status' id='activity4_1'/>
  <activity role='triage clinician' action='assess breathing' objects='breathing status' id='activity4_2'/>
  <activity role='triage clinician' action='assess circulation' objects='circulation status' id='activity4_3'/>

  <activity role='clinician' action='determine focal neurological deficits' objects='neurological signs and symptoms' id='activity5_1'/>
  <activity role='clinician' action='assess stroke severity on a standardized stroke scale' objects='NIHSS or CNS for stroke' id='activity5_2'/>
  <activity role='diagnostic imaging team' action='perform brain imaging immediately' objects='CT or MRI' id='activity6'/>
  <activity role='qualified health care provider' action='interpret brain imaging immediately' objects='CT and or MRI results' id='activity7'/>

  <activity role='clinical staff' action='ensure chest x-ray does not delay thrombolysis assessment' objects='chest x-ray request workflow, thrombolysis assessment workflow' id='activity10'/>

  <parallelGateway id='parallelgateway1'>
    <branch id='branch3'>
      <activity role='clinical staff' action='perform ECG' objects='cardiac rhythm to detect atrial fibrillation and other acute arrhythmias' id='activity8'/>
    </branch>
    <branch id='branch4'>
      <activity role='laboratory team' action='order required blood work' objects='CBC, electrolytes, creatinine, glucose, INR, partial thromboplastin time, troponin test if clinically indicated' id='activity9_1'/>
      <activity role='laboratory team' action='obtain and process blood specimens' objects='ordered blood samples and laboratory tests' id='activity9_2'/>
      <activity role='clinician' action='review blood work results when available and incorporate findings into immediate assessment and treatment decisions' objects='CBC, electrolytes, creatinine, glucose, INR, partial thromboplastin time, troponin result if ordered' id='activity9_3'/>
    </branch>
  </parallelGateway>

  <exclusiveGateway id='exclusivegateway2'>
    <branch condition='patient has stroke' id='branch8'>
      <activity role='clinical staff' action='make patient NPO initially' objects='oral medication, fluid, food' id='activity11'/>
      <activity role='clinical staff' action='apply bedside swallowing screening protocol' objects='simple, valid, reliable bedside swallowing test protocol' id='activity12_1'/>
      <activity role='clinical staff' action='withhold oral medication, fluid, or food until screening result is available' objects='oral intake restriction pending swallowing screen' id='activity12_2'/>
      <exclusiveGateway id='exclusivegateway3'>
        <branch condition='patient is not alert within the first 24 hours' id='branch9'>
          <activity role='clinical staff' action='monitor patient closely and screen swallowing ability when clinically appropriate' objects='level of alertness, swallowing ability' id='activity13'/>
        </branch>
        <branch condition='patient is alert within the first 24 hours' id='branch10'>
          <activity role='clinical staff' action='continue standard post-screen feeding and medication precautions' objects='swallow screening result' id='activity14'/>
        </branch>
      </exclusiveGateway>
      <exclusiveGateway id='exclusivegateway4'>
        <branch condition='features indicating dysphagia or pulmonary aspiration are present' id='branch11'>
          <activity role='speech-language pathologist or appropriately trained specialist' action='perform full clinical swallowing assessment and advise on swallowing ability and required consistency of diet and fluids' objects='swallowing function, diet consistency, fluid consistency' id='activity15'/>
        </branch>
        <branch condition='features indicating dysphagia or pulmonary aspiration are absent' id='branch12'>
          <activity role='clinical staff' action='proceed according to bedside swallowing screen result' objects='swallowing management plan' id='activity16'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='patient has suspected TIA or minor nondisabling stroke without stroke-related dysphagia pathway applicability' id='branch13'>
      <activity role='clinical staff' action='document that stroke dysphagia recommendations are not applicable to isolated TIA or minor nondisabling stroke as specified' objects='clinical assessment findings' id='activity17'/>
    </branch>
  </exclusiveGateway>

  <exclusiveGateway id='exclusivegateway5'>
    <branch condition='patient has TIA' id='branch14'>
      <activity role='clinician' action='apply standardized TIA triage criteria' objects='TIA triage tool, patient clinical findings' id='activity18_1'/>
      <activity role='clinician' action='assign appropriate location of care' objects='triage result and care setting decision' id='activity18_2'/>
    </branch>
    <branch condition='patient has stroke or minor nondisabling stroke rather than TIA alone' id='branch15'>
      <activity role='clinician' action='determine care setting based on stroke assessment results' objects='clinical findings, imaging results' id='activity19'/>
    </branch>
  </exclusiveGateway>

  <exclusiveGateway id='exclusivegateway6'>
    <branch condition='patient has TIA or minor nondisabling stroke presenting within 48 hours of symptom onset or with fluctuating motor or speech symptoms' id='branch16'>
      <exclusiveGateway id='exclusivegateway7'>
        <branch condition='patient is clearly not a candidate for carotid artery revascularization' id='branch17'>
          <activity role='clinician' action='document non-candidacy for carotid artery revascularization and select alternate assessment pathway' objects='revascularization candidacy assessment' id='activity20'/>
        </branch>
        <branch condition='patient is a candidate or candidacy is not excluded' id='branch23'>
          <activity role='diagnostic imaging team' action='perform immediate vascular imaging of the neck arteries' objects='carotid ultrasound, CTA, or MRA' id='activity21'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='all other patients with TIA or ischemic stroke presenting beyond 48 hours' id='branch24'>
      <activity role='diagnostic imaging team' action='perform vascular imaging of the brain and neck arteries as soon as possible' objects='brain vascular imaging, neck artery imaging' id='activity22'/>
    </branch>
  </exclusiveGateway>

  <activity role='clinical team' action='assess need for secondary prevention across care settings' objects='secondary prevention requirements for admitted or discharged patient' id='activity23_1'/>
  <activity role='clinical team' action='initiate or arrange secondary prevention therapies and referrals' objects='secondary prevention assessments and therapies, Modules 5 and 10' id='activity23_2'/>
