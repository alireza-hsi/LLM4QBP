<process>
  <activity role='ED clinician' action='triage and identify suspected acute heart failure presentation' objects='triage note, presenting symptoms' id='activity1_1'/>
  <activity role='ED clinician' action='collect standardized risk assessment inputs' objects='history, comorbidities, medications, vitals, labs/imaging as available' id='activity1_2'/>
  <activity role='ED clinician' action='apply standardized risk assessment/stratification tool consistently (multi-condition including HF) and document risk level' objects='risk stratification tool/process, documentation' id='activity1_3'/>
  <parallelGateway id='parallelgateway1_fixed'>
    <branch id='branch1_fixed'>
      <activity role='nurse' action='measure heart rate, blood pressure, and oxygen saturation frequently until patient is stabilized' objects='vital signs, pulse oximetry' id='activity3'/>
    </branch>
    <branch id='branch2_fixed'>
      <activity role='ED clinician' action='order serum creatinine and electrolyte levels' objects='blood sample, lab requisition' id='activity4'/>
    </branch>
    <branch id='branch3_fixed'>
      <activity role='ED clinician' action='order troponin measurements' objects='blood sample, lab requisition' id='activity5'/>
    </branch>
    <branch id='branch4_fixed'>
      <activity role='ED clinician' action='order complete blood count' objects='blood sample, lab requisition' id='activity6'/>
    </branch>
    <branch id='branch5_fixed'>
      <activity role='ED clinician' action='perform electrocardiogram' objects='ECG machine, leads' id='activity7'/>
    </branch>
    <branch id='branch6_fixed'>
      <activity role='radiology' action='perform chest x-ray examination' objects='x-ray imaging order' id='activity8'/>
    </branch>
  </parallelGateway>
  <activity role='ED clinician' action='administer initial diuretic therapy (e.g., IV furosemide) and begin monitoring response' objects='diuretic order, medication administration record' id='activity2_1'/>
  <activity role='ED clinician' action='reassess responsiveness to diuresis (urine output, symptoms, weight if available)' objects='urine output, clinical response' id='activity2_2'/>
  <inclusiveGateway id='inclusivegateway0_diuresisResponse'>
    <branch condition='Response adequate at reassessment' id='branch0a_incl'>
      <activity role='ED clinician' action='continue ED management plan based on current diuresis response' objects='treatment plan' id='activity2_3_modified'/>
    </branch>
    <branch condition='Response poor or insufficient at reassessment' id='branch0b_incl'>
      <activity role='ED clinician' action='escalate treatment and flag poor diuretic response as high-risk marker' objects='escalation plan, high-risk marker flag' id='activity2_4'/>
    </branch>
  </inclusiveGateway>
  <exclusiveGateway id='exclusivegateway1'>
    <branch condition='No recent echocardiogram available' id='branch7_echo'>
      <activity role='cardiology/echo lab' action='perform echocardiogram' objects='echocardiography study request' id='activity9'/>
    </branch>
    <branch condition='Recent echocardiogram available' id='branch8_echo'>
      <activity role='ED clinician' action='review most recent echocardiogram results' objects='prior echocardiogram report' id='activity10'/>
    </branch>
  </exclusiveGateway>
  <parallelGateway id='parallelgateway2'>
    <branch id='branch2a'>
      <activity role='ED clinician' action='assess respiratory status and oxygenation for high-risk markers' objects='respiratory exam, oxygen saturation, work of breathing' id='activity11_1'/>
    </branch>
    <branch id='branch2b'>
      <activity role='ED clinician' action='assess hemodynamics and rhythm for high-risk markers' objects='blood pressure trends, perfusion signs, ECG rhythm assessment' id='activity11_2'/>
    </branch>
    <branch id='branch2c'>
      <activity role='ED clinician' action='review biomarkers/imaging and diuresis response for high-risk markers' objects='troponin result, chest x-ray findings, diuresis response' id='activity11_3'/>
    </branch>
  </parallelGateway>
  <activity role='ED clinician' action='consolidate findings and screen for presence of any high-risk markers' objects='clinical exam, ECG/troponin/imaging results, response to diuresis' id='activity11_4'/>
  <activity role='ED clinician' action='calculate acute heart failure risk score (e.g., EHMRG) to predict 7-day mortality risk and support clinical decision-making' objects='EHMRG inputs, calculator' id='activity12'/>
  <exclusiveGateway id='exclusivegateway4_moved'>
    <branch condition='Transfer to higher level provider/facility required' id='branch15_moved'>
      <activity role='ED clinician/transfer coordinator' action='identify transfer trigger and contact receiving service/provider to accept patient' objects='transfer protocol, receiving service contact' id='activity19_1_moved'/>
      <activity role='ED clinician/nurse' action='stabilize patient for transport and prepare transfer documentation' objects='stabilization checklist, transfer summary, copies of results' id='activity19_2_moved'/>
      <activity role='transfer coordinator/paramedics' action='arrange transport and complete clinical handover to receiving team' objects='transport booking, handover report' id='activity19_3_moved'/>
    </branch>
    <branch condition='Transfer not required' id='branch16_moved'>
      <exclusiveGateway id='exclusivegateway1_riskDisposition'>
        <branch condition='EHMRG low-risk (quintiles 1-2) AND responded to initial ED treatment' id='branch9'>
          <activity role='ED clinician' action='assess other considerations before discharge (e.g., advanced-directives, severe dementia, estimated impact of admission on life-expectancy)' objects='patient history, goals-of-care documentation' id='activity13'/>
          <exclusiveGateway id='exclusivegateway2'>
            <branch condition='No other considerations requiring admission' id='branch10'>
              <activity role='ED clinician' action='treat in ED or outpatient setting and discharge home without inpatient admission (low intensity)' objects='discharge instructions' id='activity14'/>
              <activity role='ED clinician/care coordinator' action='arrange follow-up visit with primary care provider within days of ED discharge' objects='follow-up appointment details' id='activity15'/>
            </branch>
            <branch condition='Other considerations present requiring admission' id='branch11'>
              <activity role='ED clinician' action='admit to hospital despite low-risk score' objects='admission order' id='activity16'/>
            </branch>
          </exclusiveGateway>
        </branch>
        <branch condition='EHMRG higher-risk (quintiles 3-5) OR high-risk markers present' id='branch12'>
          <exclusiveGateway id='exclusivegateway3'>
            <branch condition='Requires ventilation (noninvasive or invasive) and/or ICU-level care' id='branch13'>
              <activity role='ED clinician/ICU team' action='admit to intensive care unit with higher nurse-to-patient staffing (high intensity) and initiate ventilation as needed' objects='ICU bed request, ventilatory support' id='activity17'/>
            </branch>
            <branch condition='Does not require ventilation/ICU but requires inpatient admission' id='branch14'>
              <activity role='ED clinician/inpatient team' action='admit to inpatient care with normal nurse-to-patient staffing (average intensity)' objects='inpatient bed request' id='activity18'/>
            </branch>
          </exclusiveGateway>
        </branch>
      </exclusiveGateway>
      <activity role='ED clinician' action='continue care at current hospital at assigned intensity level' objects='care plan' id='activity20'/>
    </branch>
  </exclusiveGateway>
</process>
