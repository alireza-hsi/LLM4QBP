<process>
  <activity role='admitting clinician' action='determine applicability of acute care admission pathway' objects='acute stroke diagnosis, admitted TIA status, minor nondisabling stroke status' id='activity1'/>
  <exclusiveGateway id='exclusivegateway1'>
    <branch condition='patient has acute stroke or admitted TIA meeting criteria for acute inpatient admission' id='branch1'>
      <activity role='ED or admitting clinicians' action='screen or assess for diabetes, dental issues, cognitive impairment, and depression upon presentation to the ED' objects='diabetes screening, dental assessment, cognitive impairment screening, depression screening' id='activity15_1'/>
      <activity role='ED or admitting clinicians' action='forward screening and assessment results to the TIA clinic for follow-up' objects='screening results, TIA clinic follow-up information' id='activity15_2'/>
      <activity role='admitting team' action='verify stroke unit bed availability' objects='co-located stroke unit bed availability status' id='activity2_1'/>
      <activity role='admitting team' action='complete admission to a specialized geographically defined stroke unit' objects='co-located stroke unit bed, stroke admission order' id='activity2_2'/>
      <activity role='stroke unit leadership' action='assign specialized interprofessional stroke team with 7 day per week access and stroke expertise maintained through regular stroke education' objects='medicine, nursing, occupational therapy, physiotherapy, speech-language pathology, social work, dietitian staffing' id='activity3'/>
      <parallelGateway id='parallelgateway1'>
        <branch id='branch2'>
          <activity role='clinicians' action='select standardized valid assessment tools' objects='standardized validated assessment tools' id='activity4_1'/>
          <activity role='clinicians' action='document stroke-related impairments and functional status findings in EHR-linked format when available' objects='assessment findings, electronic EHR-linked tools when available' id='activity4_2'/>
          <activity role='clinicians' action='complete AlphaFIM on or by day 3 after admission with target day 3 and admission day counted as day 1' objects='AlphaFIM assessment' id='activity5'/>
        </branch>
        <branch id='branch3'>
          <activity role='physiotherapist, occupational therapist, and speech-language pathologist' action='assess patient within 48 hours of hospital admission and formulate a management plan' objects='therapy assessments, management plan' id='activity6'/>
        </branch>
        <branch id='branch4'>
          <activity role='clinical team' action='assess risk of developing venous thromboembolism' objects='venous thromboembolism risk assessment' id='activity7'/>
          <activity role='clinical team' action='encourage adequate hydration to help prevent venous thromboembolism' objects='hydration plan' id='activity8'/>
        </branch>
        <branch id='branch5'>
          <activity role='nursing or designated clinician' action='assess nutrition and hydration status within the first 48 hours of admission using a valid screening tool' objects='nutrition screening tool, hydration screening tool' id='activity9'/>
          <inclusiveGateway id='inclusivegateway1'>
            <branch condition='nutritional concerns and or hydration deficits and or dysphagia and or other comorbidities affecting nutrition or fluids identified' id='branch6'>
              <activity role='clinical team' action='refer patient to a dietitian within 7 days of admission for recommendations' objects='dietitian referral' id='activity10'/>
              <activity role='dietitian' action='review patient nutrition and fluid requirements and provide recommendations' objects='dietitian assessment, nutrition and fluid recommendations' id='activity23'/>
              <exclusiveGateway id='exclusivegateway4'>
                <branch condition='patient is unable to meet nutritional and fluid requirements' id='branch18'>
                  <activity role='dietitian' action='consider enteral nutrition support' objects='enteral nutrition support recommendation' id='activity14'/>
                </branch>
                <branch condition='patient can meet nutritional and fluid requirements' id='branch19'>
                  <activity role='dietitian' action='document nutrition and fluid management recommendations without enteral nutrition support' objects='nutrition and fluid management plan' id='activity22'/>
                </branch>
              </exclusiveGateway>
            </branch>
            <branch condition='no nutritional or hydration-related concerns identified' id='branch20'>
              <activity role='nursing or designated clinician' action='document no dietitian referral required and continue routine monitoring' objects='nutrition and hydration monitoring record' id='activity24'/>
            </branch>
          </inclusiveGateway>
        </branch>
      </parallelGateway>
      <exclusiveGateway id='exclusivegateway2'>
        <branch condition='24 to 48 hours from stroke onset and no contraindication to mobilization' id='branch12'>
          <activity role='clinical team' action='mobilize patient early between 24 hours and 48 hours of stroke onset' objects='early mobilization intervention' id='activity16'/>
        </branch>
        <branch condition='within 24 hours of stroke onset' id='branch13'>
          <exclusiveGateway id='exclusivegateway3'>
            <branch condition='clinical judgement supports selected mobilization and no contraindication' id='branch14'>
              <activity role='clinical team' action='provide limited very early mobilization based on clinical judgement' objects='selected mobilization intervention' id='activity17'/>
            </branch>
            <branch condition='frequent out-of-bed activity not appropriate or contraindicated' id='branch15'>
              <activity role='clinical team' action='defer frequent out-of-bed activity within 24 hours of stroke onset and continue monitoring for later mobilization' objects='monitoring plan' id='activity18'/>
            </branch>
          </exclusiveGateway>
        </branch>
        <branch condition='mobilization contraindicated' id='branch16'>
          <activity role='clinical team' action='withhold mobilization and document contraindications while reassessing readiness' objects='contraindication documentation, reassessment plan' id='activity19'/>
        </branch>
      </exclusiveGateway>
      <activity role='stroke team' action='initiate cross-continuum secondary prevention assessment' objects='secondary prevention assessment plan, Modules 5 and 10 inputs' id='activity20_1'/>
      <activity role='stroke team' action='provide or arrange cross-continuum secondary prevention therapies and follow-up plan whether patient is transferred to in-hospital rehabilitation or discharged' objects='secondary prevention therapies, follow-up plan, Modules 5 and 10 plans' id='activity20_2'/>
    </branch>
    <branch condition='patient has non-admitted TIA or minor nondisabling stroke not applicable to Module 3a recommendations' id='branch17'>
      <activity role='clinical team' action='exclude patient from Module 3a acute care treatment recommendations and route to appropriate TIA or minor stroke management pathway' objects='alternative pathway referral' id='activity21'/>
    </branch>
  </exclusiveGateway>
</process>
