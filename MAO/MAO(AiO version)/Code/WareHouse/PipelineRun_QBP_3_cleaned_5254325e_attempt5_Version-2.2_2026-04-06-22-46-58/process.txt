        <branch id='branch7'>

        </branch>
        <branch id='branch8'>
          <activity role='stroke unit clinicians' action='Encourage early mobilization and adequate hydration to prevent venous thromboembolism' objects='patient, mobilization protocol, hydration plan' id='activity12'/>
