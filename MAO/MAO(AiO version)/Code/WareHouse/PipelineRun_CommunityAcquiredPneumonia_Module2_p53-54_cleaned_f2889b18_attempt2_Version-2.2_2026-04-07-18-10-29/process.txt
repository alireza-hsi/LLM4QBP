<process>
  <activity role='medical staff' action='Assess patient presenting with symptoms of community-acquired pneumonia (CAP)' objects='patient' id='activity1'/>
  <activity role='medical staff' action='Measure oxygen saturation' objects='patient' id='activity2'/>
  <activity role='medical staff' action='Order chest x-ray' objects='patient, chest x-ray' id='activity3-1'/>
  <activity role='medical staff' action='Do not order CT scan for diagnosis of CAP' objects='patient' id='activity6'/>
  <activity role='medical staff' action='Interpret chest x-ray results' objects='chest x-ray results' id='activity3-2'/>
  <exclusiveGateway id='exclusivegateway1'>
    <branch condition='Patient has sepsis' id='branch1'>
      <activity role='medical staff' action='Initiate antibiotics before chest x-ray results' objects='patient, antibiotics' id='activity4'/>
    </branch>
    <branch condition='Patient does not have sepsis' id='branch2'>
      <activity role='medical staff' action='Wait for chest x-ray results before initiating antibiotics' objects='patient, chest x-ray results' id='activity5'/>
    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='exclusivegateway2'>
    <branch condition='Patient admitted to hospital' id='branch3'>
      <activity role='medical staff' action='Order CBC' objects='patient, CBC' id='activity7-1'/>
      <activity role='medical staff' action='Order electrolytes' objects='patient, electrolytes' id='activity7-2'/>
      <activity role='medical staff' action='Order renal function tests' objects='patient, renal function tests' id='activity7-3'/>
      <exclusiveGateway id='exclusivegateway3'>
        <branch condition='Adequate sputum sample can be obtained' id='branch4'>
          <activity role='medical staff' action='Consider sputum culture' objects='patient, sputum sample' id='activity8'/>
          <exclusiveGateway id='exclusivegateway4'>
            <branch condition='Sputum culture obtained' id='branch5'>
              <activity role='medical staff' action='Consider Gram stain of sputum sample' objects='sputum sample' id='activity9'/>
            </branch>
            <branch condition='Sputum culture not obtained' id='branch6'>
              <activity role='medical staff' action='Proceed without Gram stain' objects='patient' id='activity10'/>
            </branch>
          </exclusiveGateway>
        </branch>
        <branch condition='Adequate sputum sample cannot be obtained' id='branch7'>
          <activity role='medical staff' action='Proceed without sputum culture and Gram stain' objects='patient' id='activity11'/>
        </branch>
      </exclusiveGateway>
      <exclusiveGateway id='exclusivegateway5'>
        <branch condition='Patient meets >2 CRB-65 criteria' id='branch8'>
          <activity role='medical staff' action='Consider blood gases, lactate, liver enzyme, and liver function tests' objects='patient, blood tests' id='activity12'/>
        </branch>
        <branch condition='Patient does not meet >2 CRB-65 criteria' id='branch9'>
          <activity role='medical staff' action='Do not order additional tests for CRB-65' objects='patient' id='activity13'/>
        </branch>
      </exclusiveGateway>
      <exclusiveGateway id='exclusivegateway6'>
        <branch condition='Patient meets >2 SIRS criteria or requires ICU admission' id='branch10'>
          <activity role='medical staff' action='Order blood cultures' objects='patient, blood cultures' id='activity14'/>
        </branch>
        <branch condition='Patient does not meet >2 SIRS criteria and does not require ICU admission' id='branch11'>
          <activity role='medical staff' action='Do not order blood cultures' objects='patient' id='activity15'/>
        </branch>
      </exclusiveGateway>
      <inclusiveGateway id='inclusivegateway1'>
        <branch condition='Patient is high-severity' id='branch12'>
          <activity role='medical staff' action='Collect urine sample' objects='patient, urine sample' id='activity16-1'/>
          <activity role='medical staff' action='Send sample for antigen testing' objects='urine sample, antigen test' id='activity16-2'/>
        </branch>
        <branch condition='Enhanced surveillance directive for Legionella in inpatients' id='branch13'>
          <activity role='medical staff' action='Collect urine sample' objects='patient, urine sample' id='activity17-1'/>
          <activity role='medical staff' action='Send sample for antigen testing' objects='urine sample, antigen test' id='activity17-2'/>
        </branch>
        <branch condition='Patient not responding to drug therapy after 48–72 hours' id='branch14'>
          <activity role='medical staff' action='Collect urine sample' objects='patient, urine sample' id='activity18-1'/>
          <activity role='medical staff' action='Send sample for antigen testing' objects='urine sample, antigen test' id='activity18-2'/>
        </branch>
        <branch condition='Inpatient during peak season (mid-June to early October)' id='branch15'>
          <activity role='medical staff' action='Collect urine sample' objects='patient, urine sample' id='activity19-1'/>
          <activity role='medical staff' action='Send sample for antigen testing' objects='urine sample, antigen test' id='activity19-2'/>
        </branch>
      </inclusiveGateway>
      <activity role='medical staff' action='Ensure only one urine antigen test is performed per patient' objects='patient, urine antigen test' id='activity24'/>
      <exclusiveGateway id='exclusivegateway7'>
        <branch condition='Patient is intubated (high severity)' id='branch16'>
          <activity role='medical staff' action='Consider PCR testing for Legionella' objects='patient, PCR test' id='activity20'/>
        </branch>
        <branch condition='Patient is not intubated' id='branch17'>
          <activity role='medical staff' action='Do not order PCR for Legionella' objects='patient' id='activity21'/>
        </branch>
      </exclusiveGateway>
      <activity role='medical staff' action='Recommend field evaluation to assess clinical utility of urine antigen testing in CAP' objects='urine antigen test, field evaluation' id='activity22'/>
    </branch>
    <branch condition='Patient not admitted to hospital' id='branch18'>
      <activity role='medical staff' action='Follow outpatient CAP diagnostic protocol' objects='patient' id='activity23'/>
    </branch>
  </exclusiveGateway>
