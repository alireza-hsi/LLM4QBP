<process>
  <activity role='patient' action='present with suspected COPD exacerbation' objects='ED, outpatient clinic, or direct admission' id='activity1'/>
  <exclusiveGateway id='exclusivegateway1'>
    <branch condition='initial presentation at the ED requires immediate resuscitation' id='branch1'>
      <activity role='ED team' action='trial immediate resuscitation on initial presentation at the ED' objects='resuscitation measures' id='activity2'/>
    </branch>
    <branch condition='no immediate resuscitation is required or patient presents in outpatient clinic or by direct admission' id='branch2'>
      <activity role='clinical team' action='proceed with initial presentation and assessment' objects='suspected COPD exacerbation' id='activity4'/>
    </branch>
  </exclusiveGateway>
  <parallelGateway id='parallelgateway1'>
    <branch id='branch3'>
      <activity role='clinical team' action='check vital signs and assess hypoventilation' objects='vital signs and respiratory status' id='activity5_1'/>
      <activity role='clinical team' action='check level of consciousness or cognition' objects='mental status' id='activity5_2'/>
      <activity role='clinical team' action='check blood-oxygen saturation via pulse oximetry' objects='pulse oximetry reading' id='activity5_3'/>
      <activity role='clinical team' action='assess whether patient has purulent sputum' objects='sputum characteristics' id='activity5_4'/>
      <activity role='clinical team' action='perform physical examination' objects='patient' id='activity6'/>
    </branch>
    <branch id='branch4'>
      <activity role='clinical team' action='check patient history' objects='symptoms, COPD history, comorbidities' id='activity7'/>
      <activity role='clinical team' action='document and reconcile medications currently used by patient' objects='current medication list' id='activity8'/>
      <activity role='clinical team' action='identify patient wishes with respect to goals of care and limitations of treatment' objects='goals of care and code status' id='activity9'/>
    </branch>
    <branch id='branch5'>
      <exclusiveGateway id='exclusivegateway4'>
        <branch condition='patient can undergo standard chest x-ray positioning and transport' id='branch14'>
          <activity role='clinical team' action='perform posteroanterior and lateral chest x-ray' objects='standard chest x-ray views' id='activity10_1'/>
        </branch>
        <branch condition='patient is too unwell to leave current care area for standard imaging' id='branch15'>
          <activity role='clinical team' action='perform portable chest x-ray' objects='portable chest x-ray' id='activity10_2'/>
        </branch>
      </exclusiveGateway>
      <exclusiveGateway id='exclusivegateway5'>
        <branch condition='there is concern with pneumothorax' id='branch16'>
          <activity role='clinical team' action='perform expiratory chest x-ray view' objects='expiratory chest x-ray view' id='activity10_3'/>
        </branch>
        <branch condition='there is no concern with pneumothorax' id='branch17'>
          <activity role='clinical team' action='continue chest imaging assessment without expiratory view' objects='available chest x-ray views' id='activity10_4'/>
        </branch>
      </exclusiveGateway>
      <activity role='clinical team' action='conduct baseline blood work' objects='complete blood count, electrolytes, creatinine, and blood urea nitrogen if available' id='activity11'/>
      <activity role='clinical team' action='perform electrocardiogram' objects='assessment for arrhythmias, myocardial ischemia, right ventricular strain, and other cardiac findings' id='activity12'/>
    </branch>
  </parallelGateway>
  <inclusiveGateway id='inclusivegateway1'>
    <branch condition='low oxygen saturation on oximetry and or acute respiratory failure is suspected' id='branch6'>
      <activity role='clinical team' action='check arterial blood gases where appropriate' objects='arterial blood gases' id='activity13'/>
    </branch>
    <branch condition='pneumonia or suspected sepsis is present' id='branch7'>
      <activity role='clinical team' action='draw blood cultures' objects='blood cultures' id='activity14'/>
    </branch>
    <branch condition='suspected cardiac disorder is present' id='branch8'>
      <activity role='clinical team' action='examine cardiac markers' objects='cardiac markers' id='activity15'/>
    </branch>
    <branch condition='other diagnoses or comorbidities need to be identified or ruled out' id='branch9'>
      <activity role='clinical team' action='perform other diagnostic interventions as appropriate' objects='additional diagnostic tests for other suspected diagnoses or comorbidities' id='activity16'/>
    </branch>
  </inclusiveGateway>
  <activity role='clinical team' action='provide initial medical management for suspected COPD exacerbation' objects='initial treatment and supportive care' id='activity16_1'/>
  <parallelGateway id='parallelgateway2'>
    <branch id='branch18'>
      <activity role='admitting clinician' action='assess response to initial medical management and clinical severity' objects='failure to respond, breathlessness, resting dyspnea, general condition, severe underlying COPD, activity level, cyanosis, peripheral edema, new physical signs, decreased level of consciousness, acute confusion, rapid onset, and frequent exacerbations' id='activity17_1'/>
    </branch>
    <branch id='branch19'>
      <activity role='admitting clinician' action='assess home support and social circumstances' objects='insufficient home support, inability to cope at home, social circumstances, and older age' id='activity17_2'/>
    </branch>
    <branch id='branch20'>
      <activity role='admitting clinician' action='assess comorbidities and objective admission indicators' objects='significant comorbidity, long-term oxygen therapy, SaO2 less than 90 percent, pH, PaO2, and chest x-ray changes' id='activity17_3'/>
    </branch>
  </parallelGateway>
  <exclusiveGateway id='exclusivegateway6'>
    <branch condition='immediate resuscitation was performed on initial presentation at the ED' id='branch21'>
      <activity role='ED team' action='re-evaluate need for admission after immediate resuscitation' objects='clinical status and initial response' id='activity3'/>
    </branch>
    <branch condition='immediate resuscitation was not performed on initial presentation at the ED, or patient presents in outpatient clinic, or patient was admitted directly' id='branch22'>
      <activity role='clinical team' action='proceed with completed assessment and initial management findings' objects='assessment findings and treatment response' id='activity22'/>
    </branch>
  </exclusiveGateway>
  <activity role='admitting clinician' action='integrate findings with local resource availability to determine level of care required' objects='clinical judgment, NICE and or GOLD criteria, and local resources' id='activity17_4'/>
  <exclusiveGateway id='exclusivegateway2'>
    <branch condition='patient was admitted directly or inpatient admission is required after initial management or re-evaluation' id='branch10'>
      <activity role='admitting clinician' action='assign or confirm the appropriate level of inpatient care' objects='admission status and level of care' id='activity18'/>
    </branch>
    <branch condition='patient was not admitted directly and hospital admission is not required after initial management or re-evaluation' id='branch11'>
      <activity role='clinical team' action='continue management without hospital admission' objects='ED or outpatient care plan' id='activity19'/>
    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='exclusivegateway3'>
    <branch condition='patient has stabilized and has no prior objective documentation of COPD through spirometry' id='branch12'>
      <activity role='clinical team' action='perform spirometry once the patient has stabilized' objects='spirometry' id='activity20'/>
    </branch>
    <branch condition='patient has prior objective documentation of COPD through spirometry or is not yet stabilized' id='branch13'>
      <activity role='clinical team' action='defer spirometry during the initial assessment phase' objects='spirometry not required at this time' id='activity21'/>
    </branch>
  </exclusiveGateway>
</process>
