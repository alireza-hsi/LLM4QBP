<process>
  <activity role='emergency department clinicians' action='identify acute stroke patients and the specific group of TIA patients who require acute inpatient admission' objects='admission criteria, clinical findings' id='activity1'/>
  <activity role='emergency department clinicians' action='screen for diabetes and dental issues upon presentation' objects='diabetes screening results, dental issue screening results' id='activity12_1'/>
  <activity role='emergency department clinicians' action='assess cognitive impairment and depression upon presentation and forward results to the TIA clinic for follow-up' objects='cognitive assessment results, depression assessment results, TIA clinic follow-up information' id='activity12_2'/>
  <exclusiveGateway id='exclusivegateway1'>
    <branch condition='patient has acute stroke or admitted TIA requiring inpatient care' id='branch1'>
      <activity role='admitting team' action='confirm admission to a specialized stroke unit dedicated to stroke management' objects='admission decision, stroke unit admission criteria' id='activity2_1'/>
      <activity role='admitting team' action='allocate a geographically defined co-located stroke unit bed' objects='co-located stroke unit bed' id='activity2_2'/>
      <activity role='stroke unit manager' action='assign patient to a stroke unit staffed by a dedicated interprofessional team with stroke expertise available 7 days a week' objects='stroke unit team, staffing resources' id='activity3'/>
      <parallelGateway id='parallelgateway2'>
        <branch id='branch7_1'>
          <activity role='physiotherapist' action='assess patient within 48 hours of hospital admission' objects='physiotherapy assessment' id='activity11_1'/>
        </branch>
        <branch id='branch7_2'>
          <activity role='occupational therapist' action='assess patient within 48 hours of hospital admission' objects='occupational therapy assessment' id='activity11_2'/>
        </branch>
        <branch id='branch7_3'>
          <activity role='speech-language pathologist' action='assess patient within 48 hours of hospital admission' objects='speech-language pathology assessment' id='activity11_3'/>
        </branch>
      </parallelGateway>
      <activity role='physiotherapist, occupational therapist, and speech-language pathologist' action='formulate a management plan' objects='integrated therapy management plan' id='activity11_4'/>
      <activity role='clinicians' action='select standardized valid assessment tools to evaluate stroke-related impairments and functional status' objects='standardized validated assessment tools' id='activity4_1'/>
      <activity role='clinicians' action='document assessment findings in EHR-linked record' objects='assessment findings, EHR-linked documentation' id='activity4_2'/>
      <activity role='stroke unit team' action='assess risk of developing venous thromboembolism' objects='venous thromboembolism risk assessment' id='activity6'/>
      <activity role='nursing staff and clinicians' action='screen nutrition status within the first 48 hours of admission using a valid screening tool' objects='nutrition screening tool' id='activity8_1'/>
      <activity role='nursing staff and clinicians' action='assess hydration status within the first 48 hours of admission using a valid screening tool' objects='hydration screening tool' id='activity8_2'/>
      <exclusiveGateway id='exclusivegateway3'>
        <branch condition='early mobilization is not contraindicated and patient is between 24 and 48 hours from stroke onset' id='branch9'>
          <activity role='stroke unit team' action='mobilize patient early between 24 hours and 48 hours of stroke onset' objects='early mobilization plan' id='activity13'/>
          <activity role='stroke unit team' action='encourage early mobilization and adequate hydration to help prevent venous thromboembolism' objects='mobilization activities, hydration plan' id='activity14'/>
        </branch>
        <branch condition='patient is within 24 hours of stroke onset' id='branch10'>
          <exclusiveGateway id='exclusivegateway4'>
            <branch condition='clinical judgement supports very early mobilization' id='branch11'>
              <activity role='stroke unit team' action='mobilize patient within 24 hours of stroke onset using clinical judgement and avoid frequent out-of-bed activity' objects='individualized mobilization plan' id='activity15'/>
            </branch>
            <branch condition='clinical judgement does not support very early mobilization' id='branch12'>
              <activity role='stroke unit team' action='defer out-of-bed mobilization until 24 to 48 hours after stroke onset and maintain hydration' objects='delayed mobilization plan, hydration plan' id='activity16'/>
            </branch>
          </exclusiveGateway>
        </branch>
        <branch condition='early mobilization is contraindicated' id='branch13'>
          <activity role='stroke unit team' action='withhold early mobilization and document contraindications while maintaining alternative preventive care including hydration' objects='contraindication documentation, preventive care plan' id='activity17'/>
        </branch>
      </exclusiveGateway>
      <activity role='stroke unit team' action='encourage adequate hydration to help prevent venous thromboembolism' objects='hydration plan' id='activity7'/>
      <exclusiveGateway id='exclusivegateway2'>
        <branch condition='nutritional concerns, hydration deficits, dysphagia, or other comorbidities are present' id='branch5'>
          <activity role='stroke unit team' action='refer patient to a dietitian within 7 days of admission for recommendations and consideration of enteral nutrition support when unable to meet nutritional and fluid requirements' objects='dietitian referral, enteral nutrition support consideration' id='activity9'/>
        </branch>
        <branch condition='no nutritional concerns, hydration deficits, dysphagia, or other comorbidities are present' id='branch6'>
          <activity role='stroke unit team' action='continue routine nutrition and hydration monitoring' objects='nutrition and hydration status' id='activity10'/>
        </branch>
      </exclusiveGateway>
      <activity role='clinicians' action='complete AlphaFIM on or by day 3 after admission with target day 3 and admission day counted as day 1' objects='AlphaFIM assessment' id='activity5'/>
      <activity role='stroke unit leadership' action='verify the core stroke unit team includes medicine, nursing, occupational therapy, physiotherapy, speech-language pathology, social work, and clinical nutrition professionals' objects='core stroke unit team composition' id='activity18_1'/>
      <activity role='stroke unit leadership' action='verify team members have stroke expertise, spend the vast majority of their time treating stroke patients, and regularly complete stroke education' objects='stroke expertise records, clinical assignment records, stroke education records' id='activity18_2'/>
      <activity role='stroke team' action='perform appropriate cross-continuum secondary prevention assessment' objects='secondary prevention assessments, Modules 5 and 10' id='activity19_1'/>
      <activity role='stroke team' action='initiate or arrange appropriate cross-continuum secondary prevention therapies whether the patient is transferred to in-hospital rehabilitation or discharged' objects='secondary prevention therapies, rehabilitation transfer plan, discharge plan' id='activity19_2'/>
    </branch>
    <branch condition='patient has TIA or minor nondisabling stroke not applicable to these recommendations' id='branch14'>
      <activity role='clinicians' action='do not apply Module 3a acute care treatment pathway and direct patient to appropriate TIA or minor stroke management pathway' objects='alternative care pathway' id='activity20'/>
    </branch>
  </exclusiveGateway>
</process>
