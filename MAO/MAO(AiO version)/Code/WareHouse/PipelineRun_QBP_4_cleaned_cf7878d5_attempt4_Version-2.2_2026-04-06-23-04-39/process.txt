  </exclusiveGateway>

  <parallelGateway id='parallelgateway1'>
    <branch id='branch3'>
      <activity role='medical staff' action='Perform ECG to detect atrial fibrillation and other acute arrhythmias (do not delay acute treatment)' objects='patient' id='activity9'/>

    </branch>
    <branch id='branch5'>

      <activity role='medical staff' action='Order troponin test if clinically indicated' objects='patient' id='activity12'/>
    </branch>

  <activity role='health care provider with imaging expertise' action='Interpret brain imaging immediately' objects='CT/CTA/MRI results' id='activity13'/>
  <exclusiveGateway id='exclusivegateway2'>

    </branch>
    <branch condition='Patient is not clinically eligible for acute ischemic stroke treatments' id='branch8'>
      <activity role='medical staff' action='Skip advanced CT imaging' objects='patient' id='activity15'/>
    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='exclusivegateway3'>

        <branch condition='ICH patient is selected for vascular imaging' id='branch10'>
          <activity role='medical staff' action='Order CTA or MRA as soon as possible' objects='patient' id='activity18'/>
        </branch>
        <branch condition='ICH patient is not selected for vascular imaging' id='branch11'>
          <activity role='medical staff' action='Continue with standard ICH management' objects='patient' id='activity19'/>

      </exclusiveGateway>
    </branch>
    <branch condition='Patient does not have ICH' id='branch12'>
      <exclusiveGateway id='exclusivegateway5'>
        <branch condition='Patient presents beyond 48 hours with TIA or ischemic stroke' id='branch13'>
          <activity role='medical staff' action='Order vascular imaging of brain and neck arteries as soon as possible' objects='patient' id='activity20'/>
        </branch>
        <branch condition='Patient does not present beyond 48 hours with TIA or ischemic stroke' id='branch14'>
          <activity role='medical staff' action='No additional vascular imaging required at this stage' objects='patient' id='activity21'/>
        </branch>
      </exclusiveGateway>
    </branch>
  </exclusiveGateway>
  <activity role='medical staff' action='Make all stroke patients NPO initially and screen swallowing ability using a simple, valid, reliable bedside protocol before oral medication, fluid, or food' objects='patient' id='activity22'/>

    </branch>
    <branch condition='Patient is alert within first 24 hours' id='branch16'>

    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='exclusivegateway7'>

    </branch>
    <branch condition='No features of dysphagia or pulmonary aspiration' id='branch18'>
      <activity role='medical staff' action='Continue monitoring swallowing status as per protocol' objects='patient' id='activity26'/>
    </branch>
  </exclusiveGateway>
  <activity role='medical staff' action='Provide all patients (admitted or discharged) with appropriate cross-continuum secondary prevention assessments and therapies' objects='patient' id='activity27'/>
