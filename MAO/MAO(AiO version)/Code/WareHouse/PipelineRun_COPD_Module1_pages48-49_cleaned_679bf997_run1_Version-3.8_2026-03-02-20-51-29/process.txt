<process>
  <activity role='ED/outpatient clinic triage nurse or admitting clinician' action='register patient and initiate suspected COPD exacerbation pathway' objects='patient' id='activity1'/>
  <activity role='ED/outpatient clinic triage nurse or admitting clinician' action='confirm presentation source' objects='ED vs outpatient clinic vs direct admission' id='activity1_1'/>
  <activity role='ED/outpatient clinic triage nurse or admitting clinician' action='initiate pathway documentation' objects='COPD assessment forms/EMR pathway order set' id='activity1_2'/>

  <parallelGateway id='parallelgateway1_split'>
    <branch id='branch1'>
      <activity role='nurse or clinician' action='check vital signs' objects='respiratory rate, heart rate, blood pressure, temperature' id='activity2'/>
      <activity role='nurse or clinician' action='assess for hypoventilation' objects='breathing pattern/ventilation' id='activity3'/>
      <activity role='nurse or clinician' action='check level of consciousness/cognition' objects='mental status' id='activity4'/>
      <activity role='nurse or clinician' action='check blood-oxygen saturation level via pulse oximetry' objects='pulse oximeter, SpO2' id='activity5'/>
      <activity role='nurse or clinician' action='assess whether patient has purulent sputum' objects='sputum characteristics' id='activity6'/>
    </branch>
    <branch id='branch2'>
      <activity role='clinician' action='perform a physical examination' objects='respiratory and cardiovascular exam' id='activity7'/>
    </branch>
    <branch id='branch3'>
      <activity role='clinician' action='check patient history' objects='symptoms, prior COPD/asthma, smoking, prior exacerbations, comorbidities' id='activity8'/>
    </branch>
    <branch id='branch4'>
      <activity role='clinician/pharmacist/nurse' action='document and reconcile medications currently used by patient' objects='home meds, inhalers, oxygen, allergies' id='activity9'/>
      <activity role='clinician/pharmacist/nurse' action='obtain best possible medication history' objects='patient/caregiver report, pharmacy records, prior lists' id='activity9_1'/>
      <activity role='clinician/pharmacist/nurse' action='reconcile medications and resolve discrepancies' objects='current vs home meds, duplications, omissions, allergies' id='activity9_2'/>
    </branch>
    <branch id='branch5'>
      <activity role='clinician' action='identify patient wishes with respect to goals of care and/or limitations of treatment (code status)' objects='advance directives/goals of care' id='activity10'/>
    </branch>
    <branch id='branch6'>
      <activity role='clinician' action='perform other diagnostic interventions as appropriate to identify and/or rule out other suspected diagnoses or comorbidities' objects='additional tests per clinical assessment and local ED processes' id='activity11'/>
    </branch>
  </parallelGateway>

  <activity role='clinician' action='confirm initial assessment bundle completed' objects='vitals/physical/history/medication reconciliation/goals of care/other diagnostics as indicated' id='activity11_confirm'/>

  <parallelGateway id='parallelgateway_tests_split_mod'>
    <branch id='branch_tests_cxr_mod'>
      <activity role='clinician' action='order chest x-ray if clinically indicated (should not delay assessment/treatment)' objects='CXR indication decision; posteroanterior and lateral views if ordered' id='activity12_mod'/>
      <exclusiveGateway id='exclusivegateway_cxr_required_mod'>
        <branch condition='chest x-ray required' id='branch_cxr_required_mod'>
          <exclusiveGateway id='exclusivegateway1_mod'>
            <branch condition='patient too unwell to leave emergency department' id='branch7_mod'>
              <activity role='radiology technologist' action='perform portable chest x-ray' objects='portable x-ray in ED' id='activity13_mod'/>
            </branch>
            <branch condition='patient stable to attend radiology' id='branch8_mod'>
              <activity role='radiology technologist' action='perform chest x-ray in radiology' objects='posteroanterior and lateral views' id='activity14_mod'/>
            </branch>
          </exclusiveGateway>
          <exclusiveGateway id='exclusivegateway2_mod'>
            <branch condition='concern for pneumothorax' id='branch9_mod'>
              <activity role='radiology technologist' action='add expiratory view chest x-ray' objects='expiratory view' id='activity15_mod'/>
            </branch>
            <branch condition='no concern for pneumothorax' id='branch10_mod'>
              <activity role='clinician' action='proceed without expiratory view' objects='chest x-ray order as placed' id='activity16_mod'/>
            </branch>
          </exclusiveGateway>
          <activity role='radiologist/clinician' action='interpret chest x-ray and document findings immediately when available' objects='CXR images; pneumonia/pneumothorax/edema findings' id='activity12_interpret_immediate_mod'/>
        </branch>
        <branch condition='chest x-ray not required' id='branch_cxr_not_required_mod'>
          <activity role='clinician' action='skip chest x-ray and continue pathway without delay' objects='documented rationale for no CXR' id='activity12_skip_mod'/>
        </branch>
      </exclusiveGateway>
    </branch>

    <branch id='branch_tests_blood_ecg_abg_mod'>
      <activity role='clinician' action='order baseline blood work' objects='complete blood count, electrolytes, creatinine, blood urea nitrogen (if available)' id='activity17'/>
      <activity role='nurse/phlebotomy' action='collect baseline blood work specimens' objects='CBC, electrolytes, creatinine, BUN (if available)' id='activity18'/>
      <activity role='clinician/lab medicine' action='interpret blood work and document results immediately when available' objects='CBC/electrolytes/creatinine/BUN results' id='activity18_interpret'/>

      <activity role='nurse/ECG technician' action='perform electrocardiogram' objects='ECG to check for arrhythmias, myocardial ischemia, right ventricular strain' id='activity19'/>
      <activity role='clinician' action='interpret ECG and document findings immediately' objects='ECG tracing and interpretation' id='activity19_interpret'/>

      <exclusiveGateway id='exclusivegateway3'>
        <branch condition='low oxygen saturation on oximetry and/or acute respiratory failure suspected' id='branch11'>
          <activity role='clinician/nurse/respiratory therapist' action='check arterial blood gases where appropriate' objects='ABG sample and analysis' id='activity20'/>
          <activity role='clinician' action='interpret ABG and document results immediately when available' objects='pH/PaCO2/PaO2/HCO3 results' id='activity20_interpret'/>
        </branch>
        <branch condition='no low oxygen saturation and no acute respiratory failure suspected' id='branch12'>
          <activity role='clinician' action='do not obtain arterial blood gases at this time' objects='ABG not indicated' id='activity21'/>
        </branch>
      </exclusiveGateway>
    </branch>
  </parallelGateway>

  <inclusiveGateway id='inclusivegateway1_with_join'>
    <branch condition='pneumonia or suspected sepsis' id='branch13_mod'>
      <activity role='clinician/nurse/phlebotomy' action='draw blood cultures' objects='blood culture sets' id='activity22_mod'/>
      <activity role='clinician/lab medicine' action='interpret blood culture status/results and document immediately when available' objects='culture collected/time-to-positivity/preliminary and final results' id='activity22_interpret_mod'/>
    </branch>
    <branch condition='suspected cardiac disorders' id='branch14_mod'>
      <activity role='clinician' action='examine cardiac markers' objects='cardiac biomarkers (e.g., troponin) as appropriate' id='activity23_mod'/>
      <activity role='clinician/lab medicine' action='interpret cardiac marker results and document immediately when available' objects='troponin/other biomarkers and trend interpretation' id='activity23_interpret_mod'/>
    </branch>
  </inclusiveGateway>

  <exclusiveGateway id='exclusivegateway4'>
    <branch condition='no prior objective documentation of COPD through spirometry AND patient stabilized (not during initial assessment)' id='branch15'>
      <activity role='clinician/respiratory therapist/pulmonary function lab' action='perform spirometry once the patient has stabilized' objects='spirometry' id='activity24'/>
    </branch>
    <branch condition='prior spirometry documented OR patient not yet stabilized' id='branch16'>
      <activity role='clinician' action='do not perform spirometry during initial assessment' objects='spirometry deferred/not required' id='activity25'/>
    </branch>
  </exclusiveGateway>

  <activity role='clinician' action='assess level of care required using clinical judgment and local resource availability' objects='overall clinical assessment and resource status' id='activity26'/>
  <parallelGateway id='parallelgateway2'>
    <branch id='branch19'>
      <activity role='clinician' action='assess clinical severity and physiologic risk' objects='work of breathing, mental status, vital signs, oxygenation/ABG if available' id='activity26_1'/>
    </branch>
    <branch id='branch20'>
      <activity role='clinician/social work as available' action='assess home supports and ability to cope at home' objects='caregiver availability, functional status, social circumstances' id='activity26_2'/>
    </branch>
    <branch id='branch21'>
      <activity role='clinician/bed management as needed' action='assess local resource availability' objects='availability of ward/step-down/ICU beds, staffing, oxygen/NIV capability' id='activity26_3'/>
    </branch>
  </parallelGateway>

  <activity role='clinician' action='apply NICE and/or GOLD admission criteria as a guide' objects='response to initial management, home support, breathlessness/resting dyspnea, general condition/severe COPD, activity level, cyanosis, edema/new signs, consciousness, LTOT, social circumstances/older age, acute confusion, rapid onset/frequent exacerbations, comorbidity (HF/arrhythmias), SaO2<90%, pH, PaO2, chest x-ray changes' id='activity27'/>

  <parallelGateway id='parallelgateway_admission_eval_split'>
    <branch id='branch_admission_eval'>
      <activity role='ED clinician' action='re-evaluate in ED for admission decision (concurrent with ongoing assessment/tests/resuscitation as needed)' objects='clinical status, response to treatment, NICE/GOLD criteria, resource availability' id='activity29_parallel'/>
    </branch>
    <branch id='branch_continue_pathway'>
      <activity role='ED team' action='trial immediate resuscitation on initial presentation at the ED' objects='resuscitation interventions per ED standards' id='activity28'/>
      <activity role='ED team' action='initiate immediate stabilization interventions' objects='airway/breathing support, oxygen titration, bronchodilators per local protocol, IV access/fluids as indicated' id='activity28_1'/>
      <exclusiveGateway id='exclusivegateway6'>
        <branch condition='clinical deterioration or inadequate response to stabilization' id='branch22'>
          <activity role='ED clinician' action='escalate level of care and consult appropriate services' objects='ICU/respiratory therapy/medicine consults; consider NIV/intubation per local protocol' id='activity28_2'/>
        </branch>
        <branch condition='clinical improvement or adequate response' id='branch23'>
          <activity role='ED clinician' action='continue current management and proceed to admission re-evaluation' objects='ongoing monitoring and treatment' id='activity28_3'/>
        </branch>
      </exclusiveGateway>
    </branch>
  </parallelGateway>

  <exclusiveGateway id='exclusivegateway5'>
    <branch condition='admission indicated based on re-evaluation and criteria' id='branch17'>
      <activity role='ED clinician/admitting clinician' action='admit patient and determine appropriate level of care (ward/step-down/ICU as locally defined)' objects='admission orders, bed request' id='activity30'/>
    </branch>
    <branch condition='admission not indicated' id='branch18'>
      <activity role='ED/outpatient clinician' action='manage as outpatient per local processes and provide disposition plan' objects='discharge/follow-up plan and safety net instructions' id='activity31'/>
    </branch>
  </exclusiveGateway>

  <activity role='hospital administration' action='on a quarterly basis, make COPD admission rates known to administration and admitting clinicians benchmarked against similar hospitals in the province' objects='quarterly COPD admission rate reports/benchmarks' id='activity32'/>
  <activity role='ED leadership/quality team' action='ensure emergency room staff use an admission checklist including non-COPD concerns (pain, anxiety, comorbidities) in addition to Module 1 practices' objects='ED admission checklist' id='activity33'/>
</process>
