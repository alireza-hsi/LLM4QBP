<process>
  <activity role='emergency department clinicians' action='screen and assess the patient for diabetes, dental issues, cognitive impairment, and depression upon presentation to the ED' objects='diabetes screening,dental assessment,cognitive assessment,depression screening' id='activity1'/>
  <exclusiveGateway id='exclusivegateway1'>
    <branch condition='patient has acute stroke and is not TIA or minor nondisabling stroke' id='branch1'>
      <activity role='admitting team' action='admit the patient to a specialized geographically defined stroke unit dedicated to stroke management' objects='stroke unit bed' id='activity2'/>
      <activity role='stroke unit coordinator' action='assign the core stroke unit team with stroke expertise in medicine, nursing, occupational therapy, physiotherapy, speech-language pathology, social work, and clinical nutrition' objects='core stroke unit team' id='activity3'/>
      <activity role='stroke unit coordinator' action='ensure 7-day access to the specialized interprofessional team' objects='7-day interprofessional team coverage' id='activity3_1'/>
      <activity role='interprofessional stroke team' action='use standardized validated assessment tools linked to the EHR where possible to evaluate stroke-related impairments and functional status' objects='standardized assessment tools,EHR' id='activity4'/>
      <parallelGateway id='parallelgateway1'>
        <branch id='branch2'>
          <activity role='physiotherapist' action='assess the patient within 48 hours of admission and formulate a physiotherapy management plan' objects='physiotherapy assessment' id='activity5'/>
        </branch>
        <branch id='branch3'>
          <activity role='occupational therapist' action='assess the patient within 48 hours of admission and formulate an occupational therapy management plan' objects='occupational therapy assessment' id='activity6'/>
        </branch>
        <branch id='branch4'>
          <activity role='speech-language pathologist' action='assess the patient within 48 hours of admission and formulate a speech-language pathology management plan' objects='speech-language assessment' id='activity7'/>
        </branch>
        <branch id='branch5'>
          <activity role='clinical team' action='complete AlphaFIM on or by day 3 after admission with target day 3 and admission day as day 1' objects='AlphaFIM' id='activity8'/>
        </branch>
        <branch id='branch6'>
          <activity role='clinical team' action='assess the patient for risk of developing venous thromboembolism' objects='VTE risk assessment' id='activity9'/>
        </branch>
        <branch id='branch7'>
          <parallelGateway id='parallelgateway2'>
            <branch id='branch7_1'>
              <activity role='nursing staff' action='assess nutrition status within the first 48 hours of admission using a valid screening tool' objects='nutrition screening tool' id='activity10'/>
            </branch>
            <branch id='branch7_2'>
              <activity role='nursing staff' action='assess hydration status within the first 48 hours of admission using a valid screening tool' objects='hydration screening tool' id='activity10_1'/>
            </branch>
          </parallelGateway>
        </branch>
      </parallelGateway>
      <exclusiveGateway id='exclusivegateway2'>
        <branch condition='mobilization is contraindicated' id='branch8'>
          <activity role='stroke physician' action='document contraindication to early mobilization and continue monitoring for later mobilization readiness' objects='contraindication record' id='activity11'/>
        </branch>
        <branch condition='within 24 hours of stroke onset, mobilization is not contraindicated, and limited mobilization may be reasonable based on clinical judgement' id='branch9'>
          <activity role='interprofessional stroke team' action='use clinical judgement to consider limited mobilization within 24 hours of stroke onset and avoid frequent out-of-bed activity' objects='early mobilization plan' id='activity12'/>
        </branch>
        <branch condition='between 24 hours and 48 hours of stroke onset and not contraindicated' id='branch10'>
          <activity role='interprofessional stroke team' action='mobilize the patient early between 24 hours and 48 hours of stroke onset' objects='mobilization plan' id='activity14'/>
        </branch>
      </exclusiveGateway>
      <activity role='nursing staff' action='encourage adequate hydration to help prevent venous thromboembolism' objects='hydration plan' id='activity15_new'/>
      <inclusiveGateway id='inclusivegateway1'>
        <branch condition='nutrition concerns, hydration deficits, dysphagia, other comorbidities affecting nutrition or hydration, or inability to meet nutritional and fluid requirements are identified' id='branch11_new'>
          <activity role='clinical team' action='refer the patient to a dietitian within 7 days of admission for recommendations' objects='dietitian referral' id='activity16_new'/>
        </branch>
        <branch condition='patient is unable to meet nutritional and fluid requirements' id='branch15_new'>
          <activity role='dietitian' action='consider enteral nutrition support within 7 days of admission' objects='enteral nutrition support plan' id='activity20_1'/>
        </branch>
      </inclusiveGateway>
      <exclusiveGateway id='exclusivegateway3'>
        <branch condition='patient is transferred to in-hospital rehabilitation' id='branch17'>
          <activity role='stroke team' action='provide appropriate cross-continuum secondary prevention assessments and therapies for patients transferred to in-hospital rehabilitation' objects='secondary prevention assessments and therapies' id='activity21'/>
        </branch>
        <branch condition='patient is being discharged' id='branch18'>
          <activity role='stroke team' action='provide appropriate cross-continuum secondary prevention assessments and therapies for patients being discharged' objects='secondary prevention assessments and therapies' id='activity21_1'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='patient has TIA or minor nondisabling stroke' id='branch16'>
      <activity role='clinical team' action='recognize that Module 3a acute care treatment recommendations are not applicable to TIA or minor nondisabling stroke' objects='applicability determination' id='activity22'/>
      <activity role='emergency department clinicians' action='forward the screening and assessment results to the TIA clinic for follow-up' objects='screening results' id='activity1_1'/>
      <activity role='clinical team' action='provide appropriate cross-continuum secondary prevention assessments and therapies and arrange follow-up with the TIA clinic as appropriate' objects='secondary prevention assessments and therapies,TIA clinic follow-up' id='activity23'/>
    </branch>
  </exclusiveGateway>
</process>
