<process>

    </branch>
    <branch condition='patient has TIA or minor nondisabling stroke' id='branch2'>

  </exclusiveGateway>
  <activity role='stroke unit coordinator' action='assign patient to stroke medicine and nursing coverage' objects='stroke unit bed assignment, medicine coverage, nursing coverage' id='activity4_1'/>
  <activity role='stroke unit coordinator' action='arrange access to allied health and nutrition members of the stroke team' objects='occupational therapy, physiotherapy, speech-language pathology, social work, dietitian services, 7 day access' id='activity4_2'/>
  <activity role='stroke unit leadership' action='ensure core stroke unit team members spend the vast majority of their time treating stroke patients and regularly complete stroke care education' objects='staffing model, stroke education records' id='activity5'/>
  <parallelGateway id='parallelgateway1'>
    <branch id='branch3'>
      <activity role='clinicians' action='evaluate stroke-related impairments using standardized valid tools' objects='standardized validated assessment tools' id='activity6_1'/>
      <activity role='clinicians' action='document assessment findings in EHR-linked standardized format' objects='functional status findings, impairment findings, EHR-linked documentation' id='activity6_2'/>
    </branch>
    <branch id='branch4'>
      <activity role='physiotherapist' action='assess patient within 48 hours of hospital admission and formulate a management plan' objects='physiotherapy assessment, management plan' id='activity8'/>
    </branch>
    <branch id='branch5'>
      <activity role='occupational therapist' action='assess patient within 48 hours of hospital admission and formulate a management plan' objects='occupational therapy assessment, management plan' id='activity9'/>
    </branch>
    <branch id='branch6'>
      <activity role='speech-language pathologist' action='assess patient within 48 hours of hospital admission and formulate a management plan' objects='speech-language pathology assessment, management plan' id='activity10'/>
    </branch>
    <branch id='branch7'>
      <activity role='nursing staff' action='assess venous thromboembolism risk for all stroke patients' objects='venous thromboembolism risk assessment' id='activity11'/>
    </branch>
    <branch id='branch8'>
      <parallelGateway id='parallelgateway2'>
        <branch id='branch8_1'>
          <activity role='nursing staff' action='screen nutrition status using a valid screening tool within 48 hours' objects='nutrition screening tool' id='activity12_1'/>
        </branch>
        <branch id='branch8_2'>
          <activity role='nursing staff' action='screen hydration status within 48 hours' objects='hydration screening tool, hydration status' id='activity12_2'/>
        </branch>
      </parallelGateway>
    </branch>
  </parallelGateway>
  <activity role='rehabilitation team' action='complete AlphaFIM on or by day 3 after admission with target day 3 and admission day counted as day 1' objects='AlphaFIM assessment' id='activity7'/>
  <exclusiveGateway id='exclusivegateway2a'>
    <branch condition='mobilization is contraindicated' id='branch10a'>
      <activity role='clinical team' action='withhold mobilization and document contraindication and reassessment plan' objects='mobilization contraindication, reassessment plan' id='activity15a'/>
    </branch>
    <branch condition='mobilization is not contraindicated' id='branch11a'>
      <exclusiveGateway id='exclusivegateway2b'>
        <branch condition='within 24 hours of stroke onset' id='branch10b'>
          <activity role='clinical team' action='avoid frequent out-of-bed activity within 24 hours of stroke onset and use clinical judgement for any very early mobilization' objects='clinical judgement, mobilization precautions' id='activity15b'/>
        </branch>
        <branch condition='between 24 hours and 48 hours of stroke onset' id='branch11b'>
          <activity role='clinical team' action='mobilize patient early between 24 hours and 48 hours of stroke onset' objects='early mobilization plan' id='activity14b'/>
        </branch>
      </exclusiveGateway>
    </branch>
  </exclusiveGateway>
  <activity role='clinical team' action='encourage early mobilization and adequate hydration for all acute stroke patients to help prevent venous thromboembolism' objects='mobilization support, hydration plan' id='activity16'/>
  <exclusiveGateway id='exclusivegateway3'>
    <branch condition='nutritional concerns or hydration deficits or dysphagia or other comorbidities are identified' id='branch12'>
      <activity role='clinical team' action='refer patient to a dietitian within 7 days of admission' objects='dietitian referral' id='activity17_1'/>

    </branch>
    <branch condition='no nutritional concerns hydration deficits dysphagia or related comorbidities are identified' id='branch13'>
      <activity role='clinical team' action='continue routine nutrition and hydration monitoring' objects='nutrition and hydration status' id='activity18'/>
    </branch>

    </branch>
