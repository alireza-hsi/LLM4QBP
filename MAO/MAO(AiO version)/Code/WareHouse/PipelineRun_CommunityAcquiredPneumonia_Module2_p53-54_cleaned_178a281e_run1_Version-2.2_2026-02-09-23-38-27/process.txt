<process>
  <activity role='medical staff' action='Assess patient presenting with symptoms of community-acquired pneumonia (CAP) in ED, inpatient ward, or ICU' objects='patient' id='activity1'/>
  <activity role='medical staff' action='Measure oxygen saturation' objects='patient' id='activity2'/>
  <activity role='medical staff' action='Request chest x-ray' objects='patient, chest x-ray order' id='activity3-1'/>
  <exclusiveGateway id='exclusivegateway1'>
    <branch condition='Patient has sepsis' id='branch1'>
      <activity role='medical staff' action='Initiate antibiotics before chest x-ray results' objects='patient, antibiotics' id='activity4'/>
      <activity role='radiology staff' action='Perform chest x-ray' objects='patient, chest x-ray' id='activity3-2a'/>
    </branch>
    <branch condition='Patient does not have sepsis' id='branch2'>
      <activity role='medical staff' action='Wait for chest x-ray results before initiating antibiotics' objects='patient, chest x-ray results' id='activity5'/>
      <activity role='radiology staff' action='Perform chest x-ray' objects='patient, chest x-ray' id='activity3-2b'/>
    </branch>
  </exclusiveGateway>
  <activity role='medical staff' action='Order CBC, electrolytes, and renal function tests' objects='patient, CBC order, electrolytes order, renal function order' id='activity7'/>
  <exclusiveGateway id='exclusivegateway2'>
    <branch condition='Patient meets >2 CRB-65 criteria' id='branch6'>
      <activity role='medical staff' action='Consider ordering blood gases, lactate, liver enzyme and liver function tests' objects='patient, blood samples' id='activity8'/>
    </branch>
    <branch condition='Patient does not meet >2 CRB-65 criteria' id='branch7'>
      <activity role='medical staff' action='Do not order additional blood gases, lactate, liver enzyme or liver function tests' objects='patient' id='activity9'/>
    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='exclusivegateway3'>
    <branch condition='Patient meets >2 SIRS criteria or requires ICU admission' id='branch8'>
      <activity role='medical staff' action='Collect blood samples for culture' objects='patient, blood sample' id='activity10-1'/>
      <activity role='medical staff' action='Send blood samples to laboratory' objects='blood sample, laboratory' id='activity10-2'/>
    </branch>
    <branch condition='Patient does not meet >2 SIRS criteria and does not require ICU admission' id='branch9'>
      <activity role='medical staff' action='Do not order blood cultures' objects='patient' id='activity11'/>
    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='exclusivegateway4'>
    <branch condition='Patient admitted to hospital and adequate sputum sample can be obtained' id='branch10'>
      <activity role='medical staff' action='Collect sputum sample' objects='patient, sputum sample' id='activity12-1'/>
      <activity role='medical staff' action='Send sputum sample for culture' objects='sputum sample, laboratory' id='activity12-2'/>
      <exclusiveGateway id='exclusivegateway5'>
        <branch condition='Sputum culture obtained' id='branch11'>
          <activity role='medical staff' action='Consider Gram staining of sputum sample' objects='sputum sample' id='activity13'/>
        </branch>
        <branch condition='Sputum culture not obtained' id='branch12'>
          <activity role='medical staff' action='Do not perform Gram stain' objects='patient' id='activity14'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='Patient not admitted or inadequate sputum sample' id='branch13'>
      <activity role='medical staff' action='Do not obtain sputum culture or Gram stain' objects='patient' id='activity15'/>
    </branch>
  </exclusiveGateway>
  <inclusiveGateway id='inclusivegateway1'>
    <branch condition='Patient is high-severity' id='branch14'>
      <activity role='medical staff' action='Collect urine sample' objects='patient, urine sample' id='activity16-1'/>
      <activity role='medical staff' action='Send urine sample for antigen testing' objects='urine sample, laboratory' id='activity16-2'/>
    </branch>
    <branch condition='Enhanced surveillance directive from Public Health Ontario on Legionella in inpatients' id='branch15'>
      <activity role='medical staff' action='Collect urine sample' objects='patient, urine sample' id='activity17-1'/>
      <activity role='medical staff' action='Send urine sample for antigen testing' objects='urine sample, laboratory' id='activity17-2'/>
    </branch>
    <branch condition='Patient not responding to drug therapy after 48–72 hours' id='branch16'>
      <activity role='medical staff' action='Collect urine sample' objects='patient, urine sample' id='activity18-1'/>
      <activity role='medical staff' action='Send urine sample for antigen testing' objects='urine sample, laboratory' id='activity18-2'/>
    </branch>
    <branch condition='Inpatient during peak season (mid-June to early October)' id='branch17'>
      <activity role='medical staff' action='Collect urine sample' objects='patient, urine sample' id='activity19-1'/>
      <activity role='medical staff' action='Send urine sample for antigen testing' objects='urine sample, laboratory' id='activity19-2'/>
    </branch>
  </inclusiveGateway>
  <activity role='medical staff' action='Consider field evaluation to assess clinical utility of urine antigen testing' objects='urine antigen test, evaluation protocol' id='activity20'/>
  <exclusiveGateway id='exclusivegateway6'>
    <branch condition='Patient is intubated (high severity)' id='branch18'>
      <activity role='medical staff' action='Collect respiratory sample' objects='patient, respiratory sample' id='activity21-1'/>
      <activity role='medical staff' action='Send sample for PCR testing' objects='respiratory sample, laboratory' id='activity21-2'/>
    </branch>
    <branch condition='Patient is not intubated' id='branch19'>
      <activity role='medical staff' action='Do not perform PCR testing for Legionella' objects='patient' id='activity22'/>
    </branch>
  </exclusiveGateway>
</process>
