<process>
  <activity role='admitting team' action='confirm whether the patient has acute stroke or TIA or minor nondisabling stroke' objects='diagnostic findings' id='activity1_2'/>
  <exclusiveGateway id='exclusivegateway1'>
    <branch condition='patient has acute stroke' id='branch1'>
      <activity role='admitting team' action='admit the patient to acute care' objects='admission record' id='activity1_1'/>
      <activity role='admitting team' action='admit the patient to a specialized geographically defined stroke unit dedicated to stroke management' objects='stroke unit bed, admission order' id='activity2'/>
      <activity role='stroke unit coordinator' action='assign a core stroke unit team with expertise in medicine, nursing, occupational therapy, physiotherapy, speech-language pathology, social work, and clinical nutrition' objects='core stroke unit team assignment' id='activity3'/>
      <activity role='stroke unit manager' action='verify that core stroke unit team members spend the vast majority of their time treating stroke patients and regularly complete stroke care education' objects='staff competency and education records' id='activity4'/>
      <activity role='stroke unit coordinator' action='ensure the patient has access to a specialized interprofessional team 7 days per week' objects='7 day team coverage plan' id='activity5'/>
      <exclusiveGateway id='exclusivegateway_activity6_1'>
        <branch condition='electronic standardized assessment tools linked to the EHR are available' id='branch_activity6_1'>
          <activity role='interprofessional stroke team' action='use standardized valid assessment tools in electronic format linked to the EHR' objects='standardized assessment tools, EHR' id='activity6_1'/>
        </branch>
        <branch condition='electronic standardized assessment tools linked to the EHR are not available' id='branch_activity6_2'>
          <activity role='interprofessional stroke team' action='use standardized valid assessment tools in non electronic format and document results in the patient record' objects='standardized assessment tools, patient record' id='activity6_2'/>
        </branch>
      </exclusiveGateway>
      <parallelGateway id='parallelgateway1'>
        <branch id='branch2'>
          <activity role='clinicians' action='evaluate stroke related impairments using standardized valid assessment tools' objects='neurological impairment assessment' id='activity7_1'/>
          <activity role='clinicians' action='evaluate functional status using standardized valid assessment tools' objects='functional status assessment' id='activity7_2'/>
          <activity role='clinicians' action='complete AlphaFIM on or by day 3 after admission with admission day counted as day 1' objects='AlphaFIM assessment' id='activity8'/>
        </branch>
        <branch id='branch3'>
          <activity role='physiotherapist' action='assess the patient within 48 hours of admission to hospital and formulate a management plan' objects='physiotherapy assessment, management plan' id='activity9'/>
        </branch>
        <branch id='branch4'>
          <activity role='occupational therapist' action='assess the patient within 48 hours of admission to hospital and formulate a management plan' objects='occupational therapy assessment, management plan' id='activity10'/>
        </branch>
        <branch id='branch5'>
          <activity role='speech-language pathologist' action='assess the patient within 48 hours of admission to hospital and formulate a management plan' objects='speech-language pathology assessment, management plan' id='activity11'/>
        </branch>
        <branch id='branch6'>
          <activity role='clinical team' action='assess the patient for risk of developing venous thromboembolism' objects='venous thromboembolism risk assessment' id='activity12'/>
          <activity role='clinical team' action='encourage adequate hydration to help prevent venous thromboembolism' objects='hydration plan' id='activity13'/>
        </branch>
        <branch id='branch7'>
          <activity role='nursing staff' action='assess the patient nutrition and hydration status within the first 48 hours of admission using a valid screening tool' objects='nutrition and hydration screening tool' id='activity14'/>
        </branch>
        <branch id='branch10'>
          <activity role='clinical team' action='assess stroke onset timing and contraindications to mobilization' objects='mobilization assessment' id='activity16_1'/>
          <exclusiveGateway id='exclusivegateway3'>
            <branch condition='mobilization is contraindicated' id='branch11'>
              <activity role='clinical team' action='defer early mobilization and reassess contraindications' objects='mobilization contraindication assessment' id='activity17'/>
            </branch>
            <branch condition='patient is within 24 hours of stroke onset and clinical judgement supports mobilization' id='branch12'>
              <activity role='clinical team' action='mobilize the patient within 24 hours of stroke onset cautiously and avoid frequent out of bed activity' objects='very early mobilization plan' id='activity18'/>
            </branch>
            <branch condition='patient is within 24 hours of stroke onset and clinical judgement does not support mobilization' id='branch15'>
              <activity role='clinical team' action='delay mobilization continue monitoring and hydration and reassess the patient when the 24 to 48 hour window is reached' objects='monitoring record, hydration plan, reassessment plan' id='activity18_1'/>
            </branch>
            <branch condition='patient is between 24 hours and 48 hours of stroke onset and has no contraindication' id='branch13'>
              <activity role='clinical team' action='mobilize the patient early between 24 hours and 48 hours of stroke onset' objects='early mobilization plan' id='activity19'/>
            </branch>
          </exclusiveGateway>
        </branch>
      </parallelGateway>
      <exclusiveGateway id='exclusivegateway2'>
        <branch condition='nutritional concerns, hydration deficits, dysphagia, or other comorbidities are identified' id='branch8'>
          <activity role='nursing staff' action='refer the patient to a dietitian within 7 days of admission' objects='dietitian referral' id='activity15_1'/>
          <activity role='dietitian' action='assess nutritional and fluid requirements and provide recommendations' objects='dietitian assessment, recommendations' id='activity15_2'/>
          <exclusiveGateway id='exclusivegateway_activity15_1'>
            <branch condition='patient is unable to meet nutritional and fluid requirements' id='branch_activity15_1'>
              <activity role='dietitian' action='request consideration of enteral nutrition support' objects='enteral nutrition support request' id='activity15_3'/>
            </branch>
            <branch condition='patient can meet nutritional and fluid requirements with oral intake' id='branch_activity15_2'>
              <activity role='dietitian' action='provide oral nutrition and hydration recommendations with monitoring' objects='diet and hydration plan, monitoring record' id='activity15_4'/>
            </branch>
          </exclusiveGateway>
        </branch>
        <branch condition='no nutritional concerns, hydration deficits, dysphagia, or other comorbidities are identified' id='branch9'>
          <activity role='nursing staff' action='continue routine nutrition and hydration monitoring' objects='nutrition and hydration monitoring record' id='activity16'/>
        </branch>
      </exclusiveGateway>
      <activity role='acute care team' action='complete cross continuum secondary prevention assessment' objects='secondary prevention assessment, Modules 5 and 10' id='activity20_1'/>
      <activity role='acute care team' action='initiate appropriate secondary prevention therapies and communicate the plan at discharge or transfer to in hospital rehabilitation' objects='secondary prevention therapies, transition plan' id='activity20_2'/>
    </branch>
    <branch condition='patient has TIA or minor nondisabling stroke' id='branch14'>
      <activity role='clinical team' action='document that these acute care treatment recommendations are not applicable and route the patient to the relevant TIA or minor stroke pathway' objects='clinical note, alternative pathway referral' id='activity21'/>
      <activity role='clinical team' action='provide appropriate cross continuum secondary prevention assessments and therapies' objects='secondary prevention assessments and therapies, Modules 5 and 10' id='activity22'/>
    </branch>
  </exclusiveGateway>
</process>
