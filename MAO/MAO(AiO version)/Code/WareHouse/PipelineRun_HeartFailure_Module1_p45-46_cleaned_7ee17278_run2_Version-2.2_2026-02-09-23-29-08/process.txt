  <exclusiveGateway id='exclusivegateway0'>

      <activity role='ED physician' action='initiate immediate resuscitation and urgent escalation pathway' objects='resuscitation protocol' id='activity1_2'/>
    </branch>
    <branch condition='no immediate life-threatening instability' id='branch0b'>
      <activity role='ED triage nurse' action='document HF red flags and triggers for early escalation review' objects='triage notes/escalation criteria' id='activity1_3'/>
    </branch>
  </exclusiveGateway>

  <activity role='ED nurse' action='frequently measure heart rate, blood pressure, and oxygen saturation until patient is stabilized (ongoing throughout ED workup)' objects='vital signs monitor/pulse oximeter' id='activity10'/>

  <activity role='ED physician' action='perform risk assessment/stratification consistently using hospital-standard tool' objects='risk stratification assessment tool' id='activity2'/>
  <activity role='ED physician' action='assess likelihood of acute heart failure and primary clinical syndrome' objects='history/physical exam findings' id='activity2_1'/>
  <activity role='ED physician' action='identify precipitants/comorbidities influencing risk and disposition (e.g., ACS, infection, renal dysfunction)' objects='problem list/initial differential' id='activity2_2'/>

  <parallelGateway id='parallelgateway1'>
    <branch id='branch1'>
      <activity role='ED nurse' action='obtain serum creatinine and electrolyte levels' objects='blood sample' id='activity3'/>
      <activity role='ED nurse' action='obtain troponin measurements' objects='blood sample' id='activity4'/>
      <activity role='ED nurse' action='obtain complete blood count' objects='blood sample' id='activity5'/>
    </branch>
    <branch id='branch2'>
      <activity role='ED staff' action='perform electrocardiogram' objects='ECG machine' id='activity6'/>
      <activity role='radiology' action='perform chest x-ray examination' objects='x-ray imaging' id='activity7'/>
    </branch>
    <branch id='branch3'>
      <exclusiveGateway id='exclusivegateway1'>
        <branch condition='no recent echocardiogram available' id='branch3a'>
          <activity role='cardiology/echo lab' action='perform echocardiogram' objects='echocardiography' id='activity8'/>
        </branch>
        <branch condition='recent echocardiogram available' id='branch3b'>
          <activity role='ED physician' action='review most recent echocardiogram results' objects='prior echocardiogram report' id='activity9'/>
        </branch>
      </exclusiveGateway>
    </branch>
  </parallelGateway>

  <activity role='ED physician' action='assess responsiveness to diuresis (including response to furosemide administered in ED)' objects='diuresis response data' id='activity11'/>

  <activity role='ED nurse' action='reassess diuresis response (urine output, symptom relief, repeat vitals) after diuretic' objects='I&O record/clinical reassessment' id='activity11_2'/>

  <activity role='ED physician' action='screen for high-risk markers (respiratory distress, hypoxemia, severity of pulmonary edema, hemodynamic compromise, significant arrhythmias, positive troponin, poor response to furosemide in ED)' objects='clinical findings and test results' id='activity12'/>
  <activity role='ED physician' action='calculate acute heart failure risk score (e.g., EHMRG) to predict 7-day mortality risk' objects='EHMRG inputs and score' id='activity13'/>

  <exclusiveGateway id='exclusivegateway2'>
    <branch condition='high intensity required (ventilation and/or ICU with higher nurse-to-patient staffing) OR high-risk markers present' id='branch5'>
      <activity role='ED physician' action='activate pathway to transfer patient to higher level provider (ICU/critical care)' objects='transfer protocol' id='activity14'/>
      <activity role='ED physician' action='notify receiving ICU/critical care team and request acceptance' objects='consult/notification request' id='activity14_1'/>
      <activity role='ED nurse' action='prepare patient for transfer and complete standardized handover package' objects='handover checklist/transfer documents' id='activity14_2'/>
      <exclusiveGateway id='exclusivegateway3'>
        <branch condition='noninvasive ventilation required' id='branch6'>
          <activity role='ED/respiratory therapy' action='initiate noninvasive ventilation' objects='NIV equipment' id='activity15'/>
        </branch>
        <branch condition='invasive ventilation required' id='branch7'>
          <activity role='ED/critical care team' action='initiate invasive ventilation' objects='airway equipment/ventilator' id='activity16'/>

      </exclusiveGateway>
      <activity role='bed management/critical care' action='admit to intensive care unit with higher nurse-to-patient staffing' objects='ICU bed' id='activity17'/>
