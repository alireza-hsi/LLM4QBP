      <activity role='ED clinician' action='perform and interpret electrocardiogram' objects='electrocardiogram' id='activity6'/>

    <branch id='branch5'>

    </branch>
    <branch id='branch6'>
      <exclusiveGateway id='exclusivegateway1'>
        <branch condition='no recent echocardiogram is available' id='branch7'>
          <activity role='cardiac imaging team' action='perform echocardiogram' objects='echocardiogram' id='activity8'/>

        <branch condition='recent echocardiogram is available' id='branch8'>

      </exclusiveGateway>
    </branch>

  <activity role='ED clinician' action='assess responsiveness to diuresis after initial emergency department treatment' objects='urine output, symptoms, congestion status, vital signs' id='activity12'/>

        <branch condition='respiratory distress is present' id='branch11'>

        </branch>
        <branch condition='hypoxemia is present' id='branch12'>
          <activity role='ED clinician' action='document hypoxemia as a high-risk marker' objects='oxygen saturation assessment' id='activity16'/>

          <activity role='ED clinician' action='document severity of pulmonary edema as a high-risk marker' objects='chest x-ray and clinical examination findings' id='activity17'/>
        </branch>
        <branch condition='response to furosemide in the emergency department is poor' id='branch14'>
          <activity role='ED clinician' action='document poor response to furosemide as a high-risk marker' objects='diuresis response assessment' id='activity18'/>
        </branch>
        <branch condition='hemodynamic compromise is present' id='branch15'>

        </branch>
        <branch condition='significant arrhythmias are present' id='branch16'>
          <activity role='ED clinician' action='document significant arrhythmias as a high-risk marker' objects='electrocardiogram and rhythm assessment' id='activity20'/>
        </branch>
        <branch condition='troponin is positive' id='branch17'>
          <activity role='ED clinician' action='document positive troponin as a high-risk marker' objects='troponin result' id='activity21'/>
        </branch>
      </inclusiveGateway>
    </branch>
    <branch condition='no high-risk markers are identified' id='branch_highrisk_no'>
      <activity role='ED clinician' action='document that no high-risk markers are identified' objects='risk marker assessment summary' id='activity22'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED clinician' action='assign the acute heart failure intensity category using investigation results, response to diuresis, high-risk markers, and EHMRG score' objects='investigation results, diuresis response assessment, high-risk marker summary, EHMRG score and quintile' id='activity22_1'/>
  <activity role='ED clinician' action='review other considerations affecting discharge decisions' objects='advanced directives, severe dementia, estimated impact of admission on life expectancy, other clinical considerations' id='activity14'/>
  <exclusiveGateway id='exclusivegateway2'>
    <branch condition='low-intensity criteria are met: EHMRG quintile 1 or 2, patient responded to initial ED treatment, and no other considerations require admission' id='branch19'>

      <activity role='ED or outpatient team' action='provide discharge instructions and return precautions before discharge home' objects='patient discharge instructions and return precautions' id='activity23_2'/>
      <activity role='ED clinician' action='arrange follow-up with the primary care provider within days of discharge from the emergency department' objects='primary care follow-up appointment' id='activity24'/>
    </branch>
    <branch condition='average-intensity criteria are met: hospital admission is required with normal nurse-to-patient staffing' id='branch20'>
      <exclusiveGateway id='exclusivegateway3'>
        <branch condition='required inpatient care can be provided at the current hospital' id='branch21'>
          <activity role='inpatient team' action='admit the patient to inpatient care with normal nurse-to-patient staffing' objects='inpatient bed and admission orders' id='activity25'/>
        </branch>
        <branch condition='a higher-level provider is required or the current hospital cannot provide the needed inpatient care' id='branch22'>

          <activity role='receiving inpatient team' action='admit the patient to inpatient care with normal nurse-to-patient staffing' objects='receiving inpatient bed and admission orders' id='activity27'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='high-intensity criteria are met: ventilation and/or intensive care unit admission with higher nurse-to-patient staffing is required' id='branch23'>
      <exclusiveGateway id='exclusivegateway4'>
        <branch condition='required high-intensity care can be provided at the current hospital' id='branch24'>
          <inclusiveGateway id='inclusivegateway2'>
            <branch condition='ventilation is required' id='branch25'>
              <activity role='critical care team' action='initiate noninvasive or invasive ventilation as clinically indicated' objects='ventilatory support' id='activity28'/>
            </branch>
            <branch condition='intensive care unit admission with higher nurse-to-patient staffing is required' id='branch26'>
              <activity role='critical care team' action='admit the patient to the intensive care unit with higher nurse-to-patient staffing' objects='ICU bed and critical care orders' id='activity29'/>
            </branch>
          </inclusiveGateway>
        </branch>
        <branch condition='a higher-level provider is required or the current hospital cannot provide the needed high-intensity care' id='branch27'>
          <activity role='hospital transfer coordinator' action='identify and confirm acceptance by the receiving higher-level critical care provider' objects='receiving critical care provider contact and acceptance confirmation' id='activity30_1'/>
          <activity role='hospital transfer coordinator' action='arrange critical care transfer logistics and transmit the clinical handover information' objects='transfer pathway, critical care transfer order, transport arrangement, clinical handover documents' id='activity30_2'/>
          <inclusiveGateway id='inclusivegateway3'>
            <branch condition='ventilation is required after transfer' id='branch28'>
              <activity role='receiving critical care team' action='initiate noninvasive or invasive ventilation as clinically indicated' objects='ventilatory support' id='activity31'/>
            </branch>
            <branch condition='intensive care unit admission with higher nurse-to-patient staffing is required after transfer' id='branch29'>
              <activity role='receiving critical care team' action='admit the patient to the intensive care unit with higher nurse-to-patient staffing' objects='receiving ICU bed and critical care orders' id='activity32'/>
            </branch>
          </inclusiveGateway>
        </branch>
      </exclusiveGateway>
    </branch>
  </exclusiveGateway>
</process>
