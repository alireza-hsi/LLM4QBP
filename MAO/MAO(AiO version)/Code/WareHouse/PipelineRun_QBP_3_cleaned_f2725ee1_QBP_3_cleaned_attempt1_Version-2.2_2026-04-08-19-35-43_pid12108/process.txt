<process>
  <activity role='ED clinicians and admitting team' action='confirm the stroke diagnosis using clinical evaluation and available investigations' objects='stroke diagnosis' id='activity1_1'/>
  <activity role='ED clinicians and admitting team' action='determine whether the presentation is acute stroke versus TIA or minor nondisabling stroke' objects='stroke syndrome classification' id='activity1_2'/>
  <exclusiveGateway id='exclusivegateway1'>
    <branch condition='patient has acute stroke and may require hospital admission' id='branch1'>
      <activity role='ED clinicians and admitting team' action='confirm that hospital admission to the Module 3a acute care pathway is required' objects='admission decision' id='activity1_3'/>
      <activity role='admitting team' action='admit the patient to a specialized geographically defined stroke unit with identifiable co-located beds occupied by stroke patients 75% of the time' objects='stroke unit bed' id='activity2'/>
      <parallelGateway id='parallelgateway1'>
        <branch id='branch3'>
          <activity role='clinicians' action='assess stroke-related impairments using standardized validated tools' objects='standardized impairment assessment tools' id='activity6_1'/>
          <activity role='clinicians' action='assess functional status using standardized validated tools' objects='standardized functional assessment tools' id='activity6_2'/>
          <activity role='clinicians' action='document assessment findings in the electronic health record where possible' objects='assessment results, electronic health record' id='activity6_3'/>
          <activity role='clinicians' action='complete AlphaFIM on or by day 3 after admission with target day 3 and admission day counted as day 1' objects='AlphaFIM assessment' id='activity7'/>
        </branch>
        <branch id='branch4'>
          <parallelGateway id='parallelgateway2'>
            <branch id='branch5'>
              <activity role='physiotherapist' action='assess the patient within 48 hours of hospital admission and formulate a physiotherapy management plan' objects='physiotherapy assessment, physiotherapy management plan' id='activity8'/>
            </branch>
            <branch id='branch6'>
              <activity role='occupational therapist' action='assess the patient within 48 hours of hospital admission and formulate an occupational therapy management plan' objects='occupational therapy assessment, occupational therapy management plan' id='activity9'/>
            </branch>
            <branch id='branch7'>
              <activity role='speech-language pathologist' action='assess the patient within 48 hours of hospital admission and formulate a speech-language pathology management plan' objects='speech-language pathology assessment, speech-language pathology management plan' id='activity10'/>
            </branch>
          </parallelGateway>
        </branch>
        <branch id='branch8'>
          <activity role='clinicians' action='assess the patient for risk of developing venous thromboembolism' objects='venous thromboembolism risk assessment' id='activity11'/>
          <activity role='nursing staff' action='encourage adequate hydration to help prevent venous thromboembolism' objects='hydration plan' id='activity12'/>
          <exclusiveGateway id='exclusivegateway2'>
            <branch condition='early mobilization is contraindicated' id='branch9'>
              <activity role='stroke unit team' action='defer early mobilization and document the contraindication' objects='mobilization contraindication record' id='activity13'/>
            </branch>
            <branch condition='early mobilization is not contraindicated' id='branch10'>
              <exclusiveGateway id='exclusivegateway3'>
                <branch condition='stroke onset is within 24 hours and clinical judgement supports very early mobilization' id='branch11'>
                  <activity role='stroke unit team' action='mobilize the patient cautiously within 24 hours of stroke onset using clinical judgement and avoid frequent out-of-bed activity' objects='very early mobilization plan' id='activity14'/>
                </branch>
                <branch condition='stroke onset is between 24 hours and 48 hours' id='branch12'>
                  <activity role='stroke unit team' action='mobilize the patient early between 24 hours and 48 hours of stroke onset' objects='early mobilization plan' id='activity15'/>
                </branch>
                <branch condition='stroke onset is within 24 hours and clinical judgement does not support very early mobilization' id='branch13'>
                  <activity role='stroke unit team' action='avoid frequent out-of-bed activity within the first 24 hours of stroke onset and reassess for mobilization at 24 to 48 hours' objects='mobilization orders' id='activity16'/>
                  <exclusiveGateway id='exclusivegateway3_1'>
                    <branch condition='at 24 to 48 hours early mobilization is not contraindicated' id='branch13_1'>
                      <activity role='stroke unit team' action='mobilize the patient early between 24 hours and 48 hours of stroke onset' objects='early mobilization plan' id='activity28'/>
                    </branch>
                    <branch condition='at 24 to 48 hours early mobilization remains contraindicated' id='branch13_2'>
                      <activity role='stroke unit team' action='defer early mobilization and document the contraindication' objects='mobilization contraindication record' id='activity29'/>
                    </branch>
                  </exclusiveGateway>
                </branch>
                <branch condition='stroke onset is more than 48 hours ago' id='branch24'>
                  <activity role='stroke unit team' action='mobilize the patient as soon as clinically safe and document the reason the recommended early mobilization window was missed' objects='mobilization plan, delayed mobilization record' id='activity27'/>
                </branch>
              </exclusiveGateway>
            </branch>
          </exclusiveGateway>
        </branch>
        <branch id='branch14'>
          <activity role='clinicians' action='screen nutrition status within 48 hours using a valid screening tool' objects='nutrition screening tool' id='activity17_1'/>
          <activity role='clinicians' action='screen hydration status within 48 hours using a valid screening tool' objects='hydration screening tool' id='activity17_2'/>
          <activity role='clinicians' action='document screening results and identify nutritional concerns, hydration deficits, dysphagia, or other comorbidities' objects='nutrition and hydration screening results' id='activity17_3'/>
          <exclusiveGateway id='exclusivegateway4'>
            <branch condition='nutritional concerns, hydration deficits, dysphagia, or other comorbidities are present' id='branch15'>
              <activity role='clinicians' action='refer the patient to a dietitian within 7 days of admission for recommendations' objects='dietitian referral' id='activity18'/>
              <exclusiveGateway id='exclusivegateway5'>
                <branch condition='patient is unable to meet nutritional and fluid requirements' id='branch16'>
                  <activity role='dietitian' action='consider enteral nutrition support' objects='enteral nutrition support plan' id='activity19'/>
                </branch>
                <branch condition='patient can meet nutritional and fluid requirements' id='branch17'>
                  <activity role='dietitian' action='provide nutrition and hydration recommendations' objects='nutrition and hydration recommendations' id='activity20'/>
                </branch>
              </exclusiveGateway>
            </branch>
            <branch condition='no nutritional concerns or hydration deficits are identified' id='branch18'>
              <activity role='clinicians' action='continue routine nutrition and hydration management' objects='nutrition and hydration care plan' id='activity21'/>
            </branch>
          </exclusiveGateway>
        </branch>
      </parallelGateway>
      <activity role='stroke unit team' action='provide appropriate cross-continuum secondary prevention assessments for ongoing care' objects='Module 5 prevention assessments' id='activity22'/>
      <activity role='stroke unit team' action='provide appropriate cross-continuum secondary prevention therapies for ongoing care' objects='Module 10 prevention therapies' id='activity23'/>
      <exclusiveGateway id='exclusivegateway6'>
        <branch condition='patient is admitted to in-hospital rehabilitation' id='branch21'>
          <activity role='acute care team' action='transfer the patient to in-hospital rehabilitation with the stroke management and secondary prevention plans' objects='rehabilitation referral, stroke care plan, secondary prevention plan' id='activity24'/>
        </branch>
        <branch condition='patient is discharged from acute care' id='branch22'>
          <activity role='acute care team' action='discharge the patient with the stroke management and secondary prevention plans' objects='discharge summary, stroke care plan, secondary prevention plan' id='activity25'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='patient has TIA or minor nondisabling stroke' id='branch23'>
      <activity role='admitting team' action='do not apply Module 3a acute care treatment recommendations and redirect the patient to the appropriate TIA or minor stroke pathway' objects='TIA or minor stroke care pathway' id='activity26'/>
    </branch>
  </exclusiveGateway>
</process>
