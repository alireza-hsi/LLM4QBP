  <parallelGateway id='parallelgateway1'>

    <branch id='branch2'>
      <activity role='clinician' action='make patient NPO initially' objects='oral medication, fluid, food restrictions' id='activity4'/>
    </branch>
    <branch id='branch3'>
      <activity role='clinician' action='perform ECG without delaying acute treatment' objects='ECG, atrial fibrillation and other acute arrhythmias assessment' id='activity5'/>

    <branch id='branch4'>
      <activity role='clinician' action='order and collect blood work' objects='CBC, electrolytes, creatinine, glucose, INR, partial thromboplastin time, troponin test if clinically indicated' id='activity6'/>
    </branch>

          <parallelGateway id='parallelgateway2'>
            <branch id='branch5_1_1'>
              <activity role='stroke team' action='perform rapid brain imaging' objects='non-contrast CT or MRI with door-to-imaging target less than 25 minutes' id='activity7_1'/>
            </branch>
            <branch id='branch5_1_2'>
              <activity role='stroke team' action='perform rapid vascular imaging' objects='CTA with door-to-CTA target less than 25 minutes' id='activity7_2'/>
            </branch>
          </parallelGateway>
          <activity role='qualified imaging expert' action='interpret brain imaging immediately' objects='CT, CTA, and/or MRI of the brain' id='activity8'/>
        </branch>
        <branch condition='chest x-ray is clinically requested' id='branch6'>
          <activity role='clinician' action='perform chest x-ray without delaying assessment for thrombolysis' objects='chest x-ray' id='activity9'/>
        </branch>
      </inclusiveGateway>
      <exclusiveGateway id='exclusivegateway1'>
        <branch condition='brain imaging confirms acute ischemic stroke' id='branch10'>
          <exclusiveGateway id='exclusivegateway2'>
            <branch condition='patient is clinically eligible for acute stroke treatments' id='branch11'>
              <exclusiveGateway id='exclusivegateway2_1'>
                <branch condition='advanced CT with multiphase CTA is selected' id='branch11_1'>
                  <activity role='stroke physician' action='perform advanced CT with multiphase CTA' objects='advanced CT imaging and multiphase CTA' id='activity13_1'/>
                </branch>
                <branch condition='Perfusion CT is selected' id='branch11_2'>
                  <activity role='stroke physician' action='perform Perfusion CT' objects='perfusion CT imaging' id='activity13_2'/>
                </branch>
              </exclusiveGateway>
              <exclusiveGateway id='exclusivegateway3'>
                <branch condition='patient has potentially disabling acute neurological symptoms within 6.0 hours of onset or treatment can be initiated within 12 hours and clinical and imaging criteria are met' id='branch12'>
                  <activity role='stroke physician and regional stroke system' action='consider endovascular therapy and determine treatment and transport pathway' objects='EVT eligibility, clinical criteria, imaging criteria' id='activity14'/>
                </branch>
                <branch condition='patient does not meet EVT timing or clinical and imaging criteria' id='branch13'>
                  <activity role='stroke physician' action='continue non-EVT acute ischemic stroke pathway' objects='ischemic stroke management plan' id='activity15'/>
                </branch>
              </exclusiveGateway>
            </branch>
            <branch condition='patient is not clinically eligible for acute stroke treatments' id='branch14'>
              <activity role='stroke physician' action='continue assessment and medical management for acute ischemic stroke' objects='ischemic stroke management plan' id='activity16'/>
            </branch>
          </exclusiveGateway>
        </branch>
        <branch condition='brain imaging confirms intracerebral hemorrhage' id='branch15'>
          <activity role='stroke physician' action='treat as a medical emergency and evaluate immediately' objects='ICH patient assessment by physician with expertise in stroke management' id='activity19'/>
          <activity role='stroke physician' action='ask about anticoagulant therapy' objects='anticoagulant medication history' id='activity20_1'/>
          <activity role='stroke physician' action='review platelet count, PTT, and INR' objects='platelet count, PTT, INR results' id='activity20_2'/>
          <exclusiveGateway id='exclusivegateway4'>
            <branch condition='patient is selected for additional vascular imaging' id='branch16'>
              <activity role='stroke team' action='perform vascular imaging as soon as possible' objects='CTA or MRA' id='activity21'/>

            </branch>
          </exclusiveGateway>
        </branch>
        <branch condition='other suspected stroke presentation or diagnosis pathway applies after acute imaging' id='branch19'>
          <activity role='clinical team' action='continue appropriate assessment pathway based on presentation and findings' objects='clinical findings and local stroke protocol' id='activity25'/>
        </branch>
      </exclusiveGateway>
    </branch>

      <exclusiveGateway id='exclusivegateway5'>

          <activity role='stroke team' action='perform vascular imaging as soon as possible' objects='vascular imaging of the brain and neck arteries' id='activity24'/>
        </branch>
        <branch condition='other diagnosis or pathway applies after initial clinical assessment' id='branch18_2'>
          <activity role='clinical team' action='continue appropriate assessment pathway based on presentation and findings' objects='clinical findings and local stroke protocol' id='activity27'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='other suspected stroke presentation or diagnosis pathway applies before acute imaging' id='branch20'>
      <activity role='clinical team' action='continue appropriate assessment pathway based on presentation and findings' objects='clinical findings and local stroke protocol' id='activity28'/>
    </branch>
  </exclusiveGateway>
