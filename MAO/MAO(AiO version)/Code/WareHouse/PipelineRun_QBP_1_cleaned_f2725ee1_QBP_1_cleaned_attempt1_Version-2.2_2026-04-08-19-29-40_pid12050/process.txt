<process>
  <exclusiveGateway id='eg1'>
    <branch condition='patient arrives by EMS' id='br1'>
      <activity role='EMS' action='identify suspected stroke/TIA and divert the patient to a regional or district stroke centre where feasible' objects='patient, regional redirect protocol' id='a1'/>
      <activity role='EMS' action='prenotify the receiving hospital of the stroke/TIA patient arrival' objects='patient status, symptom onset time, clinical details' id='a2'/>
      <activity role='receiving hospital' action='contact and call the acute stroke team to the ED for appropriate patients' objects='EMS prenotification details' id='a3'/>
    </branch>
    <branch condition='patient presents to the ED or outpatient clinic without EMS' id='br2'>
      <activity role='triage staff' action='apply a standardized stroke/TIA assessment and treatment protocol at presentation' objects='patient' id='a4'/>
      <activity role='ED or outpatient clinic staff' action='contact the acute stroke team for appropriate patients' objects='presentation details' id='a5'/>
    </branch>
    <branch condition='patient is directly admitted to acute care' id='br3'>
      <activity role='acute care team' action='activate the standardized stroke/TIA assessment pathway on admission' objects='patient' id='a6'/>
      <activity role='acute care team' action='notify the acute stroke team for appropriate patients' objects='admission details' id='a7'/>
    </branch>
  </exclusiveGateway>
  <activity role='clinical staff' action='assess airway' objects='airway patency' id='a8_1'/>
  <activity role='clinical staff' action='assess breathing' objects='respiratory status' id='a8_2'/>
  <activity role='clinical staff' action='assess circulation' objects='hemodynamic status' id='a8_3'/>
  <parallelGateway id='pg1'>
    <branch id='br4'>
      <activity role='clinical staff' action='perform a neurological examination to determine focal neurological deficits and assess stroke severity' objects='NIHSS or CNS for stroke' id='a9'/>
    </branch>
    <branch id='br5'>
      <exclusiveGateway id='eg10_1'>
        <branch condition='CT is the selected and available immediate brain imaging modality' id='br10_1'>
          <activity role='diagnostic imaging staff' action='perform immediate CT brain imaging' objects='CT brain images' id='a10_1'/>
        </branch>
        <branch condition='MRI is the selected and available immediate brain imaging modality' id='br10_2'>
          <activity role='diagnostic imaging staff' action='perform immediate MRI brain imaging' objects='MRI brain images' id='a10_2'/>
        </branch>
      </exclusiveGateway>
      <activity role='health care provider with CT and/or MRI expertise' action='interpret brain imaging immediately' objects='CT and/or MRI images' id='a11'/>
    </branch>
    <branch id='br6'>
      <activity role='clinical staff' action='perform ECG to detect atrial fibrillation and other acute arrhythmias' objects='electrocardiogram tracing' id='a12'/>
    </branch>
    <branch id='br7'>
      <activity role='clinical staff' action='obtain routine blood work' objects='CBC, electrolytes, creatinine, glucose, INR, partial thromboplastin time' id='a13_1'/>
      <exclusiveGateway id='eg13_1'>
        <branch condition='troponin test is clinically indicated' id='br13_1'>
          <activity role='clinical staff' action='obtain troponin test' objects='troponin specimen and result' id='a13_2'/>
        </branch>
        <branch condition='troponin test is not clinically indicated' id='br13_2'>
          <activity role='clinical staff' action='continue assessment without troponin test' objects='clinical indication review' id='a13_3'/>
        </branch>
      </exclusiveGateway>
    </branch>
  </parallelGateway>
  <exclusiveGateway id='eg2_new'>
    <branch condition='chest x-ray is clinically indicated after urgent stroke assessment' id='br9_new'>
      <activity role='clinical staff' action='obtain chest x-ray without delaying thrombolysis assessment' objects='chest x-ray' id='a14_new'/>
    </branch>
    <branch condition='chest x-ray is not clinically indicated after urgent stroke assessment' id='br10_new'>
      <activity role='clinical staff' action='continue care without chest x-ray' objects='clinical indication review' id='a15_new'/>
    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='eg8_4'>
    <branch condition='patient has stroke other than TIA or minor (nondisabling) stroke' id='br8_4'>
      <activity role='clinical staff' action='make the patient NPO initially' objects='oral medication, fluid, food' id='a8_4'/>
    </branch>
    <branch condition='patient has TIA or minor (nondisabling) stroke' id='br8_5'>
      <activity role='clinical staff' action='continue the initial assessment without stroke-specific NPO restriction' objects='TIA or minor stroke assessment result' id='a8_5'/>
    </branch>
  </exclusiveGateway>
  <inclusiveGateway id='ig1'>
    <branch condition='patient has stroke other than TIA or minor (nondisabling) stroke and is alert for bedside screening' id='br11'>
      <activity role='clinical staff' action='screen swallowing ability as part of the initial assessment and before initiating oral medication, fluid, or food' objects='simple, valid, reliable bedside testing protocol' id='a17'/>
      <exclusiveGateway id='eg3'>
        <branch condition='features indicate dysphagia or pulmonary aspiration' id='br12'>
          <activity role='speech-language pathologist or appropriately trained specialist' action='perform a full clinical assessment of swallowing ability and advise on swallowing ability and the required consistency of diet and fluids' objects='swallowing assessment findings' id='a18'/>
        </branch>
        <branch condition='features do not indicate dysphagia or pulmonary aspiration' id='br13'>
          <activity role='clinical staff' action='proceed according to bedside swallowing screen results' objects='swallowing screen results, oral medication, fluid, food' id='a19'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='patient has stroke other than TIA or minor (nondisabling) stroke and is not alert within the first 24 hours' id='br14'>
      <activity role='clinical staff' action='monitor the patient closely within the first 24 hours' objects='level of alertness, aspiration risk' id='a21'/>
      <activity role='clinical staff' action='screen swallowing ability when clinically appropriate and before initiating oral medication, fluid, or food' objects='simple, valid, reliable bedside testing protocol' id='a22'/>
      <exclusiveGateway id='eg4'>
        <branch condition='features indicate dysphagia or pulmonary aspiration' id='br15'>
          <activity role='speech-language pathologist or appropriately trained specialist' action='perform a full clinical assessment of swallowing ability and advise on swallowing ability and the required consistency of diet and fluids' objects='swallowing assessment findings' id='a23'/>
        </branch>
        <branch condition='features do not indicate dysphagia or pulmonary aspiration' id='br16'>
          <activity role='clinical staff' action='continue care according to clinically appropriate swallowing screen results' objects='swallowing screen results' id='a24'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='patient has TIA' id='br17'>
      <activity role='clinical staff' action='collect TIA risk and presentation information' objects='symptom onset time, symptom pattern, neurological findings, risk factors' id='a25_1'/>
      <activity role='clinical staff' action='apply a standardized triage tool to determine the appropriate location for care' objects='TIA triage tool, patient risk information' id='a25_2'/>
      <exclusiveGateway id='eg5'>
        <branch condition='triage indicates hospital admission or in-hospital stroke care' id='br18'>
          <activity role='clinical staff' action='admit the patient to the appropriate hospital stroke pathway' objects='admission order, triage results' id='a26'/>
        </branch>
        <branch condition='triage indicates discharge with rapid outpatient follow-up' id='br19'>
          <activity role='clinical staff' action='refer the patient to a rapid-assessment TIA or minor stroke unit, TIA clinic, or provincial stroke-prevention clinic' objects='referral, triage results' id='a27'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='patient has TIA or minor (nondisabling) stroke presenting within 48 hours of symptom onset or with fluctuating motor or speech symptoms' id='br20'>
      <exclusiveGateway id='eg6'>
        <branch condition='patient is not clearly excluded from carotid artery revascularization' id='br21'>
          <activity role='clinical staff' action='perform immediate vascular imaging of the neck arteries' objects='carotid ultrasound, CTA, or MRA' id='a28'/>
        </branch>
        <branch condition='patient is clearly not a candidate for carotid artery revascularization' id='br22'>
          <activity role='clinical staff' action='document that immediate neck artery vascular imaging is not required because the patient is clearly not a candidate for carotid artery revascularization' objects='clinical candidacy assessment' id='a29'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='patient has TIA or ischemic stroke presenting beyond 48 hours' id='br23'>
      <activity role='clinical staff' action='perform vascular imaging of the brain and neck arteries as soon as possible' objects='vascular imaging of brain and neck arteries' id='a30'/>
    </branch>
    <branch condition='patient has TIA or minor (nondisabling) stroke and stroke-specific dysphagia recommendations are not applicable' id='br24'>
      <activity role='clinical staff' action='exclude the patient from the stroke-specific dysphagia pathway' objects='TIA or minor stroke assessment result' id='a31'/>
    </branch>
  </inclusiveGateway>
  <exclusiveGateway id='eg32_1'>
    <branch condition='patient is admitted to hospital' id='br32_1'>
      <activity role='clinical team' action='initiate appropriate cross-continuum secondary prevention assessments and therapies during admission' objects='secondary prevention plan, Modules 5 and 10 interventions' id='a32_1'/>
    </branch>
    <branch condition='patient is discharged from the ED' id='br32_2'>
      <activity role='clinical team' action='provide appropriate cross-continuum secondary prevention assessments, therapies, and follow-up planning at discharge' objects='discharge secondary prevention plan, referrals, Modules 5 and 10 interventions' id='a32_2'/>
    </branch>
  </exclusiveGateway>
</process>
