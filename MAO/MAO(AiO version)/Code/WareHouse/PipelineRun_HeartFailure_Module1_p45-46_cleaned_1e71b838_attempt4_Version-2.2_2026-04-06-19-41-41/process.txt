<process>
  <activity role='emergency department clinician' action='Initiate acute heart failure episode of care' objects='patient presenting with suspected acute heart failure' id='activity1'/>
  <activity role='nursing staff' action='Frequently measure heart rate, blood pressure, and oxygen saturation until patient is stabilized' objects='monitoring equipment' id='activity2'/>
  <activity role='emergency department clinician' action='Apply standardized risk-stratification tool for acute heart failure and other conditions' objects='risk-stratification tool' id='activity3'/>
  <parallelGateway id='parallelgateway1'>
    <branch id='branch1'>
      <activity role='emergency department clinician' action='Order initial investigations for responsiveness to diuresis' objects='serum creatinine, electrolyte levels, troponin, complete blood count, electrocardiogram, chest x-ray, echocardiogram (if no recent available)' id='activity4'/>
    </branch>
    <branch id='branch2'>
      <activity role='emergency department clinician' action='Review results of investigations for responsiveness to diuresis' objects='investigation results' id='activity5'/>
    </branch>
  </parallelGateway>
  <activity role='emergency department clinician' action='Calculate acute heart failure risk score (e.g., EHMRG) to predict 7-day mortality risk' objects='EHMRG risk score tool' id='activity6'/>
  <exclusiveGateway id='exclusivegateway1'>
    <branch condition='Low-risk (EHMRG quintiles 1-2), responded to initial ED treatment, no complicating factors' id='branch3'>
      <activity role='emergency department clinician' action='Provide discharge instructions' objects='discharge instructions' id='activity7'/>
      <activity role='emergency department clinician' action='Complete discharge documentation' objects='discharge forms, electronic health record' id='activity8'/>
      <activity role='emergency department clinician' action='Arrange follow-up visit with primary care provider within days of discharge' objects='follow-up appointment' id='activity9'/>
    </branch>
    <branch condition='High-risk (EHMRG quintiles 3-5) or complicating factors present AND high-intensity care required (ventilation/ICU)' id='branch4'>
      <activity role='emergency department clinician' action='Admit patient to intensive care unit with high nurse-to-patient staffing' objects='ICU admission order' id='activity10'/>
    </branch>
    <branch condition='High-risk (EHMRG quintiles 3-5) or complicating factors present AND average-intensity care required (inpatient admission, normal staffing)' id='branch5'>
      <activity role='emergency department clinician' action='Admit patient to inpatient care with normal nurse-to-patient staffing' objects='inpatient admission order' id='activity11'/>
    </branch>
  </exclusiveGateway>
  <activity role='hospital administration' action='Ensure standardized risk stratification assessment tool/process is used for all acute heart failure presentations' objects='hospital policy, risk stratification tool' id='activity12'/>
  <activity role='hospital administration' action='Maintain pathway/mechanism for transfer to higher level provider as needed' objects='transfer protocols' id='activity13'/>
</process>
