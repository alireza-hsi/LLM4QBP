<process>
  <activity role='ED triage nurse' action='identify suspected acute heart failure' objects='triage assessment, symptoms/signs' id='activity1.1'/>
  <activity role='ED triage nurse' action='initiate standardized risk assessment/stratification process (used consistently across hospitals) and document initial risk data' objects='risk stratification tool, EHR' id='activity1.2'/>

  <inclusiveGateway id='inclusivegateway1'>
    <branch condition='initial investigations required' id='branch1'>
      <activity role='ED clinician' action='order and obtain serum creatinine and electrolyte levels' objects='blood sample' id='activity2'/>
      <activity role='ED clinician' action='order and obtain troponin measurements' objects='blood sample' id='activity3'/>
      <activity role='ED clinician' action='order and obtain complete blood count' objects='blood sample' id='activity4'/>
    </branch>
    <branch condition='initial investigations required' id='branch2'>
      <activity role='ED clinician' action='perform electrocardiogram' objects='ECG machine' id='activity5'/>
      <activity role='radiology' action='perform chest x-ray examination' objects='portable x-ray or radiology suite' id='activity6'/>
    </branch>
    <branch condition='echo assessment required' id='branch3'>
      <exclusiveGateway id='exclusivegateway1'>
        <branch condition='no recent echocardiogram available' id='branch3a'>
          <activity role='cardiology/echo lab' action='perform echocardiogram' objects='echocardiography equipment' id='activity7'/>
        </branch>
        <branch condition='recent echocardiogram available' id='branch3b'>
          <activity role='ED clinician' action='review recent echocardiogram results' objects='prior echocardiogram report' id='activity8'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='patient not yet stabilized; requires frequent monitoring' id='branch4'>
      <activity role='ED nurse' action='measure heart rate, blood pressure, and oxygen saturation frequently until patient is stabilized' objects='vital signs monitor, pulse oximeter' id='activity9'/>
    </branch>
  </inclusiveGateway>

  <exclusiveGateway id='exclusivegateway10'>
    <branch condition='able to take IV therapy/needs rapid effect' id='branch10a'>
      <activity role='ED clinician' action='administer IV furosemide and record time/dose' objects='IV furosemide, MAR' id='activity10.1'/>
    </branch>
    <branch condition='appropriate for oral therapy' id='branch10b'>
      <activity role='ED clinician' action='administer oral furosemide and record time/dose' objects='oral furosemide, MAR' id='activity10.2'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED clinician' action='assess responsiveness to diuresis and reassess urine output/weight/vitals; adjust diuretic plan if needed' objects='urine output, weight, vitals, medication plan' id='activity10.3'/>

  <activity role='ED clinician' action='assess high-risk markers (respiratory distress, hypoxemia, severity of pulmonary edema, poor response to furosemide in ED, hemodynamic compromise, significant arrhythmias, positive troponin)' objects='clinical exam, vitals, ECG, labs, imaging' id='activity11'/>
  <activity role='ED clinician' action='calculate acute heart failure risk score (e.g., EHMRG) to predict 7-day mortality risk' objects='EHMRG inputs and calculator' id='activity12'/>

  <exclusiveGateway id='exclusivegateway2'>
    <branch condition='low-risk (e.g., EHMRG quintiles 1–2) AND responded to initial ED treatment AND no other considerations (advanced directives, severe dementia, estimated impact of admission on life-expectancy, etc.)' id='branch5'>
      <activity role='ED clinician' action='treat in ED or outpatient setting and confirm discharge decision' objects='clinical reassessment, disposition decision' id='activity13'/>
      <parallelGateway id='parallelgateway13'>
        <branch id='branch13a'>
          <activity role='ED clinician' action='provide discharge education and return precautions' objects='discharge instructions' id='activity13.1'/>
        </branch>
        <branch id='branch13b'>
          <activity role='ED clinician' action='reconcile home medications and prescribe/adjust HF medications as appropriate' objects='medication list, prescriptions' id='activity13.2'/>
        </branch>
      </parallelGateway>
      <activity role='ED staff' action='arrange follow-up visit with primary care provider within days of ED discharge' objects='follow-up appointment request' id='activity14'/>
    </branch>
    <branch condition='higher-risk (e.g., EHMRG quintiles 3–5) OR not responding to initial ED treatment OR other considerations present' id='branch6'>
      <activity role='ED clinician' action='ensure escalation pathway/mechanism to transfer to higher level provider is available if patient deteriorates' objects='escalation protocol' id='activity18'/>
      <exclusiveGateway id='exclusivegateway3'>
        <branch condition='ICU admission required (hemodynamic/respiratory instability, need for invasive ventilation, etc.)' id='branch7'>
          <activity role='ED clinician' action='initiate noninvasive or invasive ventilation as indicated' objects='NIV/ventilator equipment' id='activity15'/>
          <activity role='ED clinician' action='activate ICU transfer/admission protocol and request ICU bed (use hospital pathway/mechanism to transfer to higher level provider)' objects='ICU bed request, transfer protocol' id='activity16.1'/>
          <activity role='ED nurse' action='perform clinical handoff and arrange transport to ICU' objects='handoff note, transport resources' id='activity16.2'/>
        </branch>
        <branch condition='ward admission appropriate (no ICU criteria met)' id='branch8'>
          <activity role='ED clinician' action='place inpatient admission orders and request ward bed' objects='admission orders, bed request' id='activity17.1'/>
          <activity role='ED nurse' action='handoff and transfer patient to inpatient unit; confirm escalation pathway available if needed' objects='handoff note, escalation protocol' id='activity17.2'/>
        </branch>
      </exclusiveGateway>
    </branch>
  </exclusiveGateway>
</process>
