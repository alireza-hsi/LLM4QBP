<process>
  <activity role='ED triage nurse' action='identify suspected acute heart failure presentation and initiate risk stratification pathway' objects='patient' id='activity1'/>
  <activity role='ED physician' action='apply standardized risk assessment/stratification tool consistently' objects='risk-stratification tool (multi-condition including HF)' id='activity2_applyTool'/>
  <activity role='ED physician' action='collect standardized risk assessment inputs' objects='history, exam findings, comorbidities, medications' id='activity2_1'/>
  <activity role='ED physician' action='determine preliminary acuity and disposition risk level from standardized tool' objects='risk assessment outputs' id='activity2_2'/>

  <parallelGateway id='parallelgateway1'>
    <branch id='branch1'>
      <activity role='nurse/physician' action='measure heart rate, blood pressure, and oxygen saturation frequently until patient is stabilized' objects='vital signs monitor' id='activity4'/>
    </branch>
    <branch id='branch2'>
      <activity role='ED nurse/phlebotomy' action='collect serum creatinine and electrolyte levels' objects='blood sample' id='activity5'/>
      <activity role='laboratory' action='analyze serum creatinine and electrolyte levels' objects='blood sample' id='activity6'/>
    </branch>
    <branch id='branch3'>
      <activity role='ED nurse/phlebotomy' action='collect troponin measurements' objects='blood sample' id='activity7'/>
      <activity role='laboratory' action='analyze troponin measurements' objects='blood sample' id='activity8'/>
    </branch>
    <branch id='branch4'>
      <activity role='ED nurse/phlebotomy' action='collect complete blood count' objects='blood sample' id='activity9'/>
      <activity role='laboratory' action='analyze complete blood count' objects='blood sample' id='activity10'/>
    </branch>
    <branch id='branch5'>
      <activity role='ED clinician' action='perform electrocardiogram' objects='ECG machine' id='activity11'/>
      <activity role='ED clinician' action='review electrocardiogram for arrhythmias/ischemic changes' objects='ECG tracing' id='activity12'/>
    </branch>
    <branch id='branch6'>
      <activity role='radiology technologist' action='perform chest x-ray examination' objects='x-ray equipment' id='activity13'/>
      <activity role='ED physician' action='review chest x-ray for pulmonary edema severity' objects='CXR images' id='activity14'/>
    </branch>
    <branch id='branch7'>
      <exclusiveGateway id='exclusivegateway1'>
        <branch condition='no recent echocardiogram available' id='branch8'>
          <activity role='cardiology/echo lab' action='perform echocardiogram' objects='echocardiography equipment' id='activity15'/>
          <activity role='ED physician/cardiology' action='review echocardiogram results' objects='echo report' id='activity16'/>
        </branch>
        <branch condition='recent echocardiogram available' id='branch9'>
          <activity role='ED physician' action='retrieve and review recent echocardiogram' objects='prior echo report' id='activity17'/>
        </branch>
      </exclusiveGateway>
    </branch>
  </parallelGateway>

  <activity role='ED physician/ED nurse' action='administer or confirm diuretic plan (e.g., furosemide dosing) and timing' objects='medication order, administration record' id='activity18_1'/>
  <activity role='ED nurse' action='monitor urine output and symptom changes after diuresis' objects='urine output record, symptom assessment' id='activity18_2'/>
  <activity role='ED physician' action='reassess renal function and electrolytes after diuresis when available' objects='repeat/returned labs, clinical status' id='activity18_3'/>
  <activity role='ED physician' action='assess responsiveness to diuresis including response to furosemide administered in ED' objects='diuresis response data' id='activity18'/>

  <parallelGateway id='parallelgateway2'>
    <branch id='branch10'>
      <activity role='ED physician' action='screen for high-risk markers: respiratory distress and hypoxemia' objects='clinical exam, oxygen saturation' id='activity19'/>
    </branch>
    <branch id='branch11'>
      <activity role='ED physician' action='screen for high-risk markers: severity of pulmonary edema' objects='CXR, clinical exam' id='activity20'/>
    </branch>
    <branch id='branch12'>
      <activity role='ED physician' action='screen for high-risk markers: hemodynamic compromise' objects='blood pressure, perfusion findings' id='activity21'/>
    </branch>
    <branch id='branch13'>
      <activity role='ED physician' action='screen for high-risk markers: significant arrhythmias' objects='ECG, telemetry' id='activity22'/>
    </branch>
    <branch id='branch14'>
      <activity role='ED physician' action='screen for high-risk markers: positive troponin' objects='troponin result' id='activity23'/>
    </branch>
    <branch id='branch15'>
      <activity role='ED physician' action='screen for high-risk markers: poor response to furosemide administered in ED' objects='urine output, symptoms, weight' id='activity24'/>
    </branch>
  </parallelGateway>

  <activity role='ED physician' action='gather required EHMRG input variables from chart, vitals, ECG, and labs' objects='patient variables for EHMRG' id='activity25_1'/>
  <activity role='ED physician' action='compute EHMRG score and assign quintile/category' objects='EHMRG calculator output' id='activity25_2'/>
  <activity role='ED physician' action='document and communicate EHMRG risk score to care team for disposition decision' objects='ED note, handover' id='activity25_3'/>
  <activity role='ED physician' action='calculate acute heart failure risk score (e.g., EHMRG) to predict 7-day mortality risk' objects='patient clinical variables, EHMRG calculator' id='activity25'/>

  <exclusiveGateway id='exclusivegateway_intensity1'>
    <branch condition='high intensity criteria present (requires noninvasive/invasive ventilation and/or ICU with higher nurse-to-patient staffing)' id='branch16'>
      <activity role='ED physician' action='assign patient to high intensity group' objects='risk stratification results' id='activity26'/>
      <activity role='ED physician' action='transfer/admit to ICU and initiate ventilation as required' objects='ICU bed request, ventilation resources' id='activity27'/>
      <activity role='ED physician' action='initiate pathway/mechanism to transfer patient to a higher level provider when indicated' objects='transfer pathway/contacts' id='activity3'/>
      <activity role='ED physician' action='identify required higher level of care and receiving service (ICU/cardiology/tertiary centre)' objects='acuity findings, hospital capabilities' id='activity3_1'/>
      <activity role='ED physician/charge nurse' action='contact receiving service and arrange transport/bed as needed' objects='consult request, transfer request, transport resources' id='activity3_2'/>
      <activity role='ED physician/ED nurse' action='communicate clinical handover and send required documentation' objects='handover note, results, medication list' id='activity3_3'/>
    </branch>
    <branch condition='no high intensity criteria AND (EHMRG quintiles 3–5 OR not responded to initial ED treatment OR other considerations present (e.g., advanced directives, severe dementia, estimated impact of admission on life-expectancy))' id='branch17'>
      <activity role='ED physician' action='assign patient to average intensity group' objects='risk stratification results' id='activity28'/>
      <activity role='bed management/inpatient team' action='admit to inpatient care with normal nurse-to-patient staffing' objects='admission order' id='activity29'/>
    </branch>
    <branch condition='no high intensity criteria AND EHMRG quintiles 1–2 AND responded to initial ED treatment AND no other considerations present (e.g., advanced directives, severe dementia, estimated impact of admission on life-expectancy)' id='branch18'>
      <activity role='ED physician' action='assign patient to low intensity group' objects='risk stratification results' id='activity30'/>
      <activity role='ED physician' action='treat in ED or outpatient setting and prepare discharge home without inpatient admission' objects='treatment plan, discharge instructions' id='activity31'/>
      <activity role='ED physician/ED nurse' action='complete ED/outpatient treatment and confirm clinical stability for discharge' objects='treatment orders, reassessment findings' id='activity31_1'/>
      <activity role='ED physician' action='provide discharge instructions and return precautions specific to heart failure' objects='patient instructions, warning signs' id='activity31_2'/>
      <activity role='ED physician/pharmacist' action='perform medication reconciliation and issue prescriptions as needed' objects='medication list, prescriptions' id='activity31_3'/>
      <activity role='ED staff' action='arrange follow-up visit with primary care provider within days of discharge from the ED' objects='follow-up appointment request' id='activity32'/>
    </branch>
  </exclusiveGateway>
</process>
