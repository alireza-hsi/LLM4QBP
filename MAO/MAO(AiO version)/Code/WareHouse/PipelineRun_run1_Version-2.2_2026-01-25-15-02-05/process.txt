<process>
  <activity role='EMS/triage nurse' action='perform EMS prenotification and activate acute stroke team (if prenotified/appropriate)' objects='prehospital stroke/TIA information' id='activity0'/>
  <activity role='EMS/triage nurse' action='receive EMS prenotification details (symptom onset/last known well, deficits, vitals, anticoagulants, glucose) and relay to ED' objects='prehospital report' id='activity1_1'/>
  <activity role='ED charge nurse/stroke team lead' action='activate acute stroke team and prepare ED/CT access (notify CT, allocate bed, prepare thrombolysis checklist if applicable)' objects='activation protocol, CT access plan' id='activity1_2'/>

  <activity role='ED triage nurse' action='register and triage suspected stroke/TIA patient (CTAS I/II access to diagnostic imaging where available)' objects='patient, triage tool' id='activity2'/>
  <activity role='ED clerk/triage nurse' action='register patient and create encounter' objects='patient identifiers, registration system' id='activity2_1'/>
  <activity role='ED triage nurse' action='perform CTAS triage and apply standardized stroke/TIA assessment protocol/tool' objects='CTAS criteria, stroke/TIA protocol' id='activity2_2'/>

  <activity role='ED physician/nurse' action='conduct rapid initial evaluation of airway, breathing, and circulation (ABC)' objects='patient' id='activity3'/>

  <activity role='ED nurse' action='make patient NPO initially and post swallow precautions (signage, chart flag)' objects='NPO order, precautions' id='activity8_1_mod'/>

  <activity role='ED physician' action='select imaging modality (CT vs MRI) based on availability, contraindications, and clinical need' objects='imaging criteria, patient history' id='activity7_1'/>
  <activity role='ED physician/charge nurse' action='place urgent imaging order and communicate stroke priority to imaging team' objects='urgent order, prenotification/priority message' id='activity7_2'/>

  <activity role='radiology/CT-MRI technologist' action='perform brain imaging immediately (CT or MRI)' objects='CT/MRI scanner, patient' id='activity9'/>

  <parallelGateway id='parallelgateway1_mod'>
    <branch id='branch1_mod'>
      <activity role='stroke team/ED physician' action='perform neurological examination to determine focal neurological deficits and assess stroke severity using standardized stroke scale (NIHSS or CNS)' objects='patient, NIHSS/CNS tool' id='activity4'/>
    </branch>
    <branch id='branch2_mod'>
      <activity role='ED nurse/physician' action='obtain ECG to detect atrial fibrillation and other acute arrhythmias' objects='ECG machine, patient' id='activity5'/>
    </branch>
    <branch id='branch3_mod'>
      <activity role='ED nurse/phlebotomy' action='collect blood work (CBC, electrolytes, creatinine, glucose, INR, partial thromboplastin time; troponin test if clinically indicated)' objects='blood samples, lab requisitions' id='activity6'/>
    </branch>
  </parallelGateway>

  <exclusiveGateway id='exclusivegateway2_mod'>
    <branch condition='Patient not alert within first 24 hours' id='branch8_mod'>
      <activity role='ED nurse/inpatient nurse' action='monitor closely within first 24 hours and screen swallowing ability when clinically appropriate' objects='patient, monitoring plan' id='activity13'/>
    </branch>
    <branch condition='Patient alert within first 24 hours' id='branch9_mod'>
      <activity role='ED nurse' action='perform bedside swallow screen before any oral medication, fluid, or food' objects='swallow screen tool, patient' id='activity8_2_mod'/>
    </branch>
  </exclusiveGateway>

  <exclusiveGateway id='exclusivegateway3_mod'>
    <branch condition='Features indicating dysphagia or pulmonary aspiration are present' id='branch10_mod'>
      <activity role='speech-language pathologist/appropriately trained specialist' action='perform full clinical swallowing assessment and advise swallowing ability and required consistency of diet and fluids' objects='patient, swallowing assessment tools' id='activity15'/>
    </branch>
    <branch condition='No features indicating dysphagia or pulmonary aspiration' id='branch11_mod'>
      <activity role='ED physician/nurse' action='continue standard diet/fluids plan per bedside screen and clinical status' objects='swallow screen result, diet orders' id='activity16'/>
    </branch>
  </exclusiveGateway>

  <activity role='radiologist/stroke physician' action='interpret brain imaging immediately (by provider with expertise reading CT/MRI)' objects='CT/MRI images' id='activity10'/>

  <exclusiveGateway id='exclusivegateway1'>
    <branch condition='Chest x-ray is clinically required' id='branch6'>
      <activity role='ED physician' action='order chest x-ray ensuring it does not delay assessment for thrombolysis' objects='chest x-ray order' id='activity11'/>
    </branch>
    <branch condition='Chest x-ray not required' id='branch7'>
      <activity role='ED physician' action='proceed without chest x-ray to avoid delaying thrombolysis assessment' objects='clinical decision' id='activity12'/>
    </branch>
  </exclusiveGateway>

  <exclusiveGateway id='exclusivegateway4'>
    <branch condition='Presentation is TIA or minor (nondisabling) stroke' id='branch12'>
      <activity role='ED physician/stroke team' action='apply standardized TIA triage tool to determine appropriate location for care' objects='TIA triage tool' id='activity17'/>
      <exclusiveGateway id='exclusivegateway5'>
        <branch condition='Within 48 hours of symptom onset OR fluctuating motor or speech symptoms' id='branch13'>
          <exclusiveGateway id='exclusivegateway6'>
            <branch condition='Clearly not a candidate for carotid artery revascularization' id='branch14'>
              <activity role='ED physician/stroke team' action='document not a candidate for carotid revascularization and do not perform immediate neck-artery vascular imaging' objects='clinical assessment, documentation' id='activity18'/>
            </branch>
            <branch condition='Candidate (or not clearly excluded) for carotid artery revascularization' id='branch15'>
              <activity role='ED physician/stroke team' action='perform immediate vascular imaging of neck arteries (carotid ultrasound, CTA, or MRA)' objects='carotid US/CTA/MRA order' id='activity19'/>
            </branch>
          </exclusiveGateway>
        </branch>
        <branch condition='Beyond 48 hours from symptom onset' id='branch16'>
          <activity role='ED physician/stroke team' action='arrange vascular imaging of brain and neck arteries as soon as possible' objects='CTA/MRA/other vascular imaging order' id='activity20'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='Presentation is stroke (not TIA/minor nondisabling stroke)' id='branch17'>
      <activity role='ED physician/stroke team' action='continue acute stroke pathway; note that TIA/minor-stroke-specific recommendations are not applicable' objects='clinical pathway documentation' id='activity21'/>
    </branch>
  </exclusiveGateway>

  <activity role='stroke team/ED physician' action='provide cross-continuum secondary prevention assessments and therapies for all patients (admitted or discharged) per Modules 5 and 10' objects='secondary prevention orders, education, referrals' id='activity22'/>
  <activity role='stroke team/ED physician' action='initiate secondary prevention orders and provide patient/caregiver education (risk factors, meds, warning signs)' objects='orderset, education materials' id='activity22_1'/>
  <exclusiveGateway id='exclusivegateway7'>
    <branch condition='Patient admitted to hospital' id='branch18'>
      <activity role='stroke team/inpatient team' action='handover and continue secondary prevention plan during admission' objects='handover note, inpatient orders' id='activity22_2'/>
    </branch>
    <branch condition='Patient discharged from ED' id='branch19'>
      <activity role='ED physician/stroke team' action='arrange outpatient follow-up/referral (TIA clinic/stroke prevention clinic) and provide discharge plan' objects='referrals, discharge instructions' id='activity22_3'/>
    </branch>
  </exclusiveGateway>
</process>
