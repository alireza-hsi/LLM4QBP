<process>
  <activity role='admitting clinician' action='determine applicability of acute care admission pathway' objects='acute stroke diagnosis, admitted TIA criteria, minor nondisabling stroke status' id='activity1'/>
  <exclusiveGateway id='exclusivegateway1'>
    <branch condition='patient has acute stroke or TIA requiring admission' id='branch1'>
      <activity role='admitting team' action='verify availability of a co-located stroke unit bed' objects='stroke unit bed availability' id='activity2_1'/>
      <exclusiveGateway id='exclusivegateway1_1'>
        <branch condition='stroke unit bed is available' id='branch1_1'>
          <activity role='admitting team' action='complete admission to a specialized geographically defined stroke unit' objects='co-located stroke unit bed, admission placement' id='activity2_2'/>
        </branch>
        <branch condition='stroke unit bed is not available' id='branch1_2'>
          <activity role='admitting team' action='place patient in an alternative inpatient bed with stroke team involvement' objects='alternative inpatient bed, interim admission placement, stroke team support' id='activity2_3'/>
          <activity role='admitting team' action='escalate for transfer to a specialized geographically defined stroke unit when a bed becomes available' objects='stroke unit transfer request, bed escalation plan' id='activity2_4'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='patient has TIA or minor nondisabling stroke not applicable to this module' id='branch2'>
      <activity role='clinician' action='redirect patient to appropriate non-acute-admission pathway' objects='TIA or minor stroke care plan' id='activity3'/>
    </branch>
  </exclusiveGateway>
  <activity role='stroke unit coordinator' action='assign patient to a dedicated interprofessional stroke team with stroke expertise' objects='medicine, nursing, occupational therapy, physiotherapy, speech-language pathology, social work, clinical nutrition' id='activity4'/>
  <activity role='hospital administration' action='ensure access to the specialized interprofessional stroke team 7 days a week' objects='7-day stroke staffing coverage' id='activity5'/>
  <activity role='stroke unit team' action='select standardized validated assessment tools for stroke-related impairments and functional status' objects='standardized stroke assessment tools' id='activity6_1'/>
  <activity role='stroke unit team' action='document assessment findings in the EHR where possible' objects='functional status assessments, electronic health record documentation' id='activity6_2'/>
  <parallelGateway id='parallelgateway1'>
    <branch id='branch3'>
      <activity role='physiotherapist' action='assess patient within 48 hours of admission and formulate a management plan' objects='physiotherapy assessment, management plan' id='activity7'/>
    </branch>
    <branch id='branch4'>
      <activity role='occupational therapist' action='assess patient within 48 hours of admission and formulate a management plan' objects='occupational therapy assessment, management plan' id='activity8'/>
    </branch>
    <branch id='branch5'>
      <activity role='speech-language pathologist' action='assess patient within 48 hours of admission and formulate a management plan' objects='speech-language pathology assessment, management plan' id='activity9'/>
    </branch>
    <branch id='branch8'>
      <activity role='clinical team' action='assess risk of developing venous thromboembolism' objects='venous thromboembolism risk assessment' id='activity12'/>
    </branch>
    <branch id='branch9'>
      <activity role='nursing staff' action='assess nutrition and hydration status within the first 48 hours of admission using a valid screening tool' objects='nutrition screening tool, hydration screening tool' id='activity13'/>
    </branch>
  </parallelGateway>
  <activity role='clinical team' action='complete AlphaFIM on or by day 3 after admission with target day 3 and admission day as day 1' objects='AlphaFIM assessment' id='activity11'/>
  <exclusiveGateway id='exclusivegateway2'>
    <branch condition='24 to 48 hours from stroke onset and no contraindication to mobilization' id='branch11'>
      <activity role='stroke unit team' action='mobilize patient early between 24 hours and 48 hours of stroke onset' objects='early mobilization plan' id='activity15'/>
    </branch>
    <branch condition='within 24 hours from stroke onset' id='branch12'>
      <activity role='clinician' action='avoid frequent out-of-bed activity within 24 hours of stroke onset and use clinical judgement for any very early mobilization' objects='mobilization precautions' id='activity16'/>
    </branch>
    <branch condition='mobilization is contraindicated' id='branch13'>
      <activity role='clinician' action='withhold mobilization and document contraindications' objects='contraindications documentation' id='activity17'/>
    </branch>
  </exclusiveGateway>
  <parallelGateway id='parallelgateway2'>
    <branch id='branch18'>
      <activity role='clinical team' action='encourage early mobilization as appropriate' objects='mobilization support' id='activity18_1'/>
    </branch>
    <branch id='branch19'>
      <activity role='clinical team' action='maintain adequate hydration to reduce venous thromboembolism risk' objects='hydration plan' id='activity18_2'/>
    </branch>
  </parallelGateway>
  <exclusiveGateway id='exclusivegateway3'>
    <branch condition='nutritional concerns or hydration deficits or dysphagia or other comorbidities identified' id='branch14'>
      <activity role='clinical team' action='refer patient to a dietitian within 7 days of admission' objects='dietitian referral' id='activity19_1'/>
      <exclusiveGateway id='exclusivegateway4'>
        <branch condition='patient is unable to meet nutritional and fluid requirements' id='branch20'>
          <activity role='dietitian and clinical team' action='consider enteral nutrition support' objects='enteral nutrition support plan' id='activity19_2'/>
        </branch>
        <branch condition='patient is able to meet nutritional and fluid requirements' id='branch21'>
          <activity role='clinical team' action='continue oral nutrition and hydration plan with monitoring' objects='oral intake plan, monitoring plan' id='activity19_3'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='no nutritional concerns or hydration deficits or dysphagia or relevant comorbidities identified' id='branch15'>
      <activity role='clinical team' action='continue routine nutrition and hydration monitoring' objects='ongoing nutrition and hydration monitoring plan' id='activity20'/>
    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='exclusivegateway5'>
    <branch condition='patient is admitted to in-hospital rehabilitation' id='branch16'>
      <activity role='stroke team' action='provide appropriate cross-continuum secondary prevention assessments and therapies' objects='Modules 5 and 10 prevention assessments and therapies' id='activity21'/>
    </branch>
    <branch condition='patient is discharged from acute care' id='branch17'>
      <activity role='stroke team' action='provide appropriate cross-continuum secondary prevention assessments and therapies' objects='Modules 5 and 10 prevention assessments and therapies' id='activity22'/>
    </branch>
  </exclusiveGateway>
</process>
