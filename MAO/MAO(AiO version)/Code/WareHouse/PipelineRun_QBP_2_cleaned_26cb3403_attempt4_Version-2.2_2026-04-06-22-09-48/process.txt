<process>
  <activity role='ED medical staff' action='Obtain patient history' objects='patient' id='activity1-1'/>
  <activity role='ED medical staff' action='Perform physical examination' objects='patient' id='activity1-2'/>
  <activity role='ED medical staff' action='Measure blood glucose' objects='patient' id='activity2-1'/>
  <exclusiveGateway id='exclusivegateway1-1'>
    <branch condition='Hypoglycemia detected' id='branch1-1'>
      <activity role='ED medical staff' action='Correct hypoglycemia immediately' objects='patient' id='activity2-2'/>
    </branch>
    <branch condition='Blood glucose > 11 mmol/L' id='branch1-2'>
      <activity role='ED medical staff' action='Repeat blood glucose measurement' objects='patient' id='activity2-3'/>
    </branch>
    <branch condition='Blood glucose normal' id='branch1-3'>
      <activity role='ED medical staff' action='Continue standard care' objects='patient' id='activity2-4'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED medical staff' action='Evaluate body temperature as part of routine vital signs every 4 hours for first 48 hours' objects='patient' id='activity3-1'/>
  <exclusiveGateway id='exclusivegateway2-1'>
    <branch condition='Temperature > 37.5°C' id='branch2-1'>
      <parallelGateway id='parallelgateway1-1'>
        <branch id='branch2-2'>
          <activity role='ED medical staff' action='Initiate temperature-reducing measures' objects='patient' id='activity3-2'/>
        </branch>
        <branch id='branch2-3'>
          <activity role='ED medical staff' action='Order infection-related laboratory tests' objects='patient' id='activity3-3'/>
          <activity role='ED medical staff' action='Assess laboratory results' objects='patient' id='activity3-4'/>
          <exclusiveGateway id='exclusivegateway2-2'>
            <branch condition='Antipyretic and/or antimicrobial therapy required' id='branch2-4'>
              <activity role='ED medical staff' action='Initiate antipyretic and antimicrobial therapy as required' objects='patient' id='activity3-5'/>
            </branch>
            <branch condition='Antipyretic and antimicrobial therapy not required' id='branch2-5'>
              <activity role='ED medical staff' action='Continue monitoring and supportive care' objects='patient' id='activity3-6'/>
            </branch>
          </exclusiveGateway>
        </branch>
      </parallelGateway>
    </branch>
    <branch condition='Temperature ≤ 37.5°C' id='branch2-6'>
      <activity role='ED medical staff' action='Continue standard monitoring' objects='patient' id='activity3-7'/>
    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='exclusivegateway3-1'>
    <branch condition='TIA with high-risk features OR minor stroke' id='branch3-1'>
      <activity role='ED medical staff' action='Order brain CT scan' objects='patient' id='activity5-1'/>
      <activity role='ED medical staff' action='Review CT results and assess for contraindications' objects='patient' id='activity5-2'/>
      <activity role='ED medical staff' action='Initiate antiplatelet therapy' objects='patient' id='activity5-3'/>
      <exclusiveGateway id='exclusivegateway4-1'>
        <branch condition='Immediate care available at specialized TIA/minor stroke clinic' id='branch4-1'>
          <activity role='ED medical staff' action='Refer patient to specialized TIA/minor stroke clinic for rapid evaluation and treatment' objects='patient' id='activity5-4'/>
        </branch>
        <branch condition='Delay to specialized clinic poses risk' id='branch4-2'>
          <activity role='ED medical staff' action='Perform evaluation in ED' objects='patient' id='activity5-5'/>
          <activity role='ED medical staff' action='Consider inpatient admission for stroke prevention' objects='patient' id='activity5-6'/>
        </branch>
        <branch condition='Medical attention sought after 48 hours from symptom onset' id='branch4-3'>
          <activity role='ED medical staff' action='Refer for evaluation at specialized TIA/minor stroke clinic or outpatient clinic with stroke-prevention services within 24 hours of initial presentation' objects='patient' id='activity5-7'/>
        </branch>
      </exclusiveGateway>
      <activity role='ED medical staff' action='Provide cross-continuum secondary prevention assessments and therapies (per Modules 5 and 10)' objects='patient' id='activity4-1'/>
    </branch>
    <branch condition='TIA without high-risk features' id='branch3-2'>
      <activity role='ED medical staff' action='Order brain CT scan' objects='patient' id='activity6-1'/>
      <activity role='ED medical staff' action='Review CT results and assess for contraindications' objects='patient' id='activity6-2'/>
      <activity role='ED medical staff' action='Initiate antiplatelet therapy' objects='patient' id='activity6-3'/>
      <activity role='ED medical staff' action='Refer to outpatient clinic with stroke-prevention services for comprehensive evaluation and management within 1 month of symptom onset' objects='patient' id='activity6-4'/>
      <activity role='ED medical staff' action='Provide cross-continuum secondary prevention assessments and therapies (per Modules 5 and 10)' objects='patient' id='activity4-2'/>
    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='exclusivegateway5-1'>
    <branch condition='TIA patient with fluctuating/crescendo hemibody motor weakness, sensory loss, or speech difficulty' id='branch5-1'>
      <activity role='ED medical staff' action='Evaluate in ED for consideration of hospital admission' objects='patient' id='activity7-1'/>
    </branch>
    <branch condition='Other TIA/minor stroke patients' id='branch5-2'>
      <activity role='ED medical staff' action='Continue with standard outpatient referral' objects='patient' id='activity7-2'/>
    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='exclusivegateway6-1'>
    <branch condition='Patient not on antiplatelet agent at presentation' id='branch6-1'>
      <exclusiveGateway id='exclusivegateway7-1'>
        <branch condition='Brain imaging excludes intracranial hemorrhage' id='branch7-1'>
          <inclusiveGateway id='inclusivegateway1-1'>
            <branch condition='ASA selected' id='branch8-1'>
              <activity role='ED medical staff' action='Administer ASA 160 mg loading dose, then enteric coated ASA (81–325 mg) daily (maintenance dose 81 mg/day for most)' objects='patient' id='activity8-1'/>
            </branch>
            <branch condition='Clopidogrel selected' id='branch8-2'>
              <activity role='ED medical staff' action='Administer clopidogrel 300 mg loading dose, then 75 mg/day' objects='patient' id='activity8-2'/>
            </branch>
            <branch condition='Extended-release dipyridamole/ASA selected' id='branch8-3'>
              <activity role='ED medical staff' action='Administer extended-release dipyridamole 200 mg/ASA 25 mg bid (load with ASA 160 mg first)' objects='patient' id='activity8-3'/>
            </branch>
          </inclusiveGateway>
        </branch>
        <branch condition='Brain imaging shows intracranial hemorrhage' id='branch7-2'>
          <activity role='ED medical staff' action='Do not initiate antiplatelet therapy; manage per hemorrhagic stroke protocol' objects='patient' id='activity8-4'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='Patient already on antiplatelet agent' id='branch6-2'>
      <activity role='ED medical staff' action='Continue current antiplatelet therapy as appropriate' objects='patient' id='activity9-1'/>
    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='exclusivegateway8-1'>
    <branch condition='Patient fails swallowing screening' id='branch9-1'>
      <activity role='ED medical staff' action='Offer rectal or gastric ASA as necessary' objects='patient' id='activity10-1'/>
    </branch>
    <branch condition='Patient passes swallowing screening' id='branch9-2'>
      <activity role='ED medical staff' action='Administer oral antiplatelet therapy as indicated' objects='patient' id='activity10-2'/>
    </branch>
  </exclusiveGateway>
</process>
