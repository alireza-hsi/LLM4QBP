<process>
  <activity role='ED clinician' action='assess patient with symptoms of community-acquired pneumonia (CAP)' objects='history, physical exam, vital signs' id='activity1'/>
  <activity role='ED clinician' action='confirm CAP symptom presentation and risk factors' objects='presenting symptoms, comorbidities, exposure history' id='activity1_0'/>
  <activity role='ED clinician' action='screen for sepsis and calculate severity scores' objects='vital signs, SIRS elements, CRB-65 elements' id='activity1_2'/>
  <exclusiveGateway id='exclusivegateway1_1'>
    <branch condition='Immediate airway/breathing/circulation instability' id='branch1_1'>
      <activity role='ED clinician' action='initiate immediate stabilization and escalate level of care' objects='resuscitation measures, ICU consult if needed' id='activity1_3a'/>
    </branch>
    <branch condition='No immediate instability' id='branch1_2'>
      <activity role='ED clinician' action='proceed with diagnostic testing pathway' objects='initial assessment complete' id='activity1_3b'/>
    </branch>
  </exclusiveGateway>

  <activity role='ED clinician' action='order chest x-ray to confirm diagnosis (recommended for all ED CAP-symptom patients)' objects='chest x-ray order' id='activity2'/>
  <exclusiveGateway id='exclusivegateway2_1'>
    <branch condition='Portable CXR needed' id='branch2_1'>
      <activity role='radiology technologist' action='perform portable chest x-ray' objects='portable chest x-ray images' id='activity2_1a'/>
    </branch>
    <branch condition='Standard CXR feasible' id='branch2_2'>
      <activity role='radiology technologist' action='perform standard chest x-ray' objects='PA/lateral chest x-ray images' id='activity2_1b'/>
    </branch>
  </exclusiveGateway>
  <activity role='radiologist' action='interpret chest x-ray and report findings' objects='chest x-ray report' id='activity2_2'/>

  <activity role='ED clinician' action='do not order CT scan (not recommended)' objects='CT scan' id='activity3'/>

  <exclusiveGateway id='exclusivegateway1'>
    <branch condition='Sepsis absent' id='branch1'>
      <activity role='ED clinician' action='wait for chest x-ray results prior to giving antibiotics' objects='chest x-ray result' id='activity4'/>
    </branch>
    <branch condition='Sepsis present' id='branch2'>
      <activity role='ED clinician' action='initiate antibiotics without waiting for chest x-ray results' objects='empiric antibiotics' id='activity5'/>
    </branch>
  </exclusiveGateway>

  <exclusiveGateway id='exclusivegateway2'>
    <branch condition='Admit to hospital (ward or ICU)' id='branch3'>
      <parallelGateway id='parallelgateway1'>
        <branch id='branch4'>
          <activity role='nursing staff' action='measure oxygen saturation' objects='pulse oximetry' id='activity6'/>
        </branch>
        <branch id='branch5'>
          <activity role='clinician' action='order routine blood work' objects='CBC, electrolytes, renal function tests' id='activity7'/>
          <parallelGateway id='parallelgateway1_1'>
            <branch id='branch5_1'>
              <activity role='nursing staff' action='collect blood sample for CBC' objects='blood sample (CBC tube)' id='activity7_1'/>
            </branch>
            <branch id='branch5_2'>
              <activity role='nursing staff' action='collect blood sample for electrolytes and renal function tests' objects='blood sample (chemistry tube)' id='activity7_2'/>
            </branch>
          </parallelGateway>
        </branch>
      </parallelGateway>

      <exclusiveGateway id='exclusivegateway3'>
        <branch condition='CRB-65 criteria count > 2' id='branch6'>
          <activity role='clinician' action='consider additional tests for higher risk' objects='blood gases, lactate, liver enzyme and liver function tests' id='activity8'/>
        </branch>
        <branch condition='CRB-65 criteria count ≤ 2' id='branch7'>
          <activity role='clinician' action='do not add CRB-65-triggered tests' objects='blood gases, lactate, liver enzymes/liver function tests' id='activity9'/>
        </branch>
      </exclusiveGateway>

      <exclusiveGateway id='exclusivegateway4'>
        <branch condition='Meets > 2 SIRS criteria OR requires admission to the ICU' id='branch8'>
          <activity role='clinician' action='obtain blood cultures (recommended)' objects='blood culture set(s)' id='activity10'/>
          <activity role='nursing staff' action='collect blood cultures prior to antibiotics when feasible' objects='two sets from separate venipuncture sites' id='activity10_1'/>
          <activity role='laboratory' action='label, transport, and process blood cultures' objects='blood culture bottles, requisition' id='activity10_2'/>
        </branch>
        <branch condition='Does not meet > 2 SIRS criteria AND does not require ICU admission' id='branch9'>
          <activity role='clinician' action='do not obtain blood cultures' objects='blood cultures' id='activity11'/>
        </branch>
      </exclusiveGateway>

      <inclusiveGateway id='inclusivegateway1'>
        <branch condition='Sputum sample can be obtained adequately (consider for inpatients)' id='branch10'>
          <activity role='nursing staff/respiratory therapist' action='collect sputum culture' objects='adequate sputum sample' id='activity12'/>
          <activity role='nursing staff/respiratory therapist' action='instruct patient and attempt sputum collection' objects='collection container, patient instructions' id='activity12_1'/>
          <activity role='laboratory' action='assess sputum sample adequacy and accept/reject for culture' objects='sputum specimen (quality check)' id='activity12_2'/>
          <exclusiveGateway id='exclusivegateway5'>
            <branch condition='Sputum culture obtained' id='branch11'>
              <activity role='laboratory' action='consider Gram stain' objects='sputum sample' id='activity13'/>
            </branch>
            <branch condition='Sputum culture not obtained' id='branch12'>
              <activity role='clinician' action='skip Gram stain' objects='sputum Gram stain' id='activity14'/>
            </branch>
          </exclusiveGateway>
        </branch>
        <branch condition='Using targeted antibiotic instead of non-targeted antibiotic (sputum culture more important)' id='branch13'>
          <activity role='clinician' action='prioritize obtaining sputum culture before/with targeted therapy when feasible' objects='sputum culture order, targeted antibiotic plan' id='activity15'/>
        </branch>
      </inclusiveGateway>

      <inclusiveGateway id='inclusivegateway2'>
        <branch condition='High-severity patient' id='branch14'>
          <activity role='clinician' action='order urine antigen testing (recommended)' objects='urine antigen test (e.g., Legionella)' id='activity16'/>
          <exclusiveGateway id='exclusivegateway16_1'>
            <branch condition='Urine specimen available now' id='branch16a_1'>
              <activity role='nursing staff' action='collect urine specimen and send to lab' objects='urine sample, requisition' id='activity16_1'/>
            </branch>
            <branch condition='Urine specimen not immediately available' id='branch16a_2'>
              <activity role='nursing staff' action='arrange urine collection and send when available' objects='urine collection plan' id='activity16_2'/>
            </branch>
          </exclusiveGateway>
        </branch>
        <branch condition='Enhanced surveillance directive from Public Health Ontario on Legionella in inpatients' id='branch15'>
          <activity role='clinician' action='order urine antigen testing per enhanced surveillance directive (recommended)' objects='urine antigen test' id='activity17'/>
        </branch>
        <branch condition='Not responding to drug therapy after 48–72 hours' id='branch16'>
          <activity role='clinician' action='order urine antigen testing after 48–72 hours of non-response (recommended)' objects='urine antigen test' id='activity18'/>
        </branch>
        <branch condition='Inpatient during peak season (mid-June to early October)' id='branch17'>
          <activity role='clinician' action='order urine antigen testing during peak season (mid-June to early October) (recommended)' objects='urine antigen test' id='activity19'/>
        </branch>
      </inclusiveGateway>

      <exclusiveGateway id='exclusivegateway6'>
        <branch condition='Intubated patient (e.g., high severity)' id='branch18'>
          <activity role='clinician' action='consider PCR testing for Legionella' objects='Legionella PCR test' id='activity20'/>
          <activity role='respiratory therapist' action='collect lower respiratory specimen (ET aspirate/BAL) for PCR when indicated' objects='lower respiratory specimen' id='activity20_1'/>
          <activity role='laboratory' action='run Legionella PCR and report result' objects='PCR result report' id='activity20_2'/>
        </branch>
        <branch condition='Not intubated' id='branch19'>
          <activity role='clinician' action='do not perform Legionella PCR based on intubation criterion' objects='Legionella PCR test' id='activity21'/>
        </branch>
      </exclusiveGateway>

      <activity role='clinical leadership/quality improvement team' action='conduct field evaluation to assess clinical utility and optimal use of urine antigen testing in CAP' objects='urine antigen test ordering/impact data' id='activity22'/>
      <activity role='clinical leadership/quality improvement team' action='define evaluation protocol and measures (utilization, yield, impact on management)' objects='evaluation plan, metrics' id='activity22_1'/>
      <activity role='clinical leadership/quality improvement team' action='collect and analyze ordering patterns and outcomes' objects='EHR extracts, lab results, treatment changes' id='activity22_2'/>
      <activity role='clinical leadership/quality improvement team' action='report findings and update local guidance if needed' objects='evaluation report, updated ordering guidance' id='activity22_3'/>
    </branch>

    <branch condition='Not admitted (ED discharge/alternative pathway)' id='branch20'>
      <activity role='ED clinician' action='document diagnostic testing completed and disposition decision' objects='chest x-ray result, clinical assessment, discharge plan' id='activity23'/>
    </branch>
  </exclusiveGateway>
</process>
