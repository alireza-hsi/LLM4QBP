<process>
  <activity role='ED clinician' action='identify patient with TIA or minor (nondisabling) stroke and activate standardized priority protocol for rapid treatment' objects='suspected stroke patient' id='activity1'/>
  <parallelGateway id='parallelgateway1'>
    <branch id='branch1'>
      <activity role='ED nurse' action='measure random blood glucose at presentation' objects='blood glucose value' id='activity2'/>
      <exclusiveGateway id='exclusivegateway1'>
        <branch condition='hypoglycemia present' id='branch2'>
          <activity role='ED clinician' action='correct hypoglycemia immediately' objects='hypoglycemia' id='activity3'/>
        </branch>
        <branch condition='no hypoglycemia' id='branch3'>
          <activity role='ED clinician' action='continue glucose management' objects='patient' id='activity4'/>
        </branch>
      </exclusiveGateway>
      <exclusiveGateway id='exclusivegateway2'>
        <branch condition='first random glucose value > 11 mmol/L' id='branch4'>
          <activity role='ED nurse' action='repeat blood glucose measurement' objects='blood glucose value' id='activity5'/>
        </branch>
        <branch condition='first random glucose value <= 11 mmol/L' id='branch5'>
          <activity role='ED clinician' action='document no repeat blood glucose required' objects='blood glucose result' id='activity6'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch id='branch6'>
      <activity role='ED nurse' action='evaluate temperature as part of routine vital signs every 4 hours for first 48 hours' objects='temperature and vital signs' id='activity7'/>
      <exclusiveGateway id='exclusivegateway3'>
        <branch condition='temperature > 37.5 C' id='branch7'>
          <parallelGateway id='parallelgateway2'>
            <branch id='branch8'>
              <activity role='ED team' action='initiate temperature-reducing measures' objects='elevated temperature' id='activity8'/>
            </branch>
            <branch id='branch9'>
              <activity role='ED clinician' action='investigate potential infection' objects='possible infection' id='activity9'/>
            </branch>
            <branch id='branch10'>
              <activity role='ED clinician' action='initiate antipyretic therapy and antimicrobial therapy as required' objects='antipyretics and antimicrobials' id='activity10'/>
            </branch>
          </parallelGateway>
        </branch>
        <branch condition='temperature <= 37.5 C' id='branch11'>
          <activity role='ED nurse' action='continue routine temperature monitoring every 4 hours for first 48 hours' objects='temperature observations' id='activity11'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch id='branch12'>
      <activity role='stroke team' action='perform cross-continuum secondary prevention assessment' objects='secondary prevention needs for admitted or discharged patient' id='activity12_1'/>
      <activity role='stroke team' action='initiate or arrange appropriate secondary prevention therapies' objects='Modules 5 and 10 prevention therapies and follow-up plan' id='activity12_2'/>
    </branch>
    <branch id='branch13'>
      <activity role='ED clinician' action='assess symptom pattern and determine whether high-risk TIA features or minor stroke are present' objects='neurological symptoms and risk features' id='activity13_1'/>
      <activity role='ED clinician' action='assess time from symptom onset, current antiplatelet use, contraindications, and swallowing status' objects='timing, medication history, contraindications, and swallowing screening status' id='activity13_2'/>
      <exclusiveGateway id='exclusivegateway6'>
        <branch condition='fluctuating or crescendo hemibody motor weakness, sensory loss, or speech difficulty present' id='branch19'>
          <activity role='ED clinician' action='evaluate severity and fluctuation of high-risk TIA symptoms in the ED' objects='high-risk TIA symptom pattern' id='activity23_1'/>
          <activity role='ED clinician' action='decide whether to admit the patient to hospital for inpatient stroke-prevention management' objects='admission decision' id='activity23_2'/>
        </branch>
        <branch condition='fluctuating or crescendo high-risk symptoms absent' id='branch20'>
          <exclusiveGateway id='exclusivegateway4'>
            <branch condition='TIA with high-risk features or minor stroke' id='branch14'>
              <exclusiveGateway id='exclusivegateway10'>
                <branch condition='medical attention sought after 48 hours from symptom onset' id='branch28'>
                  <activity role='ED clinician' action='refer patient within 24 hours of initial presentation to a specialized TIA/minor stroke clinic or outpatient clinic with stroke-prevention services' objects='urgent referral' id='activity18'/>
                </branch>
                <branch condition='medical attention sought within 48 hours from symptom onset' id='branch29'>
                  <activity role='ED clinician' action='arrange brain CT scan as soon as possible and no later than 24 hours after symptom onset' objects='brain CT scan' id='activity14'/>
                  <exclusiveGateway id='exclusivegateway5'>
                    <branch condition='specialized TIA/minor stroke clinic immediately accessible without harmful delay' id='branch15'>
                      <activity role='specialized TIA/minor stroke clinic team' action='provide immediate evaluation and initiate antiplatelet therapy if not contraindicated' objects='brain CT result and antiplatelet therapy' id='activity15'/>
                    </branch>
                    <branch condition='delay to specialized TIA/minor stroke clinic poses risk to patient health' id='branch16'>
                      <activity role='appropriately resourced ED team' action='perform urgent ED-based diagnostic evaluation' objects='brain CT result and stroke-prevention workup' id='activity16_1'/>
                      <activity role='appropriately resourced ED team' action='initiate antiplatelet therapy if not contraindicated' objects='antiplatelet therapy' id='activity16_2'/>
                      <activity role='ED clinician' action='consider inpatient evaluation and management for stroke prevention' objects='admission decision' id='activity17'/>
                    </branch>
                  </exclusiveGateway>
                  <activity role='stroke clinician' action='tailor other stroke-prevention treatments to the patient after urgent evaluation' objects='individualized stroke-prevention treatment plan' id='activity19'/>
                </branch>
              </exclusiveGateway>
            </branch>
            <branch condition='TIA without high-risk features' id='branch18'>
              <activity role='ED clinician' action='arrange brain CT scan as soon as possible and no later than 24 hours after initial presentation' objects='brain CT scan' id='activity20'/>
              <activity role='ED clinician' action='initiate antiplatelet therapy if not contraindicated' objects='antiplatelet therapy' id='activity20_1'/>
              <activity role='ED clinician' action='refer patient to an outpatient clinic with stroke-prevention services for comprehensive evaluation and management within 1 month of symptom onset' objects='outpatient stroke-prevention referral' id='activity22'/>
            </branch>
          </exclusiveGateway>
        </branch>
      </exclusiveGateway>
      <exclusiveGateway id='exclusivegateway7'>
        <branch condition='not on antiplatelet agent at presentation and brain imaging excludes intracranial hemorrhage' id='branch21'>
          <exclusiveGateway id='exclusivegateway9'>
            <branch condition='ASA selected' id='branch22'>
              <activity role='ED clinician' action='start ASA 160 mg loading dose followed by enteric coated ASA 81-325 mg daily, with most patients maintained on 81 mg/day' objects='ASA regimen' id='activity25'/>
            </branch>
            <branch condition='clopidogrel selected' id='branch23'>
              <activity role='ED clinician' action='start clopidogrel 300 mg loading dose followed by 75 mg/day' objects='clopidogrel regimen' id='activity26'/>
            </branch>
            <branch condition='extended-release dipyridamole plus ASA selected' id='branch24'>
              <activity role='ED clinician' action='load with ASA 160 mg first and then start extended-release dipyridamole 200 mg/ASA 25 mg twice daily' objects='dipyridamole plus ASA regimen' id='activity27'/>
            </branch>
          </exclusiveGateway>
        </branch>
        <branch condition='already on antiplatelet agent at presentation or brain imaging has not excluded intracranial hemorrhage' id='branch25'>
          <activity role='ED clinician' action='withhold immediate new antiplatelet initiation and continue management according to imaging result and current therapy review' objects='current antiplatelet status and brain imaging result' id='activity28'/>
        </branch>
      </exclusiveGateway>
      <exclusiveGateway id='exclusivegateway8'>
        <branch condition='patient fails swallowing screening or cannot take oral ASA' id='branch26'>
          <activity role='ED clinician' action='offer rectal or gastric ASA as necessary' objects='rectal ASA or gastric ASA' id='activity29'/>
        </branch>
        <branch condition='patient passes swallowing screening and can take oral ASA if indicated' id='branch27'>
          <activity role='ED clinician' action='administer oral antiplatelet therapy as indicated' objects='oral antiplatelet therapy' id='activity30'/>
        </branch>
      </exclusiveGateway>
    </branch>
  </parallelGateway>
</process>
