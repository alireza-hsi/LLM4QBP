    <branch id='branch18'>
      <activity role='ED physician/respiratory therapy' action='evaluate respiratory status for high-risk markers (distress, hypoxemia, pulmonary edema severity)' objects='pulse oximetry, ABG if available, chest x-ray' id='activity14_1'/>
    </branch>
    <branch id='branch19'>
      <activity role='ED physician' action='evaluate hemodynamics and rhythm for high-risk markers (hypotension/shock, significant arrhythmias)' objects='blood pressure trends, ECG/telemetry' id='activity14_2'/>

      <activity role='ED physician' action='review troponin/lab/imaging results for high-risk markers and end-organ impact' objects='troponin, creatinine/electrolytes, CBC, imaging reports' id='activity14_3'/>

    <branch condition='requires noninvasive or invasive ventilation OR needs ICU-level care with higher nurse-to-patient staffing OR high-risk markers present suggesting high intensity' id='branch13'>
