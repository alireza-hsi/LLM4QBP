<process>
  <activity role='patient' action='present to the ED, outpatient clinic, or direct admission pathway' objects='suspected COPD exacerbation' id='activity1'/>
  <parallelGateway id='parallelgateway1'>
    <branch id='branch1'>
      <activity role='nurse or clinician' action='assess for hypoventilation' objects='ventilatory status' id='activity2_1'/>
    </branch>
    <branch id='branch2'>
      <activity role='nurse or clinician' action='check level of consciousness or cognition' objects='mental status' id='activity2_2'/>
    </branch>
    <branch id='branch3'>
      <activity role='nurse or clinician' action='check blood-oxygen saturation via pulse oximetry' objects='oxygen saturation reading' id='activity2_3'/>
    </branch>
    <branch id='branch4'>
      <activity role='nurse or clinician' action='assess whether the patient has purulent sputum' objects='sputum characteristics' id='activity2_4'/>
    </branch>
  </parallelGateway>
  <exclusiveGateway id='exclusivegateway1'>
    <branch condition='Immediate resuscitation is required on initial presentation at the ED' id='branch5'>
      <activity role='ED team' action='trial immediate resuscitation' objects='resuscitation interventions' id='activity7'/>
      <activity role='ED clinician' action='re-evaluate need for admission after resuscitation' objects='patient response to resuscitation' id='activity8'/>
    </branch>
    <branch condition='Immediate resuscitation is not required or the patient is not presenting through the ED' id='branch6'>
      <activity role='clinician' action='continue diagnostic assessment without immediate resuscitation' objects='patient status' id='activity9'/>
    </branch>
  </exclusiveGateway>
  <activity role='clinician' action='identify patient wishes with respect to goals of care and limitations of treatment' objects='code status and goals of care' id='activity6'/>
  <parallelGateway id='parallelgateway1_1'>
    <branch id='branch7'>
      <activity role='clinician' action='perform a physical examination' objects='patient' id='activity3'/>
    </branch>
    <branch id='branch8'>
      <activity role='clinician' action='check patient history' objects='medical history and exacerbation history' id='activity4'/>
    </branch>
    <branch id='branch9'>
      <activity role='clinician or pharmacist' action='document and reconcile medications currently used by the patient' objects='current medication list' id='activity5'/>
    </branch>
  </parallelGateway>
  <parallelGateway id='parallelgateway2'>
    <branch id='branch11'>
      <exclusiveGateway id='exclusivegateway2_1'>
        <branch condition='The patient is well enough to leave the emergency department for standard imaging' id='branch12'>
          <activity role='clinician or radiology staff' action='perform a chest x-ray' objects='posteroanterior and lateral views' id='activity10_1'/>
        </branch>
        <branch condition='The patient is too unwell to leave the emergency department' id='branch13'>
          <activity role='clinician or radiology staff' action='perform a portable chest x-ray' objects='portable chest x-ray imaging' id='activity10_2'/>
        </branch>
      </exclusiveGateway>
      <exclusiveGateway id='exclusivegateway2_2'>
        <branch condition='There is concern for pneumothorax' id='branch14'>
          <activity role='clinician or radiology staff' action='perform an expiratory chest x-ray view' objects='expiratory chest x-ray image' id='activity10_3'/>
        </branch>
        <branch condition='There is no concern for pneumothorax' id='branch15'>
          <activity role='clinician' action='continue without an additional expiratory chest x-ray view' objects='chest x-ray assessment' id='activity10_4'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch id='branch16'>
      <activity role='clinician or laboratory staff' action='conduct baseline blood work' objects='complete blood count, electrolytes, creatinine, blood urea nitrogen if available' id='activity11'/>
    </branch>
    <branch id='branch17'>
      <activity role='clinician' action='perform an electrocardiogram' objects='assessment for arrhythmias, myocardial ischemia, right ventricular strain, and related cardiac findings' id='activity12'/>
    </branch>
  </parallelGateway>
  <inclusiveGateway id='inclusivegateway1'>
    <branch condition='Low oxygen saturation on oximetry and/or acute respiratory failure is suspected' id='branch18'>
      <activity role='clinician' action='check arterial blood gases where appropriate' objects='arterial blood gas sample' id='activity13'/>
    </branch>
    <branch condition='Pneumonia or suspected sepsis is present' id='branch19'>
      <activity role='clinician or laboratory staff' action='draw blood cultures' objects='blood culture specimens' id='activity14'/>
    </branch>
    <branch condition='Cardiac disorder is suspected' id='branch20'>
      <activity role='clinician or laboratory staff' action='examine cardiac markers' objects='cardiac marker tests' id='activity15'/>
    </branch>
    <branch condition='Other suspected diagnoses or comorbidities need to be identified and/or ruled out' id='branch21'>
      <activity role='clinician' action='perform other diagnostic interventions as appropriate' objects='tests for alternative diagnoses or comorbidities' id='activity16'/>
    </branch>
  </inclusiveGateway>
  <activity role='clinician' action='defer spirometry during the initial assessment of the exacerbation' objects='spirometry' id='activity18_0'/>
  <parallelGateway id='parallelgateway3'>
    <branch id='branch22'>
      <activity role='clinician' action='assess response to initial medical management and respiratory severity' objects='failure to respond, breathlessness, resting dyspnea, cyanosis, decreased level of consciousness, and long-term oxygen therapy use' id='activity19_1'/>
    </branch>
    <branch id='branch23'>
      <activity role='clinician' action='assess oxygenation and investigation results relevant to admission' objects='arterial oxygen saturation less than 90 percent, pH level, PaO2, and changes on chest x-ray' id='activity19_2'/>
    </branch>
    <branch id='branch24'>
      <activity role='clinician' action='assess functional and social support factors' objects='insufficient home support, inability to cope at home, decreased level of activity, social circumstances, and older age' id='activity19_3'/>
    </branch>
    <branch id='branch25'>
      <activity role='clinician' action='assess comorbidities and overall clinical risk' objects='general condition, severe underlying COPD, worsening peripheral edema or new physical signs, acute confusion, rapid onset or frequent exacerbations, and significant comorbidity' id='activity19_4'/>
    </branch>
  </parallelGateway>
  <activity role='emergency room staff' action='review COPD-specific admission checklist items' objects='Module 1 practices and admission criteria' id='activity20_1'/>
  <activity role='emergency room staff' action='review non-COPD concerns on the admission checklist' objects='pain, anxiety, comorbidities, and other non-COPD concerns' id='activity20_2'/>
  <activity role='clinician' action='synthesize the level-of-care assessment and need for admission' objects='clinical judgment and local resource availability' id='activity19_5'/>
  <exclusiveGateway id='exclusivegateway3'>
    <branch condition='Admission or higher level of care is required' id='branch26'>
      <activity role='admitting clinician' action='admit the patient to the appropriate level of care' objects='hospital bed and inpatient care plan' id='activity21'/>
    </branch>
    <branch condition='Admission is not required' id='branch27'>
      <activity role='clinician' action='manage the patient through outpatient or discharge planning' objects='non-inpatient care plan and follow-up arrangements' id='activity22'/>
    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='exclusivegateway2'>
    <branch condition='The patient has stabilized after disposition and has no prior objective documentation of COPD through spirometry' id='branch28'>
      <activity role='clinician or respiratory staff' action='perform spirometry once the patient has stabilized' objects='spirometry test' id='activity17'/>
    </branch>
    <branch condition='Prior objective documentation of COPD through spirometry exists or the patient has not yet stabilized after disposition' id='branch29'>
      <activity role='clinician' action='continue to defer spirometry and reassess later if needed' objects='spirometry plan' id='activity18'/>
    </branch>
  </exclusiveGateway>
</process>
