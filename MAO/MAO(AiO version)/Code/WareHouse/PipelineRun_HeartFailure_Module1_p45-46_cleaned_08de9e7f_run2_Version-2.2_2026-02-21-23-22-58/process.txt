<process>
  <activity role='ED triage nurse' action='identify suspected acute heart failure presentation and initiate episode-of-care pathway' objects='patient, triage information' id='activity1'/>
  <activity role='ED physician' action='perform focused history and physical exam for acute heart failure and comorbidities' objects='patient history, medication list, physical exam findings' id='activity2_1'/>
  <activity role='ED physician' action='apply standardized multi-condition risk assessment/stratification tool and assign preliminary intensity level' objects='standardized risk stratification tool, assessment findings' id='activity2_2'/>
  <activity role='ED physician' action='document risk stratification results and communicate initial plan to ED team' objects='risk stratification summary, care plan' id='activity2_3'/>

  <parallelGateway id='parallelgateway1'>
    <branch id='branch1'>
      <activity role='ED nurse' action='obtain bloodwork' objects='serum creatinine and electrolyte levels' id='activity3'/>
      <activity role='laboratory' action='run assay and report results' objects='serum creatinine and electrolyte levels' id='activity4'/>
    </branch>
    <branch id='branch2'>
      <activity role='ED nurse' action='obtain bloodwork' objects='troponin measurements' id='activity5'/>
      <activity role='laboratory' action='run assay and report results' objects='troponin measurements' id='activity6'/>
    </branch>
    <branch id='branch3'>
      <activity role='ED nurse' action='obtain bloodwork' objects='complete blood count' id='activity7'/>
      <activity role='laboratory' action='run assay and report results' objects='complete blood count' id='activity8'/>
    </branch>
    <branch id='branch4'>
      <activity role='ED staff' action='perform electrocardiogram' objects='electrocardiogram' id='activity9'/>
      <activity role='ED physician' action='interpret electrocardiogram' objects='electrocardiogram results' id='activity10'/>
    </branch>
    <branch id='branch5'>
      <activity role='radiology' action='perform chest x-ray examination' objects='chest x-ray' id='activity11'/>
      <activity role='ED physician' action='review chest x-ray examination' objects='chest x-ray report/images' id='activity12'/>
    </branch>
    <branch id='branch6'>
      <activity role='ED nurse' action='initiate frequent monitoring plan for heart rate, blood pressure, and oxygen saturation and document initial readings' objects='vital signs monitor, oxygen saturation monitor, monitoring frequency' id='activity13'/>
    </branch>
    <branch id='branch7'>
      <activity role='ED physician' action='administer initial furosemide dose in ED' objects='furosemide, medication order' id='activity14_1'/>
      <activity role='ED nurse' action='monitor urine output and symptoms after diuretic administration' objects='urine output measurement, symptom assessment' id='activity14_2'/>
      <activity role='ED physician' action='reassess and document responsiveness to diuresis' objects='vital signs, urine output trend, clinical response' id='activity14_3'/>
    </branch>
  </parallelGateway>

  <exclusiveGateway id='exclusivegateway1'>
    <branch condition='No recent echocardiogram is available' id='branch8'>
      <activity role='cardiology/echo lab' action='perform echocardiogram' objects='echocardiogram' id='activity15'/>
      <activity role='ED physician' action='review echocardiogram findings' objects='echocardiogram report' id='activity16'/>
    </branch>
    <branch condition='Recent echocardiogram is available' id='branch9'>
      <activity role='ED staff' action='retrieve and review recent echocardiogram' objects='prior echocardiogram report/images' id='activity17'/>
    </branch>
  </exclusiveGateway>

  <activity role='ED physician' action='compile inputs required for acute heart failure risk score (e.g., EHMRG)' objects='EHMRG inputs, lab/imaging/vital sign data' id='activity18_1'/>
  <activity role='ED physician' action='calculate EHMRG score and determine risk category/quintile' objects='EHMRG calculator/tool, compiled inputs' id='activity18_2'/>
  <activity role='ED physician' action='document risk score and incorporate into disposition decision-making' objects='EHMRG score, disposition rationale' id='activity18_3'/>

  <parallelGateway id='parallelgateway2'>
    <branch id='branch10'>
      <activity role='ED physician' action='assess for respiratory distress' objects='respiratory assessment' id='activity19'/>
    </branch>
    <branch id='branch11'>
      <activity role='ED physician' action='assess for hypoxemia' objects='oxygen saturation readings/ABG if available' id='activity20'/>
    </branch>
    <branch id='branch12'>
      <activity role='ED physician' action='assess severity of pulmonary edema' objects='clinical exam, chest x-ray findings' id='activity21'/>
    </branch>
    <branch id='branch13'>
      <activity role='ED physician' action='assess response to furosemide administered in ED (poor response as high-risk marker)' objects='urine output, symptoms, vital signs' id='activity22'/>
    </branch>
    <branch id='branch14'>
      <activity role='ED physician' action='assess for hemodynamic compromise' objects='blood pressure, perfusion markers' id='activity23'/>
    </branch>
    <branch id='branch15'>
      <activity role='ED physician' action='assess for significant arrhythmias' objects='ECG/telemetry' id='activity24'/>
    </branch>
    <branch id='branch16'>
      <activity role='ED physician' action='assess for positive troponin' objects='troponin results' id='activity25'/>
    </branch>
  </parallelGateway>

  <exclusiveGateway id='exclusivegateway5'>
    <branch condition='Any high-risk marker present (respiratory distress OR hypoxemia OR severe pulmonary edema OR poor diuretic response OR hemodynamic compromise OR significant arrhythmias OR troponin positive)' id='branch35'>
      <inclusiveGateway id='inclusivegateway1'>
        <branch condition='respiratory distress present' id='branch17'>
          <activity role='ED physician' action='activate higher level provider/critical care support for respiratory distress' objects='consult request' id='activity26'/>
        </branch>
        <branch condition='hypoxemia present' id='branch18'>
          <activity role='ED physician' action='initiate escalation pathway for hypoxemia (consider noninvasive or invasive ventilation)' objects='oxygen/ventilation resources' id='activity27'/>
        </branch>
        <branch condition='severe pulmonary edema present' id='branch19'>
          <activity role='ED physician' action='escalate care for severe pulmonary edema and coordinate higher-acuity monitoring' objects='care escalation plan' id='activity28'/>
        </branch>
        <branch condition='poor response to furosemide in ED' id='branch20'>
          <activity role='ED physician' action='escalate to higher-acuity evaluation for diuretic nonresponse' objects='diuresis response data' id='activity29'/>
        </branch>
        <branch condition='hemodynamic compromise present' id='branch21'>
          <activity role='ED physician' action='activate critical care/cardiology for hemodynamic compromise' objects='hemodynamic data, consult request' id='activity30'/>
        </branch>
        <branch condition='significant arrhythmias present' id='branch22'>
          <activity role='ED physician' action='request urgent cardiology support for significant arrhythmias' objects='ECG/telemetry data, consult request' id='activity31'/>
        </branch>
        <branch condition='troponin positive' id='branch23'>
          <activity role='ED physician' action='activate pathway for myocardial injury/possible ischemia and involve cardiology' objects='troponin results, ECG data' id='activity32'/>
        </branch>
      </inclusiveGateway>
      <activity role='ED physician' action='flag patient as discharge-ineligible due to high-risk marker(s) and summarize findings for disposition' objects='high-risk marker summary, escalation actions taken' id='activity41'/>
      <activity role='ED physician' action='re-evaluate and update intensity level and disposition plan in light of high-risk marker(s)' objects='updated intensity level, updated disposition rationale' id='activity41_2'/>
    </branch>
    <branch condition='No high-risk markers present' id='branch36'>
      <activity role='ED physician' action='proceed to disposition decision-making' objects='EHMRG score, response to treatment, discharge considerations' id='activity42'/>
    </branch>
  </exclusiveGateway>

  <exclusiveGateway id='exclusivegateway3'>
    <branch condition='Higher level provider/facility required (transfer pathway invoked)' id='branch29'>
      <activity role='ED physician' action='determine required higher level of care and identify appropriate receiving facility' objects='transfer criteria, receiving facility options' id='activity39_1'/>
      <parallelGateway id='parallelgateway4'>
        <branch id='branch33'>
          <activity role='ED physician' action='contact receiving team and obtain acceptance' objects='receiving physician contact, clinical summary' id='activity39_2'/>
        </branch>
        <branch id='branch34'>
          <activity role='ED team' action='prepare transfer documentation and stabilize patient for transport' objects='transfer forms, imaging/lab reports, stabilization checklist' id='activity39_3'/>
        </branch>
      </parallelGateway>
      <activity role='ED staff' action='arrange transport resources and coordinate departure time' objects='transport request, ETA, handoff details' id='activity39_4'/>
      <activity role='transport team' action='transfer patient to higher level provider/facility' objects='patient, transfer documentation' id='activity40'/>
    </branch>

    <branch condition='Higher level provider/facility not required' id='branch30'>
      <parallelGateway id='parallelgateway3'>
        <branch id='branch24'>
          <activity role='ED physician' action='confirm patient has responded to initial treatment in the ED (required for discharge consideration)' objects='clinical response, vital signs, diuresis response' id='activity33'/>
        </branch>
        <branch id='branch25'>
          <activity role='ED physician' action='assess other discharge considerations (e.g., advanced-directives, severe dementia, estimated impact of admission on life-expectancy)' objects='goals-of-care documentation, cognitive status, prognosis factors' id='activity34'/>
        </branch>
      </parallelGateway>

      <exclusiveGateway id='exclusivegateway2'>
        <branch condition='Low intensity: EHMRG quintiles 1–2 AND responded to initial ED treatment AND no other considerations AND no high-risk markers present' id='branch26'>
          <activity role='ED physician' action='finalize outpatient treatment plan and prescriptions' objects='medication reconciliation, prescriptions, outpatient plan' id='activity35_1'/>
          <activity role='ED nurse' action='provide patient/caregiver education and return-precautions' objects='discharge instructions, warning signs, self-monitoring advice' id='activity35_2'/>
          <activity role='ED physician' action='complete discharge documentation for home discharge' objects='discharge summary, risk stratification results' id='activity35_3'/>
          <activity role='ED staff' action='arrange follow-up visit with primary care provider within days of ED discharge' objects='appointment request, discharge summary' id='activity36'/>
        </branch>
        <branch condition='Average intensity: requires inpatient admission with normal nurse-to-patient staffing' id='branch27'>
          <activity role='ED physician' action='request inpatient bed and notify admitting service' objects='bed request, admission notification' id='activity37_1'/>
          <activity role='ED physician' action='write admission orders (monitoring, diuresis plan, repeat labs as needed)' objects='admission orderset, medication orders, lab orders' id='activity37_2'/>
          <activity role='ED physician' action='handover to inpatient team and receiving nurse' objects='handover summary, investigation results, risk score' id='activity37_3'/>
        </branch>
        <branch condition='High intensity: requires ventilation (noninvasive or invasive) and/or ICU admission with higher nurse-to-patient staffing' id='branch28'>
          <activity role='ED physician/critical care' action='prepare for ventilatory support and airway management as indicated' objects='airway equipment, monitoring, sedation plan if needed' id='activity38_1'/>
          <exclusiveGateway id='exclusivegateway4'>
            <branch condition='Noninvasive ventilation appropriate' id='branch31'>
              <activity role='ED physician/critical care' action='initiate noninvasive ventilation' objects='NIV device, mask/interface settings' id='activity38_2'/>
            </branch>
            <branch condition='Invasive ventilation required' id='branch32'>
              <activity role='ED physician/critical care' action='perform endotracheal intubation and initiate mechanical ventilation' objects='intubation kit, ventilator, RSI medications' id='activity38_3'/>
            </branch>
          </exclusiveGateway>
          <activity role='ED physician/critical care' action='request ICU bed and admit to ICU with higher nurse-to-patient staffing' objects='ICU bed request, ICU admission orders, handover' id='activity38_4'/>
        </branch>
      </exclusiveGateway>
    </branch>
  </exclusiveGateway>
</process>
