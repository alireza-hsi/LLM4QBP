<process>
  <activity role='ED medical staff' action='Initiate early treatment protocol for TIA or minor (nondisabling) stroke' objects='patient presenting to ED' id='activity1'/>
  <activity role='ED medical staff' action='Assess and manage blood glucose; correct hypoglycemia immediately' objects='blood glucose measurement' id='activity2'/>
  <exclusiveGateway id='exclusivegateway1'>
    <branch condition='First random glucose value > 11 mmol/L' id='branch1'>
      <activity role='ED medical staff' action='Repeat blood glucose measurement' objects='blood glucose meter' id='activity3'/>
    </branch>
    <branch condition='First random glucose value ≤ 11 mmol/L' id='branch2'>
      <activity role='ED medical staff' action='Continue with standard monitoring' objects='patient' id='activity4'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED medical staff' action='Evaluate body temperature as part of routine vital signs every 4 hours for first 48 hours' objects='thermometer, patient' id='activity5'/>
  <exclusiveGateway id='exclusivegateway2'>
    <branch condition='Temperature > 37.5°C' id='branch3'>
      <parallelGateway id='parallelgateway1'>
        <branch id='branch4'>
          <activity role='ED medical staff' action='Initiate temperature-reducing measures' objects='cooling methods' id='activity6'/>
        </branch>
        <branch id='branch5'>
          <activity role='ED medical staff' action='Investigate potential infection' objects='patient, diagnostic tests' id='activity7'/>
          <exclusiveGateway id='exclusivegateway3'>
            <branch condition='Infection suspected or confirmed' id='branch6'>
              <activity role='ED medical staff' action='Initiate antipyretic and antimicrobial therapy as required' objects='antipyretics, antimicrobials' id='activity8'/>
            </branch>
            <branch condition='Infection not suspected or confirmed' id='branch7'>
              <activity role='ED medical staff' action='Continue monitoring and supportive care' objects='patient' id='activity9'/>
            </branch>
          </exclusiveGateway>
        </branch>
      </parallelGateway>
    </branch>
    <branch condition='Temperature ≤ 37.5°C' id='branch8'>
      <activity role='ED medical staff' action='Continue routine monitoring' objects='patient' id='activity10'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED medical staff' action='Provide cross-continuum secondary prevention assessments and therapies (per Modules 5 and 10)' objects='patient' id='activity11'/>
  <inclusiveGateway id='inclusivegateway1'>
    <branch condition='TIA patient presents with fluctuating or crescendo hemibody motor weakness, sensory loss, or speech difficulty' id='branch9'>
      <activity role='ED medical staff' action='Evaluate in ED and consider for hospital admission' objects='patient, admission criteria' id='activity12'/>
    </branch>
    <branch condition='Patient with TIA or minor stroke not on antiplatelet agent at presentation' id='branch10'>
      <exclusiveGateway id='exclusivegateway4'>
        <branch condition='Patient can swallow' id='branch11'>
          <activity role='ED medical staff' action='Start antiplatelet therapy immediately after brain imaging excludes intracranial hemorrhage (choose one: ASA 160 mg loading then 81–325 mg daily, clopidogrel 300 mg loading then 75 mg daily, or extended-release dipyridamole 200 mg/ASA 25 mg bid after ASA 160 mg loading)' objects='antiplatelet agents, brain imaging results' id='activity13'/>
        </branch>
        <branch condition='Patient cannot swallow' id='branch12'>
          <activity role='ED medical staff' action='Offer rectal or gastric ASA as necessary' objects='rectal/gastric ASA' id='activity14'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='All other patients with TIA or minor (nondisabling) stroke' id='branch13'>
      <exclusiveGateway id='exclusivegateway5'>
        <branch condition='TIA with high-risk features OR minor stroke' id='branch14'>
          <activity role='ED medical staff' action='Order brain CT scan' objects='CT scan' id='activity15-1'/>
          <activity role='ED medical staff' action='Review CT scan results and assess for contraindications to antiplatelet therapy' objects='CT scan results, patient history' id='activity15-2'/>
          <activity role='ED medical staff' action='Initiate antiplatelet therapy if not contraindicated' objects='antiplatelet agents' id='activity15-3'/>
          <exclusiveGateway id='exclusivegateway6'>
            <branch condition='Immediate care available at specialized TIA/minor stroke clinic' id='branch15'>
              <activity role='ED medical staff' action='Refer patient to specialized TIA/minor stroke clinic for rapid evaluation and management' objects='referral documentation' id='activity16'/>
            </branch>
            <branch condition='Delay to specialized clinic poses risk' id='branch16'>
              <activity role='ED medical staff' action='Perform evaluation and management in appropriately resourced ED; consider inpatient evaluation for stroke prevention' objects='ED resources, patient' id='activity17'/>
            </branch>
            <branch condition='Medical attention sought after 48 hours from symptom onset' id='branch17'>
              <activity role='ED medical staff' action='Refer for evaluation at specialized TIA/minor stroke clinic or outpatient clinic with stroke-prevention services within 24 hours of initial presentation' objects='referral documentation' id='activity18'/>
            </branch>
          </exclusiveGateway>
        </branch>
        <branch condition='TIA without high-risk features' id='branch18'>
          <activity role='ED medical staff' action='Order brain CT scan' objects='CT scan' id='activity19-1'/>
          <activity role='ED medical staff' action='Review CT scan results and assess for contraindications to antiplatelet therapy' objects='CT scan results, patient history' id='activity19-2'/>
          <activity role='ED medical staff' action='Initiate antiplatelet therapy if not contraindicated' objects='antiplatelet agents' id='activity19-3'/>
          <activity role='ED medical staff' action='Refer to outpatient clinic with stroke-prevention services for comprehensive evaluation and management within 1 month of symptom onset' objects='referral documentation' id='activity20'/>
        </branch>
      </exclusiveGateway>
    </branch>
  </inclusiveGateway>
</process>
