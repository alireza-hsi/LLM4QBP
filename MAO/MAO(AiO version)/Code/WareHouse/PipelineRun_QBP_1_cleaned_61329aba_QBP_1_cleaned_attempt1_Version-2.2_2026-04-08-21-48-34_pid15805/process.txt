<process>
  <activity role='triage staff' action='receive and register the patient for early assessment' objects='suspected TIA or minor stroke presentation in the ED, outpatient clinic, or acute care setting' id='activity1'/>
  <activity role='assessing clinician' action='assess airway patency' objects='airway' id='activity2_1'/>
  <activity role='assessing clinician' action='assess breathing adequacy' objects='breathing' id='activity2_2'/>
  <activity role='assessing clinician' action='assess circulation and hemodynamic stability' objects='circulation' id='activity2_3'/>
  <parallelGateway id='parallelgateway1'>
    <branch id='branch1'>
      <activity role='assessing clinician' action='perform a neurological examination and assess stroke severity' objects='focal neurological deficits, NIHSS or CNS for stroke' id='activity3'/>
    </branch>
    <branch id='branch2'>
      <exclusiveGateway id='exclusivegateway4_1'>
        <branch condition='CT is selected based on availability or clinical appropriateness' id='branch4_1'>
          <activity role='diagnostic imaging staff' action='perform CT brain imaging immediately' objects='CT' id='activity4_1'/>
        </branch>
        <branch condition='MRI is selected based on availability or clinical appropriateness' id='branch4_2'>
          <activity role='diagnostic imaging staff' action='perform MRI brain imaging immediately' objects='MRI' id='activity4_2'/>
        </branch>
      </exclusiveGateway>
      <activity role='health care provider with expertise in reading CT and/or MRI' action='interpret brain imaging immediately' objects='CT or MRI images' id='activity5'/>
    </branch>
    <branch id='branch3'>
      <activity role='cardiac diagnostics staff' action='perform ECG' objects='atrial fibrillation and other acute arrhythmias' id='activity6'/>
    </branch>
    <branch id='branch4'>
      <activity role='laboratory staff' action='obtain blood samples for required laboratory tests' objects='CBC, electrolytes, creatinine, glucose, INR, partial thromboplastin time, troponin if clinically indicated' id='activity7_1'/>
      <activity role='laboratory staff' action='process required core blood tests' objects='CBC, electrolytes, creatinine, glucose, INR, partial thromboplastin time' id='activity7_2'/>
      <exclusiveGateway id='exclusivegateway1'>
        <branch condition='troponin test is clinically indicated' id='branch5'>
          <activity role='laboratory staff' action='perform troponin test' objects='blood sample' id='activity8'/>
        </branch>
        <branch condition='troponin test is not clinically indicated' id='branch6'>
          <activity role='laboratory staff' action='document that troponin test is not clinically indicated' objects='clinical indication assessment' id='activity9'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch id='branch8'>
      <activity role='clinical team' action='ensure chest x-ray does not delay assessment for thrombolysis' objects='chest x-ray, thrombolysis assessment' id='activity12'/>
    </branch>
  </parallelGateway>
  <inclusiveGateway id='inclusivegateway1'>
    <branch condition='patient has stroke' id='branch7a'>
      <activity role='clinical team' action='keep the patient NPO initially' objects='oral medication, fluid, food' id='activity10'/>
      <activity role='clinical team' action='review alertness status and ongoing swallowing risk after the initial stroke assessment' objects='level of alertness, swallowing risk findings' id='activity13'/>
      <exclusiveGateway id='exclusivegateway2'>
        <branch condition='patient is not alert within the first 24 hours' id='branch9'>
          <activity role='clinical team' action='monitor closely and screen swallowing ability when clinically appropriate within the first 24 hours' objects='level of alertness, swallowing ability' id='activity14'/>
        </branch>
        <branch condition='patient is alert and clinically appropriate for bedside swallowing screening' id='branch10'>
          <activity role='nurse or trained clinician' action='screen swallowing ability as part of the initial assessment and before initiating oral medication, fluid, or food' objects='simple, valid, reliable bedside swallowing test protocol' id='activity11'/>
          <activity role='clinical team' action='continue swallowing management according to bedside screening findings' objects='bedside swallowing screen results' id='activity15'/>
        </branch>
      </exclusiveGateway>
      <exclusiveGateway id='exclusivegateway3'>
        <branch condition='features indicate dysphagia or pulmonary aspiration' id='branch11'>
          <activity role='speech-language pathologist or appropriately trained specialist' action='perform a full clinical assessment of swallowing ability and advise on the required consistency' objects='swallowing ability, diet consistency, fluid consistency' id='activity16'/>
        </branch>
        <branch condition='features do not indicate dysphagia or pulmonary aspiration' id='branch12'>
          <activity role='clinical team' action='continue oral intake precautions and diet decisions according to bedside screening findings' objects='bedside swallowing screen findings' id='activity17'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='patient has TIA or minor (nondisabling) ischemic stroke' id='branch14'>
      <activity role='clinician' action='apply the standardized TIA triage tool' objects='TIA triage tool, risk and disposition criteria' id='activity18_1'/>
      <activity role='clinician' action='determine the appropriate location for care' objects='patient disposition information' id='activity18_2'/>
      <activity role='clinician' action='assess time from symptom onset, symptom fluctuation, and candidacy for carotid artery revascularization' objects='symptom onset time, motor symptoms, speech symptoms, carotid revascularization candidacy' id='activity19'/>
      <exclusiveGateway id='exclusivegateway4'>
        <branch condition='presentation is within 48 hours of symptom onset or motor or speech symptoms are fluctuating and the patient is not clearly excluded from carotid artery revascularization' id='branch15'>
          <activity role='clinician or imaging staff' action='perform immediate vascular imaging of the neck arteries' objects='carotid ultrasound, CTA, or MRA' id='activity20'/>
        </branch>
        <branch condition='presentation is within 48 hours of symptom onset or motor or speech symptoms are fluctuating and the patient is clearly not a candidate for carotid artery revascularization' id='branch16'>
          <activity role='clinician or imaging staff' action='perform vascular imaging as soon as possible' objects='brain arteries and neck arteries' id='activity21'/>
        </branch>
        <branch condition='all other patients presenting beyond 48 hours with TIA or ischemic stroke' id='branch17'>
          <activity role='clinician or imaging staff' action='perform vascular imaging as soon as possible' objects='brain arteries and neck arteries' id='activity22'/>
        </branch>
      </exclusiveGateway>
    </branch>
  </inclusiveGateway>
  <exclusiveGateway id='exclusivegateway23_1'>
    <branch condition='patient is admitted to hospital or acute care' id='branch23_1'>
      <activity role='clinical team' action='initiate inpatient secondary prevention assessments and therapies' objects='secondary prevention assessments and therapies, Modules 5 and 10' id='activity23_1'/>
    </branch>
    <branch condition='patient is discharged from the ED or outpatient setting' id='branch23_2'>
      <activity role='clinical team' action='provide discharge secondary prevention assessments, therapies, and referral' objects='secondary prevention assessments and therapies, Modules 5 and 10, referral to TIA or stroke prevention follow-up services' id='activity23_2'/>
    </branch>
  </exclusiveGateway>
</process>
