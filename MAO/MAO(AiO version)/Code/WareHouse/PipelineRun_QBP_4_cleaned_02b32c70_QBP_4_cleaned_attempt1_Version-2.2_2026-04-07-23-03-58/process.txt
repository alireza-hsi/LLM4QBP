                      <activity role='stroke team' action='arrange transfer using regional bypass or drip and ship protocols' objects='transport decision and EVT centre handoff' id='activity8_3'/>

                  </exclusiveGateway>
                </branch>
              </inclusiveGateway>
            </branch>

              <exclusiveGateway id='exclusivegateway3'>
                <branch condition='patient with ICH is selected for further vascular imaging' id='branch7'>
                  <activity role='imaging team' action='perform vascular imaging as soon as possible' objects='CTA or MRA' id='activity12'/>
                </branch>
                <branch condition='patient with ICH is not selected for further vascular imaging' id='branch8'>
                  <activity role='stroke team' action='continue urgent ICH management pathway' objects='ICH management plan' id='activity13'/>
                </branch>
              </exclusiveGateway>
            </branch>
            <branch condition='other diagnosis or stroke type is not yet classified' id='branch9'>
              <activity role='stroke team' action='continue diagnostic evaluation and determine the most appropriate pathway' objects='clinical findings and imaging findings' id='activity14'/>
            </branch>
          </exclusiveGateway>
        </branch>

          <activity role='imaging team' action='perform vascular imaging as soon as possible' objects='vascular imaging of the brain and neck arteries' id='activity15'/>
        </branch>
        <branch condition='patient is outside the acute stroke treatment time window and does not meet the beyond 48 hours TIA or ischemic stroke condition' id='branch11'>

          <activity role='nursing staff' action='screen swallowing ability before initiating oral medication fluid or food using a simple valid reliable bedside testing protocol' objects='bedside swallowing screen' id='activity22'/>
        </branch>
        <branch condition='patient is not alert within the first 24 hours' id='branch17'>
          <activity role='nursing staff' action='monitor closely and screen swallowing ability when clinically appropriate' objects='neurological status and swallowing ability' id='activity23'/>
        </branch>
      </exclusiveGateway>
