<process>
  <exclusiveGateway id='exclusivegateway1'>
    <branch condition='arrival by EMS with suspected stroke/TIA' id='branch1'>
      <activity role='EMS' action='prenotify receiving hospital and regional stroke centre when feasible' objects='suspected stroke/TIA patient information' id='activity1'/>
      <activity role='receiving hospital acute stroke team' action='prepare for patient arrival and attend the ED for appropriate patients' objects='EMS prenotification details' id='activity2'/>
    </branch>
    <branch condition='arrival by outpatient clinic or direct admission without EMS prenotification' id='branch2'>
      <activity role='receiving clinical team' action='activate standardized stroke/TIA assessment pathway on presentation' objects='suspected stroke/TIA patient' id='activity3'/>
    </branch>
  </exclusiveGateway>

  <activity role='clinical team' action='assess airway patency' objects='airway' id='activity4_1'/>
  <activity role='clinical team' action='assess breathing' objects='breathing status' id='activity4_2'/>
  <activity role='clinical team' action='assess circulation' objects='circulation status' id='activity4_3'/>

  <parallelGateway id='parallelgateway1'>
    <branch id='branch3'>
      <activity role='clinician' action='perform focused neurological examination' objects='focal neurological deficits' id='activity5_1'/>
      <activity role='clinician' action='score stroke severity using a standardized stroke scale' objects='NIHSS or CNS for stroke' id='activity5_2'/>
    </branch>
    <branch id='branch4'>
      <activity role='diagnostic imaging team' action='perform brain imaging immediately' objects='CT or MRI' id='activity6'/>
      <activity role='qualified health care provider' action='interpret brain imaging immediately' objects='CT and/or MRI results' id='activity7'/>
    </branch>
    <branch id='branch5'>
      <activity role='clinical team' action='perform ECG' objects='electrocardiogram to detect atrial fibrillation and other acute arrhythmias' id='activity8'/>
    </branch>
    <branch id='branch6'>
      <activity role='clinical team' action='collect blood work' objects='CBC, electrolytes, creatinine, glucose, INR, partial thromboplastin time' id='activity9'/>
      <exclusiveGateway id='exclusivegateway_troponin'>
        <branch condition='troponin test is clinically indicated' id='branch7'>
          <activity role='clinical team' action='order troponin test' objects='troponin blood test' id='activity10'/>
        </branch>
        <branch condition='troponin test is not clinically indicated' id='branch8'>
          <activity role='clinical team' action='continue assessment without troponin testing' objects='initial blood work results' id='activity11'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch id='branch9'>
      <activity role='clinical team' action='avoid delaying thrombolysis assessment' objects='chest x-ray' id='activity12'/>
    </branch>
  </parallelGateway>

  <exclusiveGateway id='exclusivegateway2'>
    <branch condition='patient has stroke' id='branch10'>
      <activity role='clinical team' action='make patient NPO initially' objects='oral medication, fluid, and food' id='activity13'/>
      <activity role='clinical team' action='screen swallowing ability during initial assessment and before initiating oral medication, fluid, or food' objects='simple valid reliable bedside swallowing test protocol' id='activity14'/>
      <exclusiveGateway id='exclusivegateway3'>
        <branch condition='patient is not alert within the first 24 hours' id='branch11'>
          <activity role='clinical team' action='monitor closely and screen swallowing ability when clinically appropriate within the first 24 hours' objects='level of alertness and swallowing ability' id='activity15'/>
        </branch>
        <branch condition='patient is alert within the first 24 hours' id='branch12'>
          <activity role='clinical team' action='continue standard swallowing precautions' objects='screening results' id='activity16'/>
        </branch>
      </exclusiveGateway>
      <exclusiveGateway id='exclusivegateway4'>
        <branch condition='features indicate dysphagia or pulmonary aspiration' id='branch13'>
          <activity role='speech-language pathologist or appropriately trained specialist' action='perform full clinical swallowing assessment and advise required consistency' objects='swallowing ability, diet consistency, fluid consistency' id='activity17'/>
        </branch>
        <branch condition='no features indicate dysphagia or pulmonary aspiration' id='branch14'>
          <activity role='clinical team' action='proceed according to bedside swallowing screening result' objects='oral intake plan' id='activity18'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='patient has TIA or minor nondisabling stroke and stroke dysphagia recommendations are not applicable' id='branch15'>
      <activity role='clinical team' action='document that dysphagia recommendations are not applicable to TIA or minor nondisabling stroke pathway' objects='clinical assessment record' id='activity19'/>
    </branch>
  </exclusiveGateway>

  <exclusiveGateway id='exclusivegateway5'>
    <branch condition='patient has TIA' id='branch16'>
      <activity role='clinical team' action='apply standardized TIA triage tool' objects='TIA triage criteria' id='activity20_1'/>
      <activity role='clinical team' action='determine appropriate location of care' objects='TIA triage tool result' id='activity20_2'/>
    </branch>
    <branch condition='patient does not have TIA' id='branch17'>
      <activity role='clinical team' action='continue pathway selection based on assessment findings' objects='clinical and imaging findings' id='activity21'/>
    </branch>
  </exclusiveGateway>

  <exclusiveGateway id='exclusivegateway6_pre_disposition'>
    <branch condition='patient has TIA or minor nondisabling stroke presenting within 48 hours of symptom onset or with fluctuating motor or speech symptoms' id='branch18_pre'>
      <exclusiveGateway id='exclusivegateway7_pre_disposition'>
        <branch condition='patient is clearly a candidate for carotid artery revascularization or candidacy is not excluded' id='branch19_pre'>
          <activity role='clinical team' action='perform immediate vascular imaging of the neck arteries' objects='carotid ultrasound, CTA, or MRA' id='activity22_pre'/>
        </branch>
        <branch condition='patient is clearly not a candidate for carotid artery revascularization' id='branch20_pre'>
          <activity role='clinical team' action='document that immediate neck artery vascular imaging is not required' objects='carotid revascularization candidacy assessment' id='activity23_pre'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='patient has TIA or ischemic stroke presenting beyond 48 hours' id='branch21_pre'>
      <activity role='clinical team' action='perform vascular imaging as soon as possible' objects='brain and neck arteries imaging' id='activity24_pre'/>
    </branch>
    <branch condition='patient does not have TIA or minor nondisabling stroke for the Module 1.6 vascular imaging pathway' id='branch22_pre'>
      <activity role='clinical team' action='continue management under the appropriate stroke pathway without applying the TIA or minor stroke vascular imaging rule' objects='assessment findings and care plan' id='activity24_alt'/>
    </branch>
  </exclusiveGateway>

  <exclusiveGateway id='exclusivegateway10'>
    <branch condition='patient is admitted to hospital based on assessment triage imaging and overall findings' id='branch28'>
      <activity role='clinical team' action='transfer patient to the appropriate inpatient stroke care setting' objects='admission orders and care plan' id='activity28'/>
      <activity role='clinical team' action='perform secondary prevention assessment' objects='vascular risk factors and prevention needs for admitted patient' id='activity25_1'/>
      <activity role='clinical team' action='initiate indicated secondary prevention therapies' objects='Module 5 and Module 10 interventions for admitted patient' id='activity25_2'/>
    </branch>
    <branch condition='patient is discharged from the ED or not admitted to hospital based on assessment triage imaging and overall findings' id='branch29'>
      <activity role='clinical team' action='perform secondary prevention assessment before discharge' objects='vascular risk factors and prevention needs for ED discharge patient' id='activity26_1'/>
      <activity role='clinical team' action='provide indicated secondary prevention therapies and instructions before discharge' objects='Module 5 and Module 10 interventions for ED discharge patient' id='activity26_2'/>
      <activity role='clinical team' action='identify appropriate follow-up service' objects='rapid-assessment TIA or minor stroke unit, TIA clinic, or provincial stroke-prevention clinic' id='activity27_1'/>
      <activity role='clinical team' action='send referral and follow-up information' objects='outpatient referral and patient follow-up plan' id='activity27_2'/>
    </branch>
  </exclusiveGateway>
</process>
