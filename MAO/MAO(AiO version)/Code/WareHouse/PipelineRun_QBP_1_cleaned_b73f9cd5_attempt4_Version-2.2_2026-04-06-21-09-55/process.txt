  <activity role='clinician' action='assess airway' objects='airway' id='activity3_1'/>
  <activity role='clinician' action='assess breathing and circulation' objects='breathing, circulation' id='activity3_2'/>
  <activity role='clinician' action='perform neurological examination and assess stroke severity on a standardized stroke scale' objects='focal neurological deficits, NIHSS or CNS' id='activity4'/>
  <parallelGateway id='parallelgateway1'>
    <branch id='branch1'>
      <activity role='imaging team' action='perform brain imaging immediately' objects='CT or MRI' id='activity5'/>

      <activity role='clinical staff' action='defer chest x-ray if needed so it does not delay thrombolysis assessment' objects='chest x-ray, thrombolysis assessment' id='activity6_1'/>

    <branch id='branch2'>
      <activity role='clinical staff' action='perform electrocardiogram' objects='ECG' id='activity7'/>
    </branch>
    <branch id='branch3'>
      <activity role='laboratory staff' action='obtain and process core blood work' objects='CBC, electrolytes, creatinine, glucose, INR, partial thromboplastin time' id='activity8_1'/>
      <exclusiveGateway id='exclusivegateway3'>
        <branch condition='troponin test is clinically indicated' id='branch13'>
          <activity role='laboratory staff' action='obtain and process troponin test' objects='troponin test' id='activity8_2'/>
        </branch>
        <branch condition='troponin test is not clinically indicated' id='branch14'>
          <activity role='clinical staff' action='document that troponin test is not required' objects='clinical indication assessment' id='activity8_3'/>
        </branch>
      </exclusiveGateway>

  <exclusiveGateway id='exclusivegateway1'>

      <activity role='clinical staff' action='screen swallowing ability during initial assessment and before initiating oral medication, fluid, or food' objects='bedside swallowing screening protocol' id='activity11'/>
      <inclusiveGateway id='inclusivegateway1'>
        <branch condition='patient is not alert within the first 24 hours' id='branch7'>
          <activity role='clinical staff' action='monitor closely and screen swallowing ability when clinically appropriate' objects='neurological status, swallowing ability' id='activity12'/>
        </branch>
        <branch condition='patient presents with features indicating dysphagia or pulmonary aspiration' id='branch8'>
          <activity role='speech-language pathologist or appropriately trained specialist' action='perform full clinical assessment of swallowing ability and advise required consistency of diet and fluids' objects='swallowing ability, diet consistency, fluid consistency' id='activity13'/>
        </branch>
      </inclusiveGateway>
    </branch>
    <branch condition='patient has TIA or minor nondisabling stroke' id='branch9'>
      <activity role='clinician' action='apply standardized TIA triage criteria' objects='TIA triage tool' id='activity14_1'/>
      <activity role='clinician' action='determine the appropriate location for care' objects='triage decision' id='activity14_2'/>
      <exclusiveGateway id='exclusivegateway2'>
        <branch condition='presentation within 48 hours of symptom onset or fluctuating motor or speech symptoms and patient is candidate for carotid artery revascularization' id='branch10'>

        </branch>
        <branch condition='presentation within 48 hours of symptom onset or fluctuating motor or speech symptoms and patient is clearly not a candidate for carotid artery revascularization' id='branch11'>
          <activity role='clinician' action='document that immediate neck artery vascular imaging is not required because patient is clearly not a candidate for carotid artery revascularization' objects='revascularization candidacy assessment' id='activity16'/>
        </branch>
        <branch condition='all other patients presenting beyond 48 hours with TIA or ischemic stroke' id='branch12'>
          <activity role='imaging team' action='perform vascular imaging as soon as possible' objects='brain and neck arteries imaging' id='activity17'/>
        </branch>
      </exclusiveGateway>
    </branch>
  </exclusiveGateway>
  <activity role='clinical team' action='provide secondary prevention assessments' objects='cross-continuum secondary prevention assessment' id='activity18_1'/>
