    <branch id='branch4'>

    </branch>
    <branch id='branch5'>
      <activity role='clinician' action='check patient history' objects='medical history and presenting symptoms' id='activity7'/>
    </branch>
    <branch id='branch6'>
      <activity role='clinician or pharmacist' action='document and reconcile medications currently used by patient' objects='current medication list' id='activity8'/>

    <branch id='branch7'>

        <branch condition='patient is well enough to leave emergency department' id='branch34'>
          <activity role='radiology team' action='perform standard chest x-ray' objects='posteroanterior and lateral views' id='activity9_2'/>
        </branch>
        <branch condition='patient is too unwell to leave emergency department' id='branch35'>
          <activity role='radiology team' action='perform portable chest x-ray' objects='portable chest x-ray images' id='activity9_3'/>

        <branch condition='no concern with pneumothorax' id='branch37'>
          <activity role='radiology team' action='continue without expiratory view' objects='chest x-ray study' id='activity9_5'/>
        </branch>
      </exclusiveGateway>

    <branch id='branch8'>
