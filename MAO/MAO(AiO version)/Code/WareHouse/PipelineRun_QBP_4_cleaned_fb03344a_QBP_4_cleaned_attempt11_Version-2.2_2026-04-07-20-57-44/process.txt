<process>
  <activity role='triage clinician' action='identify suspected stroke presentation in the ED, outpatient clinic, or direct acute care admission' objects='patient' id='activity1'/>
  <activity role='triage clinician' action='assign high-acuity suspected stroke triage priority and activate the early stroke assessment pathway' objects='triage priority, stroke pathway' id='activity1_1'/>
  <activity role='triage clinician' action='notify the acute stroke team for appropriate patients based on suspected stroke presentation' objects='acute stroke team notification' id='activity1_2'/>
  <activity role='clinical team' action='conduct rapid initial evaluation of airway, breathing, and circulation' objects='patient' id='activity2'/>
  <activity role='clinical team' action='perform a neurological examination to determine focal neurological deficits and assess stroke severity using NIHSS or CNS' objects='patient' id='activity3'/>
  <activity role='clinical team' action='make the patient NPO initially before any oral medication, fluid, or food' objects='patient' id='activity4'/>
  <parallelGateway id='parallelgateway1'>
    <branch id='branch1'>
      <exclusiveGateway id='exclusivegateway1'>
        <branch condition='presenting within acute stroke treatment time window' id='branch2'>
          <parallelGateway id='parallelgateway1_1'>
            <branch id='branch2_1'>
              <activity role='imaging team' action='perform rapid non-contrast brain imaging with CT or MRI with door to imaging less than 25 minutes' objects='non-contrast CT or MRI' id='activity5_1'/>
            </branch>
            <branch id='branch2_2'>
              <activity role='imaging team' action='perform rapid vascular imaging with CTA with door to CT or CTA less than 25 minutes' objects='CTA' id='activity5_2'/>
            </branch>
          </parallelGateway>
          <exclusiveGateway id='exclusivegateway2'>
            <branch condition='acute ischemic stroke is clinically eligible for acute stroke treatments' id='branch3'>
              <activity role='imaging team' action='consider advanced CT imaging including CT and multiphase CTA or Perfusion CT' objects='advanced CT imaging' id='activity6'/>
            </branch>
            <branch condition='acute ischemic stroke is not clinically eligible for acute stroke treatments' id='branch4'>
              <activity role='imaging team' action='proceed with standard urgent imaging pathway without advanced acute treatment CT selection imaging' objects='standard imaging results' id='activity7'/>
            </branch>
          </exclusiveGateway>
          <activity role='stroke imaging expert' action='interpret CT/CTA and or MRI immediately' objects='brain imaging studies' id='activity9'/>
        </branch>
        <branch condition='presenting beyond the acute stroke treatment time window' id='branch5'>
          <activity role='clinical team' action='continue nonhyperacute stroke and TIA assessment pathway after initial urgent assessment' objects='patient' id='activity8'/>
          <activity role='imaging team' action='perform appropriate brain imaging for nonhyperacute stroke and TIA assessment' objects='brain imaging studies' id='activity8_1'/>
          <activity role='stroke imaging expert' action='interpret CT/CTA and or MRI immediately' objects='brain imaging studies' id='activity9_1'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch id='branch6'>
      <activity role='cardiac assessment team' action='perform ECG to detect atrial fibrillation and other acute arrhythmias without delaying acute treatment' objects='ECG' id='activity10'/>
    </branch>
    <branch id='branch7'>
      <activity role='laboratory team' action='obtain blood work including CBC, electrolytes, creatinine, glucose, INR, and partial thromboplastin time' objects='blood samples' id='activity11'/>
      <exclusiveGateway id='exclusivegateway3'>
        <branch condition='troponin is clinically indicated' id='branch8'>
          <activity role='laboratory team' action='perform troponin test' objects='blood sample' id='activity12'/>
        </branch>
        <branch condition='troponin is not clinically indicated' id='branch9'>
          <activity role='laboratory team' action='skip troponin testing' objects='clinical indication review' id='activity13'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch id='branch10'>
      <activity role='clinical team' action='ensure chest x-ray does not delay assessment for thrombolysis' objects='chest x-ray request' id='activity14'/>
    </branch>
  </parallelGateway>
  <exclusiveGateway id='exclusivegateway6'>
    <branch condition='brain imaging confirms intracerebral hemorrhage' id='branch16'>
      <activity role='stroke physician' action='treat ICH as a medical emergency and evaluate the patient immediately' objects='patient with ICH' id='activity19'/>
      <activity role='stroke physician' action='ask about anticoagulant therapy and evaluate platelet count, PTT, and INR' objects='medication history, laboratory results' id='activity20'/>
      <exclusiveGateway id='exclusivegateway7'>
        <branch condition='selected patient with ICH requires vascular cause assessment' id='branch17'>
          <activity role='imaging team' action='perform CTA or MRA as soon as possible' objects='CTA or MRA' id='activity21'/>
        </branch>
        <branch condition='selected CTA or MRA is not required' id='branch18'>
          <activity role='stroke physician' action='continue ICH management without additional CTA or MRA' objects='ICH assessment results' id='activity22'/>
        </branch>
      </exclusiveGateway>
      <exclusiveGateway id='exclusivegateway10'>
        <branch condition='patient is alert for bedside screening after stroke confirmation' id='branch25'>
          <activity role='nursing staff' action='screen swallowing ability using a simple valid reliable bedside testing protocol before oral medication fluid or food' objects='swallow screen protocol' id='activity30'/>
          <exclusiveGateway id='exclusivegateway10_1'>
            <branch condition='bedside swallowing screen is passed' id='branch26'>
              <activity role='nursing staff' action='document bedside swallowing screen passed and proceed according to oral intake protocol' objects='swallow screen result, oral intake protocol' id='activity31'/>
            </branch>
            <branch condition='bedside swallowing screen is abnormal' id='branch27'>
              <activity role='nursing staff' action='maintain NPO and request full swallowing assessment because bedside swallowing screen is abnormal' objects='NPO order, swallowing assessment request' id='activity32'/>
            </branch>
          </exclusiveGateway>
        </branch>
        <branch condition='patient is not alert within the first 24 hours after stroke confirmation' id='branch28'>
          <activity role='nursing staff' action='monitor closely within the first 24 hours and defer swallowing screen until clinically appropriate' objects='patient status' id='activity33'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='brain imaging confirms ischemic stroke or TIA' id='branch19'>
      <exclusiveGateway id='exclusivegateway5'>
        <branch condition='potentially disabling acute neurological symptoms suggest acute stroke within 6.0 hours of symptom onset or treatment can be initiated within 12 hours and clinical and imaging criteria are met' id='branch14'>
          <activity role='stroke team' action='consider endovascular therapy' objects='clinical criteria, imaging criteria' id='activity17'/>
          <activity role='stroke team' action='confirm EVT eligibility based on clinical and imaging criteria' objects='clinical findings, imaging findings' id='activity17_1'/>
          <exclusiveGateway id='exclusivegateway5_1'>
            <branch condition='EVT is available at the current site' id='branch14_1'>
              <activity role='stroke team' action='coordinate immediate EVT destination and treatment pathway at the current site' objects='EVT treatment plan' id='activity17_2'/>
            </branch>
            <branch condition='EVT requires transfer to another EVT site' id='branch14_2'>
              <activity role='stroke team' action='coordinate transfer and transport pathway with the EVT centre through regional communication processes' objects='EVT transfer plan, transport coordination' id='activity17_3'/>
            </branch>
          </exclusiveGateway>
        </branch>
        <branch condition='EVT time or eligibility criteria are not met' id='branch15'>
          <activity role='stroke team' action='continue with non-EVT acute stroke management pathway' objects='assessment results' id='activity18'/>
        </branch>
      </exclusiveGateway>
      <exclusiveGateway id='exclusivegateway8'>
        <branch condition='TIA or ischemic stroke presents beyond 48 hours' id='branch20'>
          <activity role='imaging team' action='perform vascular imaging of the brain and neck arteries as soon as possible' objects='vascular imaging' id='activity23'/>
        </branch>
        <branch condition='TIA or ischemic stroke does not present beyond 48 hours' id='branch21'>
          <activity role='clinical team' action='follow the acute ischemic stroke pathway and document that recommendation 1.6.1 is not applicable to ischemic stroke patients' objects='ischemic stroke pathway record' id='activity24'/>
        </branch>
      </exclusiveGateway>
      <activity role='clinical team' action='document that recommendation 1.5 is not applicable to ischemic or ICH stroke patients' objects='clinical record' id='activity25'/>
      <exclusiveGateway id='exclusivegateway11'>
        <branch condition='patient is alert for bedside screening after stroke confirmation' id='branch29'>
          <activity role='nursing staff' action='screen swallowing ability using a simple valid reliable bedside testing protocol before oral medication fluid or food' objects='swallow screen protocol' id='activity34'/>
          <exclusiveGateway id='exclusivegateway11_1'>
            <branch condition='bedside swallowing screen is passed' id='branch30'>
              <activity role='nursing staff' action='document bedside swallowing screen passed and proceed according to oral intake protocol' objects='swallow screen result, oral intake protocol' id='activity35'/>
            </branch>
            <branch condition='bedside swallowing screen is abnormal' id='branch31'>
              <activity role='nursing staff' action='maintain NPO and request full swallowing assessment because bedside swallowing screen is abnormal' objects='NPO order, swallowing assessment request' id='activity36'/>
            </branch>
          </exclusiveGateway>
        </branch>
        <branch condition='patient is not alert within the first 24 hours after stroke confirmation' id='branch32'>
          <activity role='nursing staff' action='monitor closely within the first 24 hours and defer swallowing screen until clinically appropriate' objects='patient status' id='activity37'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='brain imaging confirms another non-ICH diagnosis' id='branch24'>
      <activity role='clinical team' action='continue appropriate non-stroke diagnostic and management pathway' objects='diagnostic findings, management plan' id='activity29'/>
    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='exclusivegateway9'>
    <branch condition='features indicating dysphagia or pulmonary aspiration are present' id='branch22'>
      <activity role='speech-language pathologist or appropriately trained specialist' action='perform a full clinical swallowing assessment and advise on swallowing ability and required diet and fluid consistency' objects='swallowing assessment, diet and fluid recommendations' id='activity26'/>
    </branch>
    <branch condition='no features indicating dysphagia or pulmonary aspiration are present' id='branch23'>
      <activity role='clinical team' action='document that no specialist swallowing assessment is currently indicated and continue swallowing precautions as appropriate' objects='clinical record, swallowing precautions' id='activity27'/>
    </branch>
  </exclusiveGateway>
  <activity role='clinical team' action='provide appropriate cross-continuum secondary prevention assessments and therapies whether the patient is admitted or discharged from the ED' objects='secondary prevention plan' id='activity28'/>
</process>
