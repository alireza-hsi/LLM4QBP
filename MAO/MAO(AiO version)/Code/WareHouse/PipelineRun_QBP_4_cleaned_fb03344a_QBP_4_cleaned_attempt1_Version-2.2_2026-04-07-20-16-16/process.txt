            <branch id='branch2_2'>
              <activity role='imaging team' action='perform CTA in parallel vascular imaging workflow with door to CT/CTA less than twenty five minutes' objects='CTA' id='activity6_2'/>
            </branch>
          </parallelGateway>
          <activity role='qualified imaging interpreter' action='interpret brain imaging immediately' objects='CT/CTA and or MRI of the brain' id='activity12'/>
        </branch>
        <branch condition='presenting beyond acute stroke treatment time window' id='branch3'>
          <activity role='clinician' action='continue non-hyperacute diagnostic assessment' objects='suspected stroke patient' id='activity7'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch id='branch4'>
      <activity role='nursing staff or technician' action='perform ECG without delaying acute treatment' objects='ECG, atrial fibrillation and other acute arrhythmias assessment' id='activity8'/>
    </branch>
    <branch id='branch5'>

      <activity role='clinician' action='ensure chest x-ray does not delay assessment for thrombolysis' objects='chest x-ray decision' id='activity10'/>

  </parallelGateway>
  <exclusiveGateway id='exclusivegateway2'>
    <branch condition='acute ischemic stroke identified' id='branch8_new'>
      <exclusiveGateway id='exclusivegateway2_1'>
