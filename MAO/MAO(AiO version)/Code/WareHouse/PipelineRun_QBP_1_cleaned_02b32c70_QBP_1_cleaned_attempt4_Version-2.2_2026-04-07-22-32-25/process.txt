<process>
  <activity role='EMS, outpatient clinic staff, or acute care intake staff' action='identify a patient with suspected stroke or TIA and initiate the early assessment pathway' objects='suspected stroke/TIA patient' id='activity1'/>
  <exclusiveGateway id='exclusiveGateway1'>
    <branch condition='patient arrives through EMS' id='branch1'>
      <activity role='EMS' action='divert the patient where feasible to a regional or district stroke centre' objects='suspected stroke/TIA patient' id='activity2'/>
      <activity role='EMS' action='prenotify the receiving hospital with timely and detailed information' objects='patient status and estimated arrival time' id='activity3'/>
      <activity role='receiving hospital' action='contact and call the acute stroke team to the ED for appropriate patients' objects='EMS prenotification' id='activity4'/>
    </branch>
    <branch condition='patient presents at the ED, outpatient clinic, or by direct admission without EMS prenotification' id='branch2'>
      <activity role='receiving team' action='activate the standardized stroke/TIA assessment protocol' objects='suspected stroke/TIA patient' id='activity5'/>
    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='exclusiveGateway20'>
    <branch condition='patient is in the ED' id='branch25'>
      <activity role='triage team' action='provide CTAS I or CTAS II priority access to early assessment and diagnostic imaging' objects='suspected stroke/TIA patient' id='activity6'/>
    </branch>
    <branch condition='patient is in an outpatient clinic or directly admitted to acute care' id='branch26'>
      <activity role='receiving team' action='expedite urgent stroke/TIA assessment and diagnostic imaging' objects='suspected stroke/TIA patient' id='activity6_2'/>
    </branch>
  </exclusiveGateway>
  <activity role='clinical team' action='assess' objects='airway' id='activity7_1'/>
  <activity role='clinical team' action='assess' objects='breathing' id='activity7_2'/>
  <activity role='clinical team' action='assess' objects='circulation' id='activity7_3'/>
  <parallelGateway id='parallelGateway1'>
    <branch id='branch3'>
      <activity role='clinician' action='perform a neurological examination and assess stroke severity' objects='focal neurological deficits, NIHSS or CNS for stroke' id='activity8'/>
    </branch>
    <branch id='branch4'>
      <exclusiveGateway id='exclusiveGateway9_1'>
        <branch condition='CT is selected or most immediately available' id='branch9_1'>
          <activity role='imaging team' action='perform brain imaging immediately' objects='CT' id='activity9_1'/>
        </branch>
        <branch condition='MRI is selected and immediately available' id='branch9_2'>
          <activity role='imaging team' action='perform brain imaging immediately' objects='MRI' id='activity9_2'/>
        </branch>
      </exclusiveGateway>
      <activity role='qualified health care provider' action='interpret brain imaging immediately' objects='CT and/or MRI results' id='activity10'/>
    </branch>
    <branch id='branch5'>
      <activity role='clinical team' action='perform ECG' objects='cardiac rhythm for atrial fibrillation and other acute arrhythmias' id='activity11'/>
    </branch>
    <branch id='branch6'>
      <inclusiveGateway id='inclusiveGateway12_1'>
        <branch condition='all patients require routine blood work' id='branch12_1'>
          <activity role='clinical team' action='obtain blood work' objects='CBC, electrolytes, creatinine, glucose, INR, and partial thromboplastin time' id='activity12_1'/>
        </branch>
        <branch condition='troponin test is clinically indicated' id='branch12_2'>
          <activity role='clinical team' action='obtain blood work' objects='troponin test' id='activity12_2'/>
        </branch>
      </inclusiveGateway>
    </branch>
  </parallelGateway>
  <exclusiveGateway id='exclusiveGateway6'>
    <branch condition='chest x-ray is clinically indicated and will not delay thrombolysis assessment' id='branch21'>
      <activity role='clinical team' action='perform chest x-ray without delaying assessment' objects='chest x-ray, thrombolysis assessment' id='activity13_1'/>
    </branch>
    <branch condition='chest x-ray is not clinically indicated or would delay thrombolysis assessment' id='branch22'>
      <activity role='clinical team' action='defer or skip chest x-ray and continue urgent stroke evaluation' objects='chest x-ray, thrombolysis assessment' id='activity13_2'/>
    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='exclusiveGateway2'>
    <branch condition='patient has stroke other than TIA or minor nondisabling stroke' id='branch8'>
      <activity role='clinical team' action='make the patient' objects='NPO initially' id='activity14_1'/>
      <activity role='clinical team' action='screen swallowing ability using a bedside protocol as part of the initial assessment and before initiating' objects='oral medication, fluid, or food' id='activity14_2'/>
      <inclusiveGateway id='inclusiveGateway1'>
        <branch condition='patient is not alert within the first 24 hours' id='branch9'>
          <activity role='clinical team' action='monitor closely and screen swallowing ability when clinically appropriate' objects='swallowing ability within the first 24 hours' id='activity15'/>
        </branch>
        <branch condition='features indicate dysphagia or pulmonary aspiration' id='branch10'>
          <activity role='speech-language pathologist or appropriately trained specialist' action='perform a full clinical swallowing assessment and advise on swallowing ability and required consistency of diet and fluids' objects='swallowing ability, diet consistency, fluid consistency' id='activity16'/>
        </branch>
        <branch condition='patient is alert within the first 24 hours and has no features indicating dysphagia or pulmonary aspiration' id='branch11'>
          <activity role='clinical team' action='proceed with oral medication, fluid, or food only according to the bedside swallowing screen result' objects='oral medication, fluid, food' id='activity17'/>
        </branch>
      </inclusiveGateway>
    </branch>
    <branch condition='patient has TIA or minor nondisabling ischemic stroke' id='branch12'>
      <parallelGateway id='parallelGateway2'>
        <branch id='branch13'>
          <exclusiveGateway id='exclusiveGateway18_0'>
            <branch condition='patient has TIA' id='branch13_1'>
              <activity role='clinical team' action='apply a standardized triage tool' objects='TIA triage tool' id='activity18_1'/>
              <activity role='clinical team' action='determine' objects='the appropriate location for care' id='activity18_2'/>
            </branch>
            <branch condition='patient has minor nondisabling ischemic stroke' id='branch13_2'>
              <activity role='clinical team' action='continue' objects='the minor nondisabling ischemic stroke pathway without applying the standardized TIA triage tool' id='activity18_3'/>
            </branch>
          </exclusiveGateway>
        </branch>
        <branch id='branch14'>
          <exclusiveGateway id='exclusiveGateway3'>
            <branch condition='presentation is within 48 hours of symptom onset or motor or speech symptoms are fluctuating' id='branch15'>
              <exclusiveGateway id='exclusiveGateway4'>
                <branch condition='patient is not clearly excluded from carotid artery revascularization' id='branch16'>
                  <exclusiveGateway id='exclusiveGateway19_1'>
                    <branch condition='carotid ultrasound is selected and available' id='branch19_1'>
                      <activity role='imaging team' action='perform immediate vascular imaging of the neck arteries' objects='carotid ultrasound' id='activity19_1'/>
                    </branch>
                    <branch condition='CTA is selected and available' id='branch19_2'>
                      <activity role='imaging team' action='perform immediate vascular imaging of the neck arteries' objects='CTA' id='activity19_2'/>
                    </branch>
                    <branch condition='MRA is selected and available' id='branch19_3'>
                      <activity role='imaging team' action='perform immediate vascular imaging of the neck arteries' objects='MRA' id='activity19_3'/>
                    </branch>
                  </exclusiveGateway>
                </branch>
                <branch condition='patient is clearly not a candidate for carotid artery revascularization' id='branch17'>
                  <activity role='clinical team' action='document noncandidacy for carotid artery revascularization and continue the TIA or minor stroke pathway' objects='revascularization candidacy assessment' id='activity20'/>
                </branch>
              </exclusiveGateway>
            </branch>
            <branch condition='all other patients present beyond 48 hours with TIA or ischemic stroke' id='branch18'>
              <activity role='imaging team' action='perform vascular imaging of the brain and neck arteries as soon as possible' objects='brain and neck vascular imaging' id='activity21'/>
            </branch>
          </exclusiveGateway>
        </branch>
      </parallelGateway>
    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='exclusiveGateway5'>
    <branch condition='patient is admitted to hospital' id='branch19'>
      <activity role='clinical team' action='provide cross-continuum secondary prevention assessments and therapies' objects='Modules 5 and 10 interventions' id='activity22'/>
    </branch>
    <branch condition='patient is discharged from the ED or is not admitted after assessment' id='branch20'>
      <activity role='clinical team' action='provide cross-continuum secondary prevention assessments and therapies' objects='Modules 5 and 10 interventions' id='activity23'/>
      <activity role='clinical team' action='arrange referral to a rapid-assessment TIA or minor stroke unit, TIA clinic, or stroke-prevention clinic' objects='referral' id='activity24'/>
    </branch>
  </exclusiveGateway>
</process>
