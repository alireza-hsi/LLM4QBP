      <parallelGateway id='pg1'>
        <branch id='br2'>
          <parallelGateway id='pg1_1'>
            <branch id='br2_1'>
              <activity role='physiotherapist' action='assess patient within 48 hours of hospital admission' objects='physiotherapy needs' id='a6_1'/>
            </branch>
            <branch id='br2_2'>
              <activity role='speech-language pathologist' action='assess patient within 48 hours of hospital admission' objects='speech-language pathology needs' id='a6_2'/>
            </branch>
            <branch id='br2_3'>
              <activity role='occupational therapist' action='assess patient within 48 hours of hospital admission' objects='occupational therapy needs' id='a6_3'/>
            </branch>
          </parallelGateway>
          <activity role='physiotherapy, speech-language pathology, and occupational therapy team members' action='formulate' objects='interprofessional stroke management plan' id='a6_4'/>
        </branch>
        <branch id='br3'>
          <parallelGateway id='pg1_2'>

              <activity role='clinicians' action='evaluate' objects='stroke-related impairments using standardized valid assessment tools' id='a7_1'/>
            </branch>
            <branch id='br3_2'>
              <activity role='clinicians' action='evaluate' objects='functional status using standardized valid assessment tools, preferably in electronic format linked to the EHR when possible' id='a7_2'/>
            </branch>
          </parallelGateway>
          <activity role='clinicians' action='complete AlphaFIM on or by day 3 after admission with admission day counted as day 1' objects='AlphaFIM assessment' id='a8'/>
        </branch>
        <branch id='br4'>
          <activity role='clinicians' action='assess patient for risk of developing' objects='venous thromboembolism' id='a9'/>

          <activity role='nursing staff' action='screen within the first 48 hours of admission using' objects='valid nutrition screening tool' id='a11_1'/>
          <activity role='nursing staff' action='screen within the first 48 hours of admission using' objects='valid hydration screening tool' id='a11_2'/>
          <exclusiveGateway id='eg4'>
            <branch condition='No nutritional concerns, hydration deficits, dysphagia, or other relevant comorbidities are identified' id='br16'>
              <activity role='acute care team' action='continue routine nutrition and hydration monitoring' objects='ongoing nutrition and hydration status' id='a11_3'/>
            </branch>
            <branch condition='Nutritional concerns, hydration deficits, dysphagia, or other comorbidities are identified' id='br17'>
              <activity role='acute care team' action='refer patient to a dietitian within 7 days of admission for recommendations' objects='dietitian referral' id='a12'/>
              <exclusiveGateway id='eg5'>
                <branch condition='Patient is unable to meet nutritional and fluid requirements' id='br18'>
                  <activity role='dietitian' action='provide recommendations and consider' objects='enteral nutrition support plan' id='a13'/>
                </branch>
                <branch condition='Patient is able to meet nutritional and fluid requirements' id='br19'>
                  <activity role='dietitian' action='provide' objects='nutrition and hydration recommendations' id='a13_1'/>
                </branch>
              </exclusiveGateway>
            </branch>
          </exclusiveGateway>
        </branch>
        <branch id='br8'>
          <activity role='acute care team' action='assess' objects='contraindications to mobilization' id='a14_1'/>
          <exclusiveGateway id='eg2a'>
            <branch condition='Mobilization is contraindicated' id='br9'>
              <activity role='acute care team' action='defer mobilization and document' objects='mobilization contraindication' id='a15'/>
            </branch>
            <branch condition='Mobilization is not contraindicated' id='br9_1'>
              <activity role='acute care team' action='determine' objects='time since stroke onset for mobilization planning' id='a14_2'/>
              <exclusiveGateway id='eg2b'>
                <branch condition='Stroke onset is within 24 hours and clinician judges mobilization reasonable' id='br10'>
                  <activity role='acute care team' action='initiate cautious mobilization within 24 hours using clinical judgement and avoid' objects='frequent out-of-bed activity' id='a16'/>
                </branch>
                <branch condition='Stroke onset is within 24 hours and clinician judges mobilization not reasonable' id='br11'>
                  <activity role='acute care team' action='avoid frequent out-of-bed activity within 24 hours and reassess for mobilization at 24 to 48 hours from stroke onset' objects='mobilization readiness' id='a17'/>
                </branch>
                <branch condition='Stroke onset is between 24 hours and 48 hours' id='br12'>
                  <activity role='acute care team' action='mobilize patient early between 24 hours and 48 hours of stroke onset to help prevent' objects='venous thromboembolism' id='a18'/>
                </branch>
                <branch condition='Stroke onset is more than 48 hours' id='br12_1'>
                  <activity role='acute care team' action='mobilize patient as soon as clinically feasible and document' objects='post-48-hour mobilization plan' id='a18_1'/>
                </branch>
              </exclusiveGateway>

          </exclusiveGateway>
        </branch>

    </branch>
