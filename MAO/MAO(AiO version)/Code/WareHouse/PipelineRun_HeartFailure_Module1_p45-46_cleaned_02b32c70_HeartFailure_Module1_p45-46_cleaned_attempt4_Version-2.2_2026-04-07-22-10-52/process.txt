<process>
  <activity role='ED clinician' action='collect presenting symptoms, vital status, and relevant comorbidity and disposition information' objects='initial presentation data for standardized risk stratification' id='activity1_1'/>
  <activity role='ED clinician' action='perform immediate clinical assessment of airway, breathing, circulation, and congestion status' objects='patient with suspected acute heart failure' id='activity2_1'/>
  <activity role='ED clinician' action='initiate initial ED treatment including diuresis as clinically indicated' objects='initial acute heart failure treatment plan' id='activity2_2'/>
  <activity role='ED nurse' action='measure heart rate, blood pressure, and oxygen saturation frequently until the patient is stabilized' objects='vital signs and oxygen saturation' id='activity10'/>
  <activity role='ED clinician' action='apply a common standardized heart failure-capable risk stratification tool at presentation' objects='common standardized multi-condition risk stratification tool including heart failure' id='activity1_2'/>
  <parallelGateway id='parallelgateway1'>
    <branch id='branch1'>
      <activity role='ED clinician and laboratory' action='obtain serum creatinine and electrolyte levels' objects='blood sample' id='activity3'/>
    </branch>
    <branch id='branch2'>
      <activity role='ED clinician and laboratory' action='obtain troponin measurements' objects='blood sample' id='activity4'/>
    </branch>
    <branch id='branch3'>
      <activity role='ED clinician and laboratory' action='obtain complete blood count' objects='blood sample' id='activity5'/>
    </branch>
    <branch id='branch4'>
      <activity role='ED clinician' action='perform electrocardiogram' objects='ECG' id='activity6'/>
    </branch>
    <branch id='branch5'>
      <activity role='radiology' action='perform chest x-ray examination' objects='chest x-ray' id='activity7'/>
    </branch>
    <branch id='branch6'>
      <exclusiveGateway id='exclusivegateway1'>
        <branch condition='no recent echocardiogram is available' id='branch7'>
          <activity role='cardiac imaging team' action='perform echocardiogram' objects='echocardiogram' id='activity8'/>
        </branch>
        <branch condition='recent echocardiogram is available' id='branch8'>
          <activity role='ED clinician' action='review recent echocardiogram' objects='recent echocardiogram result' id='activity9'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch id='branch10'>
      <activity role='ED clinician' action='assess responsiveness to diuresis and response to furosemide administered in the ED' objects='urine output, symptoms, and clinical response to furosemide' id='activity11'/>
    </branch>
  </parallelGateway>
  <activity role='ED clinician' action='review investigation results and assess high-risk markers' objects='respiratory distress, hypoxemia, severity of pulmonary edema, poor response to furosemide in the ED, hemodynamic compromise, significant arrhythmias, and troponin result' id='activity12'/>
  <inclusiveGateway id='inclusivegateway1'>
    <branch condition='respiratory distress is present' id='branch11'>
      <activity role='ED clinician' action='document respiratory distress as a high-risk marker' objects='respiratory assessment findings' id='activity13'/>
    </branch>
    <branch condition='hypoxemia is present' id='branch12'>
      <activity role='ED clinician' action='document hypoxemia as a high-risk marker' objects='oxygenation assessment findings' id='activity14'/>
    </branch>
    <branch condition='severe pulmonary edema is present' id='branch13'>
      <activity role='ED clinician' action='document severity of pulmonary edema as a high-risk marker' objects='clinical and imaging findings' id='activity15'/>
    </branch>
    <branch condition='poor response to furosemide administered in the ED is present' id='branch14'>
      <activity role='ED clinician' action='document poor response to furosemide as a high-risk marker' objects='diuresis response assessment' id='activity16'/>
    </branch>
    <branch condition='hemodynamic compromise is present' id='branch15'>
      <activity role='ED clinician' action='document hemodynamic compromise as a high-risk marker' objects='hemodynamic assessment findings' id='activity17'/>
    </branch>
    <branch condition='significant arrhythmias are present' id='branch16'>
      <activity role='ED clinician' action='document significant arrhythmias as a high-risk marker' objects='electrocardiogram findings' id='activity18'/>
    </branch>
    <branch condition='troponin is positive' id='branch17'>
      <activity role='ED clinician' action='document positive troponin as a high-risk marker' objects='troponin result' id='activity19'/>
    </branch>
    <branch condition='no high-risk markers are identified' id='branch27'>
      <activity role='ED clinician' action='document that no high-risk markers are identified' objects='risk stratification record' id='activity20'/>
    </branch>
  </inclusiveGateway>
  <activity role='ED clinician' action='calculate the acute heart failure risk score' objects='EHMRG or equivalent acute heart failure risk score' id='activity21_1'/>
  <activity role='ED clinician' action='interpret the risk score together with clinical status to support disposition planning' objects='7-day mortality risk estimate and disposition recommendation' id='activity21_2'/>
  <exclusiveGateway id='exclusivegateway2'>
    <branch condition='patient requires noninvasive or invasive ventilation and/or intensive care unit level care' id='branch19'>
      <activity role='ED clinician and critical care team' action='classify patient as high intensity' objects='risk stratification result' id='activity22'/>
      <exclusiveGateway id='exclusivegateway3'>
        <branch condition='high intensity services are available on site' id='branch20'>
          <activity role='critical care team' action='admit patient to intensive care unit with higher nurse-to-patient staffing and provide noninvasive or invasive ventilation as indicated' objects='ICU bed and ventilation support' id='activity23'/>
        </branch>
        <branch condition='high intensity services are not available on site' id='branch21'>
          <activity role='hospital transfer coordinator' action='activate the hospital transfer pathway and confirm acceptance by a higher level provider' objects='transfer request and receiving provider acceptance' id='activity24_1'/>
          <activity role='ED clinician and transport team' action='stabilize patient for transport and complete clinical handover for transfer' objects='transport readiness and transfer handover' id='activity24_2'/>
          <activity role='receiving critical care team' action='admit patient to intensive care unit with higher nurse-to-patient staffing and provide noninvasive or invasive ventilation as indicated' objects='receiving ICU bed and ventilation support' id='activity25'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='patient does not require ventilation or intensive care unit level care' id='branch22'>
      <exclusiveGateway id='exclusivegateway4'>
        <branch condition='EHMRG quintile 1 or 2 and patient responded to initial treatment in the ED and no high-risk markers are identified and there are no other considerations affecting discharge home' id='branch23'>
          <activity role='ED clinician' action='classify patient as low intensity' objects='risk stratification result' id='activity26'/>
          <activity role='ED clinician' action='treat patient in the ED or outpatient setting and discharge home without inpatient admission' objects='discharge plan' id='activity27'/>
          <activity role='ED clinician' action='arrange follow-up visit with the primary care provider within days of discharge from the ED' objects='follow-up appointment' id='activity28'/>
        </branch>
        <branch condition='EHMRG quintile 3 to 5 or patient is not eligible for discharge home or any high-risk marker is identified' id='branch24'>
          <activity role='ED clinician' action='classify patient as average intensity' objects='risk stratification result' id='activity29'/>
          <activity role='inpatient admitting team' action='admit patient to inpatient care with normal nurse-to-patient staffing' objects='inpatient bed' id='activity30'/>
        </branch>
      </exclusiveGateway>
    </branch>
  </exclusiveGateway>
</process>
