<process>
  <activity role='ED clinician' action='identify patient as acute stroke admission candidate' objects='stroke diagnosis, admission indication' id='activity1'/>
  <activity role='ED clinician' action='confirm that Module 3a applies rather than TIA or minor nondisabling stroke pathway' objects='stroke severity determination, pathway applicability' id='activity1_1'/>
  <activity role='ED clinician' action='screen for diabetes, dental issues, cognitive impairment, and depression upon ED presentation' objects='diabetes screen, dental screen, cognitive screen, depression screen' id='activity14'/>
  <activity role='ED clinician' action='document and forward screening results for the admitting stroke unit team or inpatient follow-up team' objects='screening results, admitting team follow-up information' id='activity14_1'/>
  <exclusiveGateway id='exclusivegateway1'>
    <branch condition='patient has acute stroke requiring admission' id='branch1'>
      <activity role='admitting team' action='admit patient to a specialized geographically defined stroke unit with identifiable co-located beds dedicated to stroke management' objects='stroke unit bed, admission order' id='activity2'/>
    </branch>
    <branch condition='patient has TIA or minor nondisabling stroke' id='branch2'>
      <activity role='stroke team' action='redirect patient to applicable TIA or minor stroke pathway because Module 3a recommendations are not applicable' objects='alternative pathway referral' id='activity3'/>
    </branch>
  </exclusiveGateway>
  <activity role='stroke unit leadership' action='assign core stroke unit professionals with stroke expertise' objects='medicine, nursing, occupational therapy, physiotherapy, speech-language pathology, social work, clinical nutrition' id='activity4'/>
  <activity role='stroke unit leadership' action='confirm 7-day access to the specialized interprofessional team' objects='7-day staffing coverage plan' id='activity4_1'/>
  <activity role='stroke unit leadership' action='ensure core stroke unit professionals spend the vast majority of their time treating stroke patients and regularly complete stroke care education' objects='staffing plan, stroke education records' id='activity5'/>
  <activity role='interprofessional stroke team' action='use standardized valid assessment tools to evaluate patient stroke-related impairments and functional status' objects='validated assessment tools, assessment findings' id='activity6'/>
  <activity role='interprofessional stroke team' action='document stroke-related impairments and functional status in EHR-linked records' objects='assessment findings, EHR-linked documentation' id='activity6_1'/>
  <parallelGateway id='parallelgateway1'>
    <branch id='branch4'>
      <activity role='physiotherapist' action='assess patient within 48 hours of hospital admission and formulate a management plan' objects='physiotherapy assessment, management plan' id='activity8'/>
    </branch>
    <branch id='branch5'>
      <activity role='speech-language pathologist' action='assess patient within 48 hours of hospital admission and formulate a management plan' objects='speech-language assessment, management plan' id='activity9'/>
    </branch>
    <branch id='branch6'>
      <activity role='occupational therapist' action='assess patient within 48 hours of hospital admission and formulate a management plan' objects='occupational therapy assessment, management plan' id='activity10'/>
    </branch>
    <branch id='branch7'>
      <activity role='nursing staff' action='assess venous thromboembolism risk for all stroke patients' objects='venous thromboembolism risk assessment' id='activity11'/>
    </branch>
    <branch id='branch8'>
      <activity role='nursing staff' action='assess nutrition and hydration status within the first 48 hours of admission using a valid screening tool' objects='nutrition screening tool, hydration screening tool' id='activity13'/>
    </branch>
  </parallelGateway>
  <activity role='rehabilitation team' action='complete AlphaFIM on or by day 3 after admission with target day 3 and admission day counted as day 1' objects='AlphaFIM assessment' id='activity7'/>
  <exclusiveGateway id='exclusivegateway2'>
    <branch condition='24 to 48 hours from stroke onset and no contraindication to mobilization' id='branch10'>
      <activity role='stroke unit team' action='mobilize patient early between 24 hours and 48 hours of stroke onset' objects='early mobilization intervention' id='activity15'/>
    </branch>
    <branch condition='within 24 hours of stroke onset or contraindication present' id='branch11'>
      <activity role='stroke unit team' action='avoid frequent out-of-bed activity within 24 hours of stroke onset and use clinical judgement regarding any very early mobilization' objects='clinical judgement, mobilization precautions' id='activity16'/>
    </branch>
  </exclusiveGateway>
  <activity role='stroke unit team' action='encourage adequate hydration to help prevent venous thromboembolism' objects='hydration plan' id='activity12_1'/>
  <exclusiveGateway id='exclusivegateway3'>
    <branch condition='nutritional concerns or hydration deficits or dysphagia or other comorbidities identified' id='branch12'>
      <activity role='stroke team' action='initiate dietitian referral within 7 days of admission' objects='dietitian referral, nutrition consultation request' id='activity17'/>
      <exclusiveGateway id='exclusivegateway4'>
        <branch condition='patient is unable to meet nutritional and fluid requirements' id='branch14'>
          <activity role='dietitian' action='consider enteral nutrition support' objects='enteral nutrition support assessment, nutrition recommendations' id='activity17_1'/>
        </branch>
        <branch condition='patient can meet nutritional and fluid requirements' id='branch15'>
          <activity role='dietitian' action='provide dietary recommendations without enteral nutrition support' objects='dietary recommendations' id='activity17_2'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='no nutritional concerns or hydration deficits or dysphagia or other comorbidities identified' id='branch13'>
      <activity role='stroke team' action='continue routine nutrition and hydration monitoring' objects='ongoing nutrition and hydration documentation' id='activity18'/>
    </branch>
  </exclusiveGateway>
  <activity role='stroke team' action='provide all patients with appropriate cross-continuum secondary prevention assessments and therapies whether transferred to in-hospital rehabilitation or discharged' objects='Modules 5 and 10 prevention assessments and therapies' id='activity19'/>
</process>
