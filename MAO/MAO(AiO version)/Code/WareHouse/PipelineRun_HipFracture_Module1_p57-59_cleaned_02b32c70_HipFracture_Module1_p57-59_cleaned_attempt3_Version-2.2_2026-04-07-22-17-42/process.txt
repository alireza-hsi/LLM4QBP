      <parallelGateway id='parallelgateway13_1'>
        <branch id='branch13_1'>

        <branch condition='additional consultation is needed' id='branch14'>

            </branch>

        </branch>
        <branch condition='no additional consultation is needed' id='branch17'>

    <branch condition='presenting hospital is a non-surgical hospital' id='branch10'>
