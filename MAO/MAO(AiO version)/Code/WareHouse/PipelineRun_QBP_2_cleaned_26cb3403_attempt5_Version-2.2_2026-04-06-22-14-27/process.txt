<process>
  <activity role='ED medical staff' action='Initiate standardized priority protocol for suspected TIA/minor stroke' objects='suspected TIA/minor stroke patient' id='activity1'/>
  <activity role='ED medical staff' action='Measure blood glucose' objects='patient, glucometer' id='activity2'/>
  <activity role='ED medical staff' action='Interpret blood glucose result' objects='patient, glucometer' id='activity3'/>
  <exclusiveGateway id='exclusivegateway1'>
    <branch condition='Hypoglycemia detected' id='branch1'>
      <activity role='ED medical staff' action='Correct hypoglycemia immediately' objects='patient, glucose' id='activity4'/>
    </branch>
    <branch condition='First random glucose > 11 mmol/L' id='branch2'>
      <activity role='ED medical staff' action='Repeat blood glucose measurement' objects='patient, glucometer' id='activity5'/>
    </branch>
    <branch condition='No hypoglycemia and glucose ≤ 11 mmol/L' id='branch3'>
      <activity role='ED medical staff' action='Continue routine care' objects='patient' id='activity6'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED medical staff' action='Measure body temperature' objects='patient, thermometer' id='activity7'/>
  <activity role='ED medical staff' action='Record temperature in patient chart' objects='patient, chart' id='activity8'/>
  <exclusiveGateway id='exclusivegateway2'>
    <branch condition='Temperature > 37.5°C' id='branch4'>
      <parallelGateway id='parallelgateway1'>
        <branch id='branch5'>
          <activity role='ED medical staff' action='Initiate temperature-reducing measures' objects='patient, cooling measures' id='activity9'/>
        </branch>
        <branch id='branch6'>
          <activity role='ED medical staff' action='Investigate potential infection' objects='patient, diagnostic tests' id='activity10'/>
          <exclusiveGateway id='exclusivegateway3'>
            <branch condition='Infection suspected or found' id='branch7'>
              <activity role='ED medical staff' action='Initiate antipyretic and antimicrobial therapy as required' objects='patient, medications' id='activity11'/>
            </branch>
            <branch condition='No infection suspected or found' id='branch8'>
              <activity role='ED medical staff' action='Continue monitoring and supportive care' objects='patient' id='activity12'/>
            </branch>
          </exclusiveGateway>
        </branch>
      </parallelGateway>
    </branch>
    <branch condition='Temperature ≤ 37.5°C' id='branch9'>
      <activity role='ED medical staff' action='Continue routine temperature monitoring' objects='patient' id='activity13'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED medical staff' action='Provide cross-continuum secondary prevention assessment and therapies (per Modules 5 and 10)' objects='patient' id='activity14'/>
  <activity role='ED medical staff' action='Evaluate for fluctuating/crescendo hemibody motor weakness, sensory loss, or speech difficulty' objects='patient' id='activity15'/>
  <exclusiveGateway id='exclusivegateway4'>
    <branch condition='Fluctuating/crescendo symptoms present' id='branch10'>
      <activity role='ED medical staff' action='Consider admission to hospital' objects='patient' id='activity16'/>
    </branch>
    <branch condition='No fluctuating/crescendo symptoms' id='branch11'>
      <activity role='ED medical staff' action='Continue with outpatient pathway' objects='patient' id='activity17'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED medical staff' action='Determine TIA/minor stroke risk features and timing of presentation' objects='patient, clinical assessment' id='activity18'/>
  <exclusiveGateway id='exclusivegateway5'>
    <branch condition='TIA with high-risk features or minor stroke, presentation ≤ 48 hours' id='branch12'>
      <exclusiveGateway id='exclusivegateway6'>
        <branch condition='Immediate access to specialized TIA/minor stroke clinic' id='branch13'>
          <activity role='ED medical staff' action='Arrange brain CT scan' objects='patient, CT scanner' id='activity19'/>
          <activity role='ED medical staff' action='Initiate antiplatelet therapy (if not contraindicated)' objects='patient, antiplatelet medication' id='activity20'/>
          <activity role='ED medical staff' action='Provide immediate care at specialized TIA/minor stroke clinic' objects='patient, clinic resources' id='activity21'/>
        </branch>
        <branch condition='Delay to specialized clinic poses risk' id='branch14'>
          <activity role='ED medical staff' action='Arrange brain CT scan' objects='patient, CT scanner' id='activity22'/>
          <activity role='ED medical staff' action='Initiate antiplatelet therapy (if not contraindicated)' objects='patient, antiplatelet medication' id='activity23'/>
          <activity role='ED medical staff' action='Perform evaluation and management in appropriately resourced ED; consider inpatient evaluation for stroke prevention' objects='patient, ED resources' id='activity24'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='TIA with high-risk features or minor stroke, presentation > 48 hours' id='branch15'>
      <activity role='ED medical staff' action='Refer for evaluation at specialized TIA/minor stroke clinic or outpatient clinic with stroke-prevention services within 24 hours of initial presentation' objects='patient, referral' id='activity25'/>
    </branch>
    <branch condition='TIA without high-risk features' id='branch16'>
      <activity role='ED medical staff' action='Arrange brain CT scan' objects='patient, CT scanner' id='activity26'/>
      <activity role='ED medical staff' action='Initiate antiplatelet therapy (if not contraindicated)' objects='patient, antiplatelet medication' id='activity27'/>
      <activity role='ED medical staff' action='Refer to outpatient clinic with stroke-prevention services for comprehensive evaluation and management within 1 month of symptom onset' objects='patient, referral' id='activity28'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED medical staff' action='Check if patient is on antiplatelet agent at presentation' objects='patient, medication history' id='activity29'/>
  <exclusiveGateway id='exclusivegateway7'>
    <branch condition='Not on antiplatelet agent' id='branch17'>
      <activity role='ED medical staff' action='Confirm exclusion of intracranial hemorrhage on brain imaging' objects='patient, brain imaging' id='activity30'/>
      <activity role='ED medical staff' action='Start antiplatelet therapy immediately (choose appropriate regimen)' objects='patient, ASA/clopidogrel/dipyridamole' id='activity31'/>
    </branch>
    <branch condition='Already on antiplatelet agent' id='branch18'>
      <activity role='ED medical staff' action='Continue current antiplatelet therapy as appropriate' objects='patient, current medication' id='activity32'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED medical staff' action='Assess swallowing ability for ASA administration' objects='patient, swallowing screen' id='activity33'/>
  <exclusiveGateway id='exclusivegateway8'>
    <branch condition='Swallowing screen failed' id='branch19'>
      <activity role='ED medical staff' action='Offer rectal or gastric ASA as necessary' objects='patient, rectal/gastric ASA' id='activity34'/>
    </branch>
    <branch condition='Swallowing screen passed' id='branch20'>
      <activity role='ED medical staff' action='Administer oral ASA as indicated' objects='patient, oral ASA' id='activity35'/>
    </branch>
  </exclusiveGateway>
</process>
