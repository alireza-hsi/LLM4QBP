<process>
  <activity role='ED medical staff' action='Obtain patient history' objects='patient' id='activity1-1'/>
  <activity role='ED medical staff' action='Perform physical examination' objects='patient' id='activity1-2'/>
  <activity role='ED medical staff' action='Assess neurological status' objects='patient' id='activity1-3'/>
  <activity role='ED medical staff' action='Measure blood glucose' objects='patient, glucose meter' id='activity2'/>
  <exclusiveGateway id='exclusivegateway1'>
    <branch condition='If hypoglycemia detected' id='branch2'>
      <activity role='ED medical staff' action='Correct hypoglycemia immediately' objects='patient, glucose correction protocol' id='activity3'/>
    </branch>
    <branch condition='If first random glucose value > 11 mmol/L' id='branch3'>
      <activity role='ED medical staff' action='Repeat blood glucose measurement' objects='patient, glucose meter' id='activity4'/>
    </branch>
    <branch condition='If glucose normal and ≤ 11 mmol/L' id='branch4'>
      <activity role='ED medical staff' action='Continue with standard care' objects='patient' id='activity5'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED medical staff' action='Evaluate body temperature as part of routine vital signs every 4 hours for first 48 hours' objects='patient, thermometer' id='activity6'/>
  <exclusiveGateway id='exclusivegateway2'>
    <branch condition='If temperature > 37.5°C' id='branch6'>
      <parallelGateway id='parallelgateway2'>
        <branch id='branch7'>
          <activity role='ED medical staff' action='Initiate temperature-reducing measures' objects='patient, cooling methods' id='activity7'/>
        </branch>
        <branch id='branch8'>
          <parallelGateway id='parallelgateway3'>
            <branch id='branch8-1'>
              <activity role='ED medical staff' action='Order laboratory tests for infection' objects='patient, lab requisition' id='activity8-1'/>
            </branch>
            <branch id='branch8-2'>
              <activity role='ED medical staff' action='Order imaging studies for infection source' objects='patient, imaging requisition' id='activity8-2'/>
            </branch>
          </parallelGateway>
        </branch>
        <branch id='branch9'>
          <activity role='ED medical staff' action='Initiate antipyretic and antimicrobial therapy as required' objects='patient, medications' id='activity9'/>
        </branch>
      </parallelGateway>
    </branch>
    <branch condition='If temperature ≤ 37.5°C' id='branch10'>
      <activity role='ED medical staff' action='Continue routine monitoring' objects='patient' id='activity10'/>
    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='exclusivegateway3'>
    <branch condition='If patient presents with TIA with high-risk features or minor stroke' id='branch12'>
      <activity role='ED medical staff' action='Order brain CT scan' objects='patient, CT scanner' id='activity12-1'/>
      <activity role='ED medical staff' action='Initiate antiplatelet therapy (if not contraindicated)' objects='patient, antiplatelet medication' id='activity12-2'/>
      <exclusiveGateway id='exclusivegateway4'>
        <branch condition='If immediate care available at specialized TIA/minor stroke clinic' id='branch13'>
          <activity role='ED medical staff' action='Provide immediate care at specialized TIA/minor stroke clinic' objects='patient, clinic resources' id='activity13'/>
        </branch>
        <branch condition='If delay to specialized clinic poses risk' id='branch14'>
          <activity role='ED medical staff' action='Evaluate at appropriately resourced ED and consider inpatient evaluation and management for stroke prevention' objects='patient, ED resources' id='activity14'/>
        </branch>
        <branch condition='If medical attention sought after 48 hours from symptom onset' id='branch15'>
          <activity role='ED medical staff' action='Refer for evaluation at specialized TIA/minor stroke clinic or outpatient clinic with stroke-prevention services within 24 hours of initial presentation' objects='patient, referral' id='activity15'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='If patient presents with TIA without high-risk features' id='branch16'>
      <activity role='ED medical staff' action='Order brain CT scan' objects='patient, CT scanner' id='activity16-1'/>
      <activity role='ED medical staff' action='Initiate antiplatelet therapy (if not contraindicated)' objects='patient, antiplatelet medication' id='activity16-2'/>
      <activity role='ED medical staff' action='Refer to outpatient clinic with stroke-prevention services for comprehensive evaluation and management within 1 month of symptom onset' objects='patient, referral' id='activity17'/>
    </branch>
  </exclusiveGateway>
  <inclusiveGateway id='inclusivegateway1'>
    <branch condition='If TIA patient presents with fluctuating or crescendo hemibody motor weakness, sensory loss, or speech difficulty' id='branch17'>
      <activity role='ED medical staff' action='Evaluate in ED to consider admission to hospital' objects='patient' id='activity18'/>
    </branch>
    <branch condition='If patient with TIA or minor stroke not on antiplatelet agent at presentation' id='branch18'>
      <exclusiveGateway id='exclusivegateway5'>
        <branch condition='If brain imaging excludes intracranial hemorrhage' id='branch19'>
          <inclusiveGateway id='inclusivegateway2'>
            <branch condition='If ASA appropriate' id='branch20'>
              <activity role='ED medical staff' action='Start ASA 160 mg loading dose, then enteric coated ASA (81–325 mg) daily (maintenance 81 mg/day for most)' objects='patient, ASA' id='activity19'/>
            </branch>
            <branch condition='If clopidogrel appropriate' id='branch21'>
              <activity role='ED medical staff' action='Start clopidogrel 300 mg loading dose, then 75 mg/day' objects='patient, clopidogrel' id='activity20'/>
            </branch>
            <branch condition='If extended-release dipyridamole/ASA appropriate' id='branch22'>
              <activity role='ED medical staff' action='Start extended-release dipyridamole 200 mg/ASA 25 mg bid (load with ASA 160 mg first)' objects='patient, dipyridamole, ASA' id='activity21'/>
            </branch>
          </inclusiveGateway>
        </branch>
        <branch condition='If brain imaging shows intracranial hemorrhage' id='branch23'>
          <activity role='ED medical staff' action='Do not start antiplatelet therapy; follow hemorrhage protocol' objects='patient' id='activity22'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='If patient fails swallowing screening and needs ASA' id='branch24'>
      <activity role='ED medical staff' action='Offer rectal or gastric ASA as necessary' objects='patient, rectal/gastric ASA' id='activity23'/>
    </branch>
  </inclusiveGateway>
  <activity role='ED medical staff' action='Provide cross-continuum secondary prevention assessments and therapies (Modules 5 and 10) to all patients, whether admitted or discharged' objects='patient, prevention protocol' id='activity11'/>
</process>
