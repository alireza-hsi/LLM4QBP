  <activity role='ED medical staff' action='Measure blood glucose' objects='patient' id='activity2'/>

      <activity role='ED medical staff' action='Correct hypoglycemia immediately' objects='patient' id='activity3'/>
    </branch>
    <branch condition='If first random glucose value > 11 mmol/L' id='branch1-2'>
      <activity role='ED medical staff' action='Repeat blood glucose measurement' objects='patient' id='activity4'/>
    </branch>
    <branch condition='If blood glucose is within normal range (not hypoglycemia and ≤ 11 mmol/L)' id='branch1-3'>
      <activity role='ED medical staff' action='Continue routine monitoring' objects='patient' id='activity4-1'/>
    </branch>
  </exclusiveGateway>
  
  <activity role='ED medical staff' action='Evaluate body temperature as part of routine vital signs every 4 hours for first 48 hours' objects='patient' id='activity5'/>
  <exclusiveGateway id='exclusivegateway2'>
    <branch condition='If temperature > 37.5°C' id='branch2-1'>
      <activity role='ED medical staff' action='Initiate temperature-reducing measures' objects='patient' id='activity6'/>
      <activity role='ED medical staff' action='Assess for clinical signs of infection' objects='patient' id='activity7-2'/>
      <activity role='ED medical staff' action='Order infection-related laboratory tests' objects='patient' id='activity7-1'/>
      <exclusiveGateway id='exclusivegateway3'>
        <branch condition='If infection suspected or confirmed' id='branch2-1-3-1'>
          <activity role='ED medical staff' action='Initiate antipyretic and antimicrobial therapy as required' objects='patient' id='activity8'/>
        </branch>
        <branch condition='If infection not suspected or confirmed' id='branch2-1-3-2'>
          <activity role='ED medical staff' action='Continue monitoring and supportive care' objects='patient' id='activity9-1'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='If temperature ≤ 37.5°C' id='branch2-2'>
      <activity role='ED medical staff' action='Continue routine monitoring' objects='patient' id='activity9'/>
    </branch>
  </exclusiveGateway>
  
  <exclusiveGateway id='exclusivegateway4'>
    <branch condition='TIA with high-risk features or minor stroke' id='branch3-1'>
      <activity role='ED medical staff' action='Order brain CT scan' objects='patient' id='activity11-1'/>
      <activity role='ED medical staff' action='Review CT scan results' objects='patient' id='activity11-2'/>
      <activity role='ED medical staff' action='Initiate antiplatelet therapy if no contraindication' objects='patient' id='activity11-3'/>
      <exclusiveGateway id='exclusivegateway5'>
        <branch condition='Immediate care available at specialized TIA/minor stroke clinic' id='branch3-1-1'>
          <activity role='ED medical staff' action='Provide immediate care at specialized TIA/minor stroke clinic' objects='patient' id='activity12'/>
        </branch>
        <branch condition='Delay to specialized clinic poses risk' id='branch3-1-2'>
          <activity role='ED medical staff' action='Perform evaluation at ED' objects='patient' id='activity13-1'/>
          <activity role='ED medical staff' action='Consider inpatient evaluation for stroke prevention' objects='patient' id='activity13-2'/>
        </branch>
        <branch condition='Medical attention sought after 48 hours from symptom onset' id='branch3-1-3'>
          <activity role='ED medical staff' action='Refer for evaluation at specialized TIA/minor stroke clinic or outpatient clinic with stroke-prevention services within 24 hours of initial presentation' objects='patient' id='activity14'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='TIA without high-risk features' id='branch3-2'>
      <activity role='ED medical staff' action='Order brain CT scan' objects='patient' id='activity15-1'/>
      <activity role='ED medical staff' action='Review CT scan results' objects='patient' id='activity15-2'/>
      <activity role='ED medical staff' action='Initiate antiplatelet therapy if no contraindication' objects='patient' id='activity15-3'/>
      <activity role='ED medical staff' action='Refer to outpatient clinic with stroke-prevention services for comprehensive evaluation and management within 1 month of symptom onset' objects='patient' id='activity16'/>
    </branch>
  </exclusiveGateway>
  
  <activity role='ED medical staff' action='Evaluate TIA patients with fluctuating or crescendo hemibody motor weakness, sensory loss (≥2 contiguous segments), or speech difficulty in ED for possible hospital admission' objects='patient' id='activity17'/>
  
  <exclusiveGateway id='exclusivegateway6'>
    <branch condition='Patient not on antiplatelet agent at presentation and brain imaging excludes intracranial hemorrhage' id='branch4-1'>
      <exclusiveGateway id='exclusivegateway8'>
        <branch condition='ASA selected' id='branch4-1-1'>
          <activity role='ED medical staff' action='Start ASA 160 mg loading dose, then enteric coated ASA (81–325 mg) daily (maintenance 81 mg/day for most)' objects='patient' id='activity18'/>
        </branch>
        <branch condition='Clopidogrel selected' id='branch4-1-2'>
          <activity role='ED medical staff' action='Start clopidogrel 300 mg loading dose, then 75 mg/day' objects='patient' id='activity19'/>
        </branch>
        <branch condition='Extended-release dipyridamole/ASA selected' id='branch4-1-3'>
          <activity role='ED medical staff' action='Start extended-release dipyridamole 200 mg/ASA 25 mg bid (load with ASA 160 mg first)' objects='patient' id='activity20'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='Patient already on antiplatelet agent or contraindicated' id='branch4-2'>
      <activity role='ED medical staff' action='Continue current therapy or consider alternatives as appropriate' objects='patient' id='activity21'/>
    </branch>
  </exclusiveGateway>
  
  <exclusiveGateway id='exclusivegateway7'>

  </exclusiveGateway>
  
  <activity role='ED medical staff' action='Provide cross-continuum secondary prevention assessments and therapies (Modules 5 and 10)' objects='patient' id='activity10'/>
