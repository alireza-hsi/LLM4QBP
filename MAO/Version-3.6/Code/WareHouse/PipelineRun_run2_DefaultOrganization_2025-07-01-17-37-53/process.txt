<process>
  <activity role='triage nurse or physician' action='Assess airway, breathing, and circulation (ABC)' objects='patient' id='activity1'/>
  <activity role='clinician' action='Perform neurological examination to determine focal deficits and assess stroke severity using NIHSS or CNS' objects='patient' id='activity2'/>
  <inclusiveGateway id='inclusivegateway1'>
    <branch condition='Patient presents with potentially disabling, acute neurological symptoms suggestive of acute stroke within 6 hours of symptom onset and may be eligible for treatment within 12 hours' id='branch1'>
      <activity role='clinician' action='Consider patient for endovascular therapy if clinical and imaging criteria are met' objects='patient' id='activity3'/>
    </branch>
    <branch condition='All patients presenting within acute stroke treatment time window' id='branch2'>
      <parallelGateway id='parallelgateway1'>
        <branch id='branch2-1'>
          <activity role='radiology team' action='Perform non-contrast CT or MRI' objects='patient, imaging equipment' id='activity4-1'/>
          <activity role='radiology team' action='Perform CTA' objects='patient, imaging equipment' id='activity4-2'/>
          <activity role='qualified health care provider' action='Interpret brain imaging immediately' objects='CT/CTA/MRI results' id='activity5'/>
        </branch>
        <branch id='branch2-2'>
          <activity role='clinical staff and healthcare expert' action='Order and interpret blood work (CBC, electrolytes, creatinine, glucose, INR, partial thromboplastin time, troponin test if clinically indicated)' objects='patient, lab requisition, blood work results' id='activity9'/>
        </branch>
      </parallelGateway>
      <exclusiveGateway id='exclusivegateway1'>
        <branch condition='Patient is clinically eligible for acute ischemic stroke treatments based on initial imaging and clinical assessment' id='exbranch1'>
          <activity role='radiology team' action='Perform advanced CT imaging including CT, multiphase CTA, or Perfusion CT' objects='patient, imaging equipment' id='activity6'/>
        </branch>
        <branch condition='Patient is not clinically eligible for acute ischemic stroke treatments' id='exbranch2'>
          <activity role='clinical staff' action='Continue with standard stroke care pathway' objects='patient' id='activity6-2'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='All patients' id='branch3'>
      <parallelGateway id='parallelgateway2'>
        <branch id='branch3-1'>
          <activity role='clinical staff' action='Perform ECG to detect atrial fibrillation and other acute arrhythmias (do not delay acute treatment)' objects='patient, ECG machine' id='activity7'/>
        </branch>
        <branch id='branch3-2'>
          <activity role='clinical staff' action='Order chest x-ray (do not delay assessment for thrombolysis)' objects='patient, x-ray machine' id='activity8'/>
        </branch>
      </parallelGateway>
    </branch>
    <branch condition='All patients with stroke' id='branch4'>
      <activity role='clinical staff' action='Make patient NPO initially and screen swallowing ability using a simple, valid, reliable bedside protocol before oral medication, fluid, or food' objects='patient, screening protocol' id='activity10'/>
    </branch>
    <branch condition='Patients with stroke not alert within first 24 hours' id='branch5'>
      <activity role='clinical staff' action='Monitor closely and screen swallowing ability when clinically appropriate' objects='patient' id='activity11'/>
    </branch>
    <branch condition='Patients with stroke presenting with features indicating dysphagia or pulmonary aspiration' id='branch6'>
      <activity role='speech-language pathologist or trained specialist' action='Conduct full clinical assessment of swallowing ability and advise on required consistency of diet and fluids' objects='patient' id='activity12'/>
    </branch>
    <branch condition='Patients presenting beyond 48 hours with TIA or ischemic stroke' id='branch8'>
      <activity role='radiology team' action='Conduct vascular imaging of brain and neck arteries as soon as possible' objects='patient, imaging equipment' id='activity14'/>
    </branch>
    <branch condition='Patients with ICH' id='branch9'>
      <activity role='stroke physician' action='Recognize ICH promptly and evaluate immediately' objects='patient' id='activity15'/>
      <activity role='clinical staff' action='Ask about anticoagulant therapy, measure platelet count, PTT, and INR' objects='patient, lab results' id='activity16'/>
      <activity role='radiology team' action='Perform CTA or MRA as soon as possible for select patients' objects='patient, imaging equipment' id='activity17'/>
    </branch>
  </inclusiveGateway>
  <activity role='clinical staff' action='Provide appropriate cross-continuum secondary prevention assessments and therapies' objects='patient, prevention protocol' id='activity13'/>
</process>
