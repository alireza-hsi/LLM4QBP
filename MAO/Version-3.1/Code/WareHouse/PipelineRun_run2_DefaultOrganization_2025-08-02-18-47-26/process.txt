<process>
  <activity role='ED staff' action='Initiate standardized priority protocol for suspected TIA/minor stroke patient and admit to stroke unit within first few hours if needed' objects='suspected TIA/minor stroke patient' id='activity1'/>
  <activity role='ED staff' action='Measure blood glucose on arrival' objects='patient, glucose meter' id='activity2'/>
  <exclusiveGateway id='exclusivegateway1'>
    <branch condition='If hypoglycemia detected' id='branch1'>
      <exclusiveGateway id='exclusivegateway2'>
        <branch condition='If patient is conscious' id='branch2'>
          <activity role='ED staff' action='Administer rapid-acting glucose' objects='patient, oral glucose' id='activity3'/>
        </branch>
        <branch condition='If patient is unconscious or unable to swallow' id='branch3'>
          <activity role='ED staff' action='Administer IV dextrose' objects='patient, IV dextrose' id='activity4'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='If first random glucose > 11 mmol/L' id='branch4'>
      <activity role='ED staff' action='Repeat blood glucose measurement' objects='patient, glucose meter' id='activity5'/>
    </branch>
    <branch condition='If glucose normal' id='branch5'>
      <activity role='ED staff' action='Continue with routine care' objects='patient' id='activity6'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED staff' action='Evaluate body temperature as part of routine vital signs every 4 hours for first 48 hours' objects='patient, thermometer' id='activity7'/>
  <exclusiveGateway id='exclusivegateway3'>
    <branch condition='If temperature > 37.5°C' id='branch6'>
      <parallelGateway id='parallelgateway1'>
        <branch id='branch7'>
          <activity role='ED staff' action='Initiate temperature-reducing measures' objects='patient, cooling protocol' id='activity8'/>
        </branch>
        <branch id='branch8'>
          <activity role='ED staff' action='Investigate potential infection' objects='patient, infection workup' id='activity9'/>
          <exclusiveGateway id='exclusivegateway4'>
            <branch condition='If infection suspected or confirmed' id='branch9'>
              <activity role='ED staff' action='Initiate antipyretic and antimicrobial therapy as required' objects='patient, antipyretics, antimicrobials' id='activity10'/>
            </branch>
            <branch condition='If no infection' id='branch10'>
              <activity role='ED staff' action='Continue monitoring temperature' objects='patient' id='activity11'/>
            </branch>
          </exclusiveGateway>
        </branch>
      </parallelGateway>
    </branch>
    <branch condition='If temperature ≤ 37.5°C' id='branch11'>
      <activity role='ED staff' action='Continue routine temperature monitoring' objects='patient' id='activity12'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED staff' action='Assess for high-risk features (including fluctuating/crescendo hemibody motor weakness, sensory loss, or speech difficulty), minor stroke, or other relevant symptoms' objects='patient, clinical assessment' id='activity23'/>
  <exclusiveGateway id='exclusivegateway9'>
    <branch condition='If TIA with high-risk features (including fluctuating/crescendo symptoms) or minor stroke' id='branch22'>
      <activity role='ED staff' action='Order urgent brain CT scan' objects='patient, CT scanner' id='activity13'/>
      <activity role='ED staff' action='Transport patient to CT scanner' objects='patient, transport team' id='activity14'/>
      <activity role='ED staff' action='Perform brain CT scan' objects='patient, CT scanner' id='activity15'/>
      <activity role='Radiologist' action='Interpret brain CT scan immediately' objects='CT images' id='activity16'/>
      <exclusiveGateway id='exclusivegateway10'>
        <branch condition='If immediate care at specialized TIA/minor stroke clinic is available' id='branch26'>
          <activity role='ED staff' action='Refer for immediate care at specialized TIA/minor stroke clinic' objects='patient, referral protocol' id='activity24'/>
        </branch>
        <branch condition='If delay to specialized clinic poses risk' id='branch27'>
          <activity role='ED staff' action='Evaluate in ED and consider inpatient management for stroke prevention' objects='patient, evaluation protocol' id='activity30'/>
        </branch>
      </exclusiveGateway>
      <exclusiveGateway id='exclusivegateway5'>
        <branch condition='If intracranial hemorrhage excluded' id='branch12'>
          <exclusiveGateway id='exclusivegateway6'>
            <branch condition='If patient not on antiplatelet agent at presentation' id='branch13'>
              <activity role='ED staff' action='Select appropriate antiplatelet agent based on patient profile' objects='patient, antiplatelet options' id='activity17'/>
              <activity role='ED staff' action='Administer loading dose of selected antiplatelet agent' objects='patient, selected antiplatelet' id='activity18'/>
              <exclusiveGateway id='exclusivegateway7'>
                <branch condition='If patient fails swallowing screening' id='branch14'>
                  <activity role='ED staff' action='Offer rectal or gastric ASA as necessary' objects='patient, rectal/gastric ASA' id='activity19'/>
                </branch>
                <branch condition='If patient passes swallowing screening' id='branch15'>
                  <activity role='ED staff' action='Administer oral antiplatelet therapy' objects='patient, oral antiplatelet' id='activity20'/>
                </branch>
              </exclusiveGateway>
            </branch>
            <branch condition='If patient already on antiplatelet agent' id='branch16'>
              <activity role='ED staff' action='Continue current antiplatelet therapy as appropriate' objects='patient, medication list' id='activity21'/>
            </branch>
          </exclusiveGateway>
        </branch>
        <branch condition='If intracranial hemorrhage present' id='branch17'>
          <activity role='ED staff' action='Manage according to hemorrhagic stroke protocol' objects='patient, hemorrhagic stroke protocol' id='activity22'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='If TIA without high-risk features' id='branch23'>
      <activity role='ED staff' action='Order brain CT scan and initiate antiplatelet therapy as soon as possible and no later than 24 hours after initial presentation' objects='patient, CT scanner, antiplatelet' id='activity29'/>
      <activity role='ED staff' action='Refer to outpatient clinic with stroke-prevention services for comprehensive evaluation and management within 1 month of symptom onset' objects='patient, referral protocol' id='activity25'/>
    </branch>
    <branch condition='If medical attention sought after 48 hours from symptom onset' id='branch25'>
      <activity role='ED staff' action='Refer for evaluation at specialized TIA/minor stroke clinic or outpatient clinic with stroke-prevention services within 24 hours of initial presentation' objects='patient, referral protocol' id='activity27'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED staff' action='Provide cross-continuum secondary prevention assessments and therapies as per Modules 5 and 10' objects='patient, prevention protocol' id='activity28'/>
</process>
