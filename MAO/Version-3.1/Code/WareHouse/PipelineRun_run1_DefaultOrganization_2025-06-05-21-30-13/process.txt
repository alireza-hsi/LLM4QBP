<process>
  <activity role='hospital admission staff' action='Verify patient eligibility for admission to stroke unit' objects='patient, eligibility criteria' id='activity1-1'/>
  <activity role='hospital admission staff' action='Assign patient to available co-located stroke unit bed' objects='patient, stroke unit beds' id='activity1-2'/>
  <activity role='hospital administration' action='Ensure core interprofessional stroke team is present (medicine, nursing, OT, PT, SLP, social work, dietitian)' objects='staff roster, stroke unit' id='activity2'/>
  <activity role='hospital administration' action='Ensure stroke unit team has stroke expertise and ongoing education' objects='training records, staff' id='activity3'/>
  <activity role='hospital administration' action='Ensure interprofessional stroke team is available 7 days a week' objects='staff schedule' id='activity4'/>
  <activity role='PT' action='Conduct physiotherapy assessment within 48 hours of admission' objects='patient, assessment tools' id='activity5-1'/>
  <activity role='OT' action='Conduct occupational therapy assessment within 48 hours of admission' objects='patient, assessment tools' id='activity5-2'/>
  <activity role='SLP' action='Conduct speech-language pathology assessment within 48 hours of admission' objects='patient, assessment tools' id='activity5-3'/>
  <activity role='PT, OT, SLP' action='Formulate management plan based on assessments' objects='patient, assessment results' id='activity5-4'/>
  <activity role='clinicians' action='Use standardized, valid assessment tools to evaluate stroke-related impairments and functional status' objects='assessment tools, patient' id='activity7'/>
  <activity role='clinicians' action='Complete AlphaFIM on or by day 3 after admission (admission day is day 1)' objects='AlphaFIM, patient' id='activity8'/>
  <activity role='clinicians' action='Assess all stroke patients for risk of developing venous thromboembolism' objects='patient, risk assessment tool' id='activity9'/>
  <activity role='clinical team' action='Encourage early mobilization and adequate hydration to prevent venous thromboembolism' objects='patient' id='activity10'/>
  <activity role='clinicians' action='Assess nutrition and hydration status within first 48 hours of admission using valid screening tool' objects='patient, screening tool' id='activity11'/>
  <exclusiveGateway id='exclusivegateway1'>
    <branch condition='Nutritional concerns, hydration deficits, dysphagia, or comorbidities present' id='branch1-1'>
      <activity role='clinicians' action='Refer patient to dietitian within 7 days of admission for recommendations and consideration of enteral nutrition support if unable to meet requirements' objects='patient, referral form' id='activity12'/>
    </branch>
    <branch condition='No nutritional concerns, hydration deficits, dysphagia, or comorbidities' id='branch1-2'>
      <activity role='clinicians' action='Continue routine monitoring' objects='patient' id='activity13'/>
    </branch>
  </exclusiveGateway>
  <activity role='clinical team' action='Avoid frequent out-of-bed activity within first 24 hours' objects='patient' id='activity14'/>
  <exclusiveGateway id='exclusivegateway2'>
    <branch condition='Early mobilization is contraindicated' id='branch2-1'>
      <activity role='clinical team' action='Document contraindication for early mobilization' objects='patient, medical record' id='activity15'/>
    </branch>
    <branch condition='Early mobilization is not contraindicated' id='branch2-2'>
      <activity role='clinical team' action='Mobilize patient between 24-48 hours of stroke onset' objects='patient' id='activity16'/>
    </branch>
  </exclusiveGateway>
  <activity role='clinical team' action='Conduct secondary prevention assessment as per Modules 5 and 10 before discharge or transfer' objects='patient, prevention protocol' id='activity17'/>
  <activity role='clinical team' action='Initiate secondary prevention therapies as per Modules 5 and 10 before discharge or transfer' objects='patient, prevention protocol' id='activity18'/>
</process>
