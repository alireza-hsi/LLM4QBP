<process>
  <activity role='EMS' action='Divert suspected TIA/stroke patient to regional/district stroke centre if feasible' objects='suspected TIA/stroke patient' id='activity1'/>
  <activity role='EMS' action='Prenotify receiving hospital about suspected TIA/stroke patient arrival' objects='patient information' id='activity2'/>
  <activity role='Hospital acute stroke team' action='Receive prenotification and prepare for patient arrival' objects='prenotification' id='activity3'/>
  <activity role='Medical staff' action='Conduct rapid initial evaluation for airway, breathing, and circulation' objects='patient' id='activity4'/>
  <activity role='Medical staff' action='Conduct neurological examination and assess stroke severity (NIHSS or CNS)' objects='patient' id='activity5'/>
  <parallelGateway id='parallelgateway1'>
    <branch id='branch1'>
      <activity role='Medical staff' action='Perform immediate brain imaging (CT or MRI)' objects='patient' id='activity6'/>
      <activity role='Healthcare provider with imaging expertise' action='Interpret brain imaging immediately' objects='CT or MRI results' id='activity7'/>
    </branch>
    <branch id='branch2'>
      <activity role='Medical staff' action='Perform ECG to detect atrial fibrillation and other acute arrhythmias' objects='patient' id='activity8'/>
    </branch>
    <branch id='branch3'>
      <exclusiveGateway id='exclusivegateway1'>
        <branch condition='Troponin test not clinically indicated' id='branch3-1'>
          <activity role='Medical staff' action='Order standard blood work (CBC, electrolytes, creatinine, glucose, INR, partial thromboplastin time)' objects='patient blood samples' id='activity9-1'/>
        </branch>
        <branch condition='Troponin test clinically indicated' id='branch3-2'>
          <activity role='Medical staff' action='Order standard blood work (CBC, electrolytes, creatinine, glucose, INR, partial thromboplastin time)' objects='patient blood samples' id='activity9-2'/>
          <activity role='Medical staff' action='Order troponin test' objects='patient blood samples' id='activity9-3'/>
        </branch>
      </exclusiveGateway>
      <activity role='Healthcare provider' action='Interpret blood work results immediately upon availability' objects='blood work results' id='activity10'/>
    </branch>
  </parallelGateway>
  <activity role='Medical staff' action='Order chest x-ray if indicated, ensuring it does not delay assessment for thrombolysis' objects='patient' id='activity11'/>
  <activity role='Medical staff' action='Make patient NPO initially and screen swallowing ability using valid bedside protocol before oral intake' objects='patient' id='activity12'/>
  <exclusiveGateway id='exclusivegateway2'>
    <branch condition='Patient not alert within first 24 hours' id='branch4-1'>
      <activity role='Medical staff' action='Monitor closely and screen swallowing ability when clinically appropriate' objects='patient' id='activity13'/>
    </branch>
    <branch condition='Patient presents with features of dysphagia or pulmonary aspiration' id='branch4-2'>
      <activity role='Speech-language pathologist or trained specialist' action='Perform full clinical assessment of swallowing ability and advise on diet/fluid consistency' objects='patient' id='activity14'/>
    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='exclusivegateway3'>
    <branch condition='Patient has TIA or minor (nondisabling) stroke' id='branch5-1'>
      <exclusiveGateway id='exclusivegateway4'>
        <branch condition='Presentation within 48 hours of symptom onset or fluctuating motor/speech symptoms' id='branch6-1'>
          <activity role='Medical staff' action='Perform immediate vascular imaging of neck arteries (carotid ultrasound, CTA, or MRA) unless not a candidate for revascularization' objects='patient' id='activity15'/>
          <activity role='Healthcare provider with imaging expertise' action='Interpret vascular imaging immediately' objects='vascular imaging results' id='activity16'/>
        </branch>
        <branch condition='Presentation beyond 48 hours' id='branch6-2'>
          <activity role='Medical staff' action='Perform vascular imaging of brain and neck arteries as soon as possible' objects='patient' id='activity17'/>
          <activity role='Healthcare provider with imaging expertise' action='Interpret vascular imaging results immediately upon availability' objects='vascular imaging results' id='activity18'/>
        </branch>
      </exclusiveGateway>
      <activity role='Medical staff' action='Use standardized triage tool to determine appropriate care location for TIA patient' objects='patient, triage tool' id='activity19'/>
    </branch>
    <branch condition='Patient does not have TIA or minor (nondisabling) stroke' id='branch5-2'>
      <activity role='Medical staff' action='Continue with appropriate stroke pathway (recommendations not applicable)' objects='patient' id='activity20'/>
    </branch>
  </exclusiveGateway>
  <activity role='Medical staff' action='Conduct secondary prevention assessment' objects='patient' id='activity21-1'/>
  <activity role='Medical staff' action='Initiate secondary prevention therapies' objects='patient' id='activity21-2'/>
</process>
