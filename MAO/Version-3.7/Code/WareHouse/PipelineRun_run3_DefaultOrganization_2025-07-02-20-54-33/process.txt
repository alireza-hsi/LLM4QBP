        <branch condition='Patient is not eligible for acute ischemic stroke treatments' id='branch2-ischemic'>

    </branch>

      <activity role='ED/Clinic/Acuity Care Staff' action='Order and collect blood work: CBC, electrolytes, creatinine, glucose, INR, partial thromboplastin time, troponin (if clinically indicated)' objects='patient' id='activity11-bloodwork'/>

    <branch id='branch4-vascular'>

        <branch condition='Patient does not meet criteria' id='branch2-vascular'>
          <activity role='ED/Clinic/Acuity Care Staff' action='No additional vascular imaging required at this stage' objects='patient' id='activity13-novascimaging'/>

      </exclusiveGateway>
    </branch>
