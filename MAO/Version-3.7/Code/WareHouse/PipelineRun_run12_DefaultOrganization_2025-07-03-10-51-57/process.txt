        <branch id='branch1-1'>
          <activity role='clinical staff' action='Order non-contrast CT or MRI' objects='suspected acute stroke patient' id='activity3'/>
        </branch>
        <branch id='branch1-2'>
          <activity role='clinical staff' action='Order CTA' objects='suspected acute stroke patient' id='activity4'/>
        </branch>
        <branch id='branch1-3'>
          <activity role='clinical staff' action='Order blood work (CBC, electrolytes, creatinine, glucose, INR, partial thromboplastin time, troponin if indicated)' objects='suspected stroke patient' id='activity6'/>

        <branch id='branch1-4'>

      <exclusiveGateway id='exclusivegateway5'>
        <branch condition='Chest x-ray is clinically indicated' id='branch1-7'>
          <activity role='clinical staff' action='Order chest x-ray (should not delay assessment for thrombolysis)' objects='suspected stroke patient' id='activity8'/>
        </branch>
        <branch condition='Chest x-ray not indicated' id='branch1-8'>
          <activity role='clinical staff' action='Continue without chest x-ray' objects='suspected stroke patient' id='activity24'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='Patient presents beyond 48 hours with TIA or ischemic stroke' id='branch2'>

    </branch>

      <exclusiveGateway id='exclusivegateway2'>
        <branch condition='Select ICH patient as per clinical criteria' id='branch3-1'>
          <activity role='clinical staff' action='Order CTA or MRA as soon as possible' objects='ICH patient' id='activity20'/>
        </branch>
        <branch condition='Other ICH patients' id='branch3-2'>
          <activity role='clinical staff' action='Continue with standard ICH management' objects='ICH patient' id='activity21'/>
        </branch>
      </exclusiveGateway>
    </branch>
  </inclusiveGateway>
  <activity role='clinical staff' action='Make patient NPO initially and screen swallowing ability using valid bedside protocol before oral medication, fluid, or food' objects='stroke patient' id='activity12'/>
  <exclusiveGateway id='exclusivegateway3'>
    <branch condition='Patient is not alert within first 24 hours' id='branch4-1'>
      <activity role='clinical staff' action='Monitor closely and screen swallowing ability when clinically appropriate' objects='stroke patient' id='activity13'/>
    </branch>
    <branch condition='Patient is alert or after 24 hours' id='branch4-2'>
      <activity role='clinical staff' action='Proceed with further swallowing assessment as indicated' objects='stroke patient' id='activity14'/>
    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='exclusivegateway4'>

    </branch>
    <branch condition='No features of dysphagia or aspiration' id='branch5-2'>
      <activity role='clinical staff' action='Continue standard care' objects='stroke patient' id='activity16'/>
    </branch>
