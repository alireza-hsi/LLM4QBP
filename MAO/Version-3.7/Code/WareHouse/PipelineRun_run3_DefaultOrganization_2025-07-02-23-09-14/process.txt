<process>
  <activity role='ED medical staff' action='Obtain patient history, perform physical examination, and review clinical records' objects='patient, clinical records' id='activity1'/>
  <activity role='ED medical staff' action='Measure blood glucose' objects='patient, glucometer' id='activity2'/>
  <exclusiveGateway id='exclusivegateway1'>
    <branch condition='Hypoglycemia detected' id='branch1'>
      <activity role='ED medical staff' action='Correct hypoglycemia immediately' objects='patient, glucose administration' id='activity3'/>
      <activity role='ED medical staff' action='Proceed to body temperature assessment' objects='patient' id='activity4'/>
    </branch>
    <branch condition='First random glucose > 11 mmol/L' id='branch2'>
      <activity role='ED medical staff' action='Repeat blood glucose measurement' objects='patient, glucometer' id='activity5'/>
      <activity role='ED medical staff' action='Proceed to body temperature assessment' objects='patient' id='activity6'/>
    </branch>
    <branch condition='Normoglycemia' id='branch3'>
      <activity role='ED medical staff' action='Proceed to body temperature assessment' objects='patient' id='activity7'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED medical staff' action='Measure body temperature as part of routine vital signs every 4 hours for first 48 hours' objects='patient, thermometer' id='activity8'/>
  <exclusiveGateway id='exclusivegateway2'>
    <branch condition='Temperature > 37.5C' id='branch4'>
      <parallelGateway id='parallelgateway1'>
        <branch id='branch5'>
          <activity role='ED medical staff' action='Initiate temperature-reducing measures' objects='patient, cooling methods' id='activity9'/>
        </branch>
        <branch id='branch6'>
          <activity role='ED medical staff' action='Investigate potential infection' objects='patient, laboratory requisition' id='activity10'/>
        </branch>
        <branch id='branch7'>
          <activity role='ED medical staff' action='Initiate antipyretic and antimicrobial therapy as required' objects='patient, medications' id='activity11'/>
        </branch>
      </parallelGateway>
    </branch>
    <branch condition='Temperature ≤ 37.5C' id='branch8'>
      <activity role='ED medical staff' action='Continue routine monitoring' objects='patient' id='activity12'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED medical staff' action='Risk stratification for TIA/minor stroke (high-risk features, minor stroke, or TIA without high-risk features)' objects='patient, clinical criteria' id='activity13'/>
  <exclusiveGateway id='exclusivegateway5'>
    <branch condition='TIA patient with fluctuating/crescendo hemibody motor weakness, sensory loss, or speech difficulty' id='branch14'>
      <activity role='ED medical staff' action='Evaluate in ED for consideration of hospital admission' objects='patient, admission criteria' id='activity22'/>
    </branch>
    <branch condition='TIA patient without above features' id='branch15'>
      <activity role='ED medical staff' action='Proceed with standard management pathway' objects='patient' id='activity23'/>
    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='exclusivegateway3'>
    <branch condition='TIA with high-risk features or minor stroke' id='branch9'>
      <activity role='ED medical staff' action='Order brain CT scan' objects='patient, CT scanner' id='activity14'/>
      <activity role='ED medical staff' action='Initiate antiplatelet therapy (if not contraindicated)' objects='patient, antiplatelet agents' id='activity15'/>
      <exclusiveGateway id='exclusivegateway4'>
        <branch condition='Immediate access to specialized TIA/minor stroke clinic' id='branch10'>
          <activity role='ED medical staff' action='Provide immediate care at specialized TIA/minor stroke clinic' objects='patient, clinic resources' id='activity16'/>
        </branch>
        <branch condition='Delay to specialized clinic poses risk' id='branch11'>
          <activity role='ED medical staff' action='Evaluate at appropriately resourced ED and consider inpatient evaluation and management for stroke prevention' objects='patient, ED resources' id='activity17'/>
        </branch>
        <branch condition='Medical attention sought after 48 hours from symptom onset' id='branch12'>
          <activity role='ED medical staff' action='Refer for evaluation at specialized TIA/minor stroke clinic or outpatient clinic with stroke-prevention services within 24 hours of initial presentation' objects='patient, referral documentation' id='activity18'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='TIA without high-risk features' id='branch13'>
      <activity role='ED medical staff' action='Order brain CT scan' objects='patient, CT scanner' id='activity19'/>
      <activity role='ED medical staff' action='Initiate antiplatelet therapy (if not contraindicated)' objects='patient, antiplatelet agents' id='activity20'/>
      <activity role='ED medical staff' action='Refer to outpatient clinic with stroke-prevention services for comprehensive evaluation and management within 1 month of symptom onset' objects='patient, referral documentation' id='activity21'/>
    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='exclusivegateway6'>
    <branch condition='Patient not on antiplatelet agent at presentation and brain imaging excludes intracranial hemorrhage' id='branch16'>
      <inclusiveGateway id='inclusivegateway1'>
        <branch condition='ASA selected' id='branch17'>
          <activity role='ED medical staff' action='Administer ASA 160 mg loading dose, then enteric coated ASA (81–325 mg) daily (maintenance dose 81 mg/day for most)' objects='patient, ASA' id='activity24'/>
        </branch>
        <branch condition='Clopidogrel selected' id='branch18'>
          <activity role='ED medical staff' action='Administer clopidogrel 300 mg loading dose, then 75 mg/day' objects='patient, clopidogrel' id='activity25'/>
        </branch>
        <branch condition='Extended-release dipyridamole/ASA selected' id='branch19'>
          <activity role='ED medical staff' action='Administer extended-release dipyridamole 200 mg/ASA 25 mg bid (load with ASA 160 mg first)' objects='patient, dipyridamole/ASA' id='activity26'/>
        </branch>
      </inclusiveGateway>
    </branch>
    <branch condition='Patient already on antiplatelet agent or contraindicated' id='branch20'>
      <activity role='ED medical staff' action='Continue or adjust antiplatelet therapy as clinically indicated' objects='patient, medication chart' id='activity27'/>
    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='exclusivegateway7'>
    <branch condition='Patient fails swallowing screening' id='branch21'>
      <activity role='ED medical staff' action='Offer rectal or gastric ASA as necessary' objects='patient, rectal/gastric ASA' id='activity28'/>
    </branch>
    <branch condition='Patient passes swallowing screening' id='branch22'>
      <activity role='ED medical staff' action='Administer oral antiplatelet therapy as indicated' objects='patient, oral medication' id='activity29'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED medical staff' action='Initiate cross-continuum secondary prevention assessments and therapies (Modules 5 and 10)' objects='patient, prevention protocol' id='activity30'/>
</process>
