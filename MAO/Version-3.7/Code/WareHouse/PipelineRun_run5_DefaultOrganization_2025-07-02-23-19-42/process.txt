<process>
  <activity role='ED medical staff' action='Obtain patient history' objects='patient' id='activity1-1'/>
  <activity role='ED medical staff' action='Perform physical examination' objects='patient' id='activity1-2'/>
  <activity role='ED medical staff' action='Assess neurological status' objects='patient' id='activity1-3'/>
  <activity role='ED medical staff' action='Measure blood glucose' objects='patient, glucometer' id='activity2'/>
  <activity role='ED medical staff' action='Evaluate body temperature as part of routine vital signs' objects='patient, thermometer' id='activity3'/>
  <exclusiveGateway id='exclusivegateway1'>
    <branch condition='Hypoglycemia detected' id='branch3'>
      <activity role='ED medical staff' action='Correct hypoglycemia immediately' objects='patient, glucose administration' id='activity4'/>
    </branch>
    <branch condition='First random glucose value > 11 mmol/L' id='branch4'>
      <activity role='ED medical staff' action='Repeat blood glucose measurement' objects='patient, glucometer' id='activity5'/>
    </branch>
    <branch condition='Normoglycemia' id='branch5'>
      <activity role='ED medical staff' action='Proceed to next assessment' objects='patient' id='activity6'/>
    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='exclusivegateway2'>
    <branch condition='Temperature > 37.5 C' id='branch6'>
      <parallelGateway id='parallelgateway1'>
        <branch id='branch7'>
          <activity role='ED medical staff' action='Initiate temperature-reducing measures' objects='patient, cooling methods' id='activity7'/>
        </branch>
        <branch id='branch8'>
          <activity role='ED medical staff' action='Order laboratory tests for infection' objects='patient, lab requisition' id='activity8'/>
        </branch>
        <branch id='branch9'>
          <activity role='ED medical staff' action='Order imaging studies for infection' objects='patient, imaging requisition' id='activity9'/>
        </branch>
      </parallelGateway>
      <exclusiveGateway id='exclusivegateway7'>
        <branch condition='Infection confirmed by laboratory or imaging studies' id='branch24'>
          <activity role='ED medical staff' action='Initiate antipyretic and antimicrobial therapy as required' objects='patient, medications' id='activity10'/>
        </branch>
        <branch condition='No infection confirmed' id='branch25'>
          <activity role='ED medical staff' action='Continue routine monitoring' objects='patient' id='activity30'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='Temperature ≤ 37.5 C' id='branch11'>
      <activity role='ED medical staff' action='Continue routine monitoring' objects='patient' id='activity11'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED medical staff' action='Assess for TIA or minor (nondisabling) stroke and risk features' objects='patient' id='activity12'/>
  <exclusiveGateway id='exclusivegateway3'>
    <branch condition='TIA with high-risk features or minor stroke' id='branch12'>
      <activity role='ED medical staff' action='Order brain CT scan' objects='patient, CT scanner' id='activity13'/>
      <activity role='ED medical staff' action='Initiate antiplatelet therapy if not contraindicated' objects='patient, antiplatelet agents' id='activity14'/>
      <exclusiveGateway id='exclusivegateway4'>
        <branch condition='Immediate care available at specialized TIA/minor stroke clinic' id='branch13'>
          <activity role='ED medical staff' action='Refer patient to specialized TIA/minor stroke clinic for rapid evaluation and treatment' objects='patient, referral documentation' id='activity15'/>
        </branch>
        <branch condition='Delay to specialized clinic poses risk' id='branch14'>
          <activity role='ED medical staff' action='Perform evaluation in ED' objects='patient, ED resources' id='activity16'/>
          <activity role='ED medical staff' action='Consider inpatient evaluation for stroke prevention' objects='patient, inpatient resources' id='activity17'/>
        </branch>
        <branch condition='Medical attention sought after 48 hours from symptom onset' id='branch15'>
          <activity role='ED medical staff' action='Refer for evaluation at specialized TIA/minor stroke clinic or outpatient clinic with stroke-prevention services within 24 hours of initial presentation' objects='patient, referral documentation' id='activity18'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='TIA without high-risk features' id='branch16'>
      <activity role='ED medical staff' action='Order brain CT scan' objects='patient, CT scanner' id='activity19'/>
      <activity role='ED medical staff' action='Initiate antiplatelet therapy if not contraindicated' objects='patient, antiplatelet agents' id='activity20'/>
      <activity role='ED medical staff' action='Refer to outpatient clinic with stroke-prevention services for comprehensive evaluation and management within 1 month of symptom onset' objects='patient, referral documentation' id='activity21'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED medical staff' action='Evaluate TIA patients with fluctuating/crescendo hemibody motor weakness, sensory loss, or speech difficulty in ED for possible hospital admission' objects='patient' id='activity22'/>
  <exclusiveGateway id='exclusivegateway5'>
    <branch condition='Patient not on antiplatelet agent at presentation and brain imaging has excluded intracranial hemorrhage' id='branch17'>
      <inclusiveGateway id='inclusivegateway1'>
        <branch condition='ASA selected' id='branch18'>
          <activity role='ED medical staff' action='Administer ASA 160 mg loading dose, then enteric coated ASA (81–325 mg) daily (maintenance 81 mg/day for most)' objects='patient, ASA' id='activity23'/>
        </branch>
        <branch condition='Clopidogrel selected' id='branch19'>
          <activity role='ED medical staff' action='Administer clopidogrel 300 mg loading dose, then 75 mg/day' objects='patient, clopidogrel' id='activity24'/>
        </branch>
        <branch condition='Extended-release dipyridamole/ASA selected' id='branch20'>
          <activity role='ED medical staff' action='Administer extended-release dipyridamole 200 mg/ASA 25 mg bid (load with ASA 160 mg first)' objects='patient, dipyridamole, ASA' id='activity25'/>
        </branch>
      </inclusiveGateway>
    </branch>
    <branch condition='Patient already on antiplatelet agent or contraindication present' id='branch21'>
      <activity role='ED medical staff' action='Continue or adjust antiplatelet therapy as clinically indicated' objects='patient, medication record' id='activity26'/>
    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='exclusivegateway6'>
    <branch condition='Patient fails swallowing screening' id='branch22'>
      <activity role='ED medical staff' action='Offer rectal or gastric ASA as necessary' objects='patient, rectal/gastric ASA' id='activity27'/>
    </branch>
    <branch condition='Patient passes swallowing screening' id='branch23'>
      <activity role='ED medical staff' action='Administer oral antiplatelet therapy as indicated' objects='patient, oral antiplatelet agents' id='activity28'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED medical staff' action='Provide cross-continuum secondary prevention assessments and therapies (see Modules 5 and 10)' objects='patient' id='activity29'/>
</process>
