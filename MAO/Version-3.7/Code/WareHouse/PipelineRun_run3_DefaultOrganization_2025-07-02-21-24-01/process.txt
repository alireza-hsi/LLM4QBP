<process>
  <activity role='ED medical staff' action='Obtain patient history' objects='patient' id='activity1-1'/>
  <activity role='ED medical staff' action='Perform physical examination' objects='patient' id='activity1-2'/>
  <activity role='ED medical staff' action='Order and collect blood works (CBC, electrolytes, creatinine, glucose, INR, partial thromboplastin time, troponin, infection labs)' objects='patient, blood collection supplies' id='activity2-1'/>
  <activity role='ED medical staff' action='Interpret all blood test results immediately' objects='patient, blood test results' id='activity2-2'/>
  <exclusiveGateway id='exclusivegateway1a'>
    <branch condition='If hypoglycemia detected' id='branch1a'>
      <activity role='ED medical staff' action='Correct hypoglycemia immediately' objects='patient, glucose correction agents' id='activity3'/>
      <activity role='ED medical staff' action='Recheck blood glucose after correction' objects='patient' id='activity3a'/>
      <exclusiveGateway id='exclusivegateway1b_after_hypo'>
        <branch condition='If repeat glucose > 11 mmol/L' id='branch1b_after_hypo'>
          <activity role='ED medical staff' action='Repeat blood glucose measurement' objects='patient' id='activity4_after_hypo'/>
        </branch>
        <branch condition='If repeat glucose ≤ 11 mmol/L' id='branch2b_after_hypo'>
          <activity role='ED medical staff' action='Continue standard care' objects='patient' id='activity5_after_hypo'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='If no hypoglycemia detected' id='branch2a'>
      <exclusiveGateway id='exclusivegateway1b'>
        <branch condition='If first random glucose > 11 mmol/L' id='branch1b'>
          <activity role='ED medical staff' action='Repeat blood glucose measurement' objects='patient' id='activity4'/>
        </branch>
        <branch condition='If first random glucose ≤ 11 mmol/L' id='branch2b'>
          <activity role='ED medical staff' action='Continue standard care' objects='patient' id='activity5'/>
        </branch>
      </exclusiveGateway>
    </branch>
  </exclusiveGateway>
  <activity role='ED medical staff' action='Evaluate body temperature as part of routine vital signs every 4 hours for first 48 hours' objects='patient, thermometer' id='activity6'/>
  <exclusiveGateway id='exclusivegateway2'>
    <branch condition='If temperature > 37.5°C' id='branch4'>
      <parallelGateway id='parallelgateway1'>
        <branch id='branch5'>
          <activity role='ED medical staff' action='Initiate temperature-reducing measures' objects='patient, cooling methods' id='activity7'/>
        </branch>
        <branch id='branch6'>
          <activity role='ED medical staff' action='Initiate antipyretic and antimicrobial therapy as required' objects='patient, antipyretics, antimicrobials' id='activity8'/>
        </branch>
      </parallelGateway>
    </branch>
    <branch condition='If temperature ≤ 37.5°C' id='branch7'>
      <activity role='ED medical staff' action='Continue routine monitoring' objects='patient' id='activity9'/>
    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='exclusivegateway3'>
    <branch condition='TIA with high-risk features or minor stroke' id='branch8'>
      <activity role='ED medical staff' action='Order brain CT scan' objects='patient, CT scanner' id='activity10'/>
      <activity role='ED medical staff' action='Review CT scan for intracranial hemorrhage' objects='patient, CT scan result' id='activity11'/>
      <exclusiveGateway id='exclusivegateway4'>
        <branch condition='No intracranial hemorrhage and not on antiplatelet at presentation and patient can swallow' id='branch9'>
          <activity role='ED medical staff' action='Initiate antiplatelet therapy immediately (choose one: ASA 160 mg loading then 81–325 mg daily, clopidogrel 300 mg loading then 75 mg daily, or extended-release dipyridamole 200 mg/ASA 25 mg bid with ASA 160 mg loading)' objects='patient, antiplatelet agents' id='activity12'/>
        </branch>
        <branch condition='No intracranial hemorrhage and not on antiplatelet at presentation and patient cannot swallow' id='branch10'>
          <activity role='ED medical staff' action='Offer rectal or gastric ASA as necessary' objects='patient, rectal/gastric ASA' id='activity13'/>
        </branch>
        <branch condition='Already on antiplatelet or contraindication' id='branch11'>
          <activity role='ED medical staff' action='Continue or adjust therapy as appropriate' objects='patient, medication chart' id='activity14'/>
        </branch>
      </exclusiveGateway>
      <exclusiveGateway id='exclusivegateway5'>
        <branch condition='Immediate access to specialized TIA/minor stroke clinic' id='branch12'>
          <activity role='ED medical staff' action='Provide immediate care at specialized TIA/minor stroke clinic' objects='patient, clinic resources' id='activity15'/>
        </branch>
        <branch condition='Delay to specialized clinic poses risk' id='branch13'>
          <activity role='ED medical staff' action='Perform evaluation in ED and consider inpatient evaluation and management for stroke prevention' objects='patient, ED resources' id='activity16'/>
        </branch>
        <branch condition='Medical attention sought after 48 hours from symptom onset' id='branch14'>
          <activity role='ED medical staff' action='Refer for evaluation at specialized TIA/minor stroke clinic or outpatient clinic with stroke-prevention services within 24 hours of initial presentation' objects='patient, referral' id='activity17'/>
        </branch>
      </exclusiveGateway>
      <exclusiveGateway id='exclusivegateway6'>
        <branch condition='TIA patient with fluctuating or crescendo hemibody motor weakness, sensory loss, or speech difficulty' id='branch15'>
          <activity role='ED medical staff' action='Evaluate in ED for possible hospital admission' objects='patient' id='activity18'/>
        </branch>
        <branch condition='Other cases' id='branch16'>
          <activity role='ED medical staff' action='Proceed with standard management' objects='patient' id='activity19'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='TIA without high-risk features' id='branch17'>
      <activity role='ED medical staff' action='Order brain CT scan' objects='patient, CT scanner' id='activity20'/>
      <activity role='ED medical staff' action='Review CT scan for intracranial hemorrhage' objects='patient, CT scan result' id='activity21'/>
      <exclusiveGateway id='exclusivegateway7'>
        <branch condition='No intracranial hemorrhage and not on antiplatelet at presentation and patient can swallow' id='branch18'>
          <activity role='ED medical staff' action='Initiate antiplatelet therapy immediately (choose one: ASA 160 mg loading then 81–325 mg daily, clopidogrel 300 mg loading then 75 mg daily, or extended-release dipyridamole 200 mg/ASA 25 mg bid with ASA 160 mg loading)' objects='patient, antiplatelet agents' id='activity22'/>
        </branch>
        <branch condition='No intracranial hemorrhage and not on antiplatelet at presentation and patient cannot swallow' id='branch19'>
          <activity role='ED medical staff' action='Offer rectal or gastric ASA as necessary' objects='patient, rectal/gastric ASA' id='activity23'/>
        </branch>
        <branch condition='Already on antiplatelet or contraindication' id='branch20'>
          <activity role='ED medical staff' action='Continue or adjust therapy as appropriate' objects='patient, medication chart' id='activity24'/>
        </branch>
      </exclusiveGateway>
      <activity role='ED medical staff' action='Refer to outpatient clinic with stroke-prevention services for comprehensive evaluation and management within 1 month of symptom onset' objects='patient, referral' id='activity25'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED medical staff' action='Initiate cross-continuum secondary prevention assessments and therapies (per Modules 5 and 10)' objects='patient' id='activity26'/>
</process>
