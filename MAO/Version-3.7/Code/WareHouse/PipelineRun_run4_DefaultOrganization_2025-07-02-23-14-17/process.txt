          <activity role='ED medical staff' action='Repeat blood glucose measurement' objects='patient' id='activity4'/>
        </branch>
        <branch condition='Normoglycemia and first glucose ≤ 11 mmol/L' id='branch4'>

        </branch>
      </exclusiveGateway>
    </branch>

        </branch>
        <branch condition='Temperature ≤ 37.5°C' id='branch10'>
          <activity role='ED medical staff' action='Continue routine monitoring of temperature' objects='patient' id='activity10'/>
        </branch>
      </exclusiveGateway>
    </branch>
  </parallelGateway>
  <activity role='ED medical staff' action='Provide cross-continuum secondary prevention assessments and therapies (Modules 5 and 10)' objects='patient' id='activity11'/>

    </branch>
    <branch condition='No fluctuating/crescendo symptoms' id='branch19'>
      <exclusiveGateway id='exclusivegateway8'>

            <branch condition='TIA with high-risk features or minor stroke' id='branch13'>
              <activity role='ED medical staff' action='Arrange brain CT scan' objects='patient' id='activity13-1'/>
              <activity role='ED medical staff' action='Initiate antiplatelet therapy (if not contraindicated)' objects='patient' id='activity13-2'/>
              <exclusiveGateway id='exclusivegateway4'>
                <branch condition='Immediate access to specialized TIA/minor stroke clinic' id='branch14'>
                  <activity role='ED medical staff' action='Provide immediate care at specialized TIA/minor stroke clinic' objects='patient' id='activity14'/>
                </branch>
                <branch condition='Delay to specialized clinic poses risk' id='branch15'>
                  <activity role='ED medical staff' action='Perform evaluation and management at appropriately resourced ED; consider inpatient evaluation for stroke prevention' objects='patient' id='activity15'/>
                </branch>
                <branch condition='Medical attention sought after 48 hours from symptom onset' id='branch16'>
                  <activity role='ED medical staff' action='Refer for evaluation at specialized TIA/minor stroke clinic or outpatient clinic with stroke-prevention services within 24 hours of initial presentation' objects='patient' id='activity16'/>
                </branch>
              </exclusiveGateway>
            </branch>
            <branch condition='TIA without high-risk features' id='branch17'>
              <activity role='ED medical staff' action='Arrange brain CT scan' objects='patient' id='activity17-1'/>

              <activity role='ED medical staff' action='Refer to outpatient clinic with stroke-prevention services for comprehensive evaluation and management within 1 month of symptom onset' objects='patient' id='activity18'/>
            </branch>
          </exclusiveGateway>
          <exclusiveGateway id='exclusivegateway7'>
            <branch condition='Swallowing screening failed' id='branch21'>
              <activity role='ED medical staff' action='Offer rectal or gastric ASA as necessary' objects='patient' id='activity23'/>
            </branch>
            <branch condition='Swallowing screening passed' id='branch22'>
              <activity role='ED medical staff' action='Administer oral antiplatelet therapy (ASA, clopidogrel, or extended-release dipyridamole/ASA as per protocol)' objects='patient' id='activity24'/>
            </branch>
          </exclusiveGateway>
        </branch>
        <branch condition='Patient already on antiplatelet agent at presentation' id='branch23'>
          <activity role='ED medical staff' action='Assess for OHTAC recommendations for TIA/minor stroke management' objects='patient' id='activity12-2'/>
          <exclusiveGateway id='exclusivegateway9'>
            <branch condition='TIA with high-risk features or minor stroke' id='branch25'>

              <activity role='ED medical staff' action='Continue current antiplatelet therapy as appropriate' objects='patient' id='activity25-2'/>
              <exclusiveGateway id='exclusivegateway10'>

                  <activity role='ED medical staff' action='Refer for evaluation at specialized TIA/minor stroke clinic or outpatient clinic with stroke-prevention services within 24 hours of initial presentation' objects='patient' id='activity28'/>
                </branch>
              </exclusiveGateway>
            </branch>

              <activity role='ED medical staff' action='Refer to outpatient clinic with stroke-prevention services for comprehensive evaluation and management within 1 month of symptom onset' objects='patient' id='activity29-3'/>
            </branch>

        </branch>
      </exclusiveGateway>
    </branch>
  </exclusiveGateway>
