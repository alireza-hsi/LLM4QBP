<process>
  <activity role='triage nurse or ED staff' action='Perform ABC (airway, breathing, circulation) assessment' objects='patient' id='activity1'/>
  <activity role='clinical staff' action='Perform neurological examination and assess stroke severity using NIHSS or CNS' objects='patient' id='activity2'/>
  <exclusiveGateway id='exclusivegateway1'>
    <branch condition='Patient presents with potentially disabling, acute neurological symptoms suggestive of acute stroke within 6.0 hours of symptom onset and may be eligible for endovascular therapy if treatment can be initiated within 12 hours' id='branch1-1'>
      <activity role='clinical staff' action='Assess EVT eligibility based on clinical and imaging criteria' objects='patient, clinical criteria, imaging criteria' id='activity3'/>
    </branch>
    <branch condition='Patient presents with suspected acute stroke within the treatment time window but is not eligible for EVT' id='branch1-2'>
      <exclusiveGateway id='exclusivegateway2'>
        <branch condition='Non-contrast CT indicated' id='branch1-2-1'>
          <activity role='clinical staff' action='Order non-contrast CT' objects='patient, imaging equipment' id='activity4'/>
        </branch>
        <branch condition='MRI indicated' id='branch1-2-2'>
          <activity role='clinical staff' action='Order MRI' objects='patient, imaging equipment' id='activity5'/>
        </branch>
      </exclusiveGateway>
      <activity role='clinical staff' action='Order CTA if indicated' objects='patient, imaging equipment' id='activity6'/>
    </branch>
    <branch condition='Patient is a clinically eligible acute ischemic stroke patient but presents outside the acute treatment time window' id='branch1-3'>
      <activity role='clinical staff' action='For clinically eligible acute ischemic stroke patients, consider advanced CT imaging (CT, multiphase CTA, or Perfusion CT)' objects='patient, imaging equipment' id='activity7'/>
    </branch>
  </exclusiveGateway>
  <parallelGateway id='parallelgateway1'>
    <branch id='branch2'>
      <activity role='clinical staff' action='Perform ECG to detect atrial fibrillation and other acute arrhythmias (do not delay acute treatment)' objects='patient, ECG machine' id='activity8'/>
    </branch>
    <branch id='branch3'>
      <activity role='clinical staff' action='Order chest x-ray if indicated (should not delay assessment for thrombolysis)' objects='patient, x-ray equipment' id='activity9'/>
    </branch>
    <branch id='branch4'>
      <activity role='clinical staff' action='Order blood work (CBC, electrolytes, creatinine, glucose, INR, partial thromboplastin time, and troponin if clinically indicated)' objects='patient, lab requisition' id='activity10'/>
    </branch>
    <branch id='branch5'>
      <activity role='clinical staff' action='Make all stroke patients NPO initially and screen swallowing ability using a simple, valid, reliable bedside protocol before oral medication, fluid, or food' objects='patient, screening protocol' id='activity11'/>
    </branch>
    <branch id='branch6'>
      <activity role='clinical staff' action='Monitor patients with stroke who are not alert within first 24 hours and screen swallowing ability when clinically appropriate' objects='patient' id='activity12'/>
    </branch>
    <branch id='branch7'>
      <activity role='speech-language pathologist or trained specialist' action='For patients with features indicating dysphagia or pulmonary aspiration, perform full clinical swallowing assessment and advise on diet/fluid consistency' objects='patient' id='activity13'/>
    </branch>
  </parallelGateway>
  <activity role='stroke physician or radiologist' action='Interpret brain imaging (CT/CTA/MRI) immediately' objects='imaging results' id='activity14'/>
  <exclusiveGateway id='exclusivegateway3'>
    <branch condition='Patient presenting beyond 48 hours with TIA or ischemic stroke' id='branch8'>
      <activity role='clinical staff' action='Order vascular imaging of brain and neck arteries as soon as possible' objects='patient, imaging equipment' id='activity15'/>
    </branch>
    <branch condition='Patient with acute ICH' id='branch9'>
      <parallelGateway id='parallelgateway2'>
        <branch id='branch9-1'>
          <activity role='stroke physician' action='Recognize ICH as medical emergency and evaluate immediately' objects='patient' id='activity16'/>
        </branch>
        <branch id='branch9-2'>
          <activity role='clinical staff' action='Evaluate for anticoagulant therapy, measure platelet count, PTT, and INR' objects='patient, lab results' id='activity17'/>
        </branch>
        <branch id='branch9-3'>
          <activity role='clinical staff' action='For select ICH patients, order CTA or MRA as soon as possible' objects='patient, imaging equipment' id='activity18'/>
        </branch>
      </parallelGateway>
    </branch>
    <branch condition='Other patients' id='branch10'>
      <activity role='clinical staff' action='Continue with standard stroke assessment and management' objects='patient' id='activity19'/>
    </branch>
  </exclusiveGateway>
  <activity role='clinical staff' action='Provide cross-continuum secondary prevention assessments and therapies to all patients (admitted or discharged)' objects='patient, prevention protocols' id='activity20'/>
