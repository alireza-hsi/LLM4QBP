<process>
  <activity role='triage nurse or physician' action='Perform ABC (airway, breathing, circulation) assessment' objects='patient' id='activity1-ABC'/>
  <activity role='clinician' action='Perform neurological examination to determine focal neurological deficits and assess stroke severity using standardized stroke scale (NIHSS or CNS)' objects='patient' id='activity2-neuro'/>
  <exclusiveGateway id='exclusivegateway1-initialAssessment'>
    <branch condition='Patient presents with potentially disabling, acute neurological symptoms suggestive of acute stroke within 6.0 hours of symptom onset and may be considered for treatment within 12-hour window, meeting clinical and imaging criteria' id='branch1-EVT'>
      <activity role='clinician' action='Consider patient for endovascular therapy (EVT)' objects='patient' id='activity3-EVT'/>
    </branch>
    <branch condition='Patient does not meet criteria for EVT consideration' id='branch2-nonEVT'>
      <activity role='clinician' action='Continue standard acute stroke assessment' objects='patient' id='activity4-nonEVT'/>
    </branch>
  </exclusiveGateway>
  <parallelGateway id='parallelgateway1-diagnostics'>
    <branch id='branch3-imaging'>
      <activity role='radiology team' action='Perform non-contrast CT or MRI' objects='patient' id='activity5-1-noncontrastCT'/>
      <activity role='radiology team' action='Perform CTA' objects='patient' id='activity5-2-CTA'/>
      <exclusiveGateway id='exclusivegateway2-advancedImaging'>
        <branch condition='Patient is clinically eligible for acute ischemic stroke treatments' id='branch4-advancedCT'>
          <activity role='radiology team' action='Perform advanced CT imaging including CT and multiphase CTA or Perfusion CT' objects='patient' id='activity6-advancedCT'/>
        </branch>
        <branch condition='Patient is not eligible for acute ischemic stroke treatments' id='branch5-skipAdvancedCT'>
          <activity role='radiology team' action='Skip advanced CT imaging' objects='patient' id='activity7-skipAdvancedCT'/>
        </branch>
      </exclusiveGateway>
      <activity role='stroke physician or radiologist' action='Interpret brain imaging immediately' objects='CT/CTA/MRI results' id='activity8-interpretImaging'/>
    </branch>
    <branch id='branch6-ECG'>
      <activity role='clinician' action='Order and perform ECG to detect atrial fibrillation and other acute arrhythmias (should not delay acute treatment)' objects='patient' id='activity9-ECG'/>
    </branch>
    <branch id='branch7-chestXray'>
      <activity role='clinician' action='Order chest x-ray (should not delay assessment for thrombolysis)' objects='patient' id='activity10-chestXray'/>
    </branch>
    <branch id='branch8-bloodwork'>
      <activity role='lab staff' action='Order, collect, and analyze blood work (CBC, electrolytes, creatinine, glucose, INR, partial thromboplastin time, troponin if indicated)' objects='patient' id='activity11-bloodwork'/>
    </branch>
  </parallelGateway>
  <exclusiveGateway id='exclusivegateway5-TIAorIschemic'>
    <branch condition='Patient presenting beyond 48 hours with TIA or ischemic stroke' id='branch14-TIAbeyond48'>
      <activity role='radiology team' action='Perform vascular imaging of brain and neck arteries as soon as possible' objects='patient' id='activity18-vascularImaging'/>
    </branch>
    <branch condition='Patient presenting within 48 hours or not TIA/ischemic stroke' id='branch15-noTIAbeyond48'>
      <activity role='clinician' action='Continue standard care' objects='patient' id='activity19-standardCareTIA'/>
    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='exclusivegateway6-ICH'>
    <branch condition='Patient has ICH (intracerebral hemorrhage)' id='branch16-ICH'>
      <activity role='stroke physician' action='Recognize ICH as medical emergency and evaluate immediately' objects='patient' id='activity20-ICHemergency'/>
      <activity role='stroke physician' action='Evaluate for anticoagulant therapy, measure platelet count, PTT, and INR' objects='patient' id='activity21-ICHlabs'/>
      <exclusiveGateway id='exclusivegateway7-ICHimaging'>
        <branch condition='Select ICH patient criteria met' id='branch17-ICHimaging'>
          <activity role='radiology team' action='Perform CTA or MRA as soon as possible' objects='patient' id='activity22-ICHCTA'/>
        </branch>
        <branch condition='Select ICH patient criteria not met' id='branch18-noICHimaging'>
          <activity role='stroke physician' action='Continue standard ICH management' objects='patient' id='activity23-ICHstandard'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='Patient does not have ICH' id='branch19-noICH'>
      <activity role='clinician' action='Continue standard acute stroke assessment' objects='patient' id='activity24-nonICH'/>
    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='exclusivegateway8-dysphagiaEligibility'>
    <branch condition='Patient has confirmed stroke diagnosis' id='branch20-dysphagiaEligible'>
      <activity role='nurse or clinician' action='Make all patients with stroke NPO initially and screen swallowing ability using a simple, valid, reliable bedside protocol before initiating oral medication, fluid, or food' objects='patient' id='activity25-dysphagiaScreen'/>
      <exclusiveGateway id='exclusivegateway9-alertness'>
        <branch condition='Patient is not alert within first 24 hours' id='branch21-notAlert'>
          <activity role='nurse or clinician' action='Monitor closely and screen swallowing ability when clinically appropriate' objects='patient' id='activity26-monitorSwallow'/>
        </branch>
        <branch condition='Patient is alert within first 24 hours' id='branch22-alert'>
          <activity role='nurse or clinician' action='Proceed with standard care' objects='patient' id='activity27-standardSwallow'/>
        </branch>
      </exclusiveGateway>
      <exclusiveGateway id='exclusivegateway10-dysphagiaFeatures'>
        <branch condition='Patient presents with features indicating dysphagia or pulmonary aspiration' id='branch23-dysphagiaFeatures'>
          <activity role='speech-language pathologist or trained specialist' action='Perform full clinical assessment of swallowing ability and advise on swallowing and required consistency of diet and fluids' objects='patient' id='activity28-fullSwallowAssessment'/>
        </branch>
        <branch condition='No features of dysphagia or pulmonary aspiration' id='branch24-noDysphagiaFeatures'>
          <activity role='nurse or clinician' action='Continue monitoring as per protocol' objects='patient' id='activity29-monitoringSwallow'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='Patient does not have confirmed stroke diagnosis' id='branch25-noDysphagiaEligible'>
      <activity role='clinician' action='Proceed with standard care' objects='patient' id='activity30-noDysphagiaScreen'/>
    </branch>
  </exclusiveGateway>
  <activity role='clinician' action='Provide all patients, whether admitted or discharged, with appropriate cross-continuum secondary prevention assessments and therapies (see Modules 5 and 10)' objects='patient' id='activity17-secondaryPrevention'/>
</process>
