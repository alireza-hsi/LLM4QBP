  <parallelGateway id='parallelgateway1'>
    <branch id='branch1'>
      <activity role='health care provider' action='Perform neurological examination and assess stroke severity using standardized stroke scale (NIHSS or CNS)' objects='patient' id='activity2'/>
    </branch>
    <branch id='branch2'>
      <activity role='health care provider and health care provider with imaging expertise' action='Order and perform brain imaging (CT or MRI) and interpret results immediately' objects='patient, CT or MRI results' id='activity3'/>
    </branch>
    <branch id='branch3'>
      <activity role='health care provider' action='Perform ECG to detect atrial fibrillation and other acute arrhythmias' objects='patient' id='activity4'/>

    <branch id='branch4'>

    <branch id='branch5'>
      <activity role='health care provider' action='Order and collect blood work (CBC, electrolytes, creatinine, glucose, INR, partial thromboplastin time, troponin if indicated)' objects='patient' id='activity6'/>
    </branch>
  </parallelGateway>
  <exclusiveGateway id='exclusivegateway1'>
    <branch condition='TIA or minor (nondisabling) stroke within 48 hours of symptom onset or with fluctuating motor/speech symptoms' id='branch6-1'>

      <activity role='health care provider' action='Order vascular imaging of brain and neck arteries as soon as possible' objects='patient' id='activity8'/>

  </exclusiveGateway>

    </branch>
    <branch condition='No features of dysphagia or pulmonary aspiration' id='branch8-2'>
      <activity role='health care provider' action='Continue standard care' objects='patient' id='activity14'/>
    </branch>
  </exclusiveGateway>
