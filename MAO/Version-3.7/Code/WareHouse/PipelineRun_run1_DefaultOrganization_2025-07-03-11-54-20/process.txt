      <exclusiveGateway id='exclusivegateway4'>
        <branch condition='Immediate care available at specialized TIA/minor stroke clinic' id='branch14'>

      </exclusiveGateway>
    </branch>
    <branch condition='TIA without high-risk features' id='branch17'>
