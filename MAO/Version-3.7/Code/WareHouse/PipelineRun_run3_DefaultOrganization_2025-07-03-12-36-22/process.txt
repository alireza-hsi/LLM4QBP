<process>
  <activity role='ED medical staff' action='Initial assessment of patient with suspected TIA or minor (nondisabling) stroke' objects='patient' id='activity1'/>
  <activity role='ED medical staff' action='Measure blood glucose' objects='patient' id='activity2'/>
  <exclusiveGateway id='exclusivegateway1'>
    <branch condition='Hypoglycemia detected' id='branch1'>
      <activity role='ED medical staff' action='Correct hypoglycemia immediately' objects='patient' id='activity3'/>
    </branch>
    <branch condition='First random glucose > 11 mmol/L' id='branch2'>
      <activity role='ED medical staff' action='Repeat blood glucose measurement' objects='patient' id='activity4'/>
    </branch>
    <branch condition='No hypoglycemia and glucose ≤ 11 mmol/L' id='branch3'>
      <activity role='ED medical staff' action='Continue with standard care' objects='patient' id='activity5'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED medical staff' action='Evaluate body temperature as part of routine vital signs every 4 hours for first 48 hours' objects='patient' id='activity6'/>
  <exclusiveGateway id='exclusivegateway2'>
    <branch condition='Temperature > 37.5°C' id='branch4'>
      <activity role='ED medical staff' action='Initiate temperature-reducing measures' objects='patient' id='activity7'/>
      <activity role='ED medical staff' action='Investigate potential infection' objects='patient' id='activity8'/>
      <activity role='ED medical staff' action='Initiate antipyretic and antimicrobial therapy as required' objects='patient' id='activity9'/>
    </branch>
    <branch condition='Temperature ≤ 37.5°C' id='branch5'>
      <activity role='ED medical staff' action='Continue routine monitoring' objects='patient' id='activity10'/>
    </branch>
  </exclusiveGateway>
  <exclusiveGateway id='exclusivegateway3'>
    <branch condition='TIA with high-risk features or minor stroke' id='branch6'>
      <activity role='ED medical staff' action='Arrange brain CT scan' objects='patient' id='activity11'/>
      <exclusiveGateway id='exclusivegateway4'>
        <branch condition='Intracranial hemorrhage excluded and patient not on antiplatelet agent at presentation' id='branch7'>
          <activity role='ED medical staff' action='Select antiplatelet regimen' objects='patient' id='activity20'/>
          <activity role='ED medical staff' action='Administer selected antiplatelet regimen' objects='patient' id='activity21'/>
        </branch>
        <branch condition='Intracranial hemorrhage excluded and patient already on antiplatelet agent at presentation' id='branch8'>
          <activity role='ED medical staff' action='Continue current antiplatelet therapy as appropriate' objects='patient' id='activity24'/>
        </branch>
        <branch condition='Intracranial hemorrhage detected' id='branch9'>
          <activity role='ED medical staff' action='Do not initiate antiplatelet therapy; manage per hemorrhagic stroke protocol' objects='patient' id='activity25'/>
        </branch>
      </exclusiveGateway>
      <exclusiveGateway id='exclusivegateway5'>
        <branch condition='Immediate access to specialized TIA/minor stroke clinic' id='branch10'>
          <activity role='ED medical staff' action='Provide immediate care at specialized TIA/minor stroke clinic' objects='patient' id='activity13'/>
        </branch>
        <branch condition='Delay to specialized clinic poses risk' id='branch11'>
          <activity role='ED medical staff' action='Perform evaluation in appropriately resourced ED and consider inpatient evaluation and management for stroke prevention' objects='patient' id='activity14'/>
        </branch>
        <branch condition='Medical attention sought after 48 hours from symptom onset' id='branch12'>
          <activity role='ED medical staff' action='Refer for evaluation at specialized TIA/minor stroke clinic or outpatient clinic with stroke-prevention services within 24 hours of initial presentation' objects='patient' id='activity15'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='TIA without high-risk features' id='branch13'>
      <activity role='ED medical staff' action='Arrange brain CT scan' objects='patient' id='activity16'/>
      <exclusiveGateway id='exclusivegateway6'>
        <branch condition='Intracranial hemorrhage excluded and patient not on antiplatelet agent at presentation' id='branch14'>
          <activity role='ED medical staff' action='Select antiplatelet regimen' objects='patient' id='activity26'/>
          <activity role='ED medical staff' action='Administer selected antiplatelet regimen' objects='patient' id='activity27'/>
        </branch>
        <branch condition='Intracranial hemorrhage excluded and patient already on antiplatelet agent at presentation' id='branch15'>
          <activity role='ED medical staff' action='Continue current antiplatelet therapy as appropriate' objects='patient' id='activity28'/>
        </branch>
        <branch condition='Intracranial hemorrhage detected' id='branch16'>
          <activity role='ED medical staff' action='Do not initiate antiplatelet therapy; manage per hemorrhagic stroke protocol' objects='patient' id='activity29'/>
        </branch>
      </exclusiveGateway>
      <activity role='ED medical staff' action='Refer to outpatient clinic with stroke-prevention services for comprehensive evaluation and management within 1 month of symptom onset' objects='patient' id='activity18'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED medical staff' action='Provide cross-continuum secondary prevention assessments and therapies (Modules 5 and 10) to all patients, whether admitted or discharged' objects='patient' id='activity23'/>
  <inclusiveGateway id='inclusivegateway1'>
    <branch condition='TIA patient presents with fluctuating or crescendo hemibody motor weakness, sensory loss, or speech difficulty' id='branch17'>
      <activity role='ED medical staff' action='Evaluate in ED for consideration of hospital admission' objects='patient' id='activity19'/>
    </branch>
    <branch condition='Patient fails swallowing screening' id='branch18'>
      <activity role='ED medical staff' action='Offer rectal or gastric ASA as necessary' objects='patient' id='activity22'/>
    </branch>
  </inclusiveGateway>
</process>
