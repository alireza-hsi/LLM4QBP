<process>
  <activity role='ED medical staff' action='Initial assessment of patient with suspected TIA or minor (nondisabling) stroke' objects='patient' id='activity1'/>
  <exclusiveGateway id='exclusivegateway1'>
    <branch condition='TIA patient with fluctuating/crescendo hemibody motor weakness, sensory loss, or speech difficulty' id='branch1'>
      <activity role='ED medical staff' action='Evaluate in ED for possible hospital admission' objects='patient' id='activity2'/>
    </branch>
    <branch condition='No fluctuating/crescendo symptoms' id='branch2'>
      <activity role='ED medical staff' action='Measure blood glucose' objects='patient, glucometer' id='activity3'/>
      <exclusiveGateway id='exclusivegateway2'>
        <branch condition='Hypoglycemia detected' id='branch3'>
          <activity role='ED medical staff' action='Correct hypoglycemia immediately' objects='patient, glucose' id='activity4'/>
        </branch>
        <branch condition='First random glucose value > 11 mmol/L' id='branch4'>
          <activity role='ED medical staff' action='Repeat blood glucose measurement' objects='patient, glucometer' id='activity5'/>
        </branch>
        <branch condition='Normoglycemia' id='branch5'>
          <activity role='ED medical staff' action='Continue with standard care' objects='patient' id='activity6'/>
        </branch>
      </exclusiveGateway>
      <activity role='ED medical staff' action='Evaluate body temperature as part of routine vital signs every 4 hours for first 48 hours' objects='patient, thermometer' id='activity7'/>
      <exclusiveGateway id='exclusivegateway3'>
        <branch condition='Temperature > 37.5°C' id='branch6'>
          <activity role='ED medical staff' action='Initiate temperature-reducing measures' objects='patient, antipyretics, cooling devices' id='activity8'/>
          <activity role='ED medical staff' action='Order laboratory tests for infection' objects='patient, laboratory tests' id='activity9'/>
          <activity role='ED medical staff' action='Review laboratory test results' objects='patient, laboratory results' id='activity10'/>
          <activity role='ED medical staff' action='Initiate antipyretic and antimicrobial therapy as required' objects='patient, medications' id='activity11'/>
        </branch>
        <branch condition='Temperature ≤ 37.5°C' id='branch7'>
          <activity role='ED medical staff' action='Continue routine monitoring' objects='patient' id='activity12'/>
        </branch>
      </exclusiveGateway>
      <exclusiveGateway id='exclusivegateway4'>
        <branch condition='TIA with high-risk features or minor stroke' id='branch8'>
          <activity role='ED medical staff' action='Order brain CT scan' objects='patient, CT scanner' id='activity14'/>
          <activity role='ED medical staff' action='Initiate antiplatelet therapy (if not contraindicated) as soon as possible and no later than 24 hours after symptom onset' objects='patient, antiplatelet agents' id='activity15'/>
          <exclusiveGateway id='exclusivegateway5'>
            <branch condition='Immediate access to specialized TIA/minor stroke clinic' id='branch9'>
              <activity role='ED medical staff' action='Provide immediate care at specialized TIA/minor stroke clinic' objects='patient, clinic resources' id='activity16'/>
            </branch>
            <branch condition='Delay to specialized clinic poses risk' id='branch10'>
              <activity role='ED medical staff' action='Perform evaluation in appropriately resourced ED and consider inpatient evaluation and management for stroke prevention' objects='patient, ED resources' id='activity17'/>
            </branch>
            <branch condition='Medical attention sought >48 hours from symptom onset' id='branch11'>
              <activity role='ED medical staff' action='Refer for evaluation at specialized TIA/minor stroke clinic or outpatient clinic with stroke-prevention services within 24 hours of initial presentation' objects='patient, referral' id='activity18'/>
            </branch>
          </exclusiveGateway>
        </branch>
        <branch condition='TIA without high-risk features' id='branch12'>
          <activity role='ED medical staff' action='Order brain CT scan' objects='patient, CT scanner' id='activity19'/>
          <activity role='ED medical staff' action='Initiate antiplatelet therapy (if not contraindicated) as soon as possible and no later than 24 hours after initial presentation' objects='patient, antiplatelet agents' id='activity20'/>
          <activity role='ED medical staff' action='Refer to outpatient clinic with stroke-prevention services for comprehensive evaluation and management within 1 month of symptom onset' objects='patient, referral' id='activity21'/>
        </branch>
      </exclusiveGateway>
      <exclusiveGateway id='exclusivegateway8'>
        <branch condition='Patient fails swallowing screening' id='branch18'>
          <activity role='ED medical staff' action='Offer rectal or gastric ASA as necessary' objects='patient, rectal/gastric ASA' id='activity26'/>
        </branch>
        <branch condition='Patient passes swallowing screening' id='branch19'>
          <exclusiveGateway id='exclusivegateway7'>
            <branch condition='ASA selected' id='branch14'>
              <activity role='ED medical staff' action='Administer ASA 160 mg loading dose, then enteric coated ASA (81–325 mg) daily (most on 81 mg/day)' objects='patient, ASA' id='activity22'/>
            </branch>
            <branch condition='Clopidogrel selected' id='branch15'>
              <activity role='ED medical staff' action='Administer clopidogrel 300 mg loading dose, then 75 mg/day' objects='patient, clopidogrel' id='activity23'/>
            </branch>
            <branch condition='Extended-release dipyridamole/ASA selected' id='branch16'>
              <activity role='ED medical staff' action='Administer extended-release dipyridamole 200 mg/ASA 25 mg bid (load with ASA 160 mg first)' objects='patient, dipyridamole, ASA' id='activity24'/>
            </branch>
          </exclusiveGateway>
        </branch>
      </exclusiveGateway>
      <exclusiveGateway id='exclusivegateway6'>
        <branch condition='Patient already on antiplatelet agent or contraindication' id='branch17'>
          <activity role='ED medical staff' action='Continue or adjust antiplatelet therapy as appropriate' objects='patient, medication history' id='activity25'/>
        </branch>
        <branch condition='Patient not on antiplatelet agent and no contraindication' id='branch20'>
          <activity role='ED medical staff' action='Proceed with antiplatelet therapy as per protocol' objects='patient, antiplatelet agents' id='activity28'/>
        </branch>
      </exclusiveGateway>
    </branch>
  </exclusiveGateway>
  <activity role='ED medical staff' action='Provide cross-continuum secondary prevention assessments and therapies (per Modules 5 and 10)' objects='patient' id='activity27'/>
</process>
