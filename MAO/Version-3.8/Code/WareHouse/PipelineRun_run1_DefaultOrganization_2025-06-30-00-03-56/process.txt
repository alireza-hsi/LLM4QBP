<process>
  <activity role='triage nurse or physician' action='Perform ABC (airway, breathing, circulation) assessment' objects='patient' id='activity1'/>
  <activity role='clinician' action='Perform neurological examination to determine focal neurological deficits and assess stroke severity using standardized stroke scale (NIHSS or CNS)' objects='patient' id='activity2'/>
  <exclusiveGateway id='exclusivegateway-eligibility-advanced-imaging'>
    <branch condition='Patient is clinically eligible for acute ischemic stroke treatments' id='branch-eligibility-1'>
      <parallelGateway id='parallelgateway-imaging-advanced'>
        <branch id='branch-imaging-1'>
          <activity role='radiology staff' action='Obtain non-contrast CT or MRI' objects='patient' id='activity3-1'/>
        </branch>
        <branch id='branch-imaging-2'>
          <activity role='radiology staff' action='Obtain CTA' objects='patient' id='activity3-2'/>
        </branch>
        <branch id='branch-imaging-3'>
          <activity role='radiology staff' action='Perform advanced CT imaging including CT and multiphase CTA or Perfusion CT' objects='patient' id='activity3-3'/>
        </branch>
      </parallelGateway>
    </branch>
    <branch condition='Patient not eligible for acute ischemic stroke treatments' id='branch-eligibility-2'>
      <parallelGateway id='parallelgateway-imaging-standard'>
        <branch id='branch-imaging-4'>
          <activity role='radiology staff' action='Obtain non-contrast CT or MRI' objects='patient' id='activity3-4'/>
        </branch>
        <branch id='branch-imaging-5'>
          <activity role='radiology staff' action='Obtain CTA' objects='patient' id='activity3-5'/>
        </branch>
      </parallelGateway>
    </branch>
  </exclusiveGateway>
  <activity role='stroke physician or radiologist' action='Interpret brain imaging immediately' objects='CT/CTA/MRI results' id='activity4'/>
  <parallelGateway id='parallelgateway-ecg-cxr-lab'>
    <branch id='branch-ecg'>
      <activity role='clinician' action='Order and perform ECG to detect atrial fibrillation and other acute arrhythmias (should not delay acute treatment)' objects='patient' id='activity5'/>
    </branch>
    <branch id='branch-cxr'>
      <activity role='clinician' action='Order chest x-ray if indicated (should not delay assessment for thrombolysis)' objects='patient' id='activity6'/>
    </branch>
    <branch id='branch-lab'>
      <activity role='lab staff' action='Obtain blood work (CBC, electrolytes, creatinine, glucose, INR, partial thromboplastin time, and troponin test if clinically indicated)' objects='patient blood samples' id='activity7'/>
    </branch>
  </parallelGateway>
  <exclusiveGateway id='exclusivegateway-main'>
    <branch condition='Patient requires dysphagia assessment and early management' id='branch-main-1'>
      <activity role='nurse or clinician' action='Make all patients with stroke NPO initially' objects='patient' id='activity9'/>
      <activity role='nurse or clinician' action='Perform bedside swallowing screening' objects='patient' id='activity10'/>
      <activity role='nurse or clinician' action='Decide on oral intake based on screening result' objects='patient' id='activity11'/>
      <exclusiveGateway id='exclusivegateway-dysphagia-alertness'>
        <branch condition='Patient not alert within first 24 hours' id='branch-dysphagia-1'>
          <activity role='nurse or clinician' action='Monitor closely and screen swallowing ability when clinically appropriate' objects='patient' id='activity12'/>
        </branch>
        <branch condition='Patient alert within first 24 hours' id='branch-dysphagia-2'>
          <activity role='nurse or clinician' action='Proceed with standard swallowing assessment protocol' objects='patient' id='activity13'/>
        </branch>
      </exclusiveGateway>
      <exclusiveGateway id='exclusivegateway-dysphagia-features'>
        <branch condition='Patient presents with features indicating dysphagia or pulmonary aspiration' id='branch-dysphagia-3'>
          <activity role='speech-language pathologist or trained specialist' action='Perform full clinical assessment of swallowing ability and advise on required consistency of diet and fluids' objects='patient' id='activity14'/>
        </branch>
        <branch condition='No features of dysphagia or aspiration' id='branch-dysphagia-4'>
          <activity role='nurse or clinician' action='Continue monitoring swallowing status as per protocol' objects='patient' id='activity15'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='Patient presents beyond 48 hours with TIA or ischemic stroke' id='branch-main-3'>
      <exclusiveGateway id='exclusivegateway-tia-ischemic'>
        <branch condition='Patient presents beyond 48 hours with TIA or ischemic stroke' id='branch-tia-1'>
          <activity role='radiology staff' action='Perform vascular imaging of brain and neck arteries as soon as possible' objects='patient' id='activity17'/>
        </branch>
        <branch condition='Patient with ischemic stroke within 48 hours or not applicable' id='branch-tia-2'>
          <activity role='clinician' action='No additional vascular imaging required at this stage' objects='patient' id='activity18'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='Patient has intracerebral hemorrhage (ICH)' id='branch-main-4'>
      <exclusiveGateway id='exclusivegateway-ich'>
        <branch condition='Patient has intracerebral hemorrhage (ICH)' id='branch-ich-1'>
          <activity role='stroke physician' action='Treat ICH as medical emergency; promptly recognize and evaluate immediately' objects='patient' id='activity19'/>
          <activity role='clinician' action='Evaluate for anticoagulant therapy, measure platelet count, PTT, and INR' objects='patient' id='activity20'/>
          <exclusiveGateway id='exclusivegateway-ich-cta'>
            <branch condition='Select ICH patients as indicated' id='branch-ich-cta-1'>
              <activity role='radiology staff' action='Perform CTA or MRA as soon as possible' objects='patient' id='activity21'/>
            </branch>
            <branch condition='Not indicated for CTA/MRA' id='branch-ich-cta-2'>
              <activity role='clinician' action='Continue standard ICH management' objects='patient' id='activity22'/>
            </branch>
          </exclusiveGateway>
        </branch>
        <branch condition='Patient does not have ICH' id='branch-ich-2'>
          <activity role='clinician' action='Continue standard stroke pathway' objects='patient' id='activity23'/>
        </branch>
      </exclusiveGateway>
    </branch>
  </exclusiveGateway>
  <activity role='clinician' action='Provide appropriate cross-continuum secondary prevention assessments and therapies (see Modules 5 and 10) to all patients, whether admitted or discharged from ED' objects='patient' id='activity16'/>
</process>
