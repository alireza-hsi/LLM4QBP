<process>
  <activity role='ED clinician' action='Screen/assess patient for diabetes, dental issues, cognitive impairment, and depression upon ED presentation' objects='patient, screening tools, ED records' id='activity1-1'/>
  <activity role='ED clinician' action='Forward screening/assessment results to TIA clinic for follow-up' objects='patient, screening results, TIA clinic records' id='activity1-2'/>
  <exclusiveGateway id='exclusivegateway1'>
    <branch condition='Meets criteria for acute inpatient admission (per recommendation 2.4.2)' id='branch1'>
      <activity role='hospital administration' action='Admit patient to specialized, geographically defined stroke unit (co-located beds, 75% stroke occupancy) AND ensure dedicated interprofessional stroke team with expertise in stroke care (nursing, physiotherapy, occupational therapy, speech-language pathology, medicine, social work, clinical nutrition) is available 7 days/week AND ensure use of standardized, validated assessment tools (preferably electronic, linked to EHR)' objects='patient, stroke unit beds, staffing, assessment tools, EHR' id='activity2-merged'/>
      <parallelGateway id='parallelgateway1'>
        <branch id='branch2'>
          <activity role='physiotherapist, occupational therapist, speech-language pathologist' action='Within 48 hours of admission, use standardized, valid assessment tools to evaluate stroke-related impairments and functional status, and formulate a management plan based on assessment' objects='patient, assessment tools, assessment results' id='activity3-merged'/>
        </branch>
        <branch id='branch3'>
          <activity role='clinical team' action='Mobilize patient early (between 24-48 hours of stroke onset) unless contraindicated; avoid frequent out-of-bed activity within first 24 hours unless clinically justified' objects='patient, mobilization protocols' id='activity4'/>
        </branch>
        <branch id='branch4'>
          <activity role='clinical team' action='Complete AlphaFIM assessment on or by day 3 after admission (admission day is day 1)' objects='patient, AlphaFIM tool' id='activity5'/>
        </branch>
        <branch id='branch5'>
          <activity role='clinical team' action='Assess all stroke patients for risk of developing venous thromboembolism' objects='patient, risk assessment tool' id='activity6'/>
        </branch>
        <branch id='branch6'>
          <activity role='clinical team' action='Encourage early mobilization and adequate hydration for all acute stroke patients to help prevent venous thromboembolism' objects='patient, hydration plan' id='activity7'/>
        </branch>
        <branch id='branch7'>
          <parallelGateway id='parallelgateway2'>
            <branch id='branch7-1'>
              <activity role='clinical team' action='Assess nutrition status within first 48 hours of admission using valid screening tool' objects='patient, screening tool' id='activity8-1'/>
            </branch>
            <branch id='branch7-2'>
              <activity role='clinical team' action='Assess hydration status within first 48 hours of admission using valid screening tool' objects='patient, screening tool' id='activity8-2'/>
            </branch>
          </parallelGateway>
          <exclusiveGateway id='exclusivegateway2'>
            <branch condition='Nutritional concerns, hydration deficits, dysphagia, or other comorbidities present' id='branch8'>
              <activity role='clinical team' action='Refer patient to dietitian within 7 days of admission for recommendations and consideration of enteral nutrition support if unable to meet nutritional/fluid requirements' objects='patient, referral form' id='activity9'/>
            </branch>
            <branch condition='No nutritional concerns, hydration deficits, dysphagia, or comorbidities' id='branch9'>
              <activity role='clinical team' action='Continue standard nutrition and hydration monitoring' objects='patient' id='activity10'/>
            </branch>
          </exclusiveGateway>
        </branch>
      </parallelGateway>
      <activity role='clinical team' action='Initiate cross-continuum secondary prevention assessments and therapies (per Modules 5 and 10), regardless of discharge or in-hospital rehabilitation plan' objects='patient, prevention protocols' id='activity11'/>
    </branch>
    <branch condition='Does not meet criteria for acute inpatient admission (TIA or minor nondisabling stroke)' id='branch10'>
      <activity role='clinical team' action='Proceed with alternative management pathway (not covered in this module)' objects='patient' id='activity12'/>
    </branch>
  </exclusiveGateway>
</process>
