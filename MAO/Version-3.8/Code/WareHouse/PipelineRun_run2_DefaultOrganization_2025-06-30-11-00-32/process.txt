<process>
  <activity role='ED medical staff' action='Obtain patient history' objects='patient' id='activity1-1'/>
  <activity role='ED medical staff' action='Perform physical examination' objects='patient' id='activity1-2'/>
  <activity role='ED medical staff' action='Order initial investigations' objects='patient' id='activity1-3'/>
  <activity role='ED medical staff' action='Measure blood glucose' objects='patient' id='activity2'/>
  <exclusiveGateway id='exclusivegateway1'>
    <branch condition='Hypoglycemia detected' id='branch3'>
      <activity role='ED medical staff' action='Correct hypoglycemia immediately' objects='patient' id='activity4'/>
    </branch>
    <branch condition='First random glucose > 11 mmol/L' id='branch4'>
      <activity role='ED medical staff' action='Repeat blood glucose measurement' objects='patient' id='activity5'/>
    </branch>
    <branch condition='No hypoglycemia and glucose ≤ 11 mmol/L' id='branch5'>
      <activity role='ED medical staff' action='Continue standard care' objects='patient' id='activity6'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED medical staff' action='Evaluate body temperature as part of routine vital signs every 4 hours for first 48 hours' objects='patient' id='activity3'/>
  <exclusiveGateway id='exclusivegateway2'>
    <branch condition='Temperature > 37.5°C' id='branch6'>
      <activity role='ED medical staff' action='Initiate temperature-reducing measures' objects='patient' id='activity7'/>
      <activity role='ED medical staff' action='Investigate potential infection' objects='patient' id='activity8'/>
      <activity role='ED medical staff' action='Initiate antipyretic and antimicrobial therapy as required' objects='patient' id='activity9'/>
    </branch>
    <branch condition='Temperature ≤ 37.5°C' id='branch7'>
      <activity role='ED medical staff' action='Continue routine monitoring' objects='patient' id='activity10'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED medical staff' action='Assess for fluctuating or crescendo hemibody motor weakness, sensory loss, or speech difficulty' objects='patient' id='activity12'/>
  <exclusiveGateway id='exclusivegateway3'>
    <branch condition='Fluctuating/crescendo symptoms present' id='branch8'>
      <activity role='ED medical staff' action='Evaluate for possible hospital admission' objects='patient' id='activity13'/>
    </branch>
    <branch condition='No fluctuating/crescendo symptoms' id='branch9'>
      <activity role='ED medical staff' action='Proceed to risk stratification and management pathway' objects='patient' id='activity14'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED medical staff' action='Determine TIA/minor stroke risk features and timing of presentation' objects='patient' id='activity15'/>
  <exclusiveGateway id='exclusivegateway4'>
    <branch condition='TIA with high-risk features or minor stroke, presentation ≤ 48 hours from symptom onset' id='branch10'>
      <activity role='ED medical staff' action='Order brain CT scan' objects='patient' id='activity16-1'/>
      <activity role='ED medical staff' action='Review CT scan for intracranial hemorrhage' objects='patient, CT scan' id='activity16-2'/>
      <exclusiveGateway id='exclusivegateway5'>
        <branch condition='No intracranial hemorrhage' id='branch11'>
          <activity role='ED medical staff' action='Initiate antiplatelet therapy' objects='patient' id='activity16-3'/>
          <exclusiveGateway id='exclusivegateway6'>
            <branch condition='Immediate care available at specialized TIA/minor stroke clinic' id='branch12'>
              <activity role='ED medical staff' action='Refer patient to specialized TIA/minor stroke clinic for rapid evaluation and treatment' objects='patient' id='activity17'/>
            </branch>
            <branch condition='Delay to specialized clinic poses risk' id='branch13'>
              <activity role='ED medical staff' action='Perform evaluation and initial management in ED' objects='patient' id='activity18-1'/>
              <activity role='ED medical staff' action='Consider inpatient evaluation for stroke prevention' objects='patient' id='activity18-2'/>
            </branch>
          </exclusiveGateway>
        </branch>
        <branch condition='Intracranial hemorrhage present' id='branch14'>
          <activity role='ED medical staff' action='Manage according to hemorrhagic stroke protocol' objects='patient' id='activity19'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='Medical attention sought > 48 hours from symptom onset' id='branch15'>
      <activity role='ED medical staff' action='Refer for evaluation at specialized TIA/minor stroke clinic or outpatient clinic with stroke-prevention services within 24 hours of initial presentation' objects='patient' id='activity20'/>
    </branch>
    <branch condition='TIA without high-risk features, presentation ≤ 48 hours from symptom onset' id='branch16'>
      <activity role='ED medical staff' action='Order brain CT scan' objects='patient' id='activity21-1'/>
      <activity role='ED medical staff' action='Review CT scan for intracranial hemorrhage' objects='patient, CT scan' id='activity21-2'/>
      <exclusiveGateway id='exclusivegateway7'>
        <branch condition='No intracranial hemorrhage' id='branch17'>
          <activity role='ED medical staff' action='Initiate antiplatelet therapy' objects='patient' id='activity21-3'/>
          <activity role='ED medical staff' action='Refer to outpatient clinic with stroke-prevention services for comprehensive evaluation and management within 1 month of symptom onset' objects='patient' id='activity22'/>
        </branch>
        <branch condition='Intracranial hemorrhage present' id='branch18'>
          <activity role='ED medical staff' action='Manage according to hemorrhagic stroke protocol' objects='patient' id='activity23'/>
        </branch>
      </exclusiveGateway>
    </branch>
  </exclusiveGateway>
  <activity role='ED medical staff' action='Check if patient is on antiplatelet agent at time of presentation' objects='patient' id='activity24'/>
  <exclusiveGateway id='exclusivegateway8'>
    <branch condition='Not on antiplatelet agent' id='branch19'>
      <activity role='ED medical staff' action='Select appropriate antiplatelet agent' objects='patient, antiplatelet options' id='activity25-1'/>
      <exclusiveGateway id='exclusivegateway9'>
        <branch condition='Patient fails swallowing screening' id='branch21'>
          <activity role='ED medical staff' action='Offer rectal or gastric ASA as necessary' objects='patient, rectal/gastric ASA' id='activity27'/>
          <activity role='ED medical staff' action='Administer loading dose' objects='patient, selected antiplatelet' id='activity25-2'/>
          <activity role='ED medical staff' action='Prescribe maintenance dose' objects='patient, selected antiplatelet' id='activity25-3'/>
        </branch>
        <branch condition='Patient passes swallowing screening' id='branch22'>
          <activity role='ED medical staff' action='Administer oral antiplatelet agent' objects='patient, selected antiplatelet' id='activity28'/>
          <activity role='ED medical staff' action='Administer loading dose' objects='patient, selected antiplatelet' id='activity25-2b'/>
          <activity role='ED medical staff' action='Prescribe maintenance dose' objects='patient, selected antiplatelet' id='activity25-3b'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch condition='Already on antiplatelet agent' id='branch20'>
      <activity role='ED medical staff' action='Continue current antiplatelet therapy as appropriate' objects='patient' id='activity26'/>
    </branch>
  </exclusiveGateway>
  <activity role='ED medical staff' action='Provide cross-continuum secondary prevention assessment and therapies (refer to Modules 5 and 10)' objects='patient' id='activity11'/>
</process>
