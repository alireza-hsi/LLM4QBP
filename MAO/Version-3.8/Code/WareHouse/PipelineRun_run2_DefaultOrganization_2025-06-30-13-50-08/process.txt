  <parallelGateway id='parallelgateway1'>

      <exclusiveGateway id='exclusivegateway1'>

          <activity role='stroke physician or radiologist' action='Interpret advanced imaging results and communicate findings to clinical team' objects='imaging results' id='activity10'/>
        </branch>
        <branch condition='Patient is not eligible for acute ischemic stroke treatments' id='branch5'>
          <activity role='clinical staff' action='Proceed with standard imaging interpretation' objects='patient' id='activity11'/>
        </branch>

    </branch>
    <branch id='branch2a'>
      <activity role='clinical staff' action='Order and perform ECG to detect atrial fibrillation and other acute arrhythmias (do not delay acute treatment)' objects='patient' id='activity12'/>

      <activity role='clinical staff' action='Order and perform blood work (CBC, electrolytes, creatinine, glucose, INR, partial thromboplastin time, troponin if indicated)' objects='patient' id='activity14_modified'/>
    </branch>
    <branch id='branch3a'>
      <activity role='clinical staff' action='Make all patients with stroke NPO initially' objects='patient' id='activity15'/>
      <activity role='clinical staff' action='Select appropriate bedside swallowing screening protocol' objects='patient' id='activity16'/>
      <activity role='clinical staff' action='Perform bedside swallowing screening' objects='patient' id='activity17'/>
      <exclusiveGateway id='exclusivegateway3'>
        <branch condition='Patient is not alert within first 24 hours' id='branch6'>
          <activity role='clinical staff' action='Monitor closely and screen swallowing ability when clinically appropriate' objects='patient' id='activity18'/>
        </branch>
        <branch condition='Patient is alert within first 24 hours' id='branch7'>
          <activity role='clinical staff' action='Continue standard care' objects='patient' id='activity19'/>
        </branch>
      </exclusiveGateway>
      <exclusiveGateway id='exclusivegateway4'>
        <branch condition='Patient presents with features indicating dysphagia or pulmonary aspiration' id='branch8'>

        </branch>
        <branch condition='No features of dysphagia or aspiration' id='branch9'>
          <activity role='clinical staff' action='Continue monitoring and reassess as needed' objects='patient' id='activity21'/>
        </branch>
      </exclusiveGateway>
    </branch>
    <branch id='branch4a'>
      <activity role='clinical staff' action='Provide cross-continuum secondary prevention assessments and therapies (per Modules 5 and 10) to all patients, whether admitted or discharged' objects='patient' id='activity22'/>
    </branch>
    <branch id='branch5a'>
      <exclusiveGateway id='exclusivegateway5'>
        <branch condition='Patient presents beyond 48 hours with TIA or ischemic stroke' id='branch10'>
          <activity role='clinical staff' action='Order vascular imaging of brain and neck arteries as soon as possible' objects='patient' id='activity23'/>
        </branch>
        <branch condition='Patient does not present beyond 48 hours with TIA or ischemic stroke' id='branch11'>
          <activity role='clinical staff' action='No additional vascular imaging required at this stage' objects='patient' id='activity24'/>

            <branch condition='No further vascular imaging required' id='branch14'>

            </branch>
          </exclusiveGateway>
        </branch>
        <branch condition='Patient does not have ICH' id='branch15'>
          <activity role='clinical staff' action='Continue with appropriate stroke pathway' objects='patient' id='activity29'/>
        </branch>
      </exclusiveGateway>
    </branch>
  </parallelGateway>
